454  // calculate brake_roll and pitch angles to counter-act velocity
455  poshold_update_brake_angle_from_velocity(poshold.brake_roll, vel_right);
456  poshold_update_brake_angle_from_velocity(poshold.brake_pitch, -vel_fw);
457 
458  // run loiter controller
459  wp_nav->update_loiter(ekfGndSpdLimit, ekfNavVelGainScaler);
460 
461  // calculate final roll and pitch output by mixing loiter and brake controls
462  poshold.roll = poshold_mix_controls(brake_to_loiter_mix, poshold.brake_roll + poshold.wind_comp_roll, wp_nav->get_roll());
463  poshold.pitch = poshold_mix_controls(brake_to_loiter_mix, poshold.brake_pitch + poshold.wind_comp_pitch, wp_nav->get_pitch());
464 
465  // check for pilot input
466  if (!is_zero(target_roll) || !is_zero(target_pitch)) {
467  // if roll input switch to pilot override for roll
468  if (!is_zero(target_roll)) {
469  // init transition to pilot override
470  poshold_roll_controller_to_pilot_override();
471  // switch pitch-mode to brake (but ready to go back to loiter anytime)
472  // no need to reset poshold.brake_pitch here as wind comp has not been updated since last brake_pitch computation
473  poshold.pitch_mode = POSHOLD_BRAKE_READY_TO_LOITER;
474  }
475  // if pitch input switch to pilot override for pitch
476  if (!is_zero(target_pitch)) {
477  // init transition to pilot override
478  poshold_pitch_controller_to_pilot_override();
479  if (is_zero(target_roll)) {
480  // switch roll-mode to brake (but ready to go back to loiter anytime)
481  // no need to reset poshold.brake_roll here as wind comp has not been updated since last brake_roll computation
482  poshold.roll_mode = POSHOLD_BRAKE_READY_TO_LOITER;
483  }
484  }
485  }
486  break;
487 
488  case POSHOLD_LOITER:
489  // run loiter controller
490  wp_nav->update_loiter(ekfGndSpdLimit, ekfNavVelGainScaler);
491 
492  // set roll angle based on loiter controller outputs
493  poshold.roll = wp_nav->get_roll();
494  poshold.pitch = wp_nav->get_pitch();
495 
496  // update wind compensation estimate
497  poshold_update_wind_comp_estimate();
498 
499  // check for pilot input
500  if (!is_zero(target_roll) || !is_zero(target_pitch)) {
501  // if roll input switch to pilot override for roll
502  if (!is_zero(target_roll)) {
503  // init transition to pilot override
504  poshold_roll_controller_to_pilot_override();
505  // switch pitch-mode to brake (but ready to go back to loiter anytime)
506  poshold.pitch_mode = POSHOLD_BRAKE_READY_TO_LOITER;
507  // reset brake_pitch because wind_comp is now different and should give the compensation of the whole previous loiter angle
508  poshold.brake_pitch = 0;
509  }
510  // if pitch input switch to pilot override for pitch
511  if (!is_zero(target_pitch)) {
512  // init transition to pilot override
513  poshold_pitch_controller_to_pilot_override();
514  // if roll not overriden switch roll-mode to brake (but be ready to go back to loiter any time)
515  if (is_zero(target_roll)) {
516  poshold.roll_mode = POSHOLD_BRAKE_READY_TO_LOITER;
517  poshold.brake_roll = 0;
518  }
519  }
520  }
521  break;
522 
523  default:
524  // do nothing for uncombined roll and pitch modes
525  break;
526  }
527  }
528 
529  // constrain target pitch/roll angles
530  poshold.roll = constrain_int16(poshold.roll, -aparm.angle_max, aparm.angle_max);
531  poshold.pitch = constrain_int16(poshold.pitch, -aparm.angle_max, aparm.angle_max);
532 
533  // update attitude controller targets
534  attitude_control->input_euler_angle_roll_pitch_euler_rate_yaw(poshold.roll, poshold.pitch, target_yaw_rate, get_smoothing_gain());
535 
536  // adjust climb rate using rangefinder
537  if (rangefinder_alt_ok()) {
538  // if rangefinder is ok, use surface tracking
539  target_climb_rate = get_surface_tracking_climb_rate(target_climb_rate, pos_control->get_alt_target(), G_Dt);
540  }
541 
542  // get avoidance adjusted climb rate
543  target_climb_rate = get_avoidance_adjusted_climbrate(target_climb_rate);
544 
545  // update altitude target and call position controller
546  pos_control->set_alt_target_from_climb_rate_ff(target_climb_rate, G_Dt, false);
547  pos_control->add_takeoff_climb_rate(takeoff_climb_rate, G_Dt);
548  pos_control->update_z_controller();
549  }

552 // poshold_update_pilot_lean_angle - update the pilot's filtered lean angle with the latest raw input received
553 void Copter::poshold_update_pilot_lean_angle(float &lean_angle_filtered, float &lean_angle_raw)
554 {
555  // if raw input is large or reversing the vehicle's lean angle immediately set the fitlered angle to the new raw angle
556  if ((lean_angle_filtered > 0 && lean_angle_raw < 0) || (lean_angle_filtered < 0 && lean_angle_raw > 0) || (fabsf(lean_angle_raw) > POSHOLD_STICK_RELEASE_SMOOTH_ANGLE)) {
557  lean_angle_filtered = lean_angle_raw;
558  } else {
559  // lean_angle_raw must be pulling lean_angle_filtered towards zero, smooth the decrease
560  if (lean_angle_filtered > 0) {
561  // reduce the filtered lean angle at 5% or the brake rate (whichever is faster).
562  lean_angle_filtered -= MAX((float)lean_angle_filtered * POSHOLD_SMOOTH_RATE_FACTOR, MAX(1, g.poshold_brake_rate/LOOP_RATE_FACTOR));
563  // do not let the filtered angle fall below the pilot's input lean angle.
564  // the above line pulls the filtered angle down and the below line acts as a catch
565  lean_angle_filtered = MAX(lean_angle_filtered, lean_angle_raw);
566  }else{
567  lean_angle_filtered += MAX(-(float)lean_angle_filtered * POSHOLD_SMOOTH_RATE_FACTOR, MAX(1, g.poshold_brake_rate/LOOP_RATE_FACTOR));
568  lean_angle_filtered = MIN(lean_angle_filtered, lean_angle_raw);
569  }
570  }
571 }

574 // mix_ratio of 1 = use first_control completely, 0 = use second_control completely, 0.5 = mix evenly
575 int16_t Copter::poshold_mix_controls(float mix_ratio, int16_t first_control, int16_t second_control)

581 // poshold_update_brake_angle_from_velocity - updates the brake_angle based on the vehicle's velocity and brake_gain
582 // brake_angle is slewed with the wpnav.poshold_brake_rate and constrained by the wpnav.poshold_braking_angle_max
583 // velocity is assumed to be in the same direction as lean angle so for pitch you should provide the velocity backwards (i.e. -ve forward velocity)
584 void Copter::poshold_update_brake_angle_from_velocity(int16_t &brake_angle, float velocity)
585 {
586  float lean_angle;
587  int16_t brake_rate = g.poshold_brake_rate;
588 
589  brake_rate /= 4;
590  if (brake_rate <= 0) {
591  brake_rate = 1;
592  }
593 
594  // calculate velocity-only based lean angle
595  if (velocity >= 0) {
596  lean_angle = -poshold.brake_gain * velocity * (1.0f+500.0f/(velocity+60.0f));
597  } else {
598  lean_angle = -poshold.brake_gain * velocity * (1.0f+500.0f/(-velocity+60.0f));
599  }
600 
601  // do not let lean_angle be too far from brake_angle
602  brake_angle = constrain_int16((int16_t)lean_angle, brake_angle - brake_rate, brake_angle + brake_rate);
603 
604  // constrain final brake_angle
605  brake_angle = constrain_int16(brake_angle, -g.poshold_brake_angle_max, g.poshold_brake_angle_max);
606 }

608 // poshold_update_wind_comp_estimate - updates wind compensation estimate
609 // should be called at the maximum loop rate when loiter is engaged
610 void Copter::poshold_update_wind_comp_estimate()
611 {
612  // check wind estimate start has not been delayed
613  if (poshold.wind_comp_start_timer > 0) {
614  poshold.wind_comp_start_timer--;
615  return;
616  }
617 
618  // check horizontal velocity is low
619  if (inertial_nav.get_velocity_xy() > POSHOLD_WIND_COMP_ESTIMATE_SPEED_MAX) {
620  return;
621  }
622 
623  // get position controller accel target
624  // To-Do: clean this up by using accessor in loiter controller (or move entire PosHold controller to a library shared with loiter)
625  const Vector3f& accel_target = pos_control->get_accel_target();
626 
627  // update wind compensation in earth-frame lean angles
628  if (is_zero(poshold.wind_comp_ef.x)) {
629  // if wind compensation has not been initialised set it immediately to the pos controller's desired accel in north direction
630  poshold.wind_comp_ef.x = accel_target.x;
631  } else {
632  // low pass filter the position controller's lean angle output
633  poshold.wind_comp_ef.x = (1.0f-TC_WIND_COMP)*poshold.wind_comp_ef.x + TC_WIND_COMP*accel_target.x;
634  }
635  if (is_zero(poshold.wind_comp_ef.y)) {
636  // if wind compensation has not been initialised set it immediately to the pos controller's desired accel in north direction
637  poshold.wind_comp_ef.y = accel_target.y;
638  } else {
639  // low pass filter the position controller's lean angle output
640  poshold.wind_comp_ef.y = (1.0f-TC_WIND_COMP)*poshold.wind_comp_ef.y + TC_WIND_COMP*accel_target.y;
641  }
642 }

644 // poshold_get_wind_comp_lean_angles - retrieve wind compensation angles in body frame roll and pitch angles
645 // should be called at the maximum loop rate
646 void Copter::poshold_get_wind_comp_lean_angles(int16_t &roll_angle, int16_t &pitch_angle)

653  poshold.wind_comp_timer = 0;
