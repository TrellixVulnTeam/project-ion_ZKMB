221 void Copter::loop()
222 {
223  // wait for an INS sample
224  ins.wait_for_sample();
225 
226  uint32_t timer = micros();
227 
228  // check loop time
229  perf_info_check_loop_time(timer - fast_loopTimer);
230 
231  // used by PI Loops
232  G_Dt = (float)(timer - fast_loopTimer) / 1000000.0f;
233  fast_loopTimer = timer;
234 
235  // for mainloop failure monitoring
236  mainLoop_count++;
237 
238  // Execute the fast loop
239  // ---------------------
240  fast_loop();
241 
242  // tell the scheduler one tick has passed
243  scheduler.tick();
244 
245  // run all the tasks that are due to run. Note that we only
246  // have to call this once per loop, as the tasks are scheduled
247  // in multiples of the main loop tick. So if they don't run on
248  // the first call to the scheduler they won't run on a later
249  // call until scheduler.tick() is called again
250  uint32_t time_available = (timer + MAIN_LOOP_MICROS) - micros();
251  scheduler.run(time_available > MAIN_LOOP_MICROS ? 0u : time_available);
252 }

526 {
527  static uint32_t last_gps_reading[GPS_MAX_INSTANCES]; // time of last gps message
528  bool gps_updated = false;
529 
530  gps.update();
531 
532  // log after every gps message

541

543  }
544  }
545 
546  if (gps_updated) {
547  // set system time if necessary
548  set_system_time_from_GPS();

579 // update_simple_mode - rotates pilot input if we are in simple mode
580 void Copter::update_simple_mode(void)
581 {
582  float rollx, pitchx;
583 
584  // exit immediately if no new radio frame or not in simple mode
585  if (ap.simple_mode == 0 || !ap.new_radio_frame) {
586  return;
587  }
588 
589  // mark radio frame as consumed
590  ap.new_radio_frame = false;
591 
592  if (ap.simple_mode == 1) {
593  // rotate roll, pitch input by -initial simple heading (i.e. north facing)
594  rollx = channel_roll->get_control_in()*simple_cos_yaw - channel_pitch->get_control_in()*simple_sin_yaw;
595  pitchx = channel_roll->get_control_in()*simple_sin_yaw + channel_pitch->get_control_in()*simple_cos_yaw;
596  }else{
597  // rotate roll, pitch input by -super simple heading (reverse of heading to home)
598  rollx = channel_roll->get_control_in()*super_simple_cos_yaw - channel_pitch->get_control_in()*super_simple_sin_yaw;
599  pitchx = channel_roll->get_control_in()*super_simple_sin_yaw + channel_pitch->get_control_in()*super_simple_cos_yaw;
600  }
601 
602  // rotate roll, pitch input from north facing to vehicle's perspective
603  channel_roll->set_control_in(rollx*ahrs.cos_yaw() + pitchx*ahrs.sin_yaw());
604  channel_pitch->set_control_in(-rollx*ahrs.sin_yaw() + pitchx*ahrs.cos_yaw());
605 }

607 // update_super_simple_bearing - adjusts simple bearing based on location
608 // should be called after home_bearing has been updated
609 void Copter::update_super_simple_bearing(bool force_update)
610 {
611  // check if we are in super simple mode and at least 10m from home
612  if(force_update || (ap.simple_mode == 2 && home_distance > SUPER_SIMPLE_RADIUS)) {
613  // check the bearing to home has changed by at least 5 degrees
614  if (labs(super_simple_last_bearing - home_bearing) > 500) {
615  super_simple_last_bearing = home_bearing;
616  float angle_rad = radians((super_simple_last_bearing+18000)/100);
617  super_simple_cos_yaw = cosf(angle_rad);
618  super_simple_sin_yaw = sinf(angle_rad);
619  }
620  }
621 }
