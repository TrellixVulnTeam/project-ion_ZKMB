423 void Copter::rtl_compute_return_target(bool terrain_following_allowed)
424 {
425  // set return target to nearest rally point or home position (Note: alt is absolute)
426 #if AC_RALLY == ENABLED
427  rtl_path.return_target = rally.calc_best_rally_or_home_location(current_loc, ahrs.get_home().alt);
428 #else
429  rtl_path.return_target = ahrs.get_home();
430 #endif
431

433  int32_t curr_alt = current_loc.alt;
434 
435  // decide if we should use terrain altitudes
436  rtl_path.terrain_used = terrain_use() && terrain_following_allowed;
437  if (rtl_path.terrain_used) {
438  // attempt to retrieve terrain alt for current location, stopping point and origin
439  int32_t origin_terr_alt, return_target_terr_alt;
440  if (!rtl_path.origin_point.get_alt_cm(Location_Class::ALT_FRAME_ABOVE_TERRAIN, origin_terr_alt) ||
441  !rtl_path.return_target.get_alt_cm(Location_Class::ALT_FRAME_ABOVE_TERRAIN, return_target_terr_alt) ||
442  !current_loc.get_alt_cm(Location_Class::ALT_FRAME_ABOVE_TERRAIN, curr_alt)) {
443  rtl_path.terrain_used = false;
444  Log_Write_Error(ERROR_SUBSYSTEM_TERRAIN, ERROR_CODE_MISSING_TERRAIN_DATA);
445  }
446  }
447 
448  // convert return-target alt (which is an absolute alt) to alt-above-home or alt-above-terrain
449  if (!rtl_path.terrain_used || !rtl_path.return_target.change_alt_frame(Location_Class::ALT_FRAME_ABOVE_TERRAIN)) {
450  if (!rtl_path.return_target.change_alt_frame(Location_Class::ALT_FRAME_ABOVE_HOME)) {
451  // this should never happen but just in case
452  rtl_path.return_target.set_alt_cm(0, Location_Class::ALT_FRAME_ABOVE_HOME);
453  }
454  rtl_path.terrain_used = false;
455  }
456 
457  // set new target altitude to return target altitude
458  // Note: this is alt-above-home or terrain-alt depending upon use_terrain
459  // Note: ignore negative altitudes which could happen if user enters negative altitude for rally point or terrain is higher at rally point compared to home
460  int32_t target_alt = MAX(rtl_path.return_target.alt, 0);
461 
462  // increase target to maximum of current altitude + climb_min and rtl altitude

468  // don't allow really shallow slopes
469  if (g.rtl_cone_slope >= RTL_MIN_CONE_SLOPE) {
470  target_alt = MAX(curr_alt, MIN(target_alt, MAX(rtl_return_dist_cm*g.rtl_cone_slope, curr_alt+RTL_ABS_MIN_CLIMB)));
471  }
472 
473  // set returned target alt to new target_alt
474  rtl_path.return_target.set_alt_cm(target_alt, rtl_path.terrain_used ? Location_Class::ALT_FRAME_ABOVE_TERRAIN : Location_Class::ALT_FRAME_ABOVE_HOME);

479  // if terrain altitudes are being used, the code below which reduces the return_target's altitude can lead to

489  }
490  }
491  }
492 #endif
493
