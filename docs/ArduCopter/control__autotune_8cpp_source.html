5 /*
6  * Init and run calls for autotune flight mode
7  *
8  * Instructions:
9  * 1) Set up one flight mode switch position to be AltHold.
10  * 2) Set the Ch7 Opt or Ch8 Opt to AutoTune to allow you to turn the auto tuning on/off with the ch7 or ch8 switch.
11  * 3) Ensure the ch7 or ch8 switch is in the LOW position.
12  * 4) Wait for a calm day and go to a large open area.
13  * 5) Take off and put the vehicle into AltHold mode at a comfortable altitude.
14  * 6) Set the ch7/ch8 switch to the HIGH position to engage auto tuning:
15  * a) You will see it twitch about 20 degrees left and right for a few minutes, then it will repeat forward and back.
16  * b) Use the roll and pitch stick at any time to reposition the copter if it drifts away (it will use the original PID gains during repositioning and between tests).
17  * When you release the sticks it will continue auto tuning where it left off.
18  * c) Move the ch7/ch8 switch into the LOW position at any time to abandon the autotuning and return to the origin PIDs.
19  * d) Make sure that you do not have any trim set on your transmitter or the autotune may not get the signal that the sticks are centered.
20  * 7) When the tune completes the vehicle will change back to the original PID gains.
21  * 8) Put the ch7/ch8 switch into the LOW position then back to the HIGH position to test the tuned PID gains.
22  * 9) Put the ch7/ch8 switch into the LOW position to fly using the original PID gains.
23  * 10) If you are happy with the autotuned PID gains, leave the ch7/ch8 switch in the HIGH position, land and disarm to save the PIDs permanently.
24  * If you DO NOT like the new PIDS, switch ch7/ch8 LOW to return to the original PIDs. The gains will not be saved when you disarm
25  *
26  * What it's doing during each "twitch":
27  * a) invokes 90 deg/sec rate request
28  * b) records maximum "forward" roll rate and bounce back rate
29  * c) when copter reaches 20 degrees or 1 second has passed, it commands level
30  * d) tries to keep max rotation rate between 80% ~ 100% of requested rate (90deg/sec) by adjusting rate P
31  * e) increases rate D until the bounce back becomes greater than 10% of requested rate (90deg/sec)
32  * f) decreases rate D until the bounce back becomes less than 10% of requested rate (90deg/sec)
33  * g) increases rate P until the max rotate rate becomes greater than the request rate (90deg/sec)
34  * h) invokes a 20deg angle request on roll or pitch
35  * i) increases stab P until the maximum angle becomes greater than 110% of the requested angle (20deg)
36  * j) decreases stab P by 25%
37  *
38  */

645  case AUTOTUNE_AXIS_ROLL:

659  if (autotune_counter >= AUTOTUNE_SUCCESS_COUNT) {

662  autotune_counter = 0;

684  }
685  break;

697  break;
698  }
699  break;
700  case AUTOTUNE_TYPE_SP_DOWN:

702  break;
703  case AUTOTUNE_TYPE_SP_UP:

705  autotune_state.tune_type = AUTOTUNE_TYPE_RD_UP;
706 
707  // advance to the next axis
708  bool autotune_complete = false;
709  switch (autotune_state.axis) {
710  case AUTOTUNE_AXIS_ROLL:

715  } else if (autotune_yaw_enabled()) {
716  autotune_state.axis = AUTOTUNE_AXIS_YAW;
717  } else {
718  autotune_complete = true;
719  }
720  break;
721  case AUTOTUNE_AXIS_PITCH:
722  tune_pitch_sp = MAX(AUTOTUNE_SP_MIN, tune_pitch_sp * AUTOTUNE_SP_BACKOFF);
723  tune_pitch_accel = MAX(AUTOTUNE_RP_ACCEL_MIN, autotune_test_accel_max * AUTOTUNE_ACCEL_RP_BACKOFF);
724  if (autotune_yaw_enabled()) {
725  autotune_state.axis = AUTOTUNE_AXIS_YAW;
726  } else {
727  autotune_complete = true;
728  }
729  break;
730  case AUTOTUNE_AXIS_YAW:
731  tune_yaw_sp = MAX(AUTOTUNE_SP_MIN, tune_yaw_sp * AUTOTUNE_SP_BACKOFF);
732  tune_yaw_accel = MAX(AUTOTUNE_Y_ACCEL_MIN, autotune_test_accel_max * AUTOTUNE_ACCEL_Y_BACKOFF);
733  autotune_complete = true;
734  break;
735  }
736 
737  // if we've just completed all axes we have successfully completed the autotune
738  // change to TESTING mode to allow user to fly with new gains
739  if (autotune_complete) {
740  autotune_state.mode = AUTOTUNE_MODE_SUCCESS;
741  autotune_update_gcs(AUTOTUNE_MESSAGE_SUCCESS);
742  Log_Write_Event(DATA_AUTOTUNE_SUCCESS);
743  AP_Notify::events.autotune_complete = 1;
744  } else {
745  AP_Notify::events.autotune_next_axis = 1;
746  }
747  break;
748  }
749  }
750 
751  // reverse direction
752  autotune_state.positive_direction = !autotune_state.positive_direction;
753 
754  if (autotune_state.axis == AUTOTUNE_AXIS_YAW) {
755  attitude_control->input_euler_angle_roll_pitch_yaw( 0.0f, 0.0f, ahrs.yaw_sensor, false, get_smoothing_gain());
756  }
757 
758  // set gains to their intra-test values (which are very close to the original gains)
759  autotune_load_intra_test_gains();
760 
761  // reset testing step
762  autotune_state.step = AUTOTUNE_STEP_WAITING_FOR_LEVEL;
763  autotune_step_start_time = millis();
764  break;
765  }
766 }

1076 // update min and max and test for end conditions
1077 void Copter::autotune_twitching_test(float measurement, float target, float &measurement_min, float &measurement_max)
1078 {
1079  // capture maximum measurement
1080  if (measurement > measurement_max) {
1081  // the measurement is continuing to increase without stopping
1082  measurement_max = measurement;
1083  measurement_min = measurement;
1084  }
1085 
1086  // capture minimum measurement after the measurement has peaked (aka "bounce back")
1087  if ((measurement < measurement_min) && (measurement_max > target * 0.5f)) {
1088  // the measurement is bouncing back
1089  measurement_min = measurement;
1090  }
1091 
1092  // calculate early stopping time based on the time it takes to get to 90%
1093  if (measurement_max < target * 0.75f) {
1094  // the measurement not reached the 90% threshold yet
1095  autotune_step_stop_time = autotune_step_start_time + (millis() - autotune_step_start_time) * 3.0f;
1096  autotune_step_stop_time = MIN(autotune_step_stop_time, autotune_step_start_time + AUTOTUNE_TESTING_STEP_TIMEOUT_MS);
1097  }
1098 
1099  if (measurement_max > target) {
1100  // the measurement has passed the target
1101  autotune_state.step = AUTOTUNE_STEP_UPDATE_GAINS;
1102  }
1103 
1104  if (measurement_max-measurement_min > measurement_max*g.autotune_aggressiveness) {
1105  // the measurement has passed 50% of the target and bounce back is larger than the threshold
1106  autotune_state.step = AUTOTUNE_STEP_UPDATE_GAINS;
1107  }
1108 
1109  if (millis() >= autotune_step_stop_time) {
1110  // we have passed the maximum stop time

1112  }

1342 // get attitude for slow position hold in autotune mode
1343 void Copter::autotune_get_poshold_attitude(float &roll_cd, float &pitch_cd, float &yaw_cd)
1344 {

1346 
1347  if (!autotune_state.use_poshold) {
1348  // we are not trying to hold position
1349  return;
1350  }
1351 
1352  // do we know where we are?
1353  if (!position_ok()) {
1354  return;
1355  }
1356 
1357  if (!autotune_state.have_position) {
1358  autotune_state.have_position = true;
1359  autotune_state.start_position = inertial_nav.get_position();
1360  }
1361 
1362  // don't go past 10 degrees, as autotune result would deteriorate too much
1363  const float angle_max_cd = 1000;
1364 
1365  // hit the 10 degree limit at 20 meters position error
1366  const float dist_limit_cm = 2000;
1367 
1368  // we only start adjusting yaw if we are more than 5m from the
1369  // target position. That corresponds to a lean angle of 2.5 degrees
1370  const float yaw_dist_limit_cm = 500;
1371

1373  pdiff.z = 0;
1374  float dist_cm = pdiff.length();
1375  if (dist_cm < 10) {
1376  // don't do anything within 10cm
1377  return;
1378  }
1379 
1380  /*
1381  very simple linear controller
1382  */
1383  float scaling = constrain_float(angle_max_cd * dist_cm / dist_limit_cm, 0, angle_max_cd);
1384  Vector2f angle_ne(pdiff.x, pdiff.y);
1385  angle_ne *= scaling / dist_cm;
1386 
1387  // rotate into body frame
1388  pitch_cd = angle_ne.x * ahrs.cos_yaw() + angle_ne.y * ahrs.sin_yaw();

1390 
1391  if (dist_cm < yaw_dist_limit_cm) {
1392  // no yaw adjustment
1393  return;
1394  }
1395 
1396  /*
1397  also point so that twitching occurs perpendicular to the wind,
1398  if we have drifted more than yaw_dist_limit_cm from the desired
1399  position. This ensures that autotune doesn't have to deal with
1400  more than 2.5 degrees of attitude on the axis it is tuning
1401  */

1403  if (autotune_state.axis == AUTOTUNE_AXIS_PITCH) {
1404  // for roll and yaw tuning we point along the wind, for pitch
1405  // we point across the wind
1406  target_yaw_cd += 9000;
1407  }

1409  if (fabsf(yaw_cd - target_yaw_cd) > 9500) {
1410  target_yaw_cd += 18000;
1411  }
1412

1414 }
