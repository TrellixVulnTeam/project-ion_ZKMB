111 float Copter::get_throttle_assist(float velz, float pilot_throttle_scaled)
112 {

114  // Only active when pilot's throttle is between 213 ~ 787
115  // Assistance is strongest when throttle is at mid, drops linearly to no assistance at 213 and 787
