75 void Copter::failsafe_gcs_check()
76 {

81  if ((!failsafe.gcs)&&(g.failsafe_gcs == FS_GCS_DISABLED || failsafe.last_heartbeat_ms == 0 || (!failsafe.rc_override_active && control_mode != GUIDED))) {
82  return;

84

88

91  // check for recovery from gcs failsafe
