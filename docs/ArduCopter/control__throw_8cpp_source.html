29  Throw_Disarmed - motors are off
30  Throw_Detecting - motors are on and we are waiting for the throw
31  Throw_Uprighting - the throw has been detected and the copter is being uprighted
32  Throw_HgtStabilise - the copter is kept level and height is stabilised about the target height
33  Throw_PosHold - the copter is kept at a constant position and height
34  */

38  // state machine entry is always from a disarmed state

40

206 bool Copter::throw_detected()
207 {
208  // Check that we have a valid navigation solution
209  nav_filter_status filt_status = inertial_nav.get_filter_status();
210  if (!filt_status.flags.attitude || !filt_status.flags.horiz_pos_abs || !filt_status.flags.vert_pos) {
211  return false;
212  }
213 
214  // Check for high speed (>500 cm/s)
215  bool high_speed = inertial_nav.get_velocity().length() > THROW_HIGH_SPEED;
216 
217  // check for upwards or downwards trajectory (airdrop) of 50cm/s
218  bool changing_height;
219  if (g2.throw_type == ThrowType_Drop) {
220  changing_height = inertial_nav.get_velocity().z < -THROW_VERTICAL_SPEED;
221  } else {
222  changing_height = inertial_nav.get_velocity().z > THROW_VERTICAL_SPEED;
223  }
224 
225  // Check the vertical acceleraton is greater than 0.25g
226  bool free_falling = ahrs.get_accel_ef().z > -0.25 * GRAVITY_MSS;
227 
228  // Check if the accel length is < 1.0g indicating that any throw action is complete and the copter has been released
229  bool no_throw_action = ins.get_accel().length() < 1.0f * GRAVITY_MSS;
230 
231  // High velocity or free-fall combined with increasing height indicate a possible air-drop or throw release
232  bool possible_throw_detected = (free_falling || high_speed) && changing_height && no_throw_action;
233 
234  // Record time and vertical velocity when we detect the possible throw
235  if (possible_throw_detected && ((AP_HAL::millis() - throw_state.free_fall_start_ms) > 500)) {
236  throw_state.free_fall_start_ms = AP_HAL::millis();
237  throw_state.free_fall_start_velz = inertial_nav.get_velocity().z;
238  }
239 
240  // Once a possible throw condition has been detected, we check for 2.5 m/s of downwards velocity change in less than 0.5 seconds to confirm
241  bool throw_condition_confirmed = ((AP_HAL::millis() - throw_state.free_fall_start_ms < 500) && ((inertial_nav.get_velocity().z - throw_state.free_fall_start_velz) < -250.0f));
242 
243  // start motors and enter the control mode if we are in continuous freefall
244  if (throw_condition_confirmed) {
245  return true;
246  } else {
247  return false;
248  }
