403 // update_auto_armed - update status of auto_armed flag
404 void Copter::update_auto_armed()
405 {
406  // disarm checks
407  if(ap.auto_armed){
408  // if motors are disarmed, auto_armed should also be false
409  if(!motors->armed()) {
410  set_auto_armed(false);
411  return;
412  }
413  // if in stabilize or acro flight mode and throttle is zero, auto-armed should become false
414  if(mode_has_manual_throttle(control_mode) && ap.throttle_zero && !failsafe.radio) {
415  set_auto_armed(false);
416  }
417 #if FRAME_CONFIG == HELI_FRAME
418  // if helicopters are on the ground, and the motor is switched off, auto-armed should be false
419  // so that rotor runup is checked again before attempting to take-off
420  if(ap.land_complete && !motors->rotor_runup_complete()) {
421  set_auto_armed(false);
422  }
423 #endif // HELI_FRAME
424  }else{
425  // arm checks
426 
427 #if FRAME_CONFIG == HELI_FRAME
428  // for tradheli if motors are armed and throttle is above zero and the motor is started, auto_armed should be true
429  if(motors->armed() && !ap.throttle_zero && motors->rotor_runup_complete()) {
430  set_auto_armed(true);
431  }
432 #else
433  // if motors are armed and throttle is above zero auto_armed should be true
434  // if motors are armed and we are in throw mode, then auto_ermed should be true
435  if(motors->armed() && (!ap.throttle_zero || control_mode == THROW)) {
436  set_auto_armed(true);
437  }
438 #endif // HELI_FRAME
439  }
