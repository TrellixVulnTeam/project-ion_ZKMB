142 // transform pilot's manual throttle input to make hover throttle mid stick
143 // used only for manual throttle modes
144 // thr_mid should be in the range 0 to 1
145 // returns throttle output 0 to 1
146 float Copter::get_pilot_desired_throttle(int16_t throttle_control, float thr_mid)
147 {
148  if (thr_mid <= 0.0f) {
149  thr_mid = motors->get_throttle_hover();
150  }
151 
152  int16_t mid_stick = channel_throttle->get_control_mid();
153  // protect against unlikely divide by zero
154  if (mid_stick <= 0) {
155  mid_stick = 500;
156  }
157 
158  // ensure reasonable throttle values
159  throttle_control = constrain_int16(throttle_control,0,1000);
160 
161  // calculate normalised throttle input
162  float throttle_in;
163  if (throttle_control < mid_stick) {
164  // below the deadband
165  throttle_in = ((float)throttle_control)*0.5f/(float)mid_stick;
166  }else if(throttle_control > mid_stick) {
167  // above the deadband
168  throttle_in = 0.5f + ((float)(throttle_control-mid_stick)) * 0.5f / (float)(1000-mid_stick);
169  }else{
170  // must be in the deadband
171  throttle_in = 0.5f;
172  }
173 
174  float expo = constrain_float(-(thr_mid-0.5)/0.375, -0.5f, 1.0f);
175  // calculate the output throttle using the given expo function
176  float throttle_out = throttle_in*(1.0f-expo) + expo*throttle_in*throttle_in*throttle_in;
177  return throttle_out;
178 }

222 // returns climb rate (in cm/s) which should be passed to the position controller
223 float Copter::get_surface_tracking_climb_rate(int16_t target_rate, float current_alt_target, float dt)
224 {
225 #if RANGEFINDER_ENABLED == ENABLED
226  static uint32_t last_call_ms = 0;
227  float distance_error;
228  float velocity_correction;
229  float current_alt = inertial_nav.get_altitude();
230

232 
233  // reset target altitude if this controller has just been engaged
234  if (now - last_call_ms > RANGEFINDER_TIMEOUT_MS) {
235  target_rangefinder_alt = rangefinder_state.alt_cm + current_alt_target - current_alt;
236  }
237  last_call_ms = now;
238 
239  // adjust rangefinder target alt if motors have not hit their limits
240  if ((target_rate<0 && !motors->limit.throttle_lower) || (target_rate>0 && !motors->limit.throttle_upper)) {
241  target_rangefinder_alt += target_rate * dt;
242  }
243 
244  /*
245  handle rangefinder glitches. When we get a rangefinder reading
246  more than RANGEFINDER_GLITCH_ALT_CM different from the current
247  rangefinder reading then we consider it a glitch and reject
248  until we get RANGEFINDER_GLITCH_NUM_SAMPLES samples in a
249  row. When that happens we reset the target altitude to the new
250  reading
251  */
252  int32_t glitch_cm = rangefinder_state.alt_cm - target_rangefinder_alt;
253  if (glitch_cm >= RANGEFINDER_GLITCH_ALT_CM) {
254  rangefinder_state.glitch_count = MAX(rangefinder_state.glitch_count+1,1);
255  } else if (glitch_cm <= -RANGEFINDER_GLITCH_ALT_CM) {
256  rangefinder_state.glitch_count = MIN(rangefinder_state.glitch_count-1,-1);
257  } else {
258  rangefinder_state.glitch_count = 0;
259  }
260  if (abs(rangefinder_state.glitch_count) >= RANGEFINDER_GLITCH_NUM_SAMPLES) {
261  // shift to the new rangefinder reading
262  target_rangefinder_alt = rangefinder_state.alt_cm;
263  rangefinder_state.glitch_count = 0;
264  }
265  if (rangefinder_state.glitch_count != 0) {
266  // we are currently glitching, just use the target rate
267  return target_rate;
268  }
269 
270  // calc desired velocity correction from target rangefinder alt vs actual rangefinder alt (remove the error already passed to Altitude controller to avoid oscillations)
271  distance_error = (target_rangefinder_alt - rangefinder_state.alt_cm) - (current_alt_target - current_alt);
272  velocity_correction = distance_error * g.rangefinder_gain;
273  velocity_correction = constrain_float(velocity_correction, -THR_SURFACE_TRACKING_VELZ_MAX, THR_SURFACE_TRACKING_VELZ_MAX);
274 
275  // return combined pilot climb rate + rate to correct rangefinder alt error
276  return (target_rate + velocity_correction);
277 #else
278  return (float)target_rate;
279 #endif
280 }
