28 {
29  uint8_t base_mode = MAV_MODE_FLAG_CUSTOM_MODE_ENABLED;
30  uint8_t system_status = ap.land_complete ? MAV_STATE_STANDBY : MAV_STATE_ACTIVE;
31  uint32_t custom_mode = control_mode;
32 
33  // set system as critical if any failsafe have triggered
34  if (failsafe.radio || failsafe.battery || failsafe.gcs || failsafe.ekf || failsafe.terrain || failsafe.adsb) {
35  system_status = MAV_STATE_CRITICAL;
36  }
37 
38  // work out the base_mode. This value is not very useful
39  // for APM, but we calculate it as best we can so a generic
40  // MAVLink enabled ground station can work out something about
41  // what the MAV is up to. The actual bit values are highly
42  // ambiguous for most of the APM flight modes. In practice, you
43  // only get useful information from the custom_mode, which maps to
44  // the APM flight mode and has a well defined meaning in the
45  // ArduPlane documentation
46  base_mode = MAV_MODE_FLAG_STABILIZE_ENABLED;
47  switch (control_mode) {
48  case AUTO:
49  case RTL:
50  case LOITER:
51  case AVOID_ADSB:
52  case GUIDED:
53  case CIRCLE:
54  case POSHOLD:
55  case BRAKE:
56  base_mode |= MAV_MODE_FLAG_GUIDED_ENABLED;
57  // note that MAV_MODE_FLAG_AUTO_ENABLED does not match what
58  // APM does in any mode, as that is defined as "system finds its own goal
59  // positions", which APM does not currently do
60  break;
61  default:
62  break;
63  }
64 
65  // all modes except INITIALISING have some form of manual
66  // override if stick mixing is enabled
67  base_mode |= MAV_MODE_FLAG_MANUAL_INPUT_ENABLED;
68 
69 #if HIL_MODE != HIL_MODE_DISABLED
70  base_mode |= MAV_MODE_FLAG_HIL_ENABLED;
71 #endif
72 
73  // we are armed if we are not initialising
74  if (motors->armed()) {
75  base_mode |= MAV_MODE_FLAG_SAFETY_ARMED;

819 void GCS_MAVLINK_Copter::handleMessage(mavlink_message_t* msg)
820 {
821  uint8_t result = MAV_RESULT_FAILED; // assume failure. Each messages id is responsible for return ACK or NAK if required
822 
823  switch (msg->msgid) {
824 
825  case MAVLINK_MSG_ID_HEARTBEAT: // MAV ID: 0
826  {
827  // We keep track of the last time we received a heartbeat from our GCS for failsafe purposes
828  if(msg->sysid != copter.g.sysid_my_gcs) break;
829  copter.failsafe.last_heartbeat_ms = AP_HAL::millis();
830  copter.pmTest1++;
831  break;
832  }
833 
834  case MAVLINK_MSG_ID_SET_MODE: // MAV ID: 11
835  {
836 #ifdef DISALLOW_GCS_MODE_CHANGE_DURING_RC_FAILSAFE
837  if (!copter.failsafe.radio) {
838  handle_set_mode(msg, FUNCTOR_BIND(&copter, &Copter::gcs_set_mode, bool, uint8_t));
839  } else {
840  // don't allow mode changes while in radio failsafe
841  mavlink_msg_command_ack_send_buf(msg, chan, MAVLINK_MSG_ID_SET_MODE, MAV_RESULT_FAILED);
842  }
843 #else
844  handle_set_mode(msg, FUNCTOR_BIND(&copter, &Copter::gcs_set_mode, bool, uint8_t));
845 #endif
846  break;
847  }
848 
849  case MAVLINK_MSG_ID_PARAM_REQUEST_LIST: // MAV ID: 21
850  {
851  // if we have not yet initialised (including allocating the motors
852  // object) we drop this request. That prevents the GCS from getting
853  // a confusing parameter count during bootup
854  if (!copter.ap.initialised) {
855  break;
856  }
857 
858  // mark the firmware version in the tlog
859  send_text(MAV_SEVERITY_INFO, FIRMWARE_STRING);
860

862  send_text(MAV_SEVERITY_INFO, "PX4: " PX4_GIT_VERSION " NuttX: " NUTTX_GIT_VERSION);
863 #endif
864  GCS_MAVLINK::send_statustext_chan(MAV_SEVERITY_INFO, chan, "Frame: %s", copter.get_frame_string());
865  handle_param_request_list(msg);
866  break;
867  }
868 
869  case MAVLINK_MSG_ID_PARAM_SET: // 23
870  {
871  handle_param_set(msg, &copter.DataFlash);
872  break;
873  }
874 
875  case MAVLINK_MSG_ID_PARAM_VALUE:

877 #if MOUNT == ENABLED
878  copter.camera_mount.handle_param_value(msg);
879 #endif
880  break;
881  }

883  case MAVLINK_MSG_ID_MISSION_WRITE_PARTIAL_LIST: // MAV ID: 38
884  {
885  handle_mission_write_partial_list(copter.mission, msg);
886  break;
887  }
888 
889  // GCS has sent us a mission item, store to EEPROM
890  case MAVLINK_MSG_ID_MISSION_ITEM: // MAV ID: 39
891  {
892  if (handle_mission_item(msg, copter.mission)) {
893  copter.DataFlash.Log_Write_EntireMission(copter.mission);
894  }
895  break;
896  }
897

899  {
900  if (handle_mission_item(msg, copter.mission)) {
901  copter.DataFlash.Log_Write_EntireMission(copter.mission);
902  }
903  break;
904  }
905

907  case MAVLINK_MSG_ID_MISSION_REQUEST_INT:
908  case MAVLINK_MSG_ID_MISSION_REQUEST: // MAV ID: 40, 51
909  {
910  handle_mission_request(copter.mission, msg);
911  break;
912  }
913

915  {
916  handle_mission_set_current(copter.mission, msg);
917  break;
918  }

922  {
923  handle_mission_request_list(copter.mission, msg);
924  break;
925  }
926

928  // individual commands will then be sent from the GCS using the MAVLINK_MSG_ID_MISSION_ITEM message

973  // and RC PWM values.

994  }
995

1000  mavlink_msg_command_int_decode(msg, &packet);

1004  // param1 : /* Region of interest mode (not used)*/
1005  // param2 : /* MISSION index/ target ID (not used)*/
1006  // param3 : /* ROI index (not used)*/
1007  // param4 : /* empty */
1008  // x : lat
1009  // y : lon
1010  // z : alt
1011  // sanity check location
1012  if (!check_latlng(packet.x, packet.y)) {
1013  break;
1014  }
1015  Location roi_loc;
1016  roi_loc.lat = packet.x;
1017  roi_loc.lng = packet.y;
1018  roi_loc.alt = (int32_t)(packet.z * 100.0f);
1019  copter.set_auto_yaw_roi(roi_loc);
1020  result = MAV_RESULT_ACCEPTED;
1021  break;
1022  }
1023  default:
1024  result = MAV_RESULT_UNSUPPORTED;
1025  break;
1026  }
1027 
1028  // send ACK or NAK
1029  mavlink_msg_command_ack_send_buf(msg, chan, packet.command, result);
