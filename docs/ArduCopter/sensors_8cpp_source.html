315 
316

318  control_sensors_enabled = control_sensors_present & (~MAV_SYS_STATUS_SENSOR_Z_ALTITUDE_CONTROL &
319  ~MAV_SYS_STATUS_SENSOR_XY_POSITION_CONTROL &

323 
324  switch (control_mode) {
325  case AUTO:
326  case AVOID_ADSB:
