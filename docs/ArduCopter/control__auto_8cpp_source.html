102 void Copter::auto_takeoff_start(const Location& dest_loc)
103 {
104  auto_mode = Auto_TakeOff;
105 
106  // convert location to class
107  Location_Class dest(dest_loc);
108 
109  // set horizontal target
110  dest.lat = current_loc.lat;
111  dest.lng = current_loc.lng;
112 
113  // get altitude target
114  int32_t alt_target;
115  if (!dest.get_alt_cm(Location_Class::ALT_FRAME_ABOVE_HOME, alt_target)) {
116  // this failure could only happen if take-off alt was specified as an alt-above terrain and we have no terrain data
117  Log_Write_Error(ERROR_SUBSYSTEM_TERRAIN, ERROR_CODE_MISSING_TERRAIN_DATA);
118  // fall back to altitude above current altitude
119  alt_target = current_loc.alt + dest.alt;
120  }
121 
122  // sanity check target
123  if (alt_target < current_loc.alt) {
124  dest.set_alt_cm(current_loc.alt, Location_Class::ALT_FRAME_ABOVE_HOME);
125  }
126  // Note: if taking off from below home this could cause a climb to an unexpectedly high altitude
127  if (alt_target < 100) {
128  dest.set_alt_cm(100, Location_Class::ALT_FRAME_ABOVE_HOME);
129  }
130 
131  // set waypoint controller target
132  if (!wp_nav->set_wp_destination(dest)) {
133  // failure to set destination can only be because of missing terrain data
134  failsafe_terrain_on_event();
135  return;
136  }
137 
138  // initialise yaw
139  set_auto_yaw_mode(AUTO_YAW_HOLD);
140 
141  // clear i term when we're taking off
142  set_throttle_takeoff();
143

169

175  }
176

194

432 // we assume the caller has performed all required GPS_ok checks
433 void Copter::auto_circle_movetoedge_start(const Location_Class &circle_center, float radius_m)
434 {
435  // convert location to vector from ekf origin
436  Vector3f circle_center_neu;
437  if (!circle_center.get_vector_from_origin_NEU(circle_center_neu)) {
438  // default to current position and log error
439  circle_center_neu = inertial_nav.get_position();
440  Log_Write_Error(ERROR_SUBSYSTEM_NAVIGATION, ERROR_CODE_FAILED_CIRCLE_INIT);
441  }
442  circle_nav->set_center(circle_center_neu);
443 
444  // set circle radius
445  if (!is_zero(radius_m)) {
446  circle_nav->set_radius(radius_m * 100.0f);
447  }
448 
449  // check our distance from edge of circle
450  Vector3f circle_edge_neu;
451  circle_nav->get_closest_point_on_circle(circle_edge_neu);
452  float dist_to_edge = (inertial_nav.get_position() - circle_edge_neu).length();
453 
454  // if more than 3m then fly to edge
455  if (dist_to_edge > 300.0f) {
456  // set the state to move to the edge of the circle
457  auto_mode = Auto_CircleMoveToEdge;
458 
459  // convert circle_edge_neu to Location_Class
460  Location_Class circle_edge(circle_edge_neu);
461 
462  // convert altitude to same as command
463  circle_edge.set_alt_cm(circle_center.alt, circle_center.get_alt_frame());
464 
465  // initialise wpnav to move to edge of circle
466  if (!wp_nav->set_wp_destination(circle_edge)) {
467  // failure to set destination can only be because of missing terrain data
468  failsafe_terrain_on_event();
469  }
470 
471  // if we are outside the circle, point at the edge, otherwise hold yaw
472  const Vector3f &curr_pos = inertial_nav.get_position();
473  float dist_to_center = norm(circle_center_neu.x - curr_pos.x, circle_center_neu.y - curr_pos.y);
474  if (dist_to_center > circle_nav->get_radius() && dist_to_center > 500) {
475  set_auto_yaw_mode(get_default_auto_yaw_mode(false));
476  } else {
477  // vehicle is within circle so hold yaw to avoid spinning as we move to edge of circle
478  set_auto_yaw_mode(AUTO_YAW_HOLD);
479  }
480  } else {
481  auto_circle_start();
482  }

653 // set_auto_yaw_look_at_heading - sets the yaw look at heading for auto mode
654 void Copter::set_auto_yaw_look_at_heading(float angle_deg, float turn_rate_dps, int8_t direction, uint8_t relative_angle)
655 {
656  // get current yaw target
657  int32_t curr_yaw_target = attitude_control->get_att_target_euler_cd().z;
658 
659  // get final angle, 1 = Relative, 0 = Absolute
660  if (relative_angle == 0) {
661  // absolute angle
662  yaw_look_at_heading = wrap_360_cd(angle_deg * 100);
663  } else {
664  // relative angle
665  if (direction < 0) {
666  angle_deg = -angle_deg;
667  }
668  yaw_look_at_heading = wrap_360_cd((angle_deg * 100) + curr_yaw_target);
669  }
670 
671  // get turn speed
672  if (is_zero(turn_rate_dps)) {
673  // default to regular auto slew rate
674  yaw_look_at_heading_slew = AUTO_YAW_SLEW_RATE;
675  }else{
676  int32_t turn_rate = (wrap_180_cd(yaw_look_at_heading - curr_yaw_target) / 100) / turn_rate_dps;
677  yaw_look_at_heading_slew = constrain_int32(turn_rate, 1, 360); // deg / sec
678  }
679 
680  // set yaw mode
681  set_auto_yaw_mode(AUTO_YAW_LOOK_AT_HEADING);
682 
683  // TO-DO: restore support for clockwise and counter clockwise rotation held in cmd.content.yaw.direction. 1 = clockwise, -1 = counterclockwise
684 }
