23  if (rc5_in < 1231) switch_position = 0;
24  else if (rc5_in < 1361) switch_position = 1;

26  else if (rc5_in < 1621) switch_position = 3;

28  else switch_position = 5;
29 
30  // store time that switch last moved

33  }
34 
35  // debounce switch

37  bool sufficient_time_elapsed = tnow_ms - control_switch_state.last_edge_time_ms > CONTROL_SWITCH_DEBOUNCE_TIME_MS;
38  bool failsafe_disengaged = !failsafe.radio && failsafe.radio_counter == 0;
39 
40  if (control_switch_changed && sufficient_time_elapsed && failsafe_disengaged) {
41  // set flight mode and simple mode setting
42  if (set_mode((control_mode_t)flight_modes[switch_position].get(), MODE_REASON_TX_COMMAND)) {
43  // play a tone
44  if (control_switch_state.debounced_switch_position != -1) {
45  // alert user to mode change failure (except if autopilot is just starting up)
46  if (ap.initialised) {
47  AP_Notify::events.user_mode_change = 1;
48  }
49  }
50

52  // if none of the Aux Switches are set to Simple or Super Simple Mode then
53  // set Simple Mode using stored parameters from EEPROM

58  }
59  }

63  AP_Notify::events.user_mode_change_failed = 1;
64  }
65
