566  if (!motors->armed() || !ap.auto_armed || !motors->get_interlock() || (ap.land_complete && guided_angle_state.climb_rate_cms <= 0.0f)) {
567 #if FRAME_CONFIG == HELI_FRAME // Helicopters always stabilize roll/pitch/yaw
568  // call attitude controller
569  attitude_control->set_yaw_target_to_current_heading();
570  attitude_control->input_euler_angle_roll_pitch_euler_rate_yaw(0.0f, 0.0f, 0.0f, get_smoothing_gain());
571  attitude_control->set_throttle_out(0.0f,false,g.throttle_filt);
572 #else
573  motors->set_desired_spool_state(AP_Motors::DESIRED_SPIN_WHEN_ARMED);
574  // multicopters do not stabilize roll/pitch/yaw when disarmed
575  attitude_control->set_throttle_out_unstabilized(0.0f,true,g.throttle_filt);
576 #endif
577  pos_control->relax_alt_hold_controllers(0.0f);
578  return;
579  }
580 
581  // constrain desired lean angles
582  float roll_in = guided_angle_state.roll_cd;
583  float pitch_in = guided_angle_state.pitch_cd;
584  float total_in = norm(roll_in, pitch_in);
585  float angle_max = MIN(attitude_control->get_althold_lean_angle_max(), aparm.angle_max);
586  if (total_in > angle_max) {
587  float ratio = angle_max / total_in;
588  roll_in *= ratio;
589  pitch_in *= ratio;
590  }
591 
592  // wrap yaw request
593  float yaw_in = wrap_180_cd(guided_angle_state.yaw_cd);
594  float yaw_rate_in = wrap_180_cd(guided_angle_state.yaw_rate_cds);
595 
596  // constrain climb rate
597  float climb_rate_cms = constrain_float(guided_angle_state.climb_rate_cms, -fabsf(wp_nav->get_speed_down()), wp_nav->get_speed_up());
598 
599  // get avoidance adjusted climb rate
600  climb_rate_cms = get_avoidance_adjusted_climbrate(climb_rate_cms);
601 
602  // check for timeout - set lean angles and climb rate to zero if no updates received for 3 seconds
603  uint32_t tnow = millis();
604  if (tnow - guided_angle_state.update_time_ms > GUIDED_ATTITUDE_TIMEOUT_MS) {
605  roll_in = 0.0f;
606  pitch_in = 0.0f;
607  climb_rate_cms = 0.0f;
608  yaw_rate_in = 0.0f;
609  }
610 
611  // set motors to full range
612  motors->set_desired_spool_state(AP_Motors::DESIRED_THROTTLE_UNLIMITED);
613 
614  // call attitude controller
615  if (guided_angle_state.use_yaw_rate) {
616  attitude_control->input_euler_angle_roll_pitch_euler_rate_yaw(roll_in, pitch_in, yaw_rate_in, get_smoothing_gain());
617  } else {
618  attitude_control->input_euler_angle_roll_pitch_yaw(roll_in, pitch_in, yaw_in, true, get_smoothing_gain());
619  }
620 
621  // call position controller
622  pos_control->set_alt_target_from_climb_rate_ff(climb_rate_cms, G_Dt, false);
623  pos_control->update_z_controller();
624 }

626 // helper function to update position controller's desired velocity while respecting acceleration limits
627 void Copter::guided_set_desired_velocity_with_accel_and_fence_limits(const Vector3f& vel_des)
628 {
629  // get current desired velocity
630  Vector3f curr_vel_des = pos_control->get_desired_velocity();
631 
632  // exit immediately if already equal
633  if (curr_vel_des == vel_des) {
634  return;
635  }
636 
637  // get change in desired velocity
638  Vector3f vel_delta = vel_des - curr_vel_des;
639 
640  // limit xy change
641  float vel_delta_xy = safe_sqrt(sq(vel_delta.x)+sq(vel_delta.y));
642  float vel_delta_xy_max = G_Dt * pos_control->get_accel_xy();
643  float ratio_xy = 1.0f;
644  if (!is_zero(vel_delta_xy) && (vel_delta_xy > vel_delta_xy_max)) {
645  ratio_xy = vel_delta_xy_max / vel_delta_xy;
646  }
647  curr_vel_des.x += (vel_delta.x * ratio_xy);
648  curr_vel_des.y += (vel_delta.y * ratio_xy);
649 
650  // limit z change
651  float vel_delta_z_max = G_Dt * pos_control->get_accel_z();
652  curr_vel_des.z += constrain_float(vel_delta.z, -vel_delta_z_max, vel_delta_z_max);
653 
654 #if AC_AVOID_ENABLED
655  // limit the velocity to prevent fence violations
656  avoid.adjust_velocity(pos_control->get_pos_xy_kP(), pos_control->get_accel_xy(), curr_vel_des);
657 #endif
658 
659  // update position controller with new target
660  pos_control->set_desired_velocity(curr_vel_des);
661 }
