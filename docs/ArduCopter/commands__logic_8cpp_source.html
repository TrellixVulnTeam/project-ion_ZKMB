710 // verify_payload_place - returns true if placing has been completed
711 bool Copter::verify_payload_place()
712 {
713  const uint16_t hover_throttle_calibrate_time = 2000; // milliseconds
714  const uint16_t descend_throttle_calibrate_time = 2000; // milliseconds
715  const float hover_throttle_placed_fraction = 0.7; // i.e. if throttle is less than 70% of hover we have placed
716  const float descent_throttle_placed_fraction = 0.9; // i.e. if throttle is less than 90% of descent throttle we have placed
717  const uint16_t placed_time = 500; // how long we have to be below a throttle threshold before considering placed
718 
719  const float current_throttle_level = motors->get_throttle();
720  const uint32_t now = AP_HAL::millis();
721 
722  // if we discover we've landed then immediately release the load:
723  if (ap.land_complete) {
724  switch (nav_payload_place.state) {
725  case PayloadPlaceStateType_FlyToLocation:
726  case PayloadPlaceStateType_Calibrating_Hover_Start:
727  case PayloadPlaceStateType_Calibrating_Hover:
728  case PayloadPlaceStateType_Descending_Start:
729  case PayloadPlaceStateType_Descending:
730  gcs_send_text_fmt(MAV_SEVERITY_INFO, "NAV_PLACE: landed");
731  nav_payload_place.state = PayloadPlaceStateType_Releasing_Start;
732  break;
733  case PayloadPlaceStateType_Releasing_Start:
734  case PayloadPlaceStateType_Releasing:
735  case PayloadPlaceStateType_Released:
736  case PayloadPlaceStateType_Ascending_Start:
737  case PayloadPlaceStateType_Ascending:
738  case PayloadPlaceStateType_Done:
739  break;
740  }
741  }
742 
743  switch (nav_payload_place.state) {
744  case PayloadPlaceStateType_FlyToLocation:
745  if (!wp_nav->reached_wp_destination()) {
746  return false;
747  }
748  // we're there; set loiter target
749  auto_payload_place_start(wp_nav->get_wp_destination());
750  nav_payload_place.state = PayloadPlaceStateType_Calibrating_Hover_Start;
751  // no break
752  case PayloadPlaceStateType_Calibrating_Hover_Start:
753  // hover for 1 second to get an idea of what our hover
754  // throttle looks like
755  debug("Calibrate start");
756  nav_payload_place.hover_start_timestamp = now;
757  nav_payload_place.state = PayloadPlaceStateType_Calibrating_Hover;
758  // no break
759  case PayloadPlaceStateType_Calibrating_Hover: {
760  if (now - nav_payload_place.hover_start_timestamp < hover_throttle_calibrate_time) {
761  // still calibrating...
762  debug("Calibrate Timer: %d", now - nav_payload_place.hover_start_timestamp);
763  return false;
764  }
765  // we have a valid calibration. Hopefully.
766  nav_payload_place.hover_throttle_level = current_throttle_level;
767  const float hover_throttle_delta = fabsf(nav_payload_place.hover_throttle_level - motors->get_throttle_hover());
768  gcs_send_text_fmt(MAV_SEVERITY_INFO, "hover throttle delta: %f", static_cast<double>(hover_throttle_delta));
769  nav_payload_place.state = PayloadPlaceStateType_Descending_Start;
770  }
771  // no break
772  case PayloadPlaceStateType_Descending_Start:
773  nav_payload_place.descend_start_timestamp = now;
774  nav_payload_place.descend_start_altitude = inertial_nav.get_altitude();
775  nav_payload_place.descend_throttle_level = 0;
776  nav_payload_place.state = PayloadPlaceStateType_Descending;
777  // no break
778  case PayloadPlaceStateType_Descending:
779  // make sure we don't descend too far:
780  debug("descended: %f cm (%f cm max)", (nav_payload_place.descend_start_altitude - inertial_nav.get_altitude()), nav_payload_place.descend_max);
781  if (!is_zero(nav_payload_place.descend_max) &&
782  nav_payload_place.descend_start_altitude - inertial_nav.get_altitude() > nav_payload_place.descend_max) {
783  nav_payload_place.state = PayloadPlaceStateType_Ascending;
784  gcs_send_text_fmt(MAV_SEVERITY_WARNING, "Reached maximum descent");
785  return false; // we'll do any cleanups required next time through the loop
786  }
787  // see if we've been descending long enough to calibrate a descend-throttle-level:
788  if (is_zero(nav_payload_place.descend_throttle_level) &&
789  now - nav_payload_place.descend_start_timestamp > descend_throttle_calibrate_time) {
790  nav_payload_place.descend_throttle_level = current_throttle_level;
791  }
792  // watch the throttle to determine whether the load has been placed
793  // debug("hover ratio: %f descend ratio: %f\n", current_throttle_level/nav_payload_place.hover_throttle_level, ((nav_payload_place.descend_throttle_level == 0) ? -1.0f : current_throttle_level/nav_payload_place.descend_throttle_level));
794  if (current_throttle_level/nav_payload_place.hover_throttle_level > hover_throttle_placed_fraction &&
795  (is_zero(nav_payload_place.descend_throttle_level) ||
796  current_throttle_level/nav_payload_place.descend_throttle_level > descent_throttle_placed_fraction)) {
797  // throttle is above both threshold ratios (or above hover threshold ration and descent threshold ratio not yet valid)
798  nav_payload_place.place_start_timestamp = 0;
799  return false;
800  }
801  if (nav_payload_place.place_start_timestamp == 0) {
802  // we've only just now hit the correct throttle level
803  nav_payload_place.place_start_timestamp = now;
804  return false;
805  } else if (now - nav_payload_place.place_start_timestamp < placed_time) {
806  // keep going down....
807  debug("Place Timer: %d", now - nav_payload_place.place_start_timestamp);
808  return false;
809  }
810  nav_payload_place.state = PayloadPlaceStateType_Releasing_Start;
811  // no break
812  case PayloadPlaceStateType_Releasing_Start:
813 #if GRIPPER_ENABLED == ENABLED
814  if (g2.gripper.valid()) {
815  gcs_send_text_fmt(MAV_SEVERITY_INFO, "Releasing the gripper");
816  g2.gripper.release();
817  } else {
818  gcs_send_text_fmt(MAV_SEVERITY_INFO, "Gripper not valid");
819  nav_payload_place.state = PayloadPlaceStateType_Ascending_Start;
820  break;
821  }
822 #else
823  gcs_send_text_fmt(MAV_SEVERITY_INFO, "Gripper code disabled");
824 #endif
825  nav_payload_place.state = PayloadPlaceStateType_Releasing;
826  // no break
827  case PayloadPlaceStateType_Releasing:
828 #if GRIPPER_ENABLED == ENABLED
829  if (g2.gripper.valid() && !g2.gripper.released()) {
830  return false;
831  }
832 #endif
833  nav_payload_place.state = PayloadPlaceStateType_Released;
834  // no break
835  case PayloadPlaceStateType_Released: {
836  nav_payload_place.state = PayloadPlaceStateType_Ascending_Start;
837  }
838  // no break
839  case PayloadPlaceStateType_Ascending_Start: {
840  Location_Class target_loc = inertial_nav.get_position();
841  target_loc.alt = nav_payload_place.descend_start_altitude;
842  auto_wp_start(target_loc);
843  nav_payload_place.state = PayloadPlaceStateType_Ascending;
844  }
845  // no break
846  case PayloadPlaceStateType_Ascending:
847  if (!wp_nav->reached_wp_destination()) {
848  return false;
849  }
850  nav_payload_place.state = PayloadPlaceStateType_Done;
851  // no break
852  case PayloadPlaceStateType_Done:
853  return true;
854  default:
855  // this should never happen
856  // TO-DO: log an error
857  return true;
858  }
859  // should never get here
860  return true;
861 }
