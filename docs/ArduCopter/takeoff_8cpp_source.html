84 // pilot_climb_rate is both an input and an output
85 // takeoff_climb_rate is only an output
86 // has side-effect of turning takeoff off when timeout as expired
87 void Copter::takeoff_get_climb_rates(float& pilot_climb_rate, float& takeoff_climb_rate)
88 {
89  // return pilot_climb_rate if take-off inactive
90  if (!takeoff_state.running) {
91  takeoff_climb_rate = 0.0f;
92  return;
93  }
94 
95  // acceleration of 50cm/s/s
96  static const float takeoff_accel = 50.0f;
97  float takeoff_minspeed = MIN(50.0f,takeoff_state.max_speed);
98  float time_elapsed = (millis()-takeoff_state.start_ms)*1.0e-3f;
99  float speed = MIN(time_elapsed*takeoff_accel+takeoff_minspeed, takeoff_state.max_speed);
100 
101  float time_to_max_speed = (takeoff_state.max_speed-takeoff_minspeed)/takeoff_accel;
102  float height_gained;
103  if (time_elapsed <= time_to_max_speed) {
104  height_gained = 0.5f*takeoff_accel*sq(time_elapsed) + takeoff_minspeed*time_elapsed;
105  } else {
106  height_gained = 0.5f*takeoff_accel*sq(time_to_max_speed) + takeoff_minspeed*time_to_max_speed +
107  (time_elapsed-time_to_max_speed)*takeoff_state.max_speed;
108  }
109 
110  // check if the takeoff is over
111  if (height_gained >= takeoff_state.alt_delta) {
112  takeoff_stop();
113  }
114 
115  // if takeoff climb rate is zero return
116  if (speed <= 0.0f) {
117  takeoff_climb_rate = 0.0f;
118  return;
119  }
120 
121  // default take-off climb rate to maximum speed
122  takeoff_climb_rate = speed;
123 
124  // if pilot's commands descent
125  if (pilot_climb_rate < 0.0f) {
126  // if overall climb rate is still positive, move to take-off climb rate
127  if (takeoff_climb_rate + pilot_climb_rate > 0.0f) {
128  takeoff_climb_rate = takeoff_climb_rate + pilot_climb_rate;
129  pilot_climb_rate = 0;
130  } else {
131  // if overall is negative, move to pilot climb rate
132  pilot_climb_rate = pilot_climb_rate + takeoff_climb_rate;
133  takeoff_climb_rate = 0.0f;
134  }
135  } else { // pilot commands climb
136  // pilot climb rate is zero until it surpasses the take-off climb rate
137  if (pilot_climb_rate > takeoff_climb_rate) {
138  pilot_climb_rate = pilot_climb_rate - takeoff_climb_rate;
139  } else {
140  pilot_climb_rate = 0.0f;
141  }
142  }
143 }
