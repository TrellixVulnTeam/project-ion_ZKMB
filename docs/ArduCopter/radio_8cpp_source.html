135 #define FS_COUNTER 3 // radio failsafe kicks in after 3 consecutive throttle values below failsafe_throttle_value
136 void Copter::set_throttle_and_failsafe(uint16_t throttle_pwm)
137 {
138  // if failsafe not enabled pass through throttle and exit
139  if(g.failsafe_throttle == FS_THR_DISABLED) {
140  channel_throttle->set_pwm(throttle_pwm);
141  return;
142  }
143 
144  //check for low throttle value
145  if (throttle_pwm < (uint16_t)g.failsafe_throttle_value) {
146 
147  // if we are already in failsafe or motors not armed pass through throttle and exit
148  if (failsafe.radio || !(ap.rc_receiver_present || motors->armed())) {
149  channel_throttle->set_pwm(throttle_pwm);
150  return;
151  }
152 
153  // check for 3 low throttle values
154  // Note: we do not pass through the low throttle until 3 low throttle values are received
155  failsafe.radio_counter++;
156  if( failsafe.radio_counter >= FS_COUNTER ) {
157  failsafe.radio_counter = FS_COUNTER; // check to ensure we don't overflow the counter
158  set_failsafe_radio(true);
159  channel_throttle->set_pwm(throttle_pwm); // pass through failsafe throttle
160  }
161  }else{
162  // we have a good throttle so reduce failsafe counter
163  failsafe.radio_counter--;
164  if( failsafe.radio_counter <= 0 ) {
165  failsafe.radio_counter = 0; // check to ensure we don't underflow the counter
166 
167  // disengage failsafe after three (nearly) consecutive valid throttle values
168  if (failsafe.radio) {
169  set_failsafe_radio(false);
170  }
171  }
172  // pass through throttle
173  channel_throttle->set_pwm(throttle_pwm);
174  }
175 }

179 // throttle_zero is used to determine if the pilot intends to shut down the motors

182 void Copter::set_throttle_zero_flag(int16_t throttle_control)
183 {
184  static uint32_t last_nonzero_throttle_ms = 0;
185  uint32_t tnow_ms = millis();
186 
187  // if not using throttle interlock and non-zero throttle and not E-stopped,

189  // and we are flying. Immediately set as non-zero

191  last_nonzero_throttle_ms = tnow_ms;
192  ap.throttle_zero = false;
