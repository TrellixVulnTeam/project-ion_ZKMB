619  if (display_failure) {
620  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Compass calibration running");
621  }
622  return false;
623  }
624 
625  //check if compass has calibrated and requires reboot
626  if (_compass.compass_cal_requires_reboot()) {
627  if (display_failure) {
628  gcs_send_text(MAV_SEVERITY_CRITICAL, "PreArm: Compass calibrated requires reboot");
629  }
630  return false;
631  }
632 
633  control_mode_t control_mode = copter.control_mode;
634 
635  // always check if the current mode allows arming
636  if (!copter.mode_allows_arming(control_mode, arming_from_gcs)) {
637  if (display_failure) {
638  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Mode not armable");
639  }
640  return false;
641  }
642 
643  // always check gps
644  if (!pre_arm_gps_checks(display_failure)) {
645  return false;
646  }
647 
648  // always check motors
649  if (!motor_checks(display_failure)) {
650  return false;
651  }
652 
653  // if we are using motor interlock switch and it's enabled, fail to arm
654  // skip check in Throw mode which takes control of the motor interlock
655  if (copter.ap.using_interlock && copter.motors->get_interlock()) {
656  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Motor Interlock Enabled");
657  return false;
658  }
659 
660  // if we are not using Emergency Stop switch option, force Estop false to ensure motors
661  // can run normally
662  if (!copter.check_if_auxsw_mode_used(AUXSW_MOTOR_ESTOP)){
663  copter.set_motor_emergency_stop(false);
664  // if we are using motor Estop switch, it must not be in Estop position
665  } else if (copter.check_if_auxsw_mode_used(AUXSW_MOTOR_ESTOP) && copter.ap.motor_emergency_stop){
666  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Motor Emergency Stopped");
667  return false;
668  }
669 
670  // succeed if arming checks are disabled
671  if (checks_to_perform == ARMING_CHECK_NONE) {
672  return true;
673  }
674 
675  // baro checks
676  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_BARO)) {
677  // baro health check
678  if (!barometer.all_healthy()) {
679  if (display_failure) {
680  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Barometer not healthy");
681  }
682  return false;
683  }
684  // Check baro & inav alt are within 1m if EKF is operating in an absolute position mode.
685  // Do not check if intending to operate in a ground relative height mode as EKF will output a ground relative height
686  // that may differ from the baro height due to baro drift.
687  nav_filter_status filt_status = _inav.get_filter_status();
688  bool using_baro_ref = (!filt_status.flags.pred_horiz_pos_rel && filt_status.flags.pred_horiz_pos_abs);
689  if (using_baro_ref && (fabsf(_inav.get_altitude() - copter.baro_alt) > PREARM_MAX_ALT_DISPARITY_CM)) {
690  if (display_failure) {
691  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Altitude disparity");
692  }
693  return false;
694  }
695  }
696 
697  #if AC_FENCE == ENABLED
698  // check vehicle is within fence
699  const char *fail_msg = nullptr;
700  if (!copter.fence.pre_arm_check(fail_msg)) {
701  if (display_failure && fail_msg != nullptr) {
702  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "Arm: %s", fail_msg);
703  }
704  return false;
705  }
706  #endif
707 
708  // check lean angle
709  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_INS)) {
710  if (degrees(acosf(ahrs.cos_roll()*ahrs.cos_pitch()))*100.0f > copter.aparm.angle_max) {
711  if (display_failure) {
712  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Leaning");
713  }
714  return false;
715  }
716  }
717 
718  // check battery voltage
719  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_VOLTAGE)) {
720  if (copter.failsafe.battery || (!copter.ap.usb_connected && copter.battery.exhausted(copter.g.fs_batt_voltage, copter.g.fs_batt_mah))) {
721  if (display_failure) {
722  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Check Battery");
723  }
724  return false;
725  }
726  }
727 
728  // check for missing terrain data
729  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_PARAMETERS)) {
730  if (!pre_arm_terrain_check(display_failure)) {
731  return false;
732  }
733  }
734 
735  // check adsb
736  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_PARAMETERS)) {
737  if (copter.failsafe.adsb) {
738  if (display_failure) {
739  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: ADSB threat detected");
740  }
741  return false;
742  }
743  }
744 
745  // check throttle
746  if ((checks_to_perform == ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_RC)) {
747  // check throttle is not too low - must be above failsafe throttle
748  if (copter.g.failsafe_throttle != FS_THR_DISABLED && copter.channel_throttle->get_radio_in() < copter.g.failsafe_throttle_value) {
749  if (display_failure) {
750  #if FRAME_CONFIG == HELI_FRAME
751  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Collective below Failsafe");
752  #else
753  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Throttle below Failsafe");
754  #endif
755  }
756  return false;
757  }
758 
759  // check throttle is not too high - skips checks if arming from GCS in Guided
760  if (!(arming_from_gcs && (control_mode == GUIDED || control_mode == GUIDED_NOGPS))) {
761  // above top of deadband is too always high
762  if (copter.get_pilot_desired_climb_rate(copter.channel_throttle->get_control_in()) > 0.0f) {
763  if (display_failure) {
764  #if FRAME_CONFIG == HELI_FRAME
765  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Collective too high");
766  #else
767  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Throttle too high");
768  #endif
769  }
770  return false;
771  }
772  // in manual modes throttle must be at zero
773  if ((copter.mode_has_manual_throttle(control_mode) || control_mode == DRIFT) && copter.channel_throttle->get_control_in() > 0) {
774  if (display_failure) {
775  #if FRAME_CONFIG == HELI_FRAME
776  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Collective too high");
777  #else
778  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Throttle too high");
779  #endif
780  }
781  return false;
782  }
783  }
784  }
785 
786  // check if safety switch has been pushed
787  if (hal.util->safety_switch_state() == AP_HAL::Util::SAFETY_DISARMED) {
788  if (display_failure) {
789  gcs_send_text(MAV_SEVERITY_CRITICAL,"Arm: Safety Switch");
790  }
791  return false;
792  }
793 
794  // if we've gotten this far all is ok
795  return true;
796 }
