166 void Copter::land_run_vertical_control(bool pause_descent)
167 {
168  bool navigating = pos_control->is_active_xy();
169 
170 #if PRECISION_LANDING == ENABLED
171  bool doing_precision_landing = !ap.land_repo_active && precland.target_acquired() && navigating;
172 #else
173  bool doing_precision_landing = false;
174 #endif
175 
176  // compute desired velocity
177  const float precland_acceptable_error = 15.0f;
178  const float precland_min_descent_speed = 10.0f;
179  int32_t alt_above_ground = land_get_alt_above_ground();
180 
181  float cmb_rate = 0;
182  if (!pause_descent) {
183  float max_land_descent_velocity;
184  if (g.land_speed_high > 0) {
185  max_land_descent_velocity = -g.land_speed_high;
186  } else {
187  max_land_descent_velocity = pos_control->get_speed_down();
188  }
189 
190  // Don't speed up for landing.
191  max_land_descent_velocity = MIN(max_land_descent_velocity, -abs(g.land_speed));
192 
193  // Compute a vertical velocity demand such that the vehicle approaches LAND_START_ALT. Without the below constraint, this would cause the vehicle to hover at LAND_START_ALT.
194  cmb_rate = AC_AttitudeControl::sqrt_controller(LAND_START_ALT-alt_above_ground, g.p_alt_hold.kP(), pos_control->get_accel_z());
195 
196  // Constrain the demanded vertical velocity so that it is between the configured maximum descent speed and the configured minimum descent speed.
197  cmb_rate = constrain_float(cmb_rate, max_land_descent_velocity, -abs(g.land_speed));
198 
199  if (doing_precision_landing && rangefinder_alt_ok() && rangefinder_state.alt_cm > 35.0f && rangefinder_state.alt_cm < 200.0f) {
200  float max_descent_speed = abs(g.land_speed)/2.0f;
201  float land_slowdown = MAX(0.0f, pos_control->get_horizontal_error()*(max_descent_speed/precland_acceptable_error));
202  cmb_rate = MIN(-precland_min_descent_speed, -max_descent_speed+land_slowdown);
203  }
204  }
205 
206  // update altitude target and call position controller
207  pos_control->set_alt_target_from_climb_rate_ff(cmb_rate, G_Dt, true);
208  pos_control->update_z_controller();
209 }
