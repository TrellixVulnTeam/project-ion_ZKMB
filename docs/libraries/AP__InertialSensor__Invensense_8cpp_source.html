356 void AP_InertialSensor_Invensense::_fifo_reset()
357 {
358  uint8_t user_ctrl = _last_stat_user_ctrl;
359  user_ctrl &= ~(BIT_USER_CTRL_FIFO_RESET | BIT_USER_CTRL_FIFO_EN);
360  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
361  _register_write(MPUREG_FIFO_EN, 0);
362  _register_write(MPUREG_USER_CTRL, user_ctrl);
363  _register_write(MPUREG_USER_CTRL, user_ctrl | BIT_USER_CTRL_FIFO_RESET);
364  _register_write(MPUREG_USER_CTRL, user_ctrl | BIT_USER_CTRL_FIFO_EN);

366  BIT_ZG_FIFO_EN | BIT_ACCEL_FIFO_EN | BIT_TEMP_FIFO_EN, true);
367  hal.scheduler->delay_microseconds(1);
368  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
369  _last_stat_user_ctrl = user_ctrl | BIT_USER_CTRL_FIFO_EN;

377 void AP_InertialSensor_Invensense::start()
378 {
379  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
380  return;
381  }
382 
383  // initially run the bus at low speed
384  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
385 
386  // only used for wake-up in accelerometer only low power mode
387  _register_write(MPUREG_PWR_MGMT_2, 0x00);
388  hal.scheduler->delay(1);
389 
390  // always use FIFO
391  _fifo_reset();
392 
393  // grab the used instances
394  enum DevTypes gdev, adev;
395  switch (_mpu_type) {
396  case Invensense_MPU9250:
397  gdev = DEVTYPE_GYR_MPU9250;
398  adev = DEVTYPE_ACC_MPU9250;
399  break;
400  case Invensense_MPU6000:
401  case Invensense_MPU6500:
402  case Invensense_ICM20608:
403  case Invensense_ICM20602:
404  default:
405  gdev = DEVTYPE_GYR_MPU6000;
406  adev = DEVTYPE_ACC_MPU6000;
407  break;
408  }
409 
410  /*
411  setup temperature sensitivity and offset. This varies
412  considerably between parts
413  */
414  switch (_mpu_type) {
415  case Invensense_MPU9250:
416  temp_zero = 21;
417  temp_sensitivity = 1.0/340;
418  break;
419 
420  case Invensense_MPU6000:
421  case Invensense_MPU6500:
422  temp_zero = 36.53;
423  temp_sensitivity = 1.0/340;
424  break;
425 
426  case Invensense_ICM20608:
427  case Invensense_ICM20602:
428  temp_zero = 25;
429  temp_sensitivity = 1.0/326.8;
430  break;
431  }
432 
433  _gyro_instance = _imu.register_gyro(1000, _dev->get_bus_id_devtype(gdev));
434  _accel_instance = _imu.register_accel(1000, _dev->get_bus_id_devtype(adev));
435 
436  // setup ODR and on-sensor filtering
437  _set_filter_register();
438 
439  // set sample rate to 1000Hz and apply a software filter
440  // In this configuration, the gyro sample rate is 8kHz
441  _register_write(MPUREG_SMPLRT_DIV, 0, true);
442  hal.scheduler->delay(1);
443 
444  // Gyro scale 2000º/s
445  _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS, true);
446  hal.scheduler->delay(1);
447 
448  // read the product ID rev c has 1/2 the sensitivity of rev d
449  uint8_t product_id = _register_read(MPUREG_PRODUCT_ID);
450 
451  if (_mpu_type == Invensense_MPU6000 &&
452  ((product_id == MPU6000ES_REV_C4) ||
453  (product_id == MPU6000ES_REV_C5) ||
454  (product_id == MPU6000_REV_C4) ||
455  (product_id == MPU6000_REV_C5))) {
456  // Accel scale 8g (4096 LSB/g)
457  // Rev C has different scaling than rev D
458  _register_write(MPUREG_ACCEL_CONFIG,1<<3, true);
459  _accel_scale = GRAVITY_MSS / 4096.f;
460  } else {
461  // Accel scale 16g (2048 LSB/g)
462  _register_write(MPUREG_ACCEL_CONFIG,3<<3, true);
463  _accel_scale = GRAVITY_MSS / 2048.f;
464  }
465  hal.scheduler->delay(1);

558 bool AP_InertialSensor_Invensense::_accumulate(uint8_t *samples, uint8_t n_samples)
559 {
560  for (uint8_t i = 0; i < n_samples; i++) {
561  const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
562  Vector3f accel, gyro;
563  bool fsync_set = false;
564 
565 #if INVENSENSE_EXT_SYNC_ENABLE
566  fsync_set = (int16_val(data, 2) & 1U) != 0;
567 #endif
568 
569  accel = Vector3f(int16_val(data, 1),
570  int16_val(data, 0),
571  -int16_val(data, 2));
572  accel *= _accel_scale;
573 
574  int16_t t2 = int16_val(data, 3);
575  if (!_check_raw_temp(t2)) {
576  debug("temp reset %d %d", _raw_temp, t2);
577  _fifo_reset();
578  return false;
579  }
580  float temp = t2 * temp_sensitivity + temp_zero;
581 
582  gyro = Vector3f(int16_val(data, 5),
583  int16_val(data, 4),
584  -int16_val(data, 6));
585  gyro *= GYRO_SCALE;
586 
587  _rotate_and_correct_accel(_accel_instance, accel);
588  _rotate_and_correct_gyro(_gyro_instance, gyro);
589 
590  _notify_new_accel_raw_sample(_accel_instance, accel, AP_HAL::micros64(), fsync_set);
591  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
592 
593  _temp_filtered = _temp_filter.apply(temp);
594  }
595  return true;
596 }

598 /*
599  when doing fast sampling the sensor gives us 8k samples/second. Every 2nd accel sample is a duplicate.
600 
601  To filter this we first apply a 1p low pass filter at 188Hz, then we
602  average over 8 samples to bring the data rate down to 1kHz. This
603  gives very good aliasing rejection at frequencies well above what
604  can be handled with 1kHz sample rates.
605  */
606 bool AP_InertialSensor_Invensense::_accumulate_fast_sampling(uint8_t *samples, uint8_t n_samples)
607 {
608  int32_t tsum = 0;
609  const int32_t clip_limit = AP_INERTIAL_SENSOR_ACCEL_CLIP_THRESH_MSS / _accel_scale;
610  bool clipped = false;
611  bool ret = true;
612 
613  for (uint8_t i = 0; i < n_samples; i++) {
614  const uint8_t *data = samples + MPU_SAMPLE_SIZE * i;
615 
616  // use temperatue to detect FIFO corruption
617  int16_t t2 = int16_val(data, 3);
618  if (!_check_raw_temp(t2)) {
619  debug("temp reset %d %d", _raw_temp, t2);
620  _fifo_reset();
621  ret = false;
622  break;
623  }
624  tsum += t2;
625 
626  if ((_accum.count & 1) == 0) {
627  // accel data is at 4kHz
628  Vector3f a(int16_val(data, 1),
629  int16_val(data, 0),
630  -int16_val(data, 2));
631  if (fabsf(a.x) > clip_limit ||
632  fabsf(a.y) > clip_limit ||
633  fabsf(a.z) > clip_limit) {
634  clipped = true;
635  }
636  _accum.accel += _accum.accel_filter.apply(a);
637  }
638 
639  Vector3f g(int16_val(data, 5),
640  int16_val(data, 4),
641  -int16_val(data, 6));
642 
643  _accum.gyro += _accum.gyro_filter.apply(g);
644  _accum.count++;
645 
646  if (_accum.count == MPU_FIFO_DOWNSAMPLE_COUNT) {
647  float ascale = _accel_scale / (MPU_FIFO_DOWNSAMPLE_COUNT/2);
648  _accum.accel *= ascale;
649 
650  float gscale = GYRO_SCALE / MPU_FIFO_DOWNSAMPLE_COUNT;
651  _accum.gyro *= gscale;
652 
653  _rotate_and_correct_accel(_accel_instance, _accum.accel);
654  _rotate_and_correct_gyro(_gyro_instance, _accum.gyro);
655 
656  _notify_new_accel_raw_sample(_accel_instance, _accum.accel, AP_HAL::micros64(), false);
657  _notify_new_gyro_raw_sample(_gyro_instance, _accum.gyro);
658 
659  _accum.accel.zero();
660  _accum.gyro.zero();
661  _accum.count = 0;
662  }
663  }
664 
665  if (clipped) {
666  increment_clip_count(_accel_instance);
667  }
668 
669  if (ret) {
670  float temp = (static_cast<float>(tsum)/n_samples)*temp_sensitivity + temp_zero;
671  _temp_filtered = _temp_filter.apply(temp);
672  }
673 
674  return ret;
675 }

677 void AP_InertialSensor_Invensense::_read_fifo()
678 {
679  uint8_t n_samples;
680  uint16_t bytes_read;
681  uint8_t *rx = _fifo_buffer;
682  bool need_reset = false;
683 
684  if (!_block_read(MPUREG_FIFO_COUNTH, rx, 2)) {
685  goto check_registers;
686  }
687 
688  bytes_read = uint16_val(rx, 0);
689  n_samples = bytes_read / MPU_SAMPLE_SIZE;
690 
691  if (n_samples == 0) {
692  /* Not enough data in FIFO */
693  goto check_registers;
694  }
695 
696  /*
697  testing has shown that if we have more than 32 samples in the
698  FIFO then some of those samples will be corrupt. It always is
699  the ones at the end of the FIFO, so clear those with a reset
700  once we've read the first 24. Reading 24 gives us the normal
701  number of samples for fast sampling at 400Hz
702  */
703  if (n_samples > 32) {
704  need_reset = true;
705  n_samples = 24;
706  }
707 
708  while (n_samples > 0) {
709  uint8_t n = MIN(n_samples, MPU_FIFO_BUFFER_LEN);
710  if (!_dev->set_chip_select(true)) {
711  if (!_block_read(MPUREG_FIFO_R_W, rx, n * MPU_SAMPLE_SIZE)) {
712  goto check_registers;
713  }
714  } else {
715  // this ensures we keep things nicely setup for DMA
716  uint8_t reg = MPUREG_FIFO_R_W | 0x80;
717  if (!_dev->transfer(&reg, 1, nullptr, 0)) {
718  _dev->set_chip_select(false);
719  goto check_registers;
720  }
721  memset(rx, 0, n * MPU_SAMPLE_SIZE);
722  if (!_dev->transfer(rx, n * MPU_SAMPLE_SIZE, rx, n * MPU_SAMPLE_SIZE)) {
723  hal.console->printf("MPU60x0: error in fifo read %u bytes\n", n * MPU_SAMPLE_SIZE);
724  _dev->set_chip_select(false);
725  goto check_registers;
726  }
727  _dev->set_chip_select(false);
728  }
729 
730  if (_fast_sampling) {
731  if (!_accumulate_fast_sampling(rx, n)) {
732  debug("stop at %u of %u", n_samples, bytes_read/MPU_SAMPLE_SIZE);
733  break;
734  }
735  } else {
736  if (!_accumulate(rx, n)) {
737  break;
738  }
739  }
740  n_samples -= n;
741  }
742 
743  if (need_reset) {
744  //debug("fifo reset n_samples %u", bytes_read/MPU_SAMPLE_SIZE);
745  _fifo_reset();
746  }
747 
748 check_registers:
749  // check next register value for correctness
750  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
751  if (!_dev->check_next_register()) {
752  _inc_gyro_error_count(_gyro_instance);
753  _inc_accel_error_count(_accel_instance);
754  }
755  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
756 }

863 bool AP_InertialSensor_Invensense::_hardware_init(void)
864 {
865  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
866  return false;
867  }
868 
869  // setup for register checking
870  _dev->setup_checked_registers(7, 20);
871 
872  // initially run the bus at low speed
873  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
874 
875  if (!_check_whoami()) {
876  _dev->get_semaphore()->give();
877  return false;
878  }
879 
880  // Chip reset
881  uint8_t tries;
882  for (tries = 0; tries < 5; tries++) {
883  _last_stat_user_ctrl = _register_read(MPUREG_USER_CTRL);
884 
885  /* First disable the master I2C to avoid hanging the slaves on the
886  * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
887  * is used */
888  if (_last_stat_user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
889  _last_stat_user_ctrl &= ~BIT_USER_CTRL_I2C_MST_EN;
890  _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
891  hal.scheduler->delay(10);
892  }
893 
894  /* reset device */
895  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
896  hal.scheduler->delay(100);
897 
898  /* bus-dependent initialization */
899  if (_dev->bus_type() == AP_HAL::Device::BUS_TYPE_SPI) {
900  /* Disable I2C bus if SPI selected (Recommended in Datasheet to be
901  * done just after the device is reset) */
902  _last_stat_user_ctrl |= BIT_USER_CTRL_I2C_IF_DIS;
903  _register_write(MPUREG_USER_CTRL, _last_stat_user_ctrl);
904  }
905 
906  /* bus-dependent initialization */
907  if ((_dev->bus_type() == AP_HAL::Device::BUS_TYPE_I2C) && (_mpu_type == Invensense_MPU9250)) {
908  /* Enable I2C bypass to access internal AK8963 */
909  _register_write(MPUREG_INT_PIN_CFG, BIT_BYPASS_EN);
910  }
911 
912  // Wake up device and select GyroZ clock. Note that the
913  // Invensense starts up in sleep mode, and it can take some time
914  // for it to come out of sleep
915  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
916  hal.scheduler->delay(5);
917 
918  // check it has woken up
919  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
920  break;
921  }
922 
923  hal.scheduler->delay(10);
924  if (_data_ready()) {
925  break;
926  }
927  }
928 
929  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
930  _dev->get_semaphore()->give();
931 
932  if (tries == 5) {
933  hal.console->printf("Failed to boot Invensense 5 times\n");
934  return false;
935  }
936 
937  if (_mpu_type == Invensense_ICM20608 ||
938  _mpu_type == Invensense_ICM20602) {
939  // this avoids a sensor bug, see description above
940  _register_write(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE, true);
941  }
942 
943  return true;
944 }
