80 void FlightAxis::parse_reply(const char *reply)
81 {
82  const char *reply0 = reply;
83  for (uint16_t i=0; i<num_keys; i++) {
84  const char *p = strstr(reply, keytable[i].key);
85  if (p == nullptr) {
86  p = strstr(reply0, keytable[i].key);
87  }
88  if (p == nullptr) {
89  printf("Failed to find key %s\n", keytable[i].key);
90  controller_started = false;
91  break;
92  }
93  p += strlen(keytable[i].key) + 1;
94  double v;
95  if (strncmp(p, "true", 4) == 0) {
96  v = 1;
97  } else if (strncmp(p, "false", 5) == 0) {
98  v = 0;
99  } else {
100  v = atof(p);
101  }
102  keytable[i].ref = v;
103  // this assumes key order and allows us to decode arrays
104  p = strchr(p, '>');
105  if (p != nullptr) {
106  reply = p;
107  }
108  }
109 }

113  make a SOAP request, returning body of reply
114  */
115 char *FlightAxis::soap_request(const char *action, const char *fmt, ...)
116 {
117  va_list ap;
118  char *req1;
119 
120  va_start(ap, fmt);
121  vasprintf(&req1, fmt, ap);
122  va_end(ap);
123 
124  //printf("%s\n", req1);
125 
126  // open SOAP socket to FlightAxis
127  SocketAPM sock(false);
128  if (!sock.connect(controller_ip, controller_port)) {
129  free(req1);
130  return nullptr;
131  }
132  sock.set_blocking(false);
133 
134  char *req;
135  asprintf(&req, R"(POST / HTTP/1.1
136 soapaction: '%s'
137 content-length: %u
138 content-type: text/xml;charset='UTF-8'
139 Connection: Keep-Alive
140 
141 %s)",
142  action,
143  (unsigned)strlen(req1), req1);
144  sock.send(req, strlen(req));
145  free(req1);
146  free(req);
147  char reply[10000];
148  memset(reply, 0, sizeof(reply));
149  ssize_t ret = sock.recv(reply, sizeof(reply)-1, 1000);
150  if (ret <= 0) {
151  printf("No data\n");
152  return nullptr;
153  }
154  char *p = strstr(reply, "Content-Length: ");
155  if (p == nullptr) {
156  printf("No Content-Length\n");
157  return nullptr;
158  }
159 
160  // get the content length
161  uint32_t content_length = strtoul(p+16, nullptr, 10);
162  char *body = strstr(p, "\r\n\r\n");
163  if (body == nullptr) {
164  printf("No body\n");
165  return nullptr;
166  }
167  body += 4;
168 
169  // get the rest of the body
170  int32_t expected_length = content_length + (body - reply);
171  if (expected_length >= (int32_t)sizeof(reply)) {
172  printf("Reply too large %i\n", expected_length);
173  return nullptr;
174  }
175  while (ret < expected_length) {
176  ssize_t ret2 = sock.recv(&reply[ret], sizeof(reply)-(1+ret), 100);
177  if (ret2 <= 0) {
178  return nullptr;
179  }
180  // nul terminate
181  reply[ret+ret2] = 0;
182  ret += ret2;
183  }
184  return strdup(reply);
185 }

190 {
191  if (!controller_started ||
192  is_zero(state.m_flightAxisControllerIsActive) ||
193  !is_zero(state.m_resetButtonHasBeenPressed)) {
194  printf("Starting controller at %s\n", controller_ip);

196  char *reply = soap_request("RestoreOriginalControllerDevice", R"(<?xml version='1.0' encoding='UTF-8'?>
197 <soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>

200 </soap:Body>
201 </soap:Envelope>)");
202  free(reply);
203  reply = soap_request("InjectUAVControllerInterface", R"(<?xml version='1.0' encoding='UTF-8'?>
204 <soap:Envelope xmlns:soap='http://schemas.xmlsoap.org/soap/envelope/' xmlns:xsd='http://www.w3.org/2001/XMLSchema' xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'>
205 <soap:Body>
206 <InjectUAVControllerInterface><a>1</a><b>2</b></InjectUAVControllerInterface>
207 </soap:Body>
208 </soap:Envelope>)");
209  free(reply);
210  activation_frame_counter = frame_counter;
211  controller_started = true;
212  }
213 
214  float scaled_servos[8];
215  for (uint8_t i=0; i<8; i++) {
216  scaled_servos[i] = (input.servos[i] - 1000) / 1000.0f;

226

229  float swash1 = scaled_servos[0];
230  float swash2 = scaled_servos[1];
231  float swash3 = scaled_servos[2];
232 
233  float roll_rate = swash1 - swash2;
234  float pitch_rate = -((swash1+swash2) / 2.0f - swash3);
235 
236  scaled_servos[0] = constrain_float(roll_rate + 0.5, 0, 1);
237  scaled_servos[1] = constrain_float(pitch_rate + 0.5, 0, 1);
238  }

240

276 /*
277  update the FlightAxis simulation by one time step
278  */
279 void FlightAxis::update(const struct sitl_input &input)
280 {
281  exchange_data(input);
282 
283  double dt_seconds = state.m_currentPhysicsTime_SEC - last_time_s;
284  if (dt_seconds < 0) {
285  // cope with restarting RealFlight while connected
286  initial_time_s = time_now_us * 1.0e-6f;
287  last_time_s = state.m_currentPhysicsTime_SEC;
288  position_offset.zero();
289  return;
290  }
291  if (dt_seconds < 0.0001f) {
292  // we probably got a repeated frame
293  time_now_us += 1;
294  return;
295  }
296  if (initial_time_s <= 0) {
297  dt_seconds = 0.001f;
298  initial_time_s = state.m_currentPhysicsTime_SEC - dt_seconds;
299  }
300 
301  /*
302  the queternion convention in realflight seems to have Z negative
303  */
304  Quaternion quat(state.m_orientationQuaternion_W,
305  state.m_orientationQuaternion_Y,
306  state.m_orientationQuaternion_X,
307  -state.m_orientationQuaternion_Z);
308  quat.rotation_matrix(dcm);
309 
310  gyro = Vector3f(radians(constrain_float(state.m_rollRate_DEGpSEC, -2000, 2000)),
311  radians(constrain_float(state.m_pitchRate_DEGpSEC, -2000, 2000)),
312  -radians(constrain_float(state.m_yawRate_DEGpSEC, -2000, 2000))) * target_speedup;
313 
314  velocity_ef = Vector3f(state.m_velocityWorldU_MPS,
315  state.m_velocityWorldV_MPS,
316  state.m_velocityWorldW_MPS);
317  position = Vector3f(state.m_aircraftPositionY_MTR,
318  state.m_aircraftPositionX_MTR,
319  -state.m_altitudeASL_MTR - home.alt*0.01);
320 
321  accel_body(state.m_accelerationBodyAX_MPS2,
322  state.m_accelerationBodyAY_MPS2,
323  state.m_accelerationBodyAZ_MPS2);
324 
325  if (rotation != ROTATION_NONE) {
326  dcm.transpose();
327  dcm = att_rotation * dcm;
328  dcm.transpose();
329  gyro.rotate(rotation);
330  accel_body.rotate(rotation);
331  }
332 
333  // accel on the ground is nasty in realflight, and prevents helicopter disarm
334  if (state.m_isTouchingGround) {
335  Vector3f accel_ef = (velocity_ef - last_velocity_ef) / dt_seconds;
336  accel_ef.z -= GRAVITY_MSS;
337  accel_body = dcm.transposed() * accel_ef;
338  }
339 
340  // limit to 16G to match pixhawk
341  float a_limit = GRAVITY_MSS*16;
342  accel_body.x = constrain_float(accel_body.x, -a_limit, a_limit);
343  accel_body.y = constrain_float(accel_body.y, -a_limit, a_limit);
344  accel_body.z = constrain_float(accel_body.z, -a_limit, a_limit);
345 
346  // offset based on first position to account for offset in RF world
347  if (position_offset.is_zero() || state.m_resetButtonHasBeenPressed) {
348  position_offset = position;
349  }
350  position -= position_offset;
351 
352  airspeed = state.m_airspeed_MPS;
353  airspeed_pitot = state.m_airspeed_MPS;
354 
355  battery_voltage = state.m_batteryVoltage_VOLTS;
356  battery_current = state.m_batteryCurrentDraw_AMPS;
357  rpm1 = state.m_heliMainRotorRPM;
358  rpm2 = state.m_propRPM;
359 
360  /*
361  the interlink interface supports 8 input channels
362  */
363  rcin_chan_count = 8;
364  for (uint8_t i=0; i<rcin_chan_count; i++) {
365  rcin[i] = state.rcin[i];
366  }
367 
368  update_position();
369  time_now_us = (state.m_currentPhysicsTime_SEC - initial_time_s)*1.0e6;
370 
371  if (frame_counter++ % 1000 == 0) {
372  if (last_frame_count_s != 0) {
373  printf("%.2f FPS\n",
374  1000 / (state.m_currentPhysicsTime_SEC - last_frame_count_s));
375  } else {
376  printf("Initial position %f %f %f\n", position.x, position.y, position.z);
377  }
378  last_frame_count_s = state.m_currentPhysicsTime_SEC;
379  }
380 
381  last_time_s = state.m_currentPhysicsTime_SEC;
382 
383  last_velocity_ef = velocity_ef;
384 
385  // update magnetic field
386  update_mag_field_bf();
387 }
