32  convert airspeed in m/s to an airspeed sensor value
33  */
34 uint16_t SITL_State::_airspeed_sensor(float airspeed)
35 {
36  const float airspeed_ratio = 1.9936f;
37  const float airspeed_offset = 2013;
38  float airspeed_pressure, airspeed_raw;
39 
40  airspeed_pressure = (airspeed*airspeed) / airspeed_ratio;
41  airspeed_raw = airspeed_pressure + airspeed_offset;
42  if (airspeed_raw/4 > 0xFFFF) {
43  return 0xFFFF;
44  }
45  // add delay
46  uint32_t now = AP_HAL::millis();
47  uint32_t best_time_delta_wind = 200; // initialise large time representing buffer entry closest to current time - delay.
48  uint8_t best_index_wind = 0; // initialise number representing the index of the entry in buffer closest to delay.
49 
50  // storing data from sensor to buffer
51  if (now - last_store_time_wind >= 10) { // store data every 10 ms.
52  last_store_time_wind = now;
53  if (store_index_wind > wind_buffer_length-1) { // reset buffer index if index greater than size of buffer
54  store_index_wind = 0;
55  }
56  buffer_wind[store_index_wind].data = airspeed_raw; // add data to current index
57  buffer_wind[store_index_wind].time = last_store_time_wind; // add time to current index
58  store_index_wind = store_index_wind + 1; // increment index
59  }
60 
61  // return delayed measurement
62  delayed_time_wind = now - _sitl->wind_delay; // get time corresponding to delay
63  // find data corresponding to delayed time in buffer
64  for (uint8_t i=0; i<=wind_buffer_length-1; i++) {
65  // find difference between delayed time and time stamp in buffer
66  time_delta_wind = abs(
67  (int32_t)(delayed_time_wind - buffer_wind[i].time));
68  // if this difference is smaller than last delta, store this time
69  if (time_delta_wind < best_time_delta_wind) {
70  best_index_wind = i;
71  best_time_delta_wind = time_delta_wind;
72  }
73  }
74  if (best_time_delta_wind < 200) { // only output stored state if < 200 msec retrieval error
75  airspeed_raw = buffer_wind[best_index_wind].data;
76  }
77 
78  return airspeed_raw/4;
79 }

85 uint16_t SITL_State::_ground_sonar(void)
86 {
87  float altitude = _sitl->height_agl;
88 
89  // sensor position offset in body frame
90  Vector3f relPosSensorBF = _sitl->rngfnd_pos_offset;
91 
92  // adjust altitude for position of the sensor on the vehicle if position offset is non-zero
93  if (!relPosSensorBF.is_zero()) {
94  // get a rotation matrix following DCM conventions (body to earth)
95  Matrix3f rotmat;
96  rotmat.from_euler(radians(_sitl->state.rollDeg),
97  radians(_sitl->state.pitchDeg),
98  radians(_sitl->state.yawDeg));
99  // rotate the offset into earth frame
100  Vector3f relPosSensorEF = rotmat * relPosSensorBF;
101  // correct the altitude at the sensor
102  altitude -= relPosSensorEF.z;
103  }
104 
105  float voltage = 5.0f;
106  if (fabs(_sitl->state.rollDeg) < 90 &&
107  fabs(_sitl->state.pitchDeg) < 90) {
108  // adjust for apparent altitude with roll
109  altitude /= cosf(radians(_sitl->state.rollDeg)) * cosf(radians(_sitl->state.pitchDeg));
110 
111  altitude += _sitl->sonar_noise * _rand_float();
112 
113  // Altitude in in m, scaler in meters/volt
114  voltage = altitude / _sitl->sonar_scale;
115  voltage = constrain_float(voltage, 0, 5.0f);
116 
117  if (_sitl->sonar_glitch >= (_rand_float() + 1.0f)/2.0f) {
118  voltage = 5.0f;
119  }
120  }
121 
122  return 1023*(voltage / 5.0f);
123 }
