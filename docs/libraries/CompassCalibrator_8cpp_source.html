16 /*
17  * The intention of a magnetometer in a compass application is to measure
18  * Earth's magnetic field. Measurements other than those of Earth's magnetic
19  * field are considered errors. This algorithm computes a set of correction
20  * parameters that null out errors from various sources:
21  *
22  * - Sensor bias error
23  * - "Hard iron" error caused by materials fixed to the vehicle body that
24  * produce static magnetic fields.
25  * - Sensor scale-factor error
26  * - Sensor cross-axis sensitivity
27  * - "Soft iron" error caused by materials fixed to the vehicle body that
28  * distort magnetic fields.
29  *
30  * This is done by taking a set of samples that are assumed to be the product
31  * of rotation in earth's magnetic field and fitting an offset ellipsoid to
32  * them, determining the correction to be applied to adjust the samples into an
33  * origin-centered sphere.
34  *
35  * The state machine of this library is described entirely by the
36  * compass_cal_status_t enum, and all state transitions are managed by the
37  * set_status function. Normally, the library is in the NOT_STARTED state. When
38  * the start function is called, the state transitions to WAITING_TO_START,
39  * until two conditions are met: the delay as elapsed, and the memory for the
40  * sample buffer has been successfully allocated.
41  * Once these conditions are met, the state transitions to RUNNING_STEP_ONE, and
42  * samples are collected via calls to the new_sample function. These samples are
43  * accepted or rejected based on distance to the nearest sample. The samples are
44  * assumed to cover the surface of a sphere, and the radius of that sphere is
45  * initialized to a conservative value. Based on a circle-packing pattern, the
46  * minimum distance is set such that some percentage of the surface of that
47  * sphere must be covered by samples.
48  *
49  * Once the sample buffer is full, a sphere fitting algorithm is run, which
50  * computes a new sphere radius. The sample buffer is thinned of samples which
51  * no longer meet the acceptance criteria, and the state transitions to
52  * RUNNING_STEP_TWO. Samples continue to be collected until the buffer is full
53  * again, the full ellipsoid fit is run, and the state transitions to either
54  * SUCCESS or FAILED.
55  *
56  * The fitting algorithm used is Levenberg-Marquardt. See also:
57  * http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm
58  */

103 float CompassCalibrator::get_completion_percent() const {
104  // first sampling step is 1/3rd of the progress bar
105  // never return more than 99% unless _status is COMPASS_CAL_SUCCESS
106  switch(_status) {
107  case COMPASS_CAL_NOT_STARTED:
108  case COMPASS_CAL_WAITING_TO_START:
109  return 0.0f;
110  case COMPASS_CAL_RUNNING_STEP_ONE:
111  return 33.3f * _samples_collected/COMPASS_CAL_NUM_SAMPLES;
112  case COMPASS_CAL_RUNNING_STEP_TWO:
113  return 33.3f + 65.7f*((float)(_samples_collected-_samples_thinned)/(COMPASS_CAL_NUM_SAMPLES-_samples_thinned));
114  case COMPASS_CAL_SUCCESS:
115  return 100.0f;
116  case COMPASS_CAL_FAILED:
117  default:
118  return 0.0f;
119  };

122 void CompassCalibrator::update_completion_mask(const Vector3f& v)
123 {
124  Matrix3f softiron{
125  _params.diag.x, _params.offdiag.x, _params.offdiag.y,
126  _params.offdiag.x, _params.diag.y, _params.offdiag.z,
127  _params.offdiag.y, _params.offdiag.z, _params.diag.z
128  };
129  Vector3f corrected = softiron * (v + _params.offset);
130  int section = AP_GeodesicGrid::section(corrected, true);
131  if (section < 0) {
132  return;
133  }
134  _completion_mask[section / 8] |= 1 << (section % 8);
135 }

174 void CompassCalibrator::update(bool &failure) {
175  failure = false;
176 
177  if(!fitting()) {
178  return;
179  }
180 
181  if(_status == COMPASS_CAL_RUNNING_STEP_ONE) {
182  if (_fit_step >= 10) {
183  if(is_equal(_fitness,_initial_fitness) || isnan(_fitness)) { //if true, means that fitness is diverging instead of converging
184  set_status(COMPASS_CAL_FAILED);
185  failure = true;
186  }
187  set_status(COMPASS_CAL_RUNNING_STEP_TWO);
188  } else {
189  if (_fit_step == 0) {
190  calc_initial_offset();
191  }
192  run_sphere_fit();
193  _fit_step++;
194  }
195  } else if(_status == COMPASS_CAL_RUNNING_STEP_TWO) {
196  if (_fit_step >= 35) {
197  if(fit_acceptable()) {
198  set_status(COMPASS_CAL_SUCCESS);
199  } else {
200  set_status(COMPASS_CAL_FAILED);
201  failure = true;
202  }
203  } else if (_fit_step < 15) {
204  run_sphere_fit();
205  _fit_step++;
206  } else {
207  run_ellipsoid_fit();
208  _fit_step++;
209  }
210  }
211 }

249 bool CompassCalibrator::set_status(compass_cal_status_t status) {
250  if (status != COMPASS_CAL_NOT_STARTED && _status == status) {
251  return true;
252  }
253 
254  switch(status) {
255  case COMPASS_CAL_NOT_STARTED:
256  reset_state();
257  _status = COMPASS_CAL_NOT_STARTED;
258 
259  if(_sample_buffer != nullptr) {
260  free(_sample_buffer);
261  _sample_buffer = nullptr;
262  }
263  return true;
264 
265  case COMPASS_CAL_WAITING_TO_START:
266  reset_state();
267  _status = COMPASS_CAL_WAITING_TO_START;
268 
269  set_status(COMPASS_CAL_RUNNING_STEP_ONE);
270  return true;
271 
272  case COMPASS_CAL_RUNNING_STEP_ONE:
273  if(_status != COMPASS_CAL_WAITING_TO_START) {
274  return false;
275  }
276 
277  if(_attempt == 1 && (AP_HAL::millis()-_start_time_ms)*1.0e-3f < _delay_start_sec) {
278  return false;
279  }
280 
281  if (_sample_buffer == nullptr) {
282  _sample_buffer =
283  (CompassSample*) malloc(sizeof(CompassSample) *
284  COMPASS_CAL_NUM_SAMPLES);
285  }
286 
287  if(_sample_buffer != nullptr) {
288  initialize_fit();
289  _status = COMPASS_CAL_RUNNING_STEP_ONE;
290  return true;
291  }
292 
293  return false;
294 
295  case COMPASS_CAL_RUNNING_STEP_TWO:
296  if(_status != COMPASS_CAL_RUNNING_STEP_ONE) {
297  return false;
298  }
299  thin_samples();
300  initialize_fit();
301  _status = COMPASS_CAL_RUNNING_STEP_TWO;
302  return true;
303 
304  case COMPASS_CAL_SUCCESS:
305  if(_status != COMPASS_CAL_RUNNING_STEP_TWO) {
306  return false;
307  }
308 
309  if(_sample_buffer != nullptr) {
310  free(_sample_buffer);
311  _sample_buffer = nullptr;
312  }
313 
314  _status = COMPASS_CAL_SUCCESS;
315  return true;
316 
317  case COMPASS_CAL_FAILED:
318  if(_status == COMPASS_CAL_NOT_STARTED) {
319  return false;
320  }
321 
322  if(_retry && set_status(COMPASS_CAL_WAITING_TO_START)) {
323  _attempt++;
324  return true;
325  }
326 
327  if(_sample_buffer != nullptr) {
328  free(_sample_buffer);
329  _sample_buffer = nullptr;
330  }
331 
332  _status = COMPASS_CAL_FAILED;
333  return true;
334 
335  default:
336  return false;
337  };

340 bool CompassCalibrator::fit_acceptable() {
341  if( !isnan(_fitness) &&
342  _params.radius > 150 && _params.radius < 950 && //Earth's magnetic field strength range: 250-850mG
343  fabsf(_params.offset.x) < _offset_max &&
344  fabsf(_params.offset.y) < _offset_max &&
345  fabsf(_params.offset.z) < _offset_max &&
346  _params.diag.x > 0.2f && _params.diag.x < 5.0f &&
347  _params.diag.y > 0.2f && _params.diag.y < 5.0f &&
348  _params.diag.z > 0.2f && _params.diag.z < 5.0f &&
349  fabsf(_params.offdiag.x) < 1.0f && //absolute of sine/cosine output cannot be greater than 1
350  fabsf(_params.offdiag.y) < 1.0f &&
351  fabsf(_params.offdiag.z) < 1.0f ){
352 
353  return _fitness <= sq(_tolerance);
354  }
355  return false;
356 }

358 void CompassCalibrator::thin_samples() {
359  if(_sample_buffer == nullptr) {
360  return;
361  }
362 
363  _samples_thinned = 0;
364  // shuffle the samples http://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle
365  // this is so that adjacent samples don't get sequentially eliminated
366  for(uint16_t i=_samples_collected-1; i>=1; i--) {
367  uint16_t j = get_random16() % (i+1);
368  CompassSample temp = _sample_buffer[i];
369  _sample_buffer[i] = _sample_buffer[j];
370  _sample_buffer[j] = temp;
371  }
372 
373  for(uint16_t i=0; i < _samples_collected; i++) {
374  if(!accept_sample(_sample_buffer[i])) {
375  _sample_buffer[i] = _sample_buffer[_samples_collected-1];
376  _samples_collected --;
377  _samples_thinned ++;
378  }
379  }
380 
381  update_completion_mask();
382 }

384 /*
385  * The sample acceptance distance is determined as follows:
386  * For any regular polyhedron with triangular faces, the angle theta subtended
387  * by two closest points is defined as
388  *
389  * theta = arccos(cos(A)/(1-cos(A)))
390  *
391  * Where:
392  * A = (4pi/F + pi)/3
393  * and
394  * F = 2V - 4 is the number of faces for the polyhedron in consideration,
395  * which depends on the number of vertices V
396  *
397  * The above equation was proved after solving for spherical triangular excess
398  * and related equations.
399  */
400 bool CompassCalibrator::accept_sample(const Vector3f& sample)
401 {
402  static const uint16_t faces = (2 * COMPASS_CAL_NUM_SAMPLES - 4);
403  static const float a = (4.0f * M_PI / (3.0f * faces)) + M_PI / 3.0f;
404  static const float theta = 0.5f * acosf(cosf(a) / (1.0f - cosf(a)));
405 
406  if(_sample_buffer == nullptr) {
407  return false;
408  }
409 
410  float min_distance = _params.radius * 2*sinf(theta/2);
411 
412  for (uint16_t i = 0; i<_samples_collected; i++){
413  float distance = (sample - _sample_buffer[i].get()).length();
414  if(distance < min_distance) {
415  return false;
416  }
417  }
418  return true;
419 }

439 float CompassCalibrator::calc_mean_squared_residuals(const param_t& params) const
440 {
441  if(_sample_buffer == nullptr || _samples_collected == 0) {
442  return 1.0e30f;
443  }

445  for(uint16_t i=0; i < _samples_collected; i++){
446  Vector3f sample = _sample_buffer[i].get();

448  sum += sq(resid);

454 void CompassCalibrator::calc_sphere_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
455  const Vector3f &offset = params.offset;
456  const Vector3f &diag = params.diag;
457  const Vector3f &offdiag = params.offdiag;
458  Matrix3f softiron(
459  diag.x , offdiag.x , offdiag.y,
460  offdiag.x , diag.y , offdiag.z,
461  offdiag.y , offdiag.z , diag.z
462  );
463 
464  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
465  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
466  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
467  float length = (softiron*(sample+offset)).length();
468 
469  // 0: partial derivative (radius wrt fitness fn) fn operated on sample
470  ret[0] = 1.0f;
471  // 1-3: partial derivative (offsets wrt fitness fn) fn operated on sample
472  ret[1] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
473  ret[2] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
474  ret[3] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
475 }

487 void CompassCalibrator::run_sphere_fit()
488 {
489  if(_sample_buffer == nullptr) {
490  return;
491  }
492 
493  const float lma_damping = 10.0f;
494 
495  float fitness = _fitness;
496  float fit1, fit2;
497  param_t fit1_params, fit2_params;
498  fit1_params = fit2_params = _params;
499 
500  float JTJ[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
501  float JTJ2[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
502  float JTFI[COMPASS_CAL_NUM_SPHERE_PARAMS] = { };
503 
504  // Gauss Newton Part common for all kind of extensions including LM
505  for(uint16_t k = 0; k<_samples_collected; k++) {
506  Vector3f sample = _sample_buffer[k].get();
507 
508  float sphere_jacob[COMPASS_CAL_NUM_SPHERE_PARAMS];
509 
510  calc_sphere_jacob(sample, fit1_params, sphere_jacob);
511 
512  for(uint8_t i = 0;i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
513  // compute JTJ
514  for(uint8_t j = 0; j < COMPASS_CAL_NUM_SPHERE_PARAMS; j++) {
515  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j];
516  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j]; //a backup JTJ for LM
517  }
518  // compute JTFI
519  JTFI[i] += sphere_jacob[i] * calc_residual(sample, fit1_params);
520  }
521  }
522 
523 
524  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
525  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
526  for(uint8_t i = 0; i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
527  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda;
528  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda/lma_damping;
529  }
530 
531  if(!inverse(JTJ, JTJ, 4)) {
532  return;
533  }
534 
535  if(!inverse(JTJ2, JTJ2, 4)) {
536  return;
537  }
538 
539  for(uint8_t row=0; row < COMPASS_CAL_NUM_SPHERE_PARAMS; row++) {
540  for(uint8_t col=0; col < COMPASS_CAL_NUM_SPHERE_PARAMS; col++) {
541  fit1_params.get_sphere_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
542  fit2_params.get_sphere_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
543  }
544  }
545 
546  fit1 = calc_mean_squared_residuals(fit1_params);
547  fit2 = calc_mean_squared_residuals(fit2_params);
548 
549  if(fit1 > _fitness && fit2 > _fitness){
550  _sphere_lambda *= lma_damping;
551  } else if(fit2 < _fitness && fit2 < fit1) {
552  _sphere_lambda /= lma_damping;
553  fit1_params = fit2_params;
554  fitness = fit2;
555  } else if(fit1 < _fitness){
556  fitness = fit1;
557  }
558  //--------------------Levenberg-Marquardt-part-ends-here--------------------------------//
559 
560  if(!isnan(fitness) && fitness < _fitness) {
561  _fitness = fitness;
562  _params = fit1_params;
563  update_completion_mask();
564  }
565 }

569 void CompassCalibrator::calc_ellipsoid_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
570  const Vector3f &offset = params.offset;
571  const Vector3f &diag = params.diag;
572  const Vector3f &offdiag = params.offdiag;
573  Matrix3f softiron(
574  diag.x , offdiag.x , offdiag.y,
575  offdiag.x , diag.y , offdiag.z,
576  offdiag.y , offdiag.z , diag.z
577  );
578 
579  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
580  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
581  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
582  float length = (softiron*(sample+offset)).length();
583 
584  // 0-2: partial derivative (offset wrt fitness fn) fn operated on sample
585  ret[0] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
586  ret[1] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
587  ret[2] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
588  // 3-5: partial derivative (diag offset wrt fitness fn) fn operated on sample
589  ret[3] = -1.0f * ((sample.x + offset.x) * A)/length;
590  ret[4] = -1.0f * ((sample.y + offset.y) * B)/length;
591  ret[5] = -1.0f * ((sample.z + offset.z) * C)/length;
592  // 6-8: partial derivative (off-diag offset wrt fitness fn) fn operated on sample
593  ret[6] = -1.0f * (((sample.y + offset.y) * A) + ((sample.x + offset.x) * B))/length;
594  ret[7] = -1.0f * (((sample.z + offset.z) * A) + ((sample.x + offset.x) * C))/length;
595  ret[8] = -1.0f * (((sample.z + offset.z) * B) + ((sample.y + offset.y) * C))/length;
596 }

598 void CompassCalibrator::run_ellipsoid_fit()
599 {
600  if(_sample_buffer == nullptr) {
601  return;
602  }
603 
604  const float lma_damping = 10.0f;
605 
606 
607  float fitness = _fitness;
608  float fit1, fit2;
609  param_t fit1_params, fit2_params;
610  fit1_params = fit2_params = _params;
611 
612 
613  float JTJ[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
614  float JTJ2[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
615  float JTFI[COMPASS_CAL_NUM_ELLIPSOID_PARAMS] = { };
616 
617  // Gauss Newton Part common for all kind of extensions including LM
618  for(uint16_t k = 0; k<_samples_collected; k++) {
619  Vector3f sample = _sample_buffer[k].get();
620 
621  float ellipsoid_jacob[COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
622 
623  calc_ellipsoid_jacob(sample, fit1_params, ellipsoid_jacob);
624 
625  for(uint8_t i = 0;i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
626  // compute JTJ
627  for(uint8_t j = 0; j < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; j++) {
628  JTJ [i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
629  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
630  }
631  // compute JTFI
632  JTFI[i] += ellipsoid_jacob[i] * calc_residual(sample, fit1_params);
633  }
634  }
635 
636 
637 
638  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
639  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
640  for(uint8_t i = 0; i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
641  JTJ[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda;
642  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda/lma_damping;
643  }
644 
645  if(!inverse(JTJ, JTJ, 9)) {
646  return;
647  }
648 
649  if(!inverse(JTJ2, JTJ2, 9)) {
650  return;
651  }
652 
653  for(uint8_t row=0; row < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; row++) {
654  for(uint8_t col=0; col < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; col++) {
655  fit1_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
656  fit2_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
657  }
658  }
659 
660  fit1 = calc_mean_squared_residuals(fit1_params);
661  fit2 = calc_mean_squared_residuals(fit2_params);
662 
663  if(fit1 > _fitness && fit2 > _fitness){
664  _ellipsoid_lambda *= lma_damping;
665  } else if(fit2 < _fitness && fit2 < fit1) {
666  _ellipsoid_lambda /= lma_damping;
667  fit1_params = fit2_params;
668  fitness = fit2;
669  } else if(fit1 < _fitness){
670  fitness = fit1;
671  }
672  //--------------------Levenberg-part-ends-here--------------------------------//
673 
674  if(fitness < _fitness) {
675  _fitness = fitness;
676  _params = fit1_params;
677  update_completion_mask();
678  }
679 }
