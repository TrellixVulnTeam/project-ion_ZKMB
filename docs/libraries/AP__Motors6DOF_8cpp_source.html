200 void AP_Motors6DOF::output_to_motors()
201 {
202  int8_t i;
203  int16_t motor_out[AP_MOTORS_MAX_NUM_MOTORS]; // final pwm values sent to the motor
204 
205  switch (_spool_mode) {
206  case SHUT_DOWN:
207  // sends minimum values out to the motors
208  // set motor output based on thrust requests
209  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
210  if (motor_enabled[i]) {
211  motor_out[i] = 1500;
212  }
213  }
214  break;
215  case SPIN_WHEN_ARMED:
216  // sends output to motors when armed but not flying
217  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
218  if (motor_enabled[i]) {
219  motor_out[i] = 1500;
220  }
221  }
222  break;
223  case SPOOL_UP:
224  case THROTTLE_UNLIMITED:
225  case SPOOL_DOWN:
226  // set motor output based on thrust requests
227  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
228  if (motor_enabled[i]) {
229  motor_out[i] = calc_thrust_to_pwm(_thrust_rpyt_out[i]);
230  }
231  }
232  break;
233  }
234 
235  // send output to each motor
236  hal.rcout->cork();
237  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
238  if (motor_enabled[i]) {
239  rc_write(i, motor_out[i]);
240  }
241  }
242  hal.rcout->push();
243 }

249 void AP_Motors6DOF::output_armed_stabilizing()
250 {
251  if ((sub_frame_t)_last_frame_class == SUB_FRAME_VECTORED) {
252  output_armed_stabilizing_vectored();
253  } else if ((sub_frame_t)_last_frame_class == SUB_FRAME_VECTORED_6DOF) {
254  output_armed_stabilizing_vectored_6dof();
255  } else {
256  uint8_t i; // general purpose counter
257  float roll_thrust; // roll thrust input value, +/- 1.0
258  float pitch_thrust; // pitch thrust input value, +/- 1.0
259  float yaw_thrust; // yaw thrust input value, +/- 1.0
260  float throttle_thrust; // throttle thrust input value, +/- 1.0
261  float forward_thrust; // forward thrust input value, +/- 1.0
262  float lateral_thrust; // lateral thrust input value, +/- 1.0
263 
264  roll_thrust = _roll_in;
265  pitch_thrust = _pitch_in;
266  yaw_thrust = _yaw_in;
267  throttle_thrust = get_throttle_bidirectional();
268  forward_thrust = _forward_in;
269  lateral_thrust = _lateral_in;
270 
271  float rpy_out[AP_MOTORS_MAX_NUM_MOTORS]; // buffer so we don't have to multiply coefficients multiple times.
272  float linear_out[AP_MOTORS_MAX_NUM_MOTORS]; // 3 linear DOF mix for each motor
273 
274  // initialize limits flags
275  limit.roll_pitch = false;
276  limit.yaw = false;
277  limit.throttle_lower = false;
278  limit.throttle_upper = false;
279 
280  // sanity check throttle is above zero and below current limited throttle
281  if (throttle_thrust <= -_throttle_thrust_max) {
282  throttle_thrust = -_throttle_thrust_max;
283  limit.throttle_lower = true;
284  }
285  if (throttle_thrust >= _throttle_thrust_max) {
286  throttle_thrust = _throttle_thrust_max;
287  limit.throttle_upper = true;
288  }
289 
290  // calculate roll, pitch and yaw for each motor
291  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
292  if (motor_enabled[i]) {
293  rpy_out[i] = roll_thrust * _roll_factor[i] +
294  pitch_thrust * _pitch_factor[i] +
295  yaw_thrust * _yaw_factor[i];
296 
297  }
298  }
299 
300  // calculate linear command for each motor
301  // linear factors should be 0.0 or 1.0 for now
302  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
303  if (motor_enabled[i]) {
304  linear_out[i] = throttle_thrust * _throttle_factor[i] +
305  forward_thrust * _forward_factor[i] +
306  lateral_thrust * _lateral_factor[i];
307  }
308  }
309 
310  // Calculate final output for each motor
311  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
312  if (motor_enabled[i]) {
313  _thrust_rpyt_out[i] = constrain_float(_motor_reverse[i]*(rpy_out[i] + linear_out[i]),-1.0f,1.0f);
314  }
315  }
316  }

323 void AP_Motors6DOF::output_armed_stabilizing_vectored()
324 {
325  uint8_t i; // general purpose counter
326  float roll_thrust; // roll thrust input value, +/- 1.0
327  float pitch_thrust; // pitch thrust input value, +/- 1.0
328  float yaw_thrust; // yaw thrust input value, +/- 1.0
329  float throttle_thrust; // throttle thrust input value, +/- 1.0
330  float forward_thrust; // forward thrust input value, +/- 1.0
331  float lateral_thrust; // lateral thrust input value, +/- 1.0
332 
333  roll_thrust = _roll_in;
334  pitch_thrust = _pitch_in;
335  yaw_thrust = _yaw_in;
336  throttle_thrust = get_throttle_bidirectional();
337  forward_thrust = _forward_in;
338  lateral_thrust = _lateral_in;
339 
340  float rpy_out[AP_MOTORS_MAX_NUM_MOTORS]; // buffer so we don't have to multiply coefficients multiple times.
341  float linear_out[AP_MOTORS_MAX_NUM_MOTORS]; // 3 linear DOF mix for each motor
342 
343  // initialize limits flags
344  limit.roll_pitch = false;
345  limit.yaw = false;
346  limit.throttle_lower = false;
347  limit.throttle_upper = false;
348 
349  // sanity check throttle is above zero and below current limited throttle
350  if (throttle_thrust <= -_throttle_thrust_max) {
351  throttle_thrust = -_throttle_thrust_max;
352  limit.throttle_lower = true;
353  }
354 
355  if (throttle_thrust >= _throttle_thrust_max) {
356  throttle_thrust = _throttle_thrust_max;
357  limit.throttle_upper = true;
358  }
359 
360  // calculate roll, pitch and yaw for each motor
361  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
362  if (motor_enabled[i]) {
363  rpy_out[i] = roll_thrust * _roll_factor[i] +
364  pitch_thrust * _pitch_factor[i] +
365  yaw_thrust * _yaw_factor[i];
366  }
367  }
368 
369  float forward_coupling_limit = 1-_forwardVerticalCouplingFactor*float(fabsf(throttle_thrust));
370  if (forward_coupling_limit < 0) {
371  forward_coupling_limit = 0;
372  }
373  int8_t forward_coupling_direction[] = {-1,-1,1,1,0,0,0,0};
374 
375  // calculate linear command for each motor
376  // linear factors should be 0.0 or 1.0 for now
377  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
378  if (motor_enabled[i]) {
379 
380  float forward_thrust_limited = forward_thrust;
381 
382  // The following statements decouple forward/vertical hydrodynamic coupling on
383  // vectored ROVs. This is done by limiting the maximum output of the "rear" vectored
384  // thruster (where "rear" depends on direction of travel).
385  if (!is_zero(forward_thrust_limited)) {
386  if ((forward_thrust < 0) == (forward_coupling_direction[i] < 0) && forward_coupling_direction[i] != 0) {
387  forward_thrust_limited = constrain_float(forward_thrust, -forward_coupling_limit, forward_coupling_limit);
388  }
389  }
390 
391  linear_out[i] = throttle_thrust * _throttle_factor[i] +
392  forward_thrust_limited * _forward_factor[i] +
393  lateral_thrust * _lateral_factor[i];
394  }
395  }
396 
397  // Calculate final output for each motor
398  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
399  if (motor_enabled[i]) {
400  _thrust_rpyt_out[i] = constrain_float(_motor_reverse[i]*(rpy_out[i] + linear_out[i]), -1.0f, 1.0f);
401  }
402  }
