140 void AP_Compass_LSM9DS1::_update(void)
141 {
142  struct sample_regs regs;
143  Vector3f raw_field;
144  uint32_t time_us = AP_HAL::micros();
145 
146  if (!_block_read(LSM9DS1M_STATUS_REG_M, (uint8_t *) &regs, sizeof(regs))) {
147  return;
148  }
149 
150  if (regs.status & 0x80) {
151  return;
152  }
153 
154  raw_field = Vector3f(regs.val[0], regs.val[1], regs.val[2]);
155 
156  if (is_zero(raw_field.x) && is_zero(raw_field.y) && is_zero(raw_field.z)) {
157  return;
158  }
159 
160  raw_field *= _scaling;
161 
162  // rotate raw_field from sensor frame to body frame
163  rotate_field(raw_field, _compass_instance);
164 
165  // publish raw_field (uncorrected point sample) for calibration use
166  publish_raw_field(raw_field, time_us, _compass_instance);
167 
168  // correct raw_field for known errors
169  correct_field(raw_field, _compass_instance);
170 
171  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
172  _mag_x_accum += raw_field.x;
173  _mag_y_accum += raw_field.y;
174  _mag_z_accum += raw_field.z;
175  _accum_count++;
176  if (_accum_count == 10) {
177  _mag_x_accum /= 2;
178  _mag_y_accum /= 2;
179  _mag_z_accum /= 2;
180  _accum_count = 5;
181  }
182  _sem->give();
183  }
184 }
