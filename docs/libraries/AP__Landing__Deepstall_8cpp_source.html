160 bool AP_Landing_Deepstall::verify_land(const Location &prev_WP_loc, Location &next_WP_loc, const Location &current_loc,
161  const float height, const float sink_rate, const float wp_proportion, const uint32_t last_flying_ms,
162  const bool is_armed, const bool is_flying, const bool rangefinder_state_in_range)
163 {
164  switch (stage) {
165  case DEEPSTALL_STAGE_FLY_TO_LANDING:
166  if (get_distance(current_loc, landing_point) > 2 * landing.aparm.loiter_radius) {
167  landing.nav_controller->update_waypoint(current_loc, landing_point);
168  return false;
169  }
170  stage = DEEPSTALL_STAGE_ESTIMATE_WIND;
171  loiter_sum_cd = 0; // reset the loiter counter
172  // no break
173  case DEEPSTALL_STAGE_ESTIMATE_WIND:
174  {
175  landing.nav_controller->update_loiter(landing_point, landing.aparm.loiter_radius, 1);
176  if (!landing.nav_controller->reached_loiter_target() || (fabsf(height) > DEEPSTALL_LOITER_ALT_TOLERANCE)) {
177  // wait until the altitude is correct before considering a breakout
178  return false;
179  }
180  // only count loiter progress when within the target altitude
181  int32_t target_bearing = landing.nav_controller->target_bearing_cd();
182  int32_t delta = wrap_180_cd(target_bearing - last_target_bearing);
183  if (delta > 0) { // only accumulate turns in the correct direction
184  loiter_sum_cd += delta;
185  }
186  last_target_bearing = target_bearing;
187  if (loiter_sum_cd < 36000) {
188  // wait until we've done at least one complete loiter at the correct altitude
189  landing.nav_controller->update_loiter(landing_point, landing.aparm.loiter_radius, 1);
190  return false;
191  }
192  stage = DEEPSTALL_STAGE_WAIT_FOR_BREAKOUT;
193  //compute optimal path for landing
194  build_approach_path();
195  // no break
196  }
197  case DEEPSTALL_STAGE_WAIT_FOR_BREAKOUT:
198  if (!verify_breakout(current_loc, arc_entry, height)) {
199  landing.nav_controller->update_loiter(landing_point, landing.aparm.loiter_radius, 1);
200  return false;
201  }
202  stage = DEEPSTALL_STAGE_FLY_TO_ARC;
203  memcpy(&breakout_location, &current_loc, sizeof(Location));
204  // no break
205  case DEEPSTALL_STAGE_FLY_TO_ARC:
206  if (get_distance(current_loc, arc_entry) > 2 * landing.aparm.loiter_radius) {
207  landing.nav_controller->update_waypoint(breakout_location, arc_entry);
208  return false;
209  }
210  stage = DEEPSTALL_STAGE_ARC;
211  // no break
212  case DEEPSTALL_STAGE_ARC:
213  {
214  Vector2f groundspeed = landing.ahrs.groundspeed_vector();
215  if (!landing.nav_controller->reached_loiter_target() ||
216  (fabsf(wrap_180(target_heading_deg -
217  degrees(atan2f(-groundspeed.y, -groundspeed.x) + M_PI))) >= 10.0f)) {
218  landing.nav_controller->update_loiter(arc, landing.aparm.loiter_radius, 1);
219  return false;
220  }
221  stage = DEEPSTALL_STAGE_APPROACH;
222  }
223  // no break
224  case DEEPSTALL_STAGE_APPROACH:
225  {
226  Location entry_point;
227  landing.nav_controller->update_waypoint(arc_exit, extended_approach);
228 
229  float relative_alt_D;
230  landing.ahrs.get_relative_position_D_home(relative_alt_D);
231 
232  const float travel_distance = predict_travel_distance(landing.ahrs.wind_estimate(), -relative_alt_D);
233 
234  memcpy(&entry_point, &landing_point, sizeof(Location));
235  location_update(entry_point, target_heading_deg + 180.0, travel_distance);
236 
237  if (!location_passed_point(current_loc, arc_exit, entry_point)) {
238  if (location_passed_point(current_loc, arc_exit, extended_approach)) {
239  // this should never happen, but prevent against an indefinite fly away
240  stage = DEEPSTALL_STAGE_FLY_TO_LANDING;
241  }
242  return false;
243  }
244  stage = DEEPSTALL_STAGE_LAND;
245  stall_entry_time = AP_HAL::millis();
246 
247  const SRV_Channel* elevator = SRV_Channels::get_channel_for(SRV_Channel::k_elevator);
248  if (elevator != nullptr) {
249  // take the last used elevator angle as the starting deflection
250  // don't worry about bailing here if the elevator channel can't be found
251  // that will be handled within override_servos
252  initial_elevator_pwm = elevator->get_output_pwm();
253  }
254  L1_xtrack_i = 0; // reset the integrators
255  }
256  // no break
257  case DEEPSTALL_STAGE_LAND:
258  // while in deepstall the only thing verify needs to keep the extended approach point sufficently far away
259  landing.nav_controller->update_waypoint(current_loc, extended_approach);
260  return false;
261  default:
262  return true;
263  }

266 bool AP_Landing_Deepstall::override_servos(void)
267 {
268  if (!(stage == DEEPSTALL_STAGE_LAND)) {
269  return false;
270  }
271 
272  SRV_Channel* elevator = SRV_Channels::get_channel_for(SRV_Channel::k_elevator);
273 
274  if (elevator == nullptr) {
275  // deepstalls are impossible without these channels, abort the process
276  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL,
277  "Deepstall: Unable to find the elevator channels");
278  request_go_around();
279  return false;
280  }
281 
282  // calculate the progress on slewing the elevator
283  float slew_progress = 1.0f;
284  if (slew_speed > 0) {
285  slew_progress = (AP_HAL::millis() - stall_entry_time) / (100.0f * slew_speed);
286  slew_progress = constrain_float (slew_progress, 0.0f, 1.0f);
287  }
288 
289  // mix the elevator to the correct value
290  elevator->set_output_pwm(linear_interpolate(initial_elevator_pwm, elevator_pwm,
291  slew_progress, 0.0f, 1.0f));
292 
293  // use the current airspeed to dictate the travel limits
294  float airspeed;
295  landing.ahrs.airspeed_estimate(&airspeed);
296 
297  // only allow the deepstall steering controller to run below the handoff airspeed
298  if (slew_progress >= 1.0f || airspeed <= handoff_airspeed) {
299  // run the steering conntroller
300  float pid = update_steering();
301 
302 
303  float travel_limit = constrain_float((handoff_airspeed - airspeed) /
304  (handoff_airspeed - handoff_lower_limit_airspeed) *
305  0.5f + 0.5f,
306  0.5f, 1.0f);
307 
308  float output = constrain_float(pid, -travel_limit, travel_limit);
309  SRV_Channels::set_output_scaled(SRV_Channel::k_aileron, output*4500);
310  SRV_Channels::set_output_scaled(SRV_Channel::k_aileron_with_input, output*4500);
311  SRV_Channels::set_output_scaled(SRV_Channel::k_rudder, output*4500);
312 
313  } else {
314  // allow the normal servo control of the channel
315  SRV_Channels::set_output_scaled(SRV_Channel::k_aileron_with_input,
316  SRV_Channels::get_output_scaled(SRV_Channel::k_aileron));
317  }
318 
319  // hand off rudder control to deepstall controlled
320  return true;
321 }

401 float AP_Landing_Deepstall::predict_travel_distance(const Vector3f wind, const float height) const
402 {
403  bool reverse = false;
404 
405  float course = radians(target_heading_deg);
406 
407  // a forward speed of 0 will result in a divide by 0
408  float forward_speed_ms = MAX(forward_speed, 0.1f);
409 
410  Vector2f wind_vec(wind.x, wind.y); // work with the 2D component of wind
411  float wind_length = MAX(wind_vec.length(), 0.05f); // always assume a slight wind to avoid divide by 0
412  Vector2f course_vec(cosf(course), sinf(course));
413 
414  float offset = course + atan2f(-wind.y, -wind.x) + M_PI;
415 
416  // estimator for how far the aircraft will travel while entering the stall
417  float stall_distance = slope_a * wind_length * cosf(offset) + slope_b;
418 
419  float theta = acosf(constrain_float((wind_vec * course_vec) / wind_length, -1.0f, 1.0f));
420  if ((course_vec % wind_vec) > 0) {
421  reverse = true;
422  theta *= -1;
423  }
424 
425  float cross_component = sinf(theta) * wind_length;
426  float estimated_crab_angle = asinf(constrain_float(cross_component / forward_speed_ms, -1.0f, 1.0f));
427  if (reverse) {
428  estimated_crab_angle *= -1;
429  }
430 
431  float estimated_forward = cosf(estimated_crab_angle) * forward_speed_ms + cosf(theta) * wind_length;
432 
433 #ifdef DEBUG_PRINTS
434  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "Predict: %f %f", stall_distance, estimated_forward * height / down_speed + stall_distance);
435 #endif // DEBUG_PRINTS
436 
437  return estimated_forward * height / down_speed + stall_distance;
438 }

456 float AP_Landing_Deepstall::update_steering()
457 {
458  Location current_loc;
459  if (!landing.ahrs.get_position(current_loc)) {
460  // panic if no position source is available
461  // continue the but target just holding the wings held level as deepstall should be a minimal energy
462  // configuration on the aircraft, and if a position isn't available aborting would be worse
463  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "Deepstall: No position available. Attempting to hold level");
464  memcpy(&current_loc, &landing_point, sizeof(Location));
465  }
466  uint32_t time = AP_HAL::millis();
467  float dt = constrain_float(time - last_time, (uint32_t)10UL, (uint32_t)200UL) / 1000.0;
468  last_time = time;
469 
470 
471  Vector2f ab = location_diff(arc_exit, extended_approach);
472  ab.normalize();
473  Vector2f a_air = location_diff(arc_exit, current_loc);
474 
475  float crosstrack_error = a_air % ab;
476  float sine_nu1 = constrain_float(crosstrack_error / MAX(L1_period, 0.1f), -0.7071f, 0.7107f);
477  float nu1 = asinf(sine_nu1);
478 
479  if (L1_i > 0) {
480  L1_xtrack_i += nu1 * L1_i / dt;
481  L1_xtrack_i = constrain_float(L1_xtrack_i, -0.5f, 0.5f);
482  nu1 += L1_xtrack_i;
483  }
484 
485  float desired_change = wrap_PI(radians(target_heading_deg) + nu1 - landing.ahrs.yaw);
486 
487  float yaw_rate = landing.ahrs.get_gyro().z;
488  float yaw_rate_limit_rps = radians(yaw_rate_limit);
489  float error = wrap_PI(constrain_float(desired_change / time_constant,
490  -yaw_rate_limit_rps, yaw_rate_limit_rps) - yaw_rate);
491 
492 #ifdef DEBUG_PRINTS
493  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "x: %f e: %f r: %f d: %f",
494  (double)crosstrack_error,
495  (double)error,
496  (double)degrees(yaw_rate),
497  (double)location_diff(current_loc, landing_point).length());
498 #endif // DEBUG_PRINTS
499 
500  return ds_PID.get_pid(error);
501 }
