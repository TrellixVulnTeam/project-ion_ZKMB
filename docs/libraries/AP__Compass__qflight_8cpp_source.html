65 void AP_Compass_QFLIGHT::timer_update(void)
66 {
67  uint32_t now = AP_HAL::millis();
68  if (now - last_check_ms < 25) {
69  return;
70  }
71  last_check_ms = now;
72 
73  if (magbuf == nullptr) {
74  magbuf = QFLIGHT_RPC_ALLOCATE(DSPBuffer::MAG);
75  if (magbuf == nullptr) {
76  AP_HAL::panic("unable to allocate MAG buffer");
77  }
78  }
79  int ret = qflight_get_mag_data((uint8_t *)magbuf, sizeof(*magbuf));
80  if (ret != 0) {
81  return;
82  }
83  uint64_t time_us = AP_HAL::micros64();
84  for (uint16_t i=0; i<magbuf->num_samples; i++) {
85  DSPBuffer::MAG::BUF &b = magbuf->buf[i];
86 
87  // get raw_field - sensor frame, uncorrected
88  Vector3f raw_field(b.mag_raw[0], b.mag_raw[1], -b.mag_raw[2]);
89 
90  // rotate raw_field from sensor frame to body frame
91  rotate_field(raw_field, instance);
92 
93  // publish raw_field (uncorrected point sample) for calibration use
94  publish_raw_field(raw_field, time_us, instance);
95 
96  // correct raw_field for known errors
97  correct_field(raw_field, instance);
98 
99  // accumulate into averaging filter
100  sum += raw_field;
101  count++;
102  }
103 }
