87 /*
88  run one tick
89  this will run as many scheduler tasks as we can in the specified time
90  */
91 void AP_Scheduler::run(uint32_t time_available)
92 {
93  uint32_t run_started_usec = AP_HAL::micros();
94  uint32_t now = run_started_usec;
95 
96  if (_debug > 3 && _perf_counters == nullptr) {
97  _perf_counters = new AP_HAL::Util::perf_counter_t[_num_tasks];
98  if (_perf_counters != nullptr) {
99  for (uint8_t i=0; i<_num_tasks; i++) {
100  _perf_counters[i] = hal.util->perf_alloc(AP_HAL::Util::PC_ELAPSED, _tasks[i].name);
101  }
102  }
103  }
104 
105  for (uint8_t i=0; i<_num_tasks; i++) {
106  uint16_t dt = _tick_counter - _last_run[i];
107  uint16_t interval_ticks = _loop_rate_hz / _tasks[i].rate_hz;
108  if (interval_ticks < 1) {
109  interval_ticks = 1;
110  }
111  if (dt >= interval_ticks) {
112  // this task is due to run. Do we have enough time to run it?
113  _task_time_allowed = _tasks[i].max_time_micros;
114 
115  if (dt >= interval_ticks*2) {
116  // we've slipped a whole run of this task!
117  if (_debug > 1) {
118  ::printf("Scheduler slip task[%u-%s] (%u/%u/%u)\n",
119  (unsigned)i,
120  _tasks[i].name,
121  (unsigned)dt,
122  (unsigned)interval_ticks,
123  (unsigned)_task_time_allowed);
124  }
125  }
126 
127  if (_task_time_allowed <= time_available) {
128  // run it
129  _task_time_started = now;
130  current_task = i;
131  if (_debug > 3 && _perf_counters && _perf_counters[i]) {
132  hal.util->perf_begin(_perf_counters[i]);
133  }
134  _tasks[i].function();
135  if (_debug > 3 && _perf_counters && _perf_counters[i]) {
136  hal.util->perf_end(_perf_counters[i]);
137  }
138  current_task = -1;
139 
140  // record the tick counter when we ran. This drives
141  // when we next run the event
142  _last_run[i] = _tick_counter;
143 
144  // work out how long the event actually took
145  now = AP_HAL::micros();
146  uint32_t time_taken = now - _task_time_started;
147 
148  if (time_taken > _task_time_allowed) {
149  // the event overran!
150  if (_debug > 4) {
151  ::printf("Scheduler overrun task[%u-%s] (%u/%u)\n",
152  (unsigned)i,
153  _tasks[i].name,
154  (unsigned)time_taken,
155  (unsigned)_task_time_allowed);
156  }
157  }
158  if (time_taken >= time_available) {
159  goto update_spare_ticks;
160  }
161  time_available -= time_taken;
162  }
163  }
164  }
165 
166  // update number of spare microseconds
167  _spare_micros += time_available;
168 
169 update_spare_ticks:
170  _spare_ticks++;
171  if (_spare_ticks == 32) {
172  _spare_ticks /= 2;
173  _spare_micros /= 2;
174  }
175 }

177 /*
178  return number of micros until the current task reaches its deadline
179  */
180 uint16_t AP_Scheduler::time_available_usec(void)
181 {
182  uint32_t dt = AP_HAL::micros() - _task_time_started;
183  if (dt > _task_time_allowed) {
184  return 0;
185  }
186  return _task_time_allowed - dt;
187 }
