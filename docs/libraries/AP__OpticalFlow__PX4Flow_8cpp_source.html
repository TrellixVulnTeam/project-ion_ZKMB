53 /*
54  look for the sensor on different buses
55  */
56 bool AP_OpticalFlow_PX4Flow::scan_buses(void)
57 {
58  for (uint8_t bus = 0; bus < 3; bus++) {
59 #ifdef HAL_OPTFLOW_PX4FLOW_I2C_BUS
60  // only one bus from HAL
61  if (bus != HAL_OPTFLOW_PX4FLOW_I2C_BUS) {
62  continue;
63  }
64 #endif
65  AP_HAL::OwnPtr<AP_HAL::Device> tdev = hal.i2c_mgr->get_device(bus, PX4FLOW_BASE_I2C_ADDR + get_bus_id());
66  if (!tdev) {
67  continue;
68  }
69  if (!tdev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
70  continue;
71  }
72  struct i2c_integral_frame frame;
73  bool ok = tdev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame));
74  tdev->get_semaphore()->give();
75  if (ok) {
76  printf("Found PX4Flow on bus %u\n", bus);
77  dev = std::move(tdev);
78  break;
79  }
80  }
81  return !!dev;
82 }

101 // timer to read sensor
102 void AP_OpticalFlow_PX4Flow::timer(void)
103 {
104  struct i2c_integral_frame frame;
105  if (!dev->read_registers(REG_INTEGRAL_FRAME, (uint8_t *)&frame, sizeof(frame))) {
106  return;
107  }
108  struct OpticalFlow::OpticalFlow_state state {};
109  state.device_id = get_bus_id();
110 
111  if (frame.integration_timespan > 0) {
112  const Vector2f flowScaler = _flowScaler();
113  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
114  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
115  float integralToRate = 1.0e6 / frame.integration_timespan;
116 
117  state.surface_quality = frame.qual;
118  state.flowRate = Vector2f(frame.pixel_flow_x_integral * flowScaleFactorX,
119  frame.pixel_flow_y_integral * flowScaleFactorY) * 1.0e-4 * integralToRate;
120  state.bodyRate = Vector2f(frame.gyro_x_rate_integral, frame.gyro_y_rate_integral) * 1.0e-4 * integralToRate;
121 
122  _applyYaw(state.flowRate);
123  _applyYaw(state.bodyRate);
124  }
125 
126  _update_frontend(state);
127 }
