83 bool Aircraft::parse_home(const char *home_str, Location &loc, float &yaw_degrees)
84 {
85  char *saveptr = nullptr;
86  char *s = strdup(home_str);
87  if (!s) {
88  free(s);
89  return false;
90  }
91  char *lat_s = strtok_r(s, ",", &saveptr);
92  if (!lat_s) {
93  free(s);
94  return false;
95  }
96  char *lon_s = strtok_r(nullptr, ",", &saveptr);
97  if (!lon_s) {
98  free(s);
99  return false;
100  }
101  char *alt_s = strtok_r(nullptr, ",", &saveptr);
102  if (!alt_s) {
103  free(s);
104  return false;
105  }
106  char *yaw_s = strtok_r(nullptr, ",", &saveptr);
107  if (!yaw_s) {
108  free(s);
109  return false;
110  }
111 
112  memset(&loc, 0, sizeof(loc));
113  loc.lat = static_cast<int32_t>(strtof(lat_s, nullptr) * 1.0e7f);
114  loc.lng = static_cast<int32_t>(strtof(lon_s, nullptr) * 1.0e7f);
115  loc.alt = static_cast<int32_t>(strtof(alt_s, nullptr) * 1.0e2f);
116 
117  yaw_degrees = strtof(yaw_s, nullptr);
118  free(s);
119 
120  return true;

183 /*
184  update body magnetic field from position and rotation
185 */
186 void Aircraft::update_mag_field_bf()
187 {
188  // get the magnetic field intensity and orientation
189  float intensity;
190  float declination;
191  float inclination;
192  get_mag_field_ef(location.lat * 1e-7f, location.lng * 1e-7f, intensity, declination, inclination);
193 
194  // create a field vector and rotate to the required orientation
195  Vector3f mag_ef(1e3f * intensity, 0.0f, 0.0f);
196  Matrix3f R;
197  R.from_euler(0.0f, -ToRad(inclination), ToRad(declination));
198  mag_ef = R * mag_ef;
199 
200  // calculate frame height above ground
201  const float frame_height_agl = fmaxf((-position.z) + home.alt * 0.01f - ground_level, 0.0f);
202 
203  // calculate scaling factor that varies from 1 at ground level to 1/8 at sitl->mag_anomaly_hgt
204  // Assume magnetic anomaly strength scales with 1/R**3
205  float anomaly_scaler = (sitl->mag_anomaly_hgt / (frame_height_agl + sitl->mag_anomaly_hgt));
206  anomaly_scaler = anomaly_scaler * anomaly_scaler * anomaly_scaler;
207 
208  // add scaled anomaly to earth field
209  mag_ef += sitl->mag_anomaly_ned.get() * anomaly_scaler;
210 
211  // Rotate into body frame
212  mag_bf = dcm.transposed() * mag_ef;
213 
214  // add motor interference
215  mag_bf += sitl->mag_mot.get() * battery_current;
216 }

246 /*
247  try to synchronise simulation time with wall clock time, taking
248  into account desired speedup
249  This tries to take account of possible granularity of
250  get_wall_time_us() so it works reasonably well on windows
251 */
252 void Aircraft::sync_frame_time(void)
253 {
254  frame_counter++;
255  uint64_t now = get_wall_time_us();
256  if (frame_counter >= 40 &&
257  now > last_wall_time_us) {
258  const float rate = frame_counter * 1.0e6f/(now - last_wall_time_us);
259  achieved_rate_hz = (0.99f*achieved_rate_hz) + (0.01f * rate);
260  if (achieved_rate_hz < rate_hz * target_speedup) {
261  scaled_frame_time_us *= 0.999f;
262  } else {
263  scaled_frame_time_us /= 0.999f;
264  }
265 #if 0
266  ::printf("achieved_rate_hz=%.3f rate=%.2f rate_hz=%.3f sft=%.1f\n",
267  static_cast<double>(achieved_rate_hz),
268  static_cast<double>(rate),
269  static_cast<double>(rate_hz),
270  static_cast<double>(scaled_frame_time_us));
271 #endif
272  const uint32_t sleep_time = static_cast<uint32_t>(scaled_frame_time_us * frame_counter);
273  if (sleep_time > min_sleep_time) {
274  usleep(sleep_time);
275  }
276  last_wall_time_us = now;
277  frame_counter = 0;
278  }

292 /*
293  normal distribution random numbers
294  See
295  http://en.literateprograms.org/index.php?title=Special:DownloadCode/Box-Muller_transform_%28C%29&oldid=7011
296 */
297 double Aircraft::rand_normal(double mean, double stddev)
298 {
299  static double n2 = 0.0;
300  static int n2_cached = 0;
301  if (!n2_cached) {
302  double x, y, r;
303  do
304  {
305  x = 2.0 * rand()/RAND_MAX - 1;
306  y = 2.0 * rand()/RAND_MAX - 1;
307  r = x*x + y*y;
308  } while (is_zero(r) || r > 1.0);
309  const double d = sqrt(-2.0 * log(r)/r);
310  const double n1 = x * d;
311  n2 = y * d;
312  const double result = n1 * stddev + mean;
313  n2_cached = 1;
314  return result;
315  } else {
316  n2_cached = 0;
317  return n2 * stddev + mean;
318  }
319 }

412 /*
413  update the simulation attitude and relative position
414  */
415 void Aircraft::update_dynamics(const Vector3f &rot_accel)
416 {
417  const float delta_time = frame_time_us * 1.0e-6f;
418 
419  // update rotational rates in body frame
420  gyro += rot_accel * delta_time;
421 
422  gyro.x = constrain_float(gyro.x, -radians(2000.0f), radians(2000.0f));
423  gyro.y = constrain_float(gyro.y, -radians(2000.0f), radians(2000.0f));
424  gyro.z = constrain_float(gyro.z, -radians(2000.0f), radians(2000.0f));
425 
426  // estimate angular acceleration using a first order difference calculation
427  // TODO the simulator interface should provide the angular acceleration
428  ang_accel = (gyro - gyro_prev) / delta_time;
429  gyro_prev = gyro;
430 
431  // update attitude
432  dcm.rotate(gyro * delta_time);
433  dcm.normalize();
434 
435  Vector3f accel_earth = dcm * accel_body;
436  accel_earth += Vector3f(0.0f, 0.0f, GRAVITY_MSS);
437 
438  // if we're on the ground, then our vertical acceleration is limited
439  // to zero. This effectively adds the force of the ground on the aircraft
440  if (on_ground() && accel_earth.z > 0) {
441  accel_earth.z = 0;
442  }
443 
444  // work out acceleration as seen by the accelerometers. It sees the kinematic
445  // acceleration (ie. real movement), plus gravity
446  accel_body = dcm.transposed() * (accel_earth + Vector3f(0.0f, 0.0f, -GRAVITY_MSS));
447 
448  // new velocity vector
449  velocity_ef += accel_earth * delta_time;
450 
451  const bool was_on_ground = on_ground();
452  // new position vector
453  position += velocity_ef * delta_time;
454 
455  // velocity relative to air mass, in earth frame
456  velocity_air_ef = velocity_ef + wind_ef;
457 
458  // velocity relative to airmass in body frame
459  velocity_air_bf = dcm.transposed() * velocity_air_ef;
460 
461  // airspeed
462  airspeed = velocity_air_ef.length();
463 
464  // airspeed as seen by a fwd pitot tube (limited to 120m/s)
465  airspeed_pitot = constrain_float(velocity_air_bf * Vector3f(1.0f, 0.0f, 0.0f), 0.0f, 120.0f);
466 
467  // constrain height to the ground
468  if (on_ground()) {
469  if (!was_on_ground && AP_HAL::millis() - last_ground_contact_ms > 1000) {
470  printf("Hit ground at %f m/s\n", velocity_ef.z);
471  last_ground_contact_ms = AP_HAL::millis();
472  }
473  position.z = -(ground_level + frame_height - home.alt * 0.01f + ground_height_difference());
474 
475  switch (ground_behavior) {
476  case GROUND_BEHAVIOR_NONE:
477  break;
478  case GROUND_BEHAVIOR_NO_MOVEMENT: {
479  // zero roll/pitch, but keep yaw
480  float r, p, y;
481  dcm.to_euler(&r, &p, &y);
482  dcm.from_euler(0.0f, 0.0f, y);
483  // no X or Y movement
484  velocity_ef.x = 0.0f;
485  velocity_ef.y = 0.0f;
486  if (velocity_ef.z > 0.0f) {
487  velocity_ef.z = 0.0f;
488  }
489  gyro.zero();
490  use_smoothing = true;
491  break;
492  }
493  case GROUND_BEHAVIOR_FWD_ONLY: {
494  // zero roll/pitch, but keep yaw
495  float r, p, y;
496  dcm.to_euler(&r, &p, &y);
497  dcm.from_euler(0.0f, 0.0f, y);
498  // only fwd movement
499  Vector3f v_bf = dcm.transposed() * velocity_ef;
500  v_bf.y = 0.0f;
501  if (v_bf.x < 0.0f) {
502  v_bf.x = 0.0f;
503  }
504  velocity_ef = dcm * v_bf;
505  if (velocity_ef.z > 0.0f) {
506  velocity_ef.z = 0.0f;
507  }
508  gyro.zero();
509  use_smoothing = true;
510  break;
511  }
512  case GROUND_BEHAVIOR_TAILSITTER: {
513  // point straight up
514  float r, p, y;
515  dcm.to_euler(&r, &p, &y);
516  dcm.from_euler(0.0f, radians(90), y);
517  // no movement
518  if (accel_earth.z > -1.1*GRAVITY_MSS) {
519  velocity_ef.zero();
520  }
521  gyro.zero();
522  use_smoothing = true;
523  break;
524  }
525  }
526  }

531 */
532 void Aircraft::update_wind(const struct sitl_input &input)
533 {
534  // wind vector in earth frame
535  wind_ef = Vector3f(cosf(radians(input.wind.direction)), sinf(radians(input.wind.direction)), 0) * input.wind.speed;
536 
537  const float wind_turb = input.wind.turbulence * 10.0f; // scale input.wind.turbulence to match standard deviation when using iir_coef=0.98
538  const float iir_coef = 0.98f; // filtering high frequencies from turbulence
539 
540  if (wind_turb > 0 && !on_ground()) {
541 
542  turbulence_azimuth = turbulence_azimuth + (2 * rand());
543 
544  turbulence_horizontal_speed =
545  static_cast<float>(turbulence_horizontal_speed * iir_coef+wind_turb * rand_normal(0, 1) * (1 - iir_coef));
546 
547  turbulence_vertical_speed = static_cast<float>((turbulence_vertical_speed * iir_coef) + (wind_turb * rand_normal(0, 1) * (1 - iir_coef)));
548 
549  wind_ef += Vector3f(
550  cosf(radians(turbulence_azimuth)) * turbulence_horizontal_speed,
551  sinf(radians(turbulence_azimuth)) * turbulence_horizontal_speed,
552  turbulence_vertical_speed);
553  }

556 /*
557  calculate magnetic field intensity and orientation
558 */
559 bool Aircraft::get_mag_field_ef(float latitude_deg, float longitude_deg, float &intensity_gauss, float &declination_deg, float &inclination_deg)
560 {
561  bool valid_input_data = true;
562 
563  /* round down to nearest sampling resolution */
564  int32_t min_lat = static_cast<int32_t>(static_cast<int32_t>(latitude_deg / SAMPLING_RES) * SAMPLING_RES);
565  int32_t min_lon = static_cast<int32_t>(static_cast<int32_t>(longitude_deg / SAMPLING_RES) * SAMPLING_RES);
566 
567  /* for the rare case of hitting the bounds exactly
568  * the rounding logic wouldn't fit, so enforce it.
569  */
570 
571  /* limit to table bounds - required for maxima even when table spans full globe range */
572  if (latitude_deg <= SAMPLING_MIN_LAT) {
573  min_lat = static_cast<int32_t>(SAMPLING_MIN_LAT);
574  valid_input_data = false;
575  }
576 
577  if (latitude_deg >= SAMPLING_MAX_LAT) {
578  min_lat = static_cast<int32_t>(static_cast<int32_t>(latitude_deg / SAMPLING_RES) * SAMPLING_RES - SAMPLING_RES);
579  valid_input_data = false;
580  }
581 
582  if (longitude_deg <= SAMPLING_MIN_LON) {
583  min_lon = static_cast<int32_t>(SAMPLING_MIN_LON);
584  valid_input_data = false;
585  }
586 
587  if (longitude_deg >= SAMPLING_MAX_LON) {
588  min_lon = static_cast<int32_t>(static_cast<int32_t>(longitude_deg / SAMPLING_RES) * SAMPLING_RES - SAMPLING_RES);
589  valid_input_data = false;
590  }
591 
592  /* find index of nearest low sampling point */
593  uint32_t min_lat_index = static_cast<uint32_t>((-(SAMPLING_MIN_LAT) + min_lat) / SAMPLING_RES);
594  uint32_t min_lon_index = static_cast<uint32_t>((-(SAMPLING_MIN_LON) + min_lon) / SAMPLING_RES);
595 
596  /* calculate intensity */
597 
598  float data_sw = intensity_table[min_lat_index][min_lon_index];
599  float data_se = intensity_table[min_lat_index][min_lon_index + 1];;
600  float data_ne = intensity_table[min_lat_index + 1][min_lon_index + 1];
601  float data_nw = intensity_table[min_lat_index + 1][min_lon_index];
602 
603  /* perform bilinear interpolation on the four grid corners */
604 
605  float data_min = ((longitude_deg - min_lon) / SAMPLING_RES) * (data_se - data_sw) + data_sw;
606  float data_max = ((longitude_deg - min_lon) / SAMPLING_RES) * (data_ne - data_nw) + data_nw;
607 
608  intensity_gauss = ((latitude_deg - min_lat) / SAMPLING_RES) * (data_max - data_min) + data_min;
609 
610  /* calculate declination */
611 
612  data_sw = declination_table[min_lat_index][min_lon_index];
613  data_se = declination_table[min_lat_index][min_lon_index + 1];;

618

620  data_max = ((longitude_deg - min_lon) / SAMPLING_RES) * (data_ne - data_nw) + data_nw;
621 
622  declination_deg = ((latitude_deg - min_lat) / SAMPLING_RES) * (data_max - data_min) + data_min;

624  /* calculate inclination */
625 
626  data_sw = inclination_table[min_lat_index][min_lon_index];
627  data_se = inclination_table[min_lat_index][min_lon_index + 1];;
628  data_ne = inclination_table[min_lat_index + 1][min_lon_index + 1];
629  data_nw = inclination_table[min_lat_index + 1][min_lon_index];

635

641 /*
642  smooth sensors for kinematic consistancy when we interact with the ground
643  */
644 void Aircraft::smooth_sensors(void)
645 {
646  uint64_t now = time_now_us;
647  Vector3f delta_pos = position - smoothing.position;
648  if (smoothing.last_update_us == 0 || delta_pos.length() > 10) {
649  smoothing.position = position;
650  smoothing.rotation_b2e = dcm;
651  smoothing.accel_body = accel_body;
652  smoothing.velocity_ef = velocity_ef;
653  smoothing.gyro = gyro;
654  smoothing.last_update_us = now;
655  smoothing.location = location;
656  printf("Smoothing reset at %.3f\n", now * 1.0e-6f);
657  return;
658  }
659  const float delta_time = (now - smoothing.last_update_us) * 1.0e-6f;
660 
661  // calculate required accel to get us to desired position and velocity in the time_constant
662  const float time_constant = 0.1f;
663  Vector3f dvel = (velocity_ef - smoothing.velocity_ef) + (delta_pos / time_constant);
664  Vector3f accel_e = dvel / time_constant + (dcm * accel_body + Vector3f(0.0f, 0.0f, GRAVITY_MSS));
665  const float accel_limit = 14 * GRAVITY_MSS;
666  accel_e.x = constrain_float(accel_e.x, -accel_limit, accel_limit);
667  accel_e.y = constrain_float(accel_e.y, -accel_limit, accel_limit);
668  accel_e.z = constrain_float(accel_e.z, -accel_limit, accel_limit);
669  smoothing.accel_body = smoothing.rotation_b2e.transposed() * (accel_e + Vector3f(0.0f, 0.0f, -GRAVITY_MSS));
670 
671  // calculate rotational rate to get us to desired attitude in time constant
672  Quaternion desired_q, current_q, error_q;
673  desired_q.from_rotation_matrix(dcm);
674  desired_q.normalize();
675  current_q.from_rotation_matrix(smoothing.rotation_b2e);
676  current_q.normalize();
677  error_q = desired_q / current_q;
678  error_q.normalize();
679 
680  Vector3f angle_differential;
681  error_q.to_axis_angle(angle_differential);
682  smoothing.gyro = gyro + angle_differential / time_constant;
683 
684  float R, P, Y;
685  smoothing.rotation_b2e.to_euler(&R, &P, &Y);
686  float R2, P2, Y2;
687  dcm.to_euler(&R2, &P2, &Y2);
688 
689 #if 0
690  DataFlash_Class::instance()->Log_Write("SMOO", "TimeUS,AEx,AEy,AEz,DPx,DPy,DPz,R,P,Y,R2,P2,Y2",
691  "Qffffffffffff",
692  AP_HAL::micros64(),
693  degrees(angle_differential.x),
694  degrees(angle_differential.y),
695  degrees(angle_differential.z),
696  delta_pos.x, delta_pos.y, delta_pos.z,
697  degrees(R), degrees(P), degrees(Y),
698  degrees(R2), degrees(P2), degrees(Y2));
699 #endif
700 
701 
702  // integrate to get new attitude
703  smoothing.rotation_b2e.rotate(smoothing.gyro * delta_time);
704  smoothing.rotation_b2e.normalize();
705 
706  // integrate to get new position
707  smoothing.velocity_ef += accel_e * delta_time;
708  smoothing.position += smoothing.velocity_ef * delta_time;
709 
710  smoothing.location = home;
711  location_offset(smoothing.location, smoothing.position.x, smoothing.position.y);
712  smoothing.location.alt = static_cast<int32_t>(home.alt - smoothing.position.z * 100.0f);
713 
714  smoothing.last_update_us = now;
715  smoothing.enabled = true;
716 }
