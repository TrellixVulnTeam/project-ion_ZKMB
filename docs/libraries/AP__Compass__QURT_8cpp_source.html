56 void AP_Compass_QURT::read(void)

67 void AP_Compass_QURT::timer_update(void)
68 {
69  // cope the data
70  struct mpu9x50_data data;
71  if (mpu9250_mag_buffer == nullptr || !mpu9250_mag_buffer->pop(data)) {
72  return;
73  }
74 
75  last_timestamp = data.timestamp;
76 
77  Vector3f raw_field(data.mag_raw[0],
78  data.mag_raw[1],
79  -data.mag_raw[2]);
80 
81  // rotate raw_field from sensor frame to body frame
82  rotate_field(raw_field, instance);
83 
84  // publish raw_field (uncorrected point sample) for calibration use
85  publish_raw_field(raw_field, data.timestamp, instance);
86 
87  // correct raw_field for known errors
88  correct_field(raw_field, instance);
89 
90  // accumulate into averaging filter
91  sum += raw_field;
92  count++;
93 }
