305 // calculate_scalars
306 void AP_MotorsHeli_Dual::calculate_scalars()
307 {
308  // range check collective min, max and mid
309  if( _collective_min >= _collective_max ) {
310  _collective_min = AP_MOTORS_HELI_COLLECTIVE_MIN;
311  _collective_max = AP_MOTORS_HELI_COLLECTIVE_MAX;
312  }
313 
314 
315  // range check collective min, max and mid for rear swashplate
316  if( _collective2_min >= _collective2_max ) {
317  _collective2_min = AP_MOTORS_HELI_DUAL_COLLECTIVE2_MIN;
318  _collective2_max = AP_MOTORS_HELI_DUAL_COLLECTIVE2_MAX;
319  }
320 
321  _collective_mid = constrain_int16(_collective_mid, _collective_min, _collective_max);
322  _collective2_mid = constrain_int16(_collective2_mid, _collective2_min, _collective2_max);
323 
324  // calculate collective mid point as a number from 0 to 1000
325  _collective_mid_pct = ((float)(_collective_mid-_collective_min))/((float)(_collective_max-_collective_min));
326  _collective2_mid_pct = ((float)(_collective2_mid-_collective2_min))/((float)(_collective2_max-_collective2_min));
327 
328  // calculate factors based on swash type and servo position
329  calculate_roll_pitch_collective_factors();
330 
331  // set mode of main rotor controller and trigger recalculation of scalars
332  _rotor.set_control_mode(static_cast<RotorControlMode>(_rsc_mode.get()));
333  calculate_armed_scalars();
334 }

431 //
432 // move_actuators - moves swash plate to attitude of parameters passed in
433 // - expected ranges:
434 // roll : -4500 ~ 4500
435 // pitch: -4500 ~ 4500
436 // collective: 0 ~ 1000
437 // yaw: -4500 ~ 4500
438 //
439 void AP_MotorsHeli_Dual::move_actuators(float roll_out, float pitch_out, float collective_in, float yaw_out)
440 {
441  // initialize limits flag
442  limit.roll_pitch = false;
443  limit.yaw = false;
444  limit.throttle_lower = false;
445  limit.throttle_upper = false;
446 
447  if (_dual_mode == AP_MOTORS_HELI_DUAL_MODE_TRANSVERSE) {
448  if (pitch_out < -_cyclic_max/4500.0f) {
449  pitch_out = -_cyclic_max/4500.0f;
450  limit.roll_pitch = true;
451  }
452 
453  if (pitch_out > _cyclic_max/4500.0f) {
454  pitch_out = _cyclic_max/4500.0f;
455  limit.roll_pitch = true;
456  }
457  } else {
458  if (roll_out < -_cyclic_max/4500.0f) {
459  roll_out = -_cyclic_max/4500.0f;
460  limit.roll_pitch = true;
461  }
462 
463  if (roll_out > _cyclic_max/4500.0f) {
464  roll_out = _cyclic_max/4500.0f;
465  limit.roll_pitch = true;
466  }
467  }
468 
469 
470  float yaw_compensation = 0.0f;
471 
472  // if servo output not in manual mode, process pre-compensation factors
473  if (_servo_mode == SERVO_CONTROL_MODE_AUTOMATED) {
474  // add differential collective pitch yaw compensation
475  if (_dual_mode == AP_MOTORS_HELI_DUAL_MODE_TRANSVERSE) {
476  yaw_compensation = _dcp_yaw_effect * roll_out;
477  } else { // AP_MOTORS_HELI_DUAL_MODE_TANDEM
478  yaw_compensation = _dcp_yaw_effect * pitch_out;
479  }
480  yaw_out = yaw_out + yaw_compensation;
481  }
482 
483  // scale yaw and update limits
484  if (yaw_out < -_cyclic_max/4500.0f) {
485  yaw_out = -_cyclic_max/4500.0f;
486  limit.yaw = true;
487  }
488  if (yaw_out > _cyclic_max/4500.0f) {
489  yaw_out = _cyclic_max/4500.0f;
490  limit.yaw = true;
491  }
492 
493  // constrain collective input
494  float collective_out = collective_in;
495  if (collective_out <= 0.0f) {
496  collective_out = 0.0f;
497  limit.throttle_lower = true;
498  }
499  if (collective_out >= 1.0f) {
500  collective_out = 1.0f;
501  limit.throttle_upper = true;
502  }
503 
504  // Set rear collective to midpoint if required
505  float collective2_out = collective_out;
506  if (_servo_mode == SERVO_CONTROL_MODE_MANUAL_CENTER) {
507  collective2_out = _collective2_mid_pct;
508  }
509 
510 
511  // ensure not below landed/landing collective
512  if (_heliflags.landing_collective && collective_out < (_land_collective_min/1000.0f)) {

514  limit.throttle_lower = true;
515  }
516

518  float collective_scaler = ((float)(_collective_max-_collective_min))/1000.0f;
519  float collective_out_scaled = collective_out * collective_scaler + (_collective_min - 1000)/1000.0f;

528 
529  // swashplate servos
530  float servo1_out = (_rollFactor[CH_1] * roll_out + _pitchFactor[CH_1] * pitch_out + _yawFactor[CH_1] * yaw_out)/0.45f + _collectiveFactor[CH_1] * collective_out_scaled;
531  float servo2_out = (_rollFactor[CH_2] * roll_out + _pitchFactor[CH_2] * pitch_out + _yawFactor[CH_2] * yaw_out)/0.45f + _collectiveFactor[CH_2] * collective_out_scaled;

538  servo1_out = 2*servo1_out - 1;
539  servo2_out = 2*servo2_out - 1;
540  servo3_out = 2*servo3_out - 1;
541  servo4_out = 2*servo4_out - 1;
542  servo5_out = 2*servo5_out - 1;
543  servo6_out = 2*servo6_out - 1;

559 // servo_test - move servos through full range of movement
560 void AP_MotorsHeli_Dual::servo_test()
561 {
562  // this test cycle is equivalent to that of AP_MotorsHeli_Single, but excluding
563  // mixing of yaw, as that physical movement is represented by pitch and roll
564 
565  _servo_test_cycle_time += 1.0f / _loop_rate;
566 
567  if ((_servo_test_cycle_time >= 0.0f && _servo_test_cycle_time < 0.5f)|| // Tilt swash back
568  (_servo_test_cycle_time >= 6.0f && _servo_test_cycle_time < 6.5f)){
569  _pitch_test += (1.0f / (_loop_rate/2));
570  _oscillate_angle += 8 * M_PI / _loop_rate;
571  } else if ((_servo_test_cycle_time >= 0.5f && _servo_test_cycle_time < 4.5f)|| // Roll swash around
572  (_servo_test_cycle_time >= 6.5f && _servo_test_cycle_time < 10.5f)){
573  _oscillate_angle += M_PI / (2 * _loop_rate);
574  _roll_test = sinf(_oscillate_angle);
575  _pitch_test = cosf(_oscillate_angle);
576  } else if ((_servo_test_cycle_time >= 4.5f && _servo_test_cycle_time < 5.0f)|| // Return swash to level
577  (_servo_test_cycle_time >= 10.5f && _servo_test_cycle_time < 11.0f)){
578  _pitch_test -= (1.0f / (_loop_rate/2));
579  _oscillate_angle += 8 * M_PI / _loop_rate;
580  } else if (_servo_test_cycle_time >= 5.0f && _servo_test_cycle_time < 6.0f){ // Raise swash to top
581  _collective_test += (1.0f / _loop_rate);
582  _oscillate_angle += 2 * M_PI / _loop_rate;
583  } else if (_servo_test_cycle_time >= 11.0f && _servo_test_cycle_time < 12.0f){ // Lower swash to bottom
584  _collective_test -= (1.0f / _loop_rate);
585  _oscillate_angle += 2 * M_PI / _loop_rate;
586  } else { // reset cycle
587  _servo_test_cycle_time = 0.0f;
588  _oscillate_angle = 0.0f;
589  _collective_test = 0.0f;
590  _roll_test = 0.0f;
591  _pitch_test = 0.0f;
592  // decrement servo test cycle counter at the end of the cycle
593  if (_servo_test_cycle_counter > 0){
594  _servo_test_cycle_counter--;
595  }
596  }
597 
598  // over-ride servo commands to move servos through defined ranges
599 
600  _throttle_in = _collective_test;
601  _roll_in = _roll_test;
602  _pitch_in = _pitch_test;
603 }
