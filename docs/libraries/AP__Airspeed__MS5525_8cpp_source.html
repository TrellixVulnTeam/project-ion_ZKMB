59 // probe and initialise the sensor
60 bool AP_Airspeed_MS5525::init()
61 {
62  const uint8_t addresses[] = { MS5525D0_I2C_ADDR_1, MS5525D0_I2C_ADDR_2 };
63  bool found = false;
64  for (uint8_t i=0; i<ARRAY_SIZE(addresses); i++) {
65  dev = hal.i2c_mgr->get_device(get_bus(), addresses[i]);
66  if (!dev) {
67  continue;
68  }
69  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
70  continue;
71  }
72 
73  // lots of retries during probe
74  dev->set_retries(5);
75 
76  found = read_prom();
77 
78  if (found) {
79  printf("MS5525: Found sensor on bus %u address 0x%02x\n", get_bus(), addresses[i]);
80  break;
81  }
82  dev->get_semaphore()->give();
83  }
84  if (!found) {
85  printf("MS5525: no sensor found\n");
86  return false;
87  }
88 
89  // Send a command to read temperature first
90  uint8_t reg = REG_CONVERT_TEMPERATURE;
91  dev->transfer(&reg, 1, nullptr, 0);
92  state = 0;
93 
94  // drop to 2 retries for runtime
95  dev->set_retries(2);
96 
97  dev->get_semaphore()->give();
98 
99  // read at 80Hz
100  dev->register_periodic_callback(1000000UL/80U,
101  FUNCTOR_BIND_MEMBER(&AP_Airspeed_MS5525::timer, void));
102  return true;
103 }

109 uint16_t AP_Airspeed_MS5525::crc4_prom(void)
110 {
111  uint16_t n_rem = 0;
112  uint8_t n_bit;
113 
114  for (uint8_t cnt = 0; cnt < sizeof(prom); cnt++) {
115  /* uneven bytes */
116  if (cnt & 1) {
117  n_rem ^= (uint8_t)((prom[cnt >> 1]) & 0x00FF);
118  } else {
119  n_rem ^= (uint8_t)(prom[cnt >> 1] >> 8);
120  }
121 
122  for (n_bit = 8; n_bit > 0; n_bit--) {
123  if (n_rem & 0x8000) {
124  n_rem = (n_rem << 1) ^ 0x3000;
125  } else {
126  n_rem = (n_rem << 1);
127  }
128  }
129  }
130 
131  return (n_rem >> 12) & 0xF;
132 }

134 bool AP_Airspeed_MS5525::read_prom(void)
135 {
136  // reset the chip to ensure it has correct prom values loaded
137  uint8_t reg = REG_RESET;
138  if (!dev->transfer(&reg, 1, nullptr, 0)) {
139  return false;
140  }
141  hal.scheduler->delay(5);
142 
143  bool all_zero = true;
144  for (uint8_t i = 0; i < 8; i++) {
145  be16_t val;
146  if (!dev->read_registers(REG_PROM_BASE+i*2, (uint8_t *) &val,
147  sizeof(uint16_t))) {
148  return false;
149  }
150  prom[i] = be16toh(val);
151  if (prom[i] != 0) {
152  all_zero = false;
153  }
154  }
155 
156  if (all_zero) {
157  return false;
158  }
159 
160  /* save the read crc */
161  const uint16_t crc_read = prom[7] & 0xf;
162 
163  /* remove CRC byte */
164  prom[7] &= 0xff00;
165 
166  uint16_t crc_calc = crc4_prom();
167  if (crc_read != crc_calc) {
168  printf("MS5525: CRC mismatch 0x%04x 0x%04x\n", crc_read, crc_calc);
169  }
170  return crc_read == crc_calc;
171 }

177 int32_t AP_Airspeed_MS5525::read_adc()
178 {
179  uint8_t val[3];
180  if (!dev->read_registers(REG_ADC_READ, val, 3)) {

186 /*
187  calculate pressure and temperature
188  */
189 void AP_Airspeed_MS5525::calculate(void)
190 {
191  // table for the 001DS part, 1PSI range
192  const uint8_t Q1 = 15;
193  const uint8_t Q2 = 17;
194  const uint8_t Q3 = 7;
195  const uint8_t Q4 = 5;
196  const uint8_t Q5 = 7;
197  const uint8_t Q6 = 21;
198 
199  int64_t dT = D2 - int64_t(prom[5]) * (1UL<<Q5);
200  int64_t TEMP = 2000 + (dT*int64_t(prom[6]))/(1UL<<Q6);
201  int64_t OFF = int64_t(prom[2])*(1UL<<Q2) + (int64_t(prom[4])*dT)/(1UL<<Q4);
202  int64_t SENS = int64_t(prom[1])*(1UL<<Q1) + (int64_t(prom[3])*dT)/(1UL<<Q3);
203  int64_t P = (D1*SENS/(1UL<<21)-OFF)/(1UL<<15);
204  const float PSI_to_Pa = 6894.757f;
205  float P_Pa = PSI_to_Pa * 1.0e-4 * P;
206  float Temp_C = TEMP * 0.01;
207 
208 #if 0
209  static uint16_t counter;
210  if (counter++ == 100) {
211  printf("P=%.6f T=%.2f D1=%d D2=%d\n", P_Pa, Temp_C, D1, D2);
212  counter=0;
213  }
214 #endif
215 
216  if (sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
217  pressure_sum += P_Pa;
218  temperature_sum += Temp_C;
219  press_count++;
220  temp_count++;
221  last_sample_time_ms = AP_HAL::millis();
222  sem->give();
223  }
224 }

226 // 50Hz timer
227 void AP_Airspeed_MS5525::timer()
228 {
229  uint32_t adc_val = read_adc();
230 
231  /*
232  * If read fails, re-initiate a read command for current state or we are
233  * stuck
234  */
235  uint8_t next_state = state;
236  if (adc_val != 0) {
237  next_state = (state + 1) % 5;
238 
239  if (state == 0) {
240  D2 = adc_val;
241  } else {
242  D1 = adc_val;
243  calculate();
244  }
245  }
246 
247  uint8_t next_cmd = next_state == 0 ? REG_CONVERT_TEMPERATURE : REG_CONVERT_PRESSURE;
248  if (!dev->transfer(&next_cmd, 1, nullptr, 0)) {
249  return;
250  }
251 
252  state = next_state;
253 }

255 // return the current differential_pressure in Pascal
256 bool AP_Airspeed_MS5525::get_differential_pressure(float &_pressure)
257 {
258  if ((AP_HAL::millis() - last_sample_time_ms) > 100) {
259  return false;
260  }
261  if (sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
262  if (press_count > 0) {
263  pressure = pressure_sum / press_count;
264  press_count = 0;
265  pressure_sum = 0;
266  }
267  sem->give();
268  }
