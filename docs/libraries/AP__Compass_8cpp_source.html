857 /*
858  calculate a compass heading given the attitude from DCM and the mag vector
859  */
860 float
861 Compass::calculate_heading(const Matrix3f &dcm_matrix, uint8_t i) const
862 {
863  float cos_pitch_sq = 1.0f-(dcm_matrix.c.x*dcm_matrix.c.x);
864 
865  // Tilt compensated magnetic field Y component:
866  const Vector3f &field = get_field(i);
867 
868  float headY = field.y * dcm_matrix.c.z - field.z * dcm_matrix.c.y;
869 
870  // Tilt compensated magnetic field X component:
871  float headX = field.x * cos_pitch_sq - dcm_matrix.c.x * (field.y * dcm_matrix.c.y + field.z * dcm_matrix.c.z);
872 
873  // magnetic heading
874  // 6/4/11 - added constrain to keep bad values from ruining DCM Yaw - Jason S.
875  float heading = constrain_float(atan2f(-headY,headX), -3.15f, 3.15f);
876 
877  // Declination correction (if supplied)
878  if( fabsf(_declination) > 0.0f )
879  {
880  heading = heading + _declination;
881  if (heading > M_PI) // Angle normalization (-180 deg, 180 deg)
882  heading -= (2.0f * M_PI);
883  else if (heading < -M_PI)
884  heading += (2.0f * M_PI);
885  }
886 
887  return heading;
888 }
