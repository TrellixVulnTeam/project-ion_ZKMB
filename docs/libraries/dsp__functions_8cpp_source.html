121 /*
122  thread gathering sensor data from mpu9250
123  */
124 static void *mpu_data_ready(void *ctx)
125 {
126  struct mpu9x50_data data;
127  memset(&data, 0, sizeof(data));
128 
129  int ret = mpu9x50_get_data(&data);
130  if (ret != 0) {
131  return nullptr;
132  }
133  DSPBuffer::IMU::BUF b;
134  b.timestamp = data.timestamp;
135  b.accel[0] = data.accel_raw[0]*ACCEL_SCALE_1G;
136  b.accel[1] = data.accel_raw[1]*ACCEL_SCALE_1G;
137  b.accel[2] = data.accel_raw[2]*ACCEL_SCALE_1G;
138  b.gyro[0] = data.gyro_raw[0]*GYRO_SCALE;
139  b.gyro[1] = data.gyro_raw[1]*GYRO_SCALE;
140  b.gyro[2] = data.gyro_raw[2]*GYRO_SCALE;
141  imu_buffer.push(b);
142 
143  if (data.mag_data_ready) {
144  DSPBuffer::MAG::BUF m;
145  m.mag_raw[0] = data.mag_raw[0];
146  m.mag_raw[1] = data.mag_raw[1];
147  m.mag_raw[2] = data.mag_raw[2];
148  m.timestamp = data.timestamp;
149  mag_buffer.push(m);
150  }
151 
152  if (bmp280_handle != 0 && baro_counter++ % 10 == 0) {
153  struct bmp280_sensor_data data;
154  memset(&data, 0, sizeof(data));
155  int ret = bmp280_get_sensor_data(bmp280_handle, &data, false);
156  if (ret == 0) {
157  DSPBuffer::BARO::BUF b;
158  b.pressure_pa = data.pressure_in_pa;
159  b.temperature_C = data.temperature_in_c;
160  b.timestamp = data.last_read_time_in_usecs;
161  baro_buffer.push(b);
162  }
163  }
164 
165  return nullptr;
166 }

268 /*
269  open a UART
270  */
271 int qflight_UART_open(const char *device, int32_t *_fd)
272 {
273  if (num_open_uarts == max_uarts) {
274  return -1;
275  }
276  struct uartbuf &b = uarts[num_open_uarts];
277  int fd = open(device, O_RDWR | O_NONBLOCK|O_CLOEXEC);
278  if (fd == -1) {
279  return -1;
280  }
281  b.fd = fd;
282  b.readbuffer = new ByteBuffer(16384);
283 
284  struct dspal_serial_open_options options;
285  options.bit_rate = DSPAL_SIO_BITRATE_57600;
286  options.tx_flow = DSPAL_SIO_FCTL_OFF;
287  options.rx_flow = DSPAL_SIO_FCTL_OFF;
288  options.rx_data_callback = nullptr;
289  options.tx_data_callback = nullptr;
290  options.is_tx_data_synchronous = false;
291  int ret = ioctl(fd, SERIAL_IOCTL_OPEN_OPTIONS, (void *)&options);
292  if (ret != 0) {
293  HAP_PRINTF("Failed to setup UART flow control options");
294  }
295 
296  struct dspal_serial_ioctl_receive_data_callback callback {};
297  callback.context = b.readbuffer;
298  callback.rx_data_callback_func_ptr = read_callback_trampoline;
299  ret = ioctl(fd, SERIAL_IOCTL_SET_RECEIVE_DATA_CALLBACK, (void *)&callback);
300  if (ret != 0) {
301  HAP_PRINTF("Failed to setup UART read trampoline");
302  delete b.readbuffer;
303  close(fd);
304  return -1;
305  }
306 
307  HAP_PRINTF("UART open %s fd=%d num_open=%u",
308  device, fd, num_open_uarts);
309  num_open_uarts++;
310  *_fd = fd;
311  return 0;
312 }

314 /*
315  close a UART
316  */
317 int qflight_UART_close(int32_t fd)
318 {
319  uint8_t i;
320  for (i=0; i<num_open_uarts; i++) {
321  if (fd == uarts[i].fd) break;
322  }
323  if (i == num_open_uarts) {
324  return -1;
325  }
326  close(fd);
327  delete uarts[i].readbuffer;
328  if (i < num_open_uarts-1) {
329  memmove(&uarts[i], &uarts[i+1], ((num_open_uarts-1)-i)*sizeof(uarts[0]));
330  }
331  num_open_uarts--;
332  return 0;
333 }
