53 // update internal state if new GPS information is available
54 bool
55 AP_GPS_QURT::read(void)
56 {
57  if (!initialised) {
58  return false;
59  }
60  struct osp_geo_data data {};
61  if (csr_gps_get_geo_data(&data) != 0) {
62  return false;
63  }
64  state.last_gps_time_ms = AP_HAL::millis();
65  if (data.tow == last_tow) {
66  // same data again
67  return false;
68  }
69 
70  if (data.nav_type & NAV_TYPE_4SV_OR_MORE_KF_SOLUTION ||
71  data.nav_type & NAV_TYPE_3SV_KF_SOLUTION) {
72  state.status = AP_GPS::GPS_OK_FIX_3D;
73  } else if (data.nav_type & NAV_TYPE_2SV_KF_SOLUTION) {
74  state.status = AP_GPS::GPS_OK_FIX_2D;
75  } else {
76  state.status = AP_GPS::NO_FIX;
77  }
78 
79  state.num_sats = data.sv_in_fix;
80  state.hdop = data.HDOP;
81  state.vdop = 0;
82 
83  state.location.lat = bswap_32(data.lat);
84  state.location.lng = bswap_32(data.lon);
85  state.location.alt = data.alt_from_MSL;
86 
87  state.ground_speed = data.speed_over_ground;
88  state.ground_course = data.course_over_ground*0.01f;
89 
90  // convert epoch timestamp back to gps epoch - evil hack until we get the genuine
91  // raw week information (or APM switches to Posix epoch ;-) )
92  state.time_week = data.ext_week_num;
93  state.time_week_ms = data.tow;
94 
95  if (state.time_week == 0) {
96  // reject bad time
97  state.status = AP_GPS::NO_FIX;
98  }
99 
100  state.have_vertical_velocity = true;
101  float gps_heading = radians(state.ground_course);
102  state.velocity.x = state.ground_speed * cosf(gps_heading);
103  state.velocity.y = state.ground_speed * sinf(gps_heading);
104  state.velocity.z = -data.climb_rate;
105  state.speed_accuracy = data.est_hor_vel_error * 0.01f;
106  state.horizontal_accuracy = data.est_hor_pos_error * 0.01f;
107  state.vertical_accuracy = data.est_vert_pos_error * 0.01f;
108  state.have_speed_accuracy = true;
109  return true;
110 }
