36  handle interrupt on an instance

38 void AP_RPM_Pin::irq_handler(uint8_t instance)
39 {
40  uint32_t now = AP_HAL::micros();
41  uint32_t dt = now - irq_state[instance].last_pulse_us;
42  irq_state[instance].last_pulse_us = now;
43  // we don't accept pulses less than 100us. Using an irq for such
44  // high RPM is too inaccurate, and it is probably just bounce of
45  // the signal which we should ignore
46  if (dt > 100 && dt < 1000*1000) {
47  irq_state[instance].dt_sum += dt;
48  irq_state[instance].dt_count++;
49  }
50 }

70 void AP_RPM_Pin::update(void)
71 {
72  if (last_pin != get_pin()) {
73  last_pin = get_pin();
74  uint32_t gpio = 0;
75 
76 #ifdef GPIO_GPIO0_INPUT
77  switch (last_pin) {
78  case 50:
79  gpio = GPIO_GPIO0_INPUT;
80  break;
81  case 51:
82  gpio = GPIO_GPIO1_INPUT;
83  break;
84  case 52:
85  gpio = GPIO_GPIO2_INPUT;
86  break;
87  case 53:
88  gpio = GPIO_GPIO3_INPUT;
89  break;
90  case 54:
91  gpio = GPIO_GPIO4_INPUT;
92  break;
93  case 55:
94  gpio = GPIO_GPIO5_INPUT;
95  break;
96  }
97 #endif // GPIO_GPIO5_INPUT
98 
99  // uninstall old handler if installed
100  if (last_gpio != 0) {
101  stm32_gpiosetevent(last_gpio, false, false, false, nullptr);
102  }
103  irq_state[state.instance].dt_count = 0;
104  irq_state[state.instance].dt_sum = 0;
105 
106  last_gpio = gpio;
107 
108  if (gpio == 0) {
109  return;
110  }
111 
112  // install interrupt handler on rising edge of pin. This works
113  // for either polarity of pulse, as all we want is the period
114  stm32_gpiosetevent(gpio, true, false, false,
115  state.instance==0?irq_handler0:irq_handler1);
116  }
117 
118  if (irq_state[state.instance].dt_count > 0) {
119  float dt_avg;
120 
121  // disable interrupts to prevent race with irq_handler
122  irqstate_t istate = irqsave();
123  dt_avg = irq_state[state.instance].dt_sum / irq_state[state.instance].dt_count;
124  irq_state[state.instance].dt_count = 0;
125  irq_state[state.instance].dt_sum = 0;
126  irqrestore(istate);
127 
128  const float scaling = ap_rpm._scaling[state.instance];
129  float maximum = ap_rpm._maximum[state.instance];
130  float minimum = ap_rpm._minimum[state.instance];
131  float quality = 0;
132  float rpm = scaling * (1.0e6 / dt_avg) * 60;
133  float filter_value = signal_quality_filter.get();
134 
135  state.rate_rpm = signal_quality_filter.apply(rpm);
136 
137  if ((maximum <= 0 || rpm <= maximum) && (rpm >= minimum)) {
138  if (is_zero(filter_value)){
139  quality = 0;
140  } else {
141  quality = 1 - constrain_float((fabsf(rpm-filter_value))/filter_value, 0.0, 1.0);
142  quality = powf(quality, 2.0);
143  }
144  state.last_reading_ms = AP_HAL::millis();
145  } else {
146  quality = 0;
147  }
148  state.signal_quality = (0.1 * quality) + (0.9 * state.signal_quality);
149  }
150 
151  // assume we get readings at at least 1Hz, otherwise reset quality to zero
152  if (AP_HAL::millis() - state.last_reading_ms > 1000) {
153  state.signal_quality = 0;
154  state.rate_rpm = 0;
155  }
156 }
