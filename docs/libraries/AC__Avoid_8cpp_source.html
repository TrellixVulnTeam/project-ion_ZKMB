183 {
184  // exit if circular fence is not enabled
185  if ((_fence.get_enabled_fences() & AC_FENCE_TYPE_CIRCLE) == 0) {
186  return;
187  }
188 
189  // exit if the circular fence has already been breached
190  if ((_fence.get_breaches() & AC_FENCE_TYPE_CIRCLE) != 0) {
191  return;
192  }
193 
194  // get position as a 2D offset in cm from ahrs home
195  const Vector2f position_xy = get_position();
196 
197  float speed = desired_vel.length();
198  // get the fence radius in cm
199  const float fence_radius = _fence.get_radius() * 100.0f;
200  // get the margin to the fence in cm
201  const float margin = get_margin();
202 
203  if (!is_zero(speed) && position_xy.length() <= fence_radius) {
204  // Currently inside circular fence
205  Vector2f stopping_point = position_xy + desired_vel*(get_stopping_distance(kP, accel_cmss, speed)/speed);
206  float stopping_point_length = stopping_point.length();
207  if (stopping_point_length > fence_radius - margin) {
208  // Unsafe desired velocity - will not be able to stop before fence breach
209  // Project stopping point radially onto fence boundary
210  // Adjusted velocity will point towards this projected point at a safe speed
211  Vector2f target = stopping_point * ((fence_radius - margin) / stopping_point_length);
212  Vector2f target_direction = target - position_xy;
213  float distance_to_target = target_direction.length();
214  float max_speed = get_max_speed(kP, accel_cmss, distance_to_target);
215  desired_vel = target_direction * (MIN(speed,max_speed) / distance_to_target);
216  }
217  }

273 void AC_Avoid::adjust_velocity_polygon(float kP, float accel_cmss, Vector2f &desired_vel, const Vector2f* boundary, uint16_t num_points, bool earth_frame)
274 {
275  // exit if there are no points
276  if (boundary == nullptr || num_points == 0) {
277  return;
278  }
279 
280  // do not adjust velocity if vehicle is outside the polygon fence
281  Vector3f position;
282  if (earth_frame) {
283  position = _inav.get_position();
284  }
285  Vector2f position_xy(position.x, position.y);
286  if (_fence.boundary_breached(position_xy, num_points, boundary)) {
287  return;
288  }
289 
290  // Safe_vel will be adjusted to remain within fence.
291  // We need a separate vector in case adjustment fails,
292  // e.g. if we are exactly on the boundary.
293  Vector2f safe_vel(desired_vel);
294 
295  // if boundary points are in body-frame, rotate velocity vector from earth frame to body-frame
296  if (!earth_frame) {
297  safe_vel.x = desired_vel.y * _ahrs.sin_yaw() + desired_vel.x * _ahrs.cos_yaw(); // right
298  safe_vel.y = desired_vel.y * _ahrs.cos_yaw() - desired_vel.x * _ahrs.sin_yaw(); // forward
299  }
300 
301  uint16_t i, j;
302  for (i = 1, j = num_points-1; i < num_points; j = i++) {
303  // end points of current edge
304  Vector2f start = boundary[j];
305  Vector2f end = boundary[i];
306  // vector from current position to closest point on current edge
307  Vector2f limit_direction = Vector2f::closest_point(position_xy, start, end) - position_xy;
308  // distance to closest point
309  const float limit_distance = limit_direction.length();
310  if (!is_zero(limit_distance)) {
311  // We are strictly inside the given edge.
312  // Adjust velocity to not violate this edge.
313  limit_direction /= limit_distance;
314  limit_velocity(kP, accel_cmss, safe_vel, limit_direction, MAX(limit_distance - get_margin(),0.0f));
315  } else {
316  // We are exactly on the edge - treat this as a fence breach.
317  // i.e. do not adjust velocity.
318  return;
319  }
320  }
321 
322  // set modified desired velocity vector
323  if (earth_frame) {
324  desired_vel = safe_vel;
325  } else {
326  // if points were in body-frame, rotate resulting vector back to earth-frame
327  desired_vel.x = safe_vel.x * _ahrs.cos_yaw() - safe_vel.y * _ahrs.sin_yaw();
328  desired_vel.y = safe_vel.x * _ahrs.sin_yaw() + safe_vel.y * _ahrs.cos_yaw();
329  }
330 }

384 float AC_Avoid::get_stopping_distance(float kP, float accel_cmss, float speed) const
385 {
386  // avoid divide by zero by using current position if the velocity is below 10cm/s, kP is very low or acceleration is zero
387  if (kP <= 0.0f || accel_cmss <= 0.0f || is_zero(speed)) {
388  return 0.0f;
389  }
390 
391  // calculate distance within which we can stop
392  // accel_cmss/kP is the point at which velocity switches from linear to sqrt

394  return speed/kP;
395  } else {
396  // accel_cmss/(2.0f*kP*kP) is the distance at which we switch from linear to sqrt response
397  return accel_cmss/(2.0f*kP*kP) + (speed*speed)/(2.0f*accel_cmss);
398  }
