138 uint8_t AC_Fence::check_fence(float curr_alt)
139 {
140  uint8_t ret = AC_FENCE_TYPE_NONE;
141 
142  // return immediately if disabled
143  if (!_enabled || _enabled_fences == AC_FENCE_TYPE_NONE) {
144  return AC_FENCE_TYPE_NONE;
145  }
146 
147  // check if pilot is attempting to recover manually
148  if (_manual_recovery_start_ms != 0) {
149  // we ignore any fence breaches during the manual recovery period which is about 10 seconds
150  if ((AP_HAL::millis() - _manual_recovery_start_ms) < AC_FENCE_MANUAL_RECOVERY_TIME_MIN) {
151  return AC_FENCE_TYPE_NONE;
152  } else {
153  // recovery period has passed so reset manual recovery time and continue with fence breach checks
154  _manual_recovery_start_ms = 0;
155  }
156  }
157 
158  // altitude fence check
159  if ((_enabled_fences & AC_FENCE_TYPE_ALT_MAX) != 0) {
160 
161  // check if we are over the altitude fence
162  if( curr_alt >= _alt_max ) {
163 
164  // record distance above breach
165  _alt_max_breach_distance = curr_alt - _alt_max;
166 
167  // check for a new breach or a breach of the backup fence
168  if ((_breached_fences & AC_FENCE_TYPE_ALT_MAX) == 0 || (!is_zero(_alt_max_backup) && curr_alt >= _alt_max_backup)) {
169 
170  // record that we have breached the upper limit
171  record_breach(AC_FENCE_TYPE_ALT_MAX);
172  ret |= AC_FENCE_TYPE_ALT_MAX;
173 
174  // create a backup fence 20m higher up
175  _alt_max_backup = curr_alt + AC_FENCE_ALT_MAX_BACKUP_DISTANCE;
176  }
177  }else{
178  // clear alt breach if present
179  if ((_breached_fences & AC_FENCE_TYPE_ALT_MAX) != 0) {
180  clear_breach(AC_FENCE_TYPE_ALT_MAX);
181  _alt_max_backup = 0.0f;
182  _alt_max_breach_distance = 0.0f;
183  }
184  }
185  }
186 
187  // circle fence check
188  if ((_enabled_fences & AC_FENCE_TYPE_CIRCLE) != 0 ) {
189 
190  // check if we are outside the fence
191  if (_home_distance >= _circle_radius) {
192 
193  // record distance outside the fence
194  _circle_breach_distance = _home_distance - _circle_radius;
195 
196  // check for a new breach or a breach of the backup fence
197  if ((_breached_fences & AC_FENCE_TYPE_CIRCLE) == 0 || (!is_zero(_circle_radius_backup) && _home_distance >= _circle_radius_backup)) {
198 
199  // record that we have breached the circular distance limit
200  record_breach(AC_FENCE_TYPE_CIRCLE);
201  ret |= AC_FENCE_TYPE_CIRCLE;
202 
203  // create a backup fence 20m further out
204  _circle_radius_backup = _home_distance + AC_FENCE_CIRCLE_RADIUS_BACKUP_DISTANCE;
205  }
206  }else{
207  // clear circle breach if present
208  if ((_breached_fences & AC_FENCE_TYPE_CIRCLE) != 0) {
209  clear_breach(AC_FENCE_TYPE_CIRCLE);
210  _circle_radius_backup = 0.0f;
211  _circle_breach_distance = 0.0f;
212  }
213  }
214  }
215 
216  // polygon fence check
217  if ((_enabled_fences & AC_FENCE_TYPE_POLYGON) != 0 ) {
218  // check consistency of number of points
219  if (_boundary_num_points != _total) {
220  _boundary_loaded = false;
221  }
222  // load fence if necessary
223  if (!_boundary_loaded) {
224  load_polygon_from_eeprom();
225  } else if (_boundary_valid) {
226  // check if vehicle is outside the polygon fence
227  const Vector3f& position = _inav.get_position();
228  if (_poly_loader.boundary_breached(Vector2f(position.x, position.y), _boundary_num_points, _boundary, true)) {
229  // check if this is a new breach
230  if ((_breached_fences & AC_FENCE_TYPE_POLYGON) == 0) {
231  // record that we have breached the polygon
232  record_breach(AC_FENCE_TYPE_POLYGON);
233  ret |= AC_FENCE_TYPE_POLYGON;
234  }
235  } else {
236  // clear breach if present
237  if ((_breached_fences & AC_FENCE_TYPE_POLYGON) != 0) {
238  clear_breach(AC_FENCE_TYPE_POLYGON);
239  }
240  }
241  }
242  }
243 
244  // return any new breaches that have occurred
245  return ret;

358 void AC_Fence::handle_msg(mavlink_channel_t chan, mavlink_message_t* msg)
359 {
360  // exit immediately if null message
361  if (msg == nullptr) {
362  return;
363  }
364 
365  switch (msg->msgid) {
366  // receive a fence point from GCS and store in EEPROM
367  case MAVLINK_MSG_ID_FENCE_POINT: {
368  mavlink_fence_point_t packet;
369  mavlink_msg_fence_point_decode(msg, &packet);
370  if (!check_latlng(packet.lat,packet.lng)) {
371  GCS_MAVLINK::send_statustext_chan(MAV_SEVERITY_WARNING, chan, "Invalid fence point, lat or lng too large");
372  } else {
373  Vector2l point;
374  point.x = packet.lat*1.0e7f;
375  point.y = packet.lng*1.0e7f;
376  if (!_poly_loader.save_point_to_eeprom(packet.idx, point)) {
377  GCS_MAVLINK::send_statustext_chan(MAV_SEVERITY_WARNING, chan, "Failed to save polygon point, too many points?");
378  } else {
379  // trigger reload of points
380  _boundary_loaded = false;
381  }
382  }
383  break;
384  }
385 
386  // send a fence point to GCS
387  case MAVLINK_MSG_ID_FENCE_FETCH_POINT: {
388  mavlink_fence_fetch_point_t packet;
389  mavlink_msg_fence_fetch_point_decode(msg, &packet);
390  // attempt to retrieve from eeprom
391  Vector2l point;
392  if (_poly_loader.load_point_from_eeprom(packet.idx, point)) {
393  mavlink_msg_fence_point_send_buf(msg, chan, msg->sysid, msg->compid, packet.idx, _total, point.x*1.0e-7f, point.y*1.0e-7f);
394  } else {
395  GCS_MAVLINK::send_statustext_chan(MAV_SEVERITY_WARNING, chan, "Bad fence point");
396  }
397  break;
398  }
399 
400  default:
401  // do nothing
402  break;
403  }

407 bool AC_Fence::load_polygon_from_eeprom(bool force_reload)
408 {
409  // exit immediately if already loaded
410  if (_boundary_loaded && !force_reload) {
411  return true;
412  }
413 
414  // check if we need to create array
415  if (!_boundary_create_attempted) {
416  _boundary = (Vector2f *)_poly_loader.create_point_array(sizeof(Vector2f));
417  _boundary_create_attempted = true;
418  }
419 
420  // exit if we could not allocate RAM for the boundary
421  if (_boundary == nullptr) {
422  return false;
423  }
424 
425  // get current location from EKF
426  Location temp_loc;
427  if (!_inav.get_location(temp_loc)) {
428  return false;
429  }
430  const struct Location &ekf_origin = _inav.get_origin();
431 
432  // sanity check total
433  _total = constrain_int16(_total, 0, _poly_loader.max_points());
434 
435  // load each point from eeprom
436  Vector2l temp_latlon;
437  for (uint16_t index=0; index<_total; index++) {
438  // load boundary point as lat/lon point
439  _poly_loader.load_point_from_eeprom(index, temp_latlon);

441  temp_loc.lat = temp_latlon.x;
442  temp_loc.lng = temp_latlon.y;

444  }
445  _boundary_num_points = _total;

447
