158 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BEBOP || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
159  // force monitor for bebop

163  // create each instance
164  for (uint8_t instance=0; instance<AP_BATT_MONITOR_MAX_INSTANCES; instance++) {
165  uint8_t monitor_type = _monitoring[instance];
166  switch (monitor_type) {

171  _num_instances++;

176  hal.i2c_mgr->get_device(AP_BATTMONITOR_SMBUS_BUS_INTERNAL, AP_BATTMONITOR_SMBUS_I2C_ADDR));
177  _num_instances++;
178  break;
179  case BattMonitor_TYPE_MAXELL:
180  state[instance].instance = instance;
181  drivers[instance] = new AP_BattMonitor_SMBus_Maxell(*this, instance, state[instance],
182  hal.i2c_mgr->get_device(AP_BATTMONITOR_SMBUS_BUS_EXTERNAL, AP_BATTMONITOR_SMBUS_I2C_ADDR));
183  _num_instances++;
184  break;
185  case BattMonitor_TYPE_BEBOP:
186 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BEBOP || CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_DISCO
187  state[instance].instance = instance;
188  drivers[instance] = new AP_BattMonitor_Bebop(*this, instance, state[instance]);
189  _num_instances++;
190 #endif
191  break;
192  }
193 
194  // call init function for each backend
195  if (drivers[instance] != nullptr) {
196  drivers[instance]->init();
197  }
198  }

285 bool AP_BattMonitor::exhausted(uint8_t instance, float low_voltage, float min_capacity_mah)
286 {
287  // exit immediately if no monitors setup
288  if (_num_instances == 0 || instance >= _num_instances) {
289  return false;
290  }
291 
292  // check voltage
293  if ((state[instance].voltage > 0) && (low_voltage > 0) && (state[instance].voltage < low_voltage)) {
294  // this is the first time our voltage has dropped below minimum so start timer
295  if (state[instance].low_voltage_start_ms == 0) {
296  state[instance].low_voltage_start_ms = AP_HAL::millis();
297  } else if (AP_HAL::millis() - state[instance].low_voltage_start_ms > AP_BATT_LOW_VOLT_TIMEOUT_MS) {
298  return true;
299  }
300  } else {
301  // acceptable voltage so reset timer
302  state[instance].low_voltage_start_ms = 0;
303  }
304 
305  // check capacity if current monitoring is enabled
306  if (has_current(instance) && (min_capacity_mah > 0) && (_pack_capacity[instance] - state[instance].current_total_mah < min_capacity_mah)) {
307  return true;
308  }
309 
310  // if we've gotten this far then battery is ok
311  return false;
312 }
