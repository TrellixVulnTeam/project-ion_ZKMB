115 bool Location_Class::get_alt_cm(ALT_FRAME desired_frame, int32_t &ret_alt_cm) const
116 {
117  Location_Class::ALT_FRAME frame = get_alt_frame();
118 
119  // shortcut if desired and underlying frame are the same
120  if (desired_frame == frame) {
121  ret_alt_cm = alt;
122  return true;
123  }
124 
125  // check for terrain altitude
126  float alt_terr_cm = 0;
127  if (frame == ALT_FRAME_ABOVE_TERRAIN || desired_frame == ALT_FRAME_ABOVE_TERRAIN) {
128 #if AP_TERRAIN_AVAILABLE
129  if (_ahrs == nullptr || _terrain == nullptr || !_terrain->height_amsl(*(Location *)this, alt_terr_cm, true)) {
130  return false;
131  }
132  // convert terrain alt to cm
133  alt_terr_cm *= 100.0f;
134 #else
135  return false;
136 #endif
137  }
138 
139  // convert alt to absolute
140  int32_t alt_abs;
141  switch (frame) {
142  case ALT_FRAME_ABSOLUTE:
143  alt_abs = alt;
144  break;
145  case ALT_FRAME_ABOVE_HOME:
146  alt_abs = alt + _ahrs->get_home().alt;
147  break;
148  case ALT_FRAME_ABOVE_ORIGIN:
149  {
150  // fail if we cannot get ekf origin
151  Location ekf_origin;
152  if (_ahrs == nullptr || !_ahrs->get_origin(ekf_origin)) {
153  return false;
154  }
155  alt_abs = alt + ekf_origin.alt;
156  }
157  break;
158  case ALT_FRAME_ABOVE_TERRAIN:
159  alt_abs = alt + alt_terr_cm;
160  break;
161  default:
162  // unknown conversion to absolute alt, this should never happen
163  return false;
164  }
165 
166  // convert absolute to desired frame
167  switch (desired_frame) {
168  case ALT_FRAME_ABSOLUTE:
169  ret_alt_cm = alt_abs;
170  return true;
171  case ALT_FRAME_ABOVE_HOME:
172  ret_alt_cm = alt_abs - _ahrs->get_home().alt;
173  return true;
174  case ALT_FRAME_ABOVE_ORIGIN:
175  {
176  // fail if we cannot get ekf origin
177  Location ekf_origin;
178  if (_ahrs == nullptr || !_ahrs->get_origin(ekf_origin)) {
179  return false;
180  }
181  ret_alt_cm = alt_abs - ekf_origin.alt;
182  return true;
183  }
184  case ALT_FRAME_ABOVE_TERRAIN:
185  ret_alt_cm = alt_abs - alt_terr_cm;
186  return true;
