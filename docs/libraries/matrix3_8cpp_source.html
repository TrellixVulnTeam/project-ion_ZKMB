23 // create a rotation matrix given some euler angles
24 // this is based on http://gentlenav.googlecode.com/files/EulerAngles.pdf
25 template <typename T>
26 void Matrix3<T>::from_euler(float roll, float pitch, float yaw)
27 {
28  float cp = cosf(pitch);
29  float sp = sinf(pitch);
30  float sr = sinf(roll);
31  float cr = cosf(roll);
32  float sy = sinf(yaw);
33  float cy = cosf(yaw);
34 
35  a.x = cp * cy;
36  a.y = (sr * sp * cy) - (cr * sy);
37  a.z = (cr * sp * cy) + (sr * sy);
38  b.x = cp * sy;
39  b.y = (sr * sp * sy) + (cr * cy);
40  b.z = (cr * sp * sy) - (sr * cy);
41  c.x = -sp;
42  c.y = sr * cp;
43  c.z = cr * cp;
44 }

75 /*
76  fill the matrix from Euler angles in radians in 312 convention
77 */
78 template <typename T>
79 void Matrix3<T>::from_euler312(float roll, float pitch, float yaw)
80 {
81  float c3 = cosf(pitch);
82  float s3 = sinf(pitch);
83  float s2 = sinf(roll);
84  float c2 = cosf(roll);
85  float s1 = sinf(yaw);
86  float c1 = cosf(yaw);
87 
88  a.x = c1 * c3 - s1 * s2 * s3;
89  b.y = c1 * c2;
90  c.z = c3 * c2;
91  a.y = -c2*s1;
92  a.z = s3*c1 + c3*s2*s1;
93  b.x = c3*s1 + s3*s2*c1;
94  b.z = s1*s3 - s2*c1*c3;
95  c.x = -s3*c2;
96  c.y = s2;
97 }

100 // to a rotation matrix.

102 void Matrix3<T>::rotate(const Vector3<T> &g)
103 {

106  temp_matrix.a.y = a.z * g.x - a.x * g.z;
107  temp_matrix.a.z = a.x * g.y - a.y * g.x;
108  temp_matrix.b.x = b.y * g.z - b.z * g.y;
109  temp_matrix.b.y = b.z * g.x - b.x * g.z;
110  temp_matrix.b.z = b.x * g.y - b.y * g.x;
111  temp_matrix.c.x = c.y * g.z - c.z * g.y;
112  temp_matrix.c.y = c.z * g.x - c.x * g.z;
113  temp_matrix.c.z = c.x * g.y - c.y * g.x;

121 template <typename T>
122 void Matrix3<T>::normalize(void)
123 {
124  float error = a * b;
125  Vector3<T> t0 = a - (b * (0.5f * error));
126  Vector3<T> t1 = b - (a * (0.5f * error));
127  Vector3<T> t2 = t0 % t1;
128  a = t0 * (1.0f / t0.length());
129  b = t1 * (1.0f / t1.length());
130  c = t2 * (1.0f / t2.length());
131 }

232 // create rotation matrix for rotation about the vector v by angle theta
233 // See: http://www.euclideanspace.com/maths/geometry/rotations/conversions/angleToMatrix/
234 template <typename T>
235 void Matrix3<T>::from_axis_angle(const Vector3<T> &v, float theta)
236 {
237  float C = cosf(theta);
238  float S = sinf(theta);
239  float t = 1.0f - C;
240  Vector3f normv = v.normalized();
241  float x = normv.x;
242  float y = normv.y;
243  float z = normv.z;
244 
245  a.x = t*x*x + C;
246  a.y = t*x*y - z*S;
247  a.z = t*x*z + y*S;
248  b.x = t*x*y + z*S;
249  b.y = t*y*y + C;
250  b.z = t*y*z - x*S;
251  c.x = t*x*z - y*S;
252  c.y = t*y*z + x*S;
253  c.z = t*z*z + C;
254 }
