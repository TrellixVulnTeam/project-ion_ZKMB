47 void AP_AHRS_View::update(bool skip_ins_update)
48 {
49  rot_body_to_ned = ahrs.get_rotation_body_to_ned();
50  gyro = ahrs.get_gyro();
51 
52  if (rotation != ROTATION_NONE) {
53  Matrix3f &r = rot_body_to_ned;
54  r.transpose();
55  r = rot_view * r;
56  r.transpose();
57  gyro.rotate(rotation);
58  }

60  rot_body_to_ned.to_euler(&roll, &pitch, &yaw);

64  yaw_sensor = degrees(yaw) * 100;

66  ahrs.calc_trig(rot_body_to_ned,
