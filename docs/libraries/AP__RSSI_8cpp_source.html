114 // Read the receiver RSSI value as a float 0.0f - 1.0f.
115 // 0.0 represents weakest signal, 1.0 represents maximum signal.
116 float AP_RSSI::read_receiver_rssi()
117 {
118  // Default to 0 RSSI
119  float receiver_rssi = 0.0f;
120 
121  switch (rssi_type) {
122  case RssiType::RSSI_DISABLED :
123  receiver_rssi = 0.0f;
124  break;
125  case RssiType::RSSI_ANALOG_PIN :
126  receiver_rssi = read_pin_rssi();
127  break;
128  case RssiType::RSSI_RC_CHANNEL_VALUE :
129  receiver_rssi = read_channel_rssi();
130  break;
131  default :
132  receiver_rssi = 0.0f;
133  break;
134  }
135 
136  return receiver_rssi;
137 }

166 // Scale and constrain a float rssi value to 0.0 to 1.0 range
167 float AP_RSSI::scale_and_constrain_float_rssi(float current_rssi_value, float low_rssi_range, float high_rssi_range)
168 {
169  float rssi_value_range = fabsf(high_rssi_range - low_rssi_range);
170  if (is_zero(rssi_value_range)) {
171  // User range isn't meaningful, return 0 for RSSI (and avoid divide by zero)
172  return 0.0f;
173  }
174  // Note that user-supplied ranges may be inverted and we accommodate that here.
175  // (Some radio receivers put out inverted ranges for RSSI-type values).
176  bool range_is_inverted = (high_rssi_range < low_rssi_range);
177  // Constrain to the possible range - values outside are clipped to ends
178  current_rssi_value = constrain_float(current_rssi_value,
179  range_is_inverted ? high_rssi_range : low_rssi_range,
180  range_is_inverted ? low_rssi_range : high_rssi_range);
181 
182  if (range_is_inverted)
183  {
184  // Swap values so we can treat them as low->high uniformly in the code that follows
185  current_rssi_value = high_rssi_range + fabsf(current_rssi_value - low_rssi_range);
186  std::swap(low_rssi_range, high_rssi_range);
187  }
188 
189  // Scale the value down to a 0.0 - 1.0 range
190  float rssi_value_scaled = (current_rssi_value - low_rssi_range) / rssi_value_range;
191  // Make absolutely sure the value is clipped to the 0.0 - 1.0 range. This should handle things if the
192  // value retrieved falls outside the user-supplied range.
193  return constrain_float(rssi_value_scaled, 0.0f, 1.0f);
194 }
