140 int PX4CAN::computeTimings(const uint32_t target_bitrate, Timings& out_timings)
141 {
142  if (target_bitrate < 1) {
143  return -ErrInvalidBitRate;
144  }
145 
146  /*
147  * Hardware configuration
148  */
149  const uint32_t pclk = STM32_PCLK1_FREQUENCY;
150 
151  static const uint8_t MaxBS1 = 16;
152  static const uint8_t MaxBS2 = 8;
153 
154  /*
155  * Ref. "Automatic Baudrate Detection in CANopen Networks", U. Koppe, MicroControl GmbH & Co. KG
156  * CAN in Automation, 2003
157  *
158  * According to the source, optimal quanta per bit are:
159  * Bitrate Optimal Maximum
160  * 1000 kbps 8 10
161  * 500 kbps 16 17
162  * 250 kbps 16 17
163  * 125 kbps 16 17
164  */
165  const uint8_t max_quanta_per_bit = (target_bitrate >= 1000000) ? 10 : 17;
166 
167  if (max_quanta_per_bit > (MaxBS1 + MaxBS2)) {
168  if (AP_BoardConfig::get_can_debug() >= 1) {
169  printf("PX4CAN::computeTimings max_quanta_per_bit problem\n\r");
170  }
171  }
172 
173  static const uint16_t MaxSamplePointLocation = 900;
174 
175  /*
176  * Computing (prescaler * BS):
177  * BITRATE = 1 / (PRESCALER * (1 / PCLK) * (1 + BS1 + BS2)) -- See the Reference Manual
178  * BITRATE = PCLK / (PRESCALER * (1 + BS1 + BS2)) -- Simplified
179  * let:
180  * BS = 1 + BS1 + BS2 -- Number of time quanta per bit
181  * PRESCALER_BS = PRESCALER * BS
182  * ==>
183  * PRESCALER_BS = PCLK / BITRATE
184  */
185  const uint32_t prescaler_bs = pclk / target_bitrate;
186 
187  /*
188  * Searching for such prescaler value so that the number of quanta per bit is highest.
189  */
190  uint8_t bs1_bs2_sum = uint8_t(max_quanta_per_bit - 1);
191 
192  while ((prescaler_bs % (1 + bs1_bs2_sum)) != 0) {
193  if (bs1_bs2_sum <= 2) {
194  return -ErrInvalidBitRate; // No solution
195  }
196  bs1_bs2_sum--;
197  }
198 
199  const uint32_t prescaler = prescaler_bs / (1 + bs1_bs2_sum);
200  if ((prescaler < 1U) || (prescaler > 1024U)) {
201  return -ErrInvalidBitRate; // No solution
202  }
203 
204  /*
205  * Now we have a constraint: (BS1 + BS2) == bs1_bs2_sum.
206  * We need to find the values so that the sample point is as close as possible to the optimal value.
207  *
208  * Solve[(1 + bs1)/(1 + bs1 + bs2) == 7/8, bs2] (* Where 7/8 is 0.875, the recommended sample point location *)
209  * {{bs2 -> (1 + bs1)/7}}
210  *
211  * Hence:
212  * bs2 = (1 + bs1) / 7
213  * bs1 = (7 * bs1_bs2_sum - 1) / 8
214  *
215  * Sample point location can be computed as follows:
216  * Sample point location = (1 + bs1) / (1 + bs1 + bs2)
217  *
218  * Since the optimal solution is so close to the maximum, we prepare two solutions, and then pick the best one:
219  * - With rounding to nearest
220  * - With rounding to zero
221  */
222  struct BsPair {
223  uint8_t bs1;
224  uint8_t bs2;
225  uint16_t sample_point_permill;
226 
227  BsPair() :
228  bs1(0), bs2(0), sample_point_permill(0)
229  {
230  }
231 
232  BsPair(uint8_t bs1_bs2_sum, uint8_t arg_bs1) :
233  bs1(arg_bs1), bs2(uint8_t(bs1_bs2_sum - bs1)), sample_point_permill(
234  uint16_t(1000 * (1 + bs1) / (1 + bs1 + bs2)))
235  {
236  if (bs1_bs2_sum <= arg_bs1) {
237  if (AP_BoardConfig::get_can_debug() >= 1) {
238  AP_HAL::panic("PX4CAN::computeTimings bs1_bs2_sum <= arg_bs1");
239  }
240  }
241  }
242 
243  bool isValid() const
244  {
245  return (bs1 >= 1) && (bs1 <= MaxBS1) && (bs2 >= 1) && (bs2 <= MaxBS2);
246  }
247  };
248 
249  // First attempt with rounding to nearest
250  BsPair solution(bs1_bs2_sum, uint8_t(((7 * bs1_bs2_sum - 1) + 4) / 8));
251 
252  if (solution.sample_point_permill > MaxSamplePointLocation || !solution.isValid()) {
253  // Second attempt with rounding to zero
254  solution = BsPair(bs1_bs2_sum, uint8_t((7 * bs1_bs2_sum - 1) / 8));
255 
256  if (!solution.isValid())
257  {
258  printf("PX4CAN::computeTimings second solution invalid\n\r");
259  return -ErrLogic;
260  }
261  }
262 
263  /*
264  * Final validation
265  */
266  if ((target_bitrate != (pclk / (prescaler * (1 + solution.bs1 + solution.bs2)))) || !solution.isValid()) {
267  if (AP_BoardConfig::get_can_debug() >= 1) {
268  printf("PX4CAN::computeTimings target_bitrate error\n\r");
269  }
270  return -ErrLogic;
271  }
272 
273  if (AP_BoardConfig::get_can_debug() >= 2) {
274  printf("PX4CAN::computeTimings Timings: quanta/bit: %d, sample point location: %.1f%%\n\r",
275  int(1 + solution.bs1 + solution.bs2), double(solution.sample_point_permill / 10.0));
276  }
277 
278  out_timings.prescaler = uint16_t(prescaler - 1U);
279  out_timings.sjw = 0; // Which means one
280  out_timings.bs1 = uint8_t(solution.bs1 - 1);
281  out_timings.bs2 = uint8_t(solution.bs2 - 1);
282 
283  return 0;
284 }

286 int16_t PX4CAN::send(const uavcan::CanFrame& frame, uavcan::MonotonicTime tx_deadline, uavcan::CanIOFlags flags)
287 {
288  if (frame.isErrorFrame() || frame.dlc > 8) {
289  return -ErrUnsupportedFrame;
290  }
291 
292  /*
293  * Normally we should perform the same check as in @ref canAcceptNewTxFrame(), because
294  * it is possible that the highest-priority frame between select() and send() could have been
295  * replaced with a lower priority one due to TX timeout. But we don't do this check because:
296  *
297  * - It is a highly unlikely scenario.
298  *
299  * - Frames do not timeout on a properly functioning bus. Since frames do not timeout, the new
300  * frame can only have higher priority, which doesn't break the logic.
301  *
302  * - If high-priority frames are timing out in the TX queue, there's probably a lot of other
303  * issues to take care of before this one becomes relevant.
304  *
305  * - It takes CPU time. Not just CPU time, but critical section time, which is expensive.
306  */
307  CriticalSectionLocker lock;
308 
309  /*
310  * Seeking for an empty slot
311  */
312  uint8_t txmailbox = 0xFF;
313  if ((can_->TSR & bxcan::TSR_TME0) == bxcan::TSR_TME0) {
314  txmailbox = 0;
315  } else if ((can_->TSR & bxcan::TSR_TME1) == bxcan::TSR_TME1) {
316  txmailbox = 1;
317  } else if ((can_->TSR & bxcan::TSR_TME2) == bxcan::TSR_TME2) {
318  txmailbox = 2;
319  } else {
320  return 0; // No transmission for you.
321  }
322 
323  peak_tx_mailbox_index_ = uavcan::max(peak_tx_mailbox_index_, txmailbox); // Statistics
324 
325  /*
326  * Setting up the mailbox
327  */
328  bxcan::TxMailboxType& mb = can_->TxMailbox[txmailbox];
329  if (frame.isExtended()) {
330  mb.TIR = ((frame.id & uavcan::CanFrame::MaskExtID) << 3) | bxcan::TIR_IDE;
331  } else {
332  mb.TIR = ((frame.id & uavcan::CanFrame::MaskStdID) << 21);
333  }
334 
335  if (frame.isRemoteTransmissionRequest()) {
336  mb.TIR |= bxcan::TIR_RTR;
337  }
338 
339  mb.TDTR = frame.dlc;
340 
341  mb.TDHR = (uint32_t(frame.data[7]) << 24) | (uint32_t(frame.data[6]) << 16) | (uint32_t(frame.data[5]) << 8)
342  | (uint32_t(frame.data[4]) << 0);
343  mb.TDLR = (uint32_t(frame.data[3]) << 24) | (uint32_t(frame.data[2]) << 16) | (uint32_t(frame.data[1]) << 8)
344  | (uint32_t(frame.data[0]) << 0);
345 
346  mb.TIR |= bxcan::TIR_TXRQ; // Go.
347 
348  /*
349  * Registering the pending transmission so we can track its deadline and loopback it as needed
350  */
351  TxItem& txi = pending_tx_[txmailbox];
352  txi.deadline = tx_deadline;
353  txi.frame = frame;
354  txi.loopback = (flags & uavcan::CanIOFlagLoopback) != 0;
355  txi.abort_on_error = (flags & uavcan::CanIOFlagAbortOnError) != 0;
356  txi.pending = true;
357  return 1;
358 }

403 int PX4CAN::init(const uint32_t bitrate, const OperatingMode mode)
404 {
405  /* We need to silence the controller in the first order, otherwise it may interfere with the following operations. */
406 
407  {
408  CriticalSectionLocker lock;
409 
410  can_->MCR &= ~bxcan::MCR_SLEEP; // Exit sleep mode
411  can_->MCR |= bxcan::MCR_INRQ; // Request init
412 
413  can_->IER = 0; // Disable CAN interrupts while initialization is in progress
414  }
415 
416  if (!waitMsrINakBitStateChange(true)) {
417  if (AP_BoardConfig::get_can_debug() >= 1) {
418  printf("PX4CAN::init MSR INAK not set\n\r");
419  }
420  can_->MCR = bxcan::MCR_RESET;
421  return -ErrMsrInakNotSet;
422  }
423 
424  /*
425  * Object state - CAN interrupts are disabled, so it's safe to modify it now
426  */
427  rx_queue_.clear();
428  error_cnt_ = 0;
429  served_aborts_cnt_ = 0;
430  uavcan::fill_n(pending_tx_, NumTxMailboxes, TxItem());
431  peak_tx_mailbox_index_ = 0;
432  had_activity_ = false;
433 
434  /*
435  * CAN timings for this bitrate
436  */
437  Timings timings;
438  const int timings_res = computeTimings(bitrate, timings);
439  if (timings_res < 0) {
440  can_->MCR = bxcan::MCR_RESET;
441  return timings_res;
442  }
443  if (AP_BoardConfig::get_can_debug() >= 2) {
444  printf("PX4CAN::init Timings: presc=%u sjw=%u bs1=%u bs2=%u\n\r", unsigned(timings.prescaler),
445  unsigned(timings.sjw), unsigned(timings.bs1), unsigned(timings.bs2));
446  }
447 
448  /*
449  * Hardware initialization (the hardware has already confirmed initialization mode, see above)
450  */
451  can_->MCR = bxcan::MCR_ABOM | bxcan::MCR_AWUM | bxcan::MCR_INRQ; // RM page 648
452 
453  can_->BTR = ((timings.sjw & 3U) << 24) | ((timings.bs1 & 15U) << 16) | ((timings.bs2 & 7U) << 20)
454  | (timings.prescaler & 1023U) | ((mode == SilentMode) ? bxcan::BTR_SILM : 0);
455 
456  can_->IER = bxcan::IER_TMEIE | // TX mailbox empty
457  bxcan::IER_FMPIE0 | // RX FIFO 0 is not empty
458  bxcan::IER_FMPIE1; // RX FIFO 1 is not empty
459 
460  can_->MCR &= ~bxcan::MCR_INRQ; // Leave init mode
461 
462  if (!waitMsrINakBitStateChange(false)) {
463  if (AP_BoardConfig::get_can_debug() >= 1) {
464  printf("PX4CAN::init MSR INAK not cleared\n\r");
465  }
466  can_->MCR = bxcan::MCR_RESET;
467  return -ErrMsrInakNotCleared;
468  }
469 
470  /*
471  * Default filter configuration
472  */
473  if (self_index_ == 0) {
474  can_->FMR |= bxcan::FMR_FINIT;
475 
476  can_->FMR &= 0xFFFFC0F1;
477  can_->FMR |= static_cast<uint32_t>(NumFilters) << 8; // Slave (CAN2) gets half of the filters
478 
479  can_->FFA1R = 0; // All assigned to FIFO0 by default
480  can_->FM1R = 0; // Indentifier Mask mode
481 
482  can_->FS1R = 0x7ffffff; // Single 32-bit for all
483  can_->FilterRegister[0].FR1 = 0; // CAN1 accepts everything
484  can_->FilterRegister[0].FR2 = 0;
485  can_->FilterRegister[NumFilters].FR1 = 0; // CAN2 accepts everything
486  can_->FilterRegister[NumFilters].FR2 = 0;
487  can_->FA1R = 1 | (1 << NumFilters); // One filter per each iface
488 
489  can_->FMR &= ~bxcan::FMR_FINIT;
490  }
491 
492  return 0;
493 }

538 void PX4CAN::handleRxInterrupt(uint8_t fifo_index, uint64_t utc_usec)
539 {
540  if (fifo_index < 2) {
541 
542  volatile uint32_t* const rfr_reg = (fifo_index == 0) ? &can_->RF0R : &can_->RF1R;
543  if ((*rfr_reg & bxcan::RFR_FMP_MASK) == 0) {
544  return;
545  }
546 
547  /*
548  * Register overflow as a hardware error
549  */
550  if ((*rfr_reg & bxcan::RFR_FOVR) != 0) {
551  error_cnt_++;
552  }
553 
554  /*
555  * Read the frame contents
556  */
557  uavcan::CanFrame frame;
558  const bxcan::RxMailboxType& rf = can_->RxMailbox[fifo_index];
559 
560  if ((rf.RIR & bxcan::RIR_IDE) == 0) {
561  frame.id = uavcan::CanFrame::MaskStdID & (rf.RIR >> 21);
562  } else {
563  frame.id = uavcan::CanFrame::MaskExtID & (rf.RIR >> 3);
564  frame.id |= uavcan::CanFrame::FlagEFF;
565  }
566 
567  if ((rf.RIR & bxcan::RIR_RTR) != 0) {
568  frame.id |= uavcan::CanFrame::FlagRTR;
569  }
570 
571  frame.dlc = rf.RDTR & 15;
572 
573  frame.data[0] = uint8_t(0xFF & (rf.RDLR >> 0));
574  frame.data[1] = uint8_t(0xFF & (rf.RDLR >> 8));
575  frame.data[2] = uint8_t(0xFF & (rf.RDLR >> 16));
576  frame.data[3] = uint8_t(0xFF & (rf.RDLR >> 24));
577  frame.data[4] = uint8_t(0xFF & (rf.RDHR >> 0));
578  frame.data[5] = uint8_t(0xFF & (rf.RDHR >> 8));
579  frame.data[6] = uint8_t(0xFF & (rf.RDHR >> 16));
580  frame.data[7] = uint8_t(0xFF & (rf.RDHR >> 24));
581 
582  *rfr_reg = bxcan::RFR_RFOM | bxcan::RFR_FOVR | bxcan::RFR_FULL; // Release FIFO entry we just read
583 
584  /*
585  * Store with timeout into the FIFO buffer and signal update event
586  */
587  CanRxItem frm;
588  frm.frame = frame;
589  frm.flags = 0;
590  frm.utc_usec = utc_usec;
591  rx_queue_.push(frm);
592 
593  had_activity_ = true;
594  update_event_.signalFromInterrupt();
595 
596  pollErrorFlagsFromISR();
597  }

600 void PX4CAN::pollErrorFlagsFromISR()
601 {
602  const uint8_t lec = uint8_t((can_->ESR & bxcan::ESR_LEC_MASK) >> bxcan::ESR_LEC_SHIFT);
603  if (lec != 0) {
604  can_->ESR = 0;
605  error_cnt_++;

632 bool PX4CAN::canAcceptNewTxFrame(const uavcan::CanFrame& frame) const
633 {
634  /*
635  * We can accept more frames only if the following conditions are satisfied:
636  * - There is at least one TX mailbox free (obvious enough);
637  * - The priority of the new frame is higher than priority of all TX mailboxes.
638  */
639  {
640  static const uint32_t TME = bxcan::TSR_TME0 | bxcan::TSR_TME1 | bxcan::TSR_TME2;
641  const uint32_t tme = can_->TSR & TME;
642 
643  if (tme == TME) { // All TX mailboxes are free (as in freedom).
644  return true;
645  }
646 
647  if (tme == 0) { // All TX mailboxes are busy transmitting.
648  return false;
649  }
650  }
651 
652  /*
653  * The second condition requires a critical section.
654  */
655  CriticalSectionLocker lock;
656 
657  for (int mbx = 0; mbx < NumTxMailboxes; mbx++) {
658  if (pending_tx_[mbx].pending && !frame.priorityHigherThan(pending_tx_[mbx].frame)) {
659  return false; // There's a mailbox whose priority is higher or equal the priority of the new frame.
660  }
661  }
662 
663  return true; // This new frame will be added to a free TX mailbox in the next @ref send().
664 }

788 int16_t PX4CANManager::select(uavcan::CanSelectMasks& inout_masks,
789  const uavcan::CanFrame* (&pending_tx)[uavcan::MaxCanIfaces], const uavcan::MonotonicTime blocking_deadline)
790 {
791  const uavcan::CanSelectMasks in_masks = inout_masks;
792  const uavcan::MonotonicTime time = clock::getMonotonic();
793 
794  if (can_number_ >= 1) {
795  if0_.discardTimedOutTxMailboxes(time); // Check TX timeouts - this may release some TX slots
796  {
797  CriticalSectionLocker cs_locker;
798  if0_.pollErrorFlagsFromISR();
799  }
800  }
801 
802 #if CAN_STM32_NUM_IFACES > 1
803  if (can_number_ >= 2) {
804  if1_.discardTimedOutTxMailboxes(time);
805  {
806  CriticalSectionLocker cs_locker;
807  if1_.pollErrorFlagsFromISR();
808  }
809  }
810 #endif
811 
812  inout_masks = makeSelectMasks(pending_tx); // Check if we already have some of the requested events
813  if ((inout_masks.read & in_masks.read) != 0 || (inout_masks.write & in_masks.write) != 0) {
814  return 1;
815  }
816 
817  (void) update_event_.wait(blocking_deadline - time); // Block until timeout expires or any iface updates
818  inout_masks = makeSelectMasks(pending_tx); // Return what we got even if none of the requested events are set
819  return 1; // Return value doesn't matter as long as it is non-negative
820 }
