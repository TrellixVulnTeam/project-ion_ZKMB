39 void RCInput_SBUS::init()
40 {
41  fd = open(device_path, O_RDWR | O_NONBLOCK | O_CLOEXEC);
42  if (fd != -1) {
43  printf("Opened SBUS input %s fd=%d\n", device_path, (int)fd);
44  fflush(stdout);
45  struct termios2 tio {};
46 
47  if (ioctl(fd, TCGETS2, &tio) != 0) {
48  close(fd);
49  fd = -1;
50  return;
51  }
52  tio.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR
53  | IGNCR | ICRNL | IXON);
54  tio.c_iflag |= (INPCK | IGNPAR);
55  tio.c_oflag &= ~OPOST;
56  tio.c_lflag &= ~(ECHO | ECHONL | ICANON | ISIG | IEXTEN);
57  tio.c_cflag &= ~(CSIZE | CRTSCTS | PARODD | CBAUD);
58  // use BOTHER to specify speed directly in c_[io]speed member
59  tio.c_cflag |= (CS8 | CSTOPB | CLOCAL | PARENB | BOTHER | CREAD);
60  tio.c_ispeed = 100000;
61  tio.c_ospeed = 100000;
62  // see select() comment below
63  tio.c_cc[VMIN] = SBUS_FRAME_SIZE;
64  tio.c_cc[VTIME] = 0;
65  if (ioctl(fd, TCSETS2, &tio) != 0) {
66  close(fd);
67  fd = -1;
68  return;
69  }
70  }
71 }

82 void RCInput_SBUS::_timer_tick(void)
83 {
84  if (fd == -1) {
85  return;
86  }
87 
88  // read up to 10 frames at a time
89  uint8_t bytes[SBUS_FRAME_SIZE*10];
90  int nread;
91 
92  fd_set fds;
93  struct timeval tv;
94 
95  FD_ZERO(&fds);
96  FD_SET(fd, &fds);
97 
98  tv.tv_sec = 0;
99  tv.tv_usec = 0;
100 
101  // as VMIN is SBUS_FRAME_SIZE the select won't return unless there is
102  // at least SBUS_FRAME_SIZE bytes available
103  if (select(fd+1, &fds, nullptr, nullptr, &tv) != 1) {
104  return;
105  }
106 
107 #if SBUS_DEBUG_LOG
108  static int logfd = -1;
109  if (logfd == -1) {
110  logfd = open("sbus.log", O_WRONLY|O_CREAT|O_TRUNC|O_CLOEXEC, 0644);
111  }
112 #endif
113 
114 #if SBUS_CAUSE_CORRUPTION
115  // deliberately lose bytes from the port
116  static unsigned corruption_counter;
117  if (corruption_counter++ % 1000 == 0) {
118  uint8_t nn = corruption_counter/1000;
119  int n2 = ::read(fd, bytes, nn);
120  dprintf(logfd, "throw %u\n", (unsigned)n2);
121  }
122 #endif
123 
124  // cope with there being a large number of pending bytes at
125  // the start
126  do {
127  nread = ::read(fd, bytes, sizeof(bytes));
128  } while (nread == sizeof(bytes));
129 
130  if (nread % SBUS_FRAME_SIZE != 0) {
131  /*
132  SBUS frames are 25 bytes long, and always start with
133  0x0f, but there is no other framing information to
134  prevent us getting out of sync. All we have are the
135  timing guarantees
136 
137  In this case we have read a partial frame, or lost some
138  bytes. A 25 bytes frame at 100000 baud takes 2.5ms. By
139  delaying 3.5ms here we will get any remaining bytes for
140  this frame. We shouldn't get the start of the next frame
141  as frames happen at most every 7ms
142 
143  This strategy allows us to resync even if we lose
144  bytes. It assumes an interframe gap of more than
145  3.5ms. If SBUS is run at very high rate (like 300Hz)
146  then this won't work
147  */
148  hal.scheduler->delay_microseconds(3500);
149  int n2 = ::read(fd, bytes+nread, sizeof(bytes)-nread);
150  if (n2 > 0) {
151  nread += n2;
152  }
153  }
154  if (nread % SBUS_FRAME_SIZE != 0) {
155  // if we don't have a multuple of 25 then throw away the lot
156 #if SBUS_DEBUG_LOG
157  dprintf(logfd, "discard %u\n", (unsigned)nread);
158 #endif
159  return;
160  }
161  if (nread <= 0) {
162  return;
163  }
164 #if SBUS_DEBUG_LOG
165  if (logfd != -1) {
166  dprintf(logfd, "%06u %u: ", (unsigned)AP_HAL::millis(), (unsigned)nread);
167  for (uint8_t i=0; i<nread; i++) {
168  dprintf(logfd, "%02x ", (unsigned)bytes[i]);
169  }
170  dprintf(logfd, "\n");
171  }
172 #endif
173  // only process the last SBUS_FRAME_SIZE bytes. Only the latest
174  // frame matters
175  add_sbus_input(&bytes[nread-SBUS_FRAME_SIZE], SBUS_FRAME_SIZE);
176 }
177 #endif // CONFIG_HAL_BOARD_SUBTYPE
