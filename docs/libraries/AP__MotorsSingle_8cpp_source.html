31 void AP_MotorsSingle::init(motor_frame_class frame_class, motor_frame_type frame_type)
32 {
33  // set update rate for the 3 motors (but not the servo on channel 7)

35 
36  // set the motor_enabled flag so that the main ESC can be calibrated like other frame types
37  motor_enabled[AP_MOTORS_MOT_5] = true;
38  motor_enabled[AP_MOTORS_MOT_6] = true;
39 
40  _servo1 = SRV_Channels::get_channel_for(SRV_Channel::k_motor1, CH_1);
41  _servo2 = SRV_Channels::get_channel_for(SRV_Channel::k_motor2, CH_2);
42  _servo3 = SRV_Channels::get_channel_for(SRV_Channel::k_motor3, CH_3);
43  _servo4 = SRV_Channels::get_channel_for(SRV_Channel::k_motor4, CH_4);
44  if (!_servo1 || !_servo2 || !_servo3 || !_servo4) {
45  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_ERROR, "MotorsSingle: unable to setup output channels");
46  // don't set initialised_ok
47  return;
48  }
49 
50  // we set four servos to angle
51  _servo1->set_angle(AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
52  _servo2->set_angle(AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
53  _servo3->set_angle(AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
54  _servo4->set_angle(AP_MOTORS_SINGLE_SERVO_INPUT_RANGE);
55 
56  // allow mapping of motor7
57  add_motor_num(CH_7);
58 
59  // record successful initialisation if what we setup was the desired frame_class
60  _flags.initialised_ok = (frame_class == MOTOR_FRAME_SINGLE);

93 void AP_MotorsSingle::output_to_motors()
94 {
95  if (!_flags.initialised_ok) {
96  return;
97  }
98  switch (_spool_mode) {
99  case SHUT_DOWN:
100  // sends minimum values out to the motors
101  hal.rcout->cork();
102  rc_write(AP_MOTORS_MOT_1, calc_pwm_output_1to1(_roll_radio_passthrough - _yaw_radio_passthrough, _servo1));
103  rc_write(AP_MOTORS_MOT_2, calc_pwm_output_1to1(_pitch_radio_passthrough - _yaw_radio_passthrough, _servo2));
104  rc_write(AP_MOTORS_MOT_3, calc_pwm_output_1to1(-_roll_radio_passthrough - _yaw_radio_passthrough, _servo3));
105  rc_write(AP_MOTORS_MOT_4, calc_pwm_output_1to1(-_pitch_radio_passthrough - _yaw_radio_passthrough, _servo4));
106  rc_write(AP_MOTORS_MOT_5, get_pwm_output_min());
107  rc_write(AP_MOTORS_MOT_6, get_pwm_output_min());
108  hal.rcout->push();
109  break;
110  case SPIN_WHEN_ARMED:
111  // sends output to motors when armed but not flying
112  hal.rcout->cork();
113  rc_write(AP_MOTORS_MOT_1, calc_pwm_output_1to1(_spin_up_ratio * _actuator_out[0], _servo1));
114  rc_write(AP_MOTORS_MOT_2, calc_pwm_output_1to1(_spin_up_ratio * _actuator_out[1], _servo2));
115  rc_write(AP_MOTORS_MOT_3, calc_pwm_output_1to1(_spin_up_ratio * _actuator_out[2], _servo3));
116  rc_write(AP_MOTORS_MOT_4, calc_pwm_output_1to1(_spin_up_ratio * _actuator_out[3], _servo4));
117  rc_write(AP_MOTORS_MOT_5, calc_spin_up_to_pwm());
118  rc_write(AP_MOTORS_MOT_6, calc_spin_up_to_pwm());
119  hal.rcout->push();
120  break;
121  case SPOOL_UP:
122  case THROTTLE_UNLIMITED:
123  case SPOOL_DOWN:
124  // set motor output based on thrust requests
125  hal.rcout->cork();
126  rc_write(AP_MOTORS_MOT_1, calc_pwm_output_1to1(_actuator_out[0], _servo1));
127  rc_write(AP_MOTORS_MOT_2, calc_pwm_output_1to1(_actuator_out[1], _servo2));
128  rc_write(AP_MOTORS_MOT_3, calc_pwm_output_1to1(_actuator_out[2], _servo3));
129  rc_write(AP_MOTORS_MOT_4, calc_pwm_output_1to1(_actuator_out[3], _servo4));
130  rc_write(AP_MOTORS_MOT_5, calc_thrust_to_pwm(_thrust_out));
131  rc_write(AP_MOTORS_MOT_6, calc_thrust_to_pwm(_thrust_out));
132  hal.rcout->push();
133  break;
134  }
135 }

151 // sends commands to the motors
152 void AP_MotorsSingle::output_armed_stabilizing()
153 {
154  float roll_thrust; // roll thrust input value, +/- 1.0
155  float pitch_thrust; // pitch thrust input value, +/- 1.0
156  float yaw_thrust; // yaw thrust input value, +/- 1.0
157  float throttle_thrust; // throttle thrust input value, 0.0 - 1.0
158  float thrust_min_rpy; // the minimum throttle setting that will not limit the roll and pitch output
159  float thr_adj; // the difference between the pilot's desired throttle and throttle_thrust_best_rpy
160  float rp_scale = 1.0f; // this is used to scale the roll, pitch and yaw to fit within the motor limits
161  float actuator_allowed = 0.0f; // amount of yaw we can fit in
162  float actuator[NUM_ACTUATORS]; // combined roll, pitch and yaw thrusts for each actuator
163  float actuator_max = 0.0f; // maximum actuator value
164 
165  // apply voltage and air pressure compensation
166  roll_thrust = _roll_in * get_compensation_gain();
167  pitch_thrust = _pitch_in * get_compensation_gain();
168  yaw_thrust = _yaw_in * get_compensation_gain();
169  throttle_thrust = get_throttle() * get_compensation_gain();
170 
171  // sanity check throttle is above zero and below current limited throttle
172  if (throttle_thrust <= 0.0f) {
173  throttle_thrust = 0.0f;
174  limit.throttle_lower = true;
175  }
176  if (throttle_thrust >= _throttle_thrust_max) {
177  throttle_thrust = _throttle_thrust_max;
178  limit.throttle_upper = true;
179  }
180 
181  _throttle_avg_max = constrain_float(_throttle_avg_max, throttle_thrust, _throttle_thrust_max);
182 
183  float rp_thrust_max = MAX(fabsf(roll_thrust), fabsf(pitch_thrust));
184 
185  // calculate how much roll and pitch must be scaled to leave enough range for the minimum yaw
186  if (is_zero(rp_thrust_max)) {
187  rp_scale = 1.0f;
188  } else {
189  rp_scale = constrain_float((1.0f - MIN(fabsf(yaw_thrust), (float)_yaw_headroom/1000.0f)) / rp_thrust_max, 0.0f, 1.0f);
190  if (rp_scale < 1.0f) {
191  limit.roll_pitch = true;
192  }
193  }
194 
195  actuator_allowed = 1.0f - rp_scale * rp_thrust_max;
196  if (fabsf(yaw_thrust) > actuator_allowed) {
197  yaw_thrust = constrain_float(yaw_thrust, -actuator_allowed, actuator_allowed);
198  limit.yaw = true;
199  }
200 
201  // combine roll, pitch and yaw on each actuator
202  // front servo
203 
204  actuator[0] = rp_scale * roll_thrust - yaw_thrust;
205  // right servo
206  actuator[1] = rp_scale * pitch_thrust - yaw_thrust;
207  // rear servo
208  actuator[2] = -rp_scale * roll_thrust - yaw_thrust;
209  // left servo
210  actuator[3] = -rp_scale * pitch_thrust - yaw_thrust;
211 
212  // calculate the minimum thrust that doesn't limit the roll, pitch and yaw forces
213  thrust_min_rpy = MAX(MAX(fabsf(actuator[0]), fabsf(actuator[1])), MAX(fabsf(actuator[2]), fabsf(actuator[3])));
214 
215  thr_adj = throttle_thrust - _throttle_avg_max;
216  if (thr_adj < (thrust_min_rpy - _throttle_avg_max)) {
217  // Throttle can't be reduced to the desired level because this would mean roll or pitch control
218  // would not be able to reach the desired level because of lack of thrust.
219  thr_adj = MIN(thrust_min_rpy, _throttle_avg_max) - _throttle_avg_max;
220  }
221 
222  // calculate the throttle setting for the lift fan
223  _thrust_out = _throttle_avg_max + thr_adj;
224 
225  if (is_zero(_thrust_out)) {
226  limit.roll_pitch = true;
227  limit.yaw = true;
228  }
229 
230  // limit thrust out for calculation of actuator gains
231  float thrust_out_actuator = constrain_float(MAX(_throttle_hover*0.5f,_thrust_out), 0.1f, 1.0f);
232 
233  // calculate the maximum allowed actuator output and maximum requested actuator output
234  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
235  if (actuator_max > fabsf(actuator[i])) {
236  actuator_max = fabsf(actuator[i]);
237  }
238  }
239  if (actuator_max > thrust_out_actuator && !is_zero(actuator_max)) {
240  // roll, pitch and yaw request can not be achieved at full servo defection
241  // reduce roll, pitch and yaw to reduce the requested defection to maximum
242  limit.roll_pitch = true;
243  limit.yaw = true;
244  rp_scale = thrust_out_actuator/actuator_max;
245  } else {
246  rp_scale = 1.0f;
247  }
248 
249  // force of a lifting surface is approximately equal to the angle of attack times the airflow velocity squared
250  // static thrust is proportional to the airflow velocity squared
251  // therefore the torque of the roll and pitch actuators should be approximately proportional to
252  // the angle of attack multiplied by the static thrust.
253  for (uint8_t i=0; i<NUM_ACTUATORS; i++) {
254  _actuator_out[i] = constrain_float(rp_scale*actuator[i]/thrust_out_actuator, -1.0f, 1.0f);
255  }
256 }
