66  initialise sensor
67  */
68 bool AP_RangeFinder_trone::init(void)
69 {
70  if (!dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
71  return false;
72  }
73 
74  dev->set_retries(10);
75 
76  // check WHOAMI
77  uint8_t whoami;
78  if (!dev->read_registers(TRONE_WHOAMI, &whoami, 1) ||
79  whoami != TRONE_WHOAMI_VALUE) {
80  return false;
81  }
82 
83  if (!measure()) {
84  dev->get_semaphore()->give();
85  return false;
86  }
87 
88  // give time for the sensor to process the request
89  hal.scheduler->delay(70);
90 
91  uint16_t distance_cm;
92  if (!collect(distance_cm)) {
93  dev->get_semaphore()->give();
94  return false;
95  }
96 
97  dev->get_semaphore()->give();
98 
99  dev->set_retries(1);
100 
101  dev->register_periodic_callback(50000,
102  FUNCTOR_BIND_MEMBER(&AP_RangeFinder_trone::timer, void));
103 
104  return true;
105 }

114 // collect - return last value measured by sensor
115 bool AP_RangeFinder_trone::collect(uint16_t &distance_cm)
116 {
117  uint8_t d[3];
118 
119  // take range reading and read back results
120  if (!dev->transfer(nullptr, 0, d, sizeof(d))) {
121  return false;
122  }
123 
124  if (d[2] != crc_crc8(d, 2)) {
125  // bad CRC
126  return false;
127  }
128 
129  distance_cm = ((uint16_t(d[0]) << 8) | d[1]) / 10;
130 
131  return true;
132 }

134 /*
135  timer called at 20Hz
136 */
137 void AP_RangeFinder_trone::timer(void)
138 {
139  // take a reading
140  uint16_t distance_cm;
141  if (collect(distance_cm) && _sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
142  accum.sum += distance_cm;
143  accum.count++;
144  _sem->give();
145  }
146 
147  // and immediately ask for a new reading
148  measure();
149 }
