101 /*
102  write some bytes from the simulated GPS
103  */
104 void SITL_State::_gps_write(const uint8_t *p, uint16_t size, uint8_t instance)
105 {
106  while (size--) {
107  if (_sitl->gps_byteloss > 0.0f) {
108  float r = ((((unsigned)random()) % 1000000)) / 1.0e4;
109  if (r < _sitl->gps_byteloss) {
110  // lose the byte
111  p++;
112  continue;
113  }
114  }
115  if (instance == 0 && gps_state.gps_fd != 0) {
116  write(gps_state.gps_fd, p, 1);
117  }
118  if (instance == 1 && _sitl->gps2_enable) {
119  if (gps2_state.gps_fd != 0) {
120  write(gps2_state.gps_fd, p, 1);
121  }
122  }
123  p++;
124  }
125 }

127 /*
128  get timeval using simulation time
129  */
130 static void simulation_timeval(struct timeval *tv)
131 {
132  uint64_t now = AP_HAL::micros64();
133  static uint64_t first_usec;
134  static struct timeval first_tv;
135  if (first_usec == 0) {
136  first_usec = now;
137  gettimeofday(&first_tv, nullptr);
138  }
139  *tv = first_tv;
140  tv->tv_sec += now / 1000000ULL;
141  uint64_t new_usec = tv->tv_usec + (now % 1000000ULL);
142  tv->tv_sec += new_usec / 1000000ULL;
143  tv->tv_usec = new_usec % 1000000ULL;
144 }

146 /*
147  send a UBLOX GPS message
148  */
149 void SITL_State::_gps_send_ubx(uint8_t msgid, uint8_t *buf, uint16_t size, uint8_t instance)
150 {
151  const uint8_t PREAMBLE1 = 0xb5;
152  const uint8_t PREAMBLE2 = 0x62;
153  const uint8_t CLASS_NAV = 0x1;
154  uint8_t hdr[6], chk[2];
155  hdr[0] = PREAMBLE1;
156  hdr[1] = PREAMBLE2;
157  hdr[2] = CLASS_NAV;
158  hdr[3] = msgid;
159  hdr[4] = size & 0xFF;
160  hdr[5] = size >> 8;
161  chk[0] = chk[1] = hdr[2];
162  chk[1] += (chk[0] += hdr[3]);
163  chk[1] += (chk[0] += hdr[4]);
164  chk[1] += (chk[0] += hdr[5]);
165  for (uint8_t i=0; i<size; i++) {
166  chk[1] += (chk[0] += buf[i]);
167  }
168  _gps_write(hdr, sizeof(hdr), instance);
169  _gps_write(buf, size, instance);
170  _gps_write(chk, sizeof(chk), instance);
171 }

173 /*
174  return GPS time of week in milliseconds
175  */
176 static void gps_time(uint16_t *time_week, uint32_t *time_week_ms)
177 {
178  struct timeval tv;
179  simulation_timeval(&tv);
180  const uint32_t epoch = 86400*(10*365 + (1980-1969)/4 + 1 + 6 - 2) - (GPS_LEAPSECONDS_MILLIS / 1000ULL);
181  uint32_t epoch_seconds = tv.tv_sec - epoch;
182  *time_week = epoch_seconds / SEC_PER_WEEK;
183  uint32_t t_ms = tv.tv_usec / 1000;
184  // round time to nearest 200ms
185  *time_week_ms = (epoch_seconds % SEC_PER_WEEK) * MSEC_PER_SEC + ((t_ms/200) * 200);
186 }

400  send a new GPS MTK packet

402 void SITL_State::_update_gps_mtk(const struct gps_data *d)
403 {
404  struct PACKED mtk_msg {
405  uint8_t preamble1;
406  uint8_t preamble2;
407  uint8_t msg_class;
408  uint8_t msg_id;
409  int32_t latitude;
410  int32_t longitude;
411  int32_t altitude;
412  int32_t ground_speed;
413  int32_t ground_course;
414  uint8_t satellites;
415  uint8_t fix_type;
416  uint32_t utc_time;
417  uint8_t ck_a;
418  uint8_t ck_b;
419  } p;
420 
421  p.preamble1 = 0xb5;
422  p.preamble2 = 0x62;
423  p.msg_class = 1;
424  p.msg_id = 5;
425  p.latitude = d->latitude * 1.0e6;
426  p.longitude = d->longitude * 1.0e6;
427  p.altitude = d->altitude * 100;
428  p.ground_speed = norm(d->speedN, d->speedE) * 100;
429  p.ground_course = ToDeg(atan2f(d->speedE, d->speedN)) * 1000000.0f;
430  if (p.ground_course < 0.0f) {
431  p.ground_course += 360.0f * 1000000.0f;
432  }
433  p.satellites = d->have_lock?_sitl->gps_numsats:3;
434  p.fix_type = d->have_lock?3:1;
435 
436  // the spec is not very clear, but the time field seems to be
437  // milliseconds since the start of the day in UTC time,
438  // done in powers of 100.
439  // The date is powers of 100 as well, but in days since 1/1/2000
440  struct tm tm;
441  struct timeval tv;
442 
443  simulation_timeval(&tv);
444  tm = *gmtime(&tv.tv_sec);
445  uint32_t hsec = (tv.tv_usec / (10000*20)) * 20; // always multiple of 20
446 
447  p.utc_time = hsec + tm.tm_sec*100 + tm.tm_min*100*100 + tm.tm_hour*100*100*100;
448 
449  swap_uint32((uint32_t *)&p.latitude, 5);
450  swap_uint32((uint32_t *)&p.utc_time, 1);
451  mtk_checksum(&p.msg_class, sizeof(p)-4, &p.ck_a, &p.ck_b);
452 
453  _gps_write((uint8_t*)&p, sizeof(p));
454 }
