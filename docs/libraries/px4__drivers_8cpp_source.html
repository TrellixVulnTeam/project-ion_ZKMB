65 /*
66  setup PWM pins
67  */
68 void AP_BoardConfig::px4_setup_pwm()
69 {
70  /* configure the FMU driver for the right number of PWMs */
71  static const struct {
72  uint8_t mode_parm;
73  uint8_t mode_value;
74  uint8_t num_gpios;
75  } mode_table[] = {
76  /* table mapping BRD_PWM_COUNT to ioctl arguments */
77  { 0, PWM_SERVO_MODE_NONE, 6 },
78  { 2, PWM_SERVO_MODE_2PWM, 4 },
79  { 4, PWM_SERVO_MODE_4PWM, 2 },
80  { 6, PWM_SERVO_MODE_6PWM, 0 },
81  { 7, PWM_SERVO_MODE_3PWM1CAP, 2 },
82 #if CONFIG_HAL_BOARD == HAL_BOARD_VRBRAIN
83  { 8, PWM_SERVO_MODE_12PWM, 0 },
84 #endif
85  };
86  uint8_t mode_parm = (uint8_t)px4.pwm_count.get();
87  uint8_t i;
88  for (i=0; i<ARRAY_SIZE(mode_table); i++) {
89  if (mode_table[i].mode_parm == mode_parm) {
90  break;
91  }
92  }
93  if (i == ARRAY_SIZE(mode_table)) {
94  hal.console->printf("RCOutput: invalid BRD_PWM_COUNT %u\n", mode_parm);
95  } else {
96  int fd = open("/dev/px4fmu", 0);
97  if (fd == -1) {
98  AP_HAL::panic("Unable to open /dev/px4fmu");
99  }
100  if (ioctl(fd, PWM_SERVO_SET_MODE, mode_table[i].mode_value) != 0) {
101  hal.console->printf("RCOutput: unable to setup AUX PWM with BRD_PWM_COUNT %u\n", mode_parm);
102  }
103  close(fd);
104 #if CONFIG_HAL_BOARD == HAL_BOARD_PX4
105  if (mode_table[i].num_gpios < 2) {
106  // reduce change of config mistake where relay and PWM interfere
107  AP_Param::set_default_by_name("RELAY_PIN", -1);
108  AP_Param::set_default_by_name("RELAY_PIN2", -1);
109  }
110 #endif
111  }

127 /*
128  setup safety switch
129  */
130 void AP_BoardConfig::px4_setup_safety_mask()
131 {
132 #if CONFIG_HAL_BOARD == HAL_BOARD_PX4
133  // setup channels to ignore the armed state
134  int px4io_fd = open("/dev/px4io", 0);
135  if (px4io_fd != -1) {
136  if (ioctl(px4io_fd, PWM_SERVO_IGNORE_SAFETY, (uint16_t)px4.ignore_safety_channels) != 0) {
137  hal.console->printf("IGNORE_SAFETY failed\n");
138  }
139  }
140  int px4fmu_fd = open("/dev/px4fmu", 0);
141  if (px4fmu_fd != -1) {
142  uint16_t mask = px4.ignore_safety_channels;
143  if (px4io_fd != -1) {
144  mask >>= 8;
145  }
146  if (ioctl(px4fmu_fd, PWM_SERVO_IGNORE_SAFETY, (uint16_t)mask) != 0) {
147  hal.console->printf("IGNORE_SAFETY failed\n");
148  }
149  close(px4fmu_fd);
150  }
151  if (px4io_fd != -1) {
152  close(px4io_fd);
153  }
154 #endif
155 }

252 /*
253  start one px4 driver
254  */
255 bool AP_BoardConfig::px4_start_driver(main_fn_t main_function, const char *name, const char *arguments)
256 {
257  char *s = strdup(arguments);
258  char *args[10];
259  uint8_t nargs = 0;
260  char *saveptr = nullptr;
261 
262  // parse into separate arguments
263  for (char *tok=strtok_r(s, " ", &saveptr); tok; tok=strtok_r(nullptr, " ", &saveptr)) {
264  args[nargs++] = tok;
265  if (nargs == ARRAY_SIZE(args)-1) {
266  break;
267  }
268  }
269  args[nargs++] = nullptr;
270 
271  printf("Starting driver %s %s\n", name, arguments);
272  pid_t pid;
273 
274  if (task_spawn(&pid, name, main_function, nullptr, nullptr,
275  args, nullptr) != 0) {
276  free(s);
277  printf("Failed to spawn %s\n", name);
278  return false;
279  }
280 
281  // wait for task to exit and gather status
282  int status = -1;
283  if (waitpid(pid, &status, 0) != pid) {
284  printf("waitpid failed for %s\n", name);
285  free(s);
286  return false;
287  }
288  free(s);
289  return (status >> 8) == 0;
290 }

292 void AP_BoardConfig::px4_setup_drivers(void)
293 {
294 #if defined(CONFIG_ARCH_BOARD_PX4FMU_V4)
295  /*
296  this works around an issue with some FMUv4 hardware (eg. copies
297  of the Pixracer) which have incorrect components leading to
298  sensor brownout on boot
299  */
300  if (px4_start_driver(fmu_main, "fmu", "sensor_reset 20")) {
301  printf("FMUv4 sensor reset complete\n");
302  }
303 #endif
304 
305  if (px4.board_type == PX4_BOARD_OLDDRIVERS) {
306  printf("Old drivers no longer supported\n");
307  px4.board_type = PX4_BOARD_AUTO;
308  }
309 
310  // run board auto-detection
311  px4_autodetect();
312 
313  if (px4.board_type == PX4_BOARD_PH2SLIM ||
314  px4.board_type == PX4_BOARD_PIXHAWK2) {
315  _imu_target_temperature.set_default(60);
316  }
317 
318  px4_configured_board = (enum px4_board_type)px4.board_type.get();
319 
320  switch (px4_configured_board) {
321  case PX4_BOARD_PX4V1:
322  case PX4_BOARD_PIXHAWK:
323  case PX4_BOARD_PIXHAWK2:
324  case PX4_BOARD_PIXRACER:
325  case PX4_BOARD_PHMINI:
326  case PX4_BOARD_AUAV21:
327  case PX4_BOARD_PH2SLIM:
328  case PX4_BOARD_AEROFC:

344 /*
345  setup px4io, possibly updating firmware
346  */
347 void AP_BoardConfig::px4_setup_px4io(void)
348 {
349  if (px4_start_driver(px4io_main, "px4io", "start norc")) {
350  printf("px4io started OK\n");
351  } else {
352  // might be in bootloader mode if user held down safety switch
353  // at power on
354  printf("Loading /etc/px4io/px4io.bin\n");
355  px4_tone_alarm("MBABGP");
356  if (px4_start_driver(px4io_main, "px4io", "update /etc/px4io/px4io.bin")) {
357  printf("upgraded PX4IO firmware OK\n");
358  px4_tone_alarm("MSPAA");
359  } else {
360  printf("Failed to upgrade PX4IO firmware\n");
361  px4_tone_alarm("MNGGG");
362  }
363  hal.scheduler->delay(1000);
364  if (px4_start_driver(px4io_main, "px4io", "start norc")) {
365  printf("px4io started OK\n");
366  } else {
367  px4_sensor_error("px4io start failed");
368  }
369  }
370 
371  /*
372  see if we need to update px4io firmware
373  */
374  if (px4_start_driver(px4io_main, "px4io", "checkcrc /etc/px4io/px4io.bin")) {
375  printf("PX4IO CRC OK\n");
376  } else {
377  printf("PX4IO CRC failure\n");
378  px4_tone_alarm("MBABGP");
379  if (px4_start_driver(px4io_main, "px4io", "safety_on")) {
380  printf("PX4IO disarm OK\n");
381  } else {
382  printf("PX4IO disarm failed\n");
383  }
384  hal.scheduler->delay(1000);
385 
386  if (px4_start_driver(px4io_main, "px4io", "forceupdate 14662 /etc/px4io/px4io.bin")) {
387  hal.scheduler->delay(1000);
388  if (px4_start_driver(px4io_main, "px4io", "start norc")) {
389  printf("px4io restart OK\n");
390  px4_tone_alarm("MSPAA");
391  } else {
392  px4_tone_alarm("MNGGG");
393  px4_sensor_error("PX4IO restart failed");
394  }
395  } else {
396  printf("PX4IO update failed\n");
397  px4_tone_alarm("MNGGG");
398  }
399  }

439 /*
440  check a SPI device for a register value
441  */
442 bool AP_BoardConfig::spi_check_register(const char *devname, uint8_t regnum, uint8_t value, uint8_t read_flag)
443 {
444  auto dev = hal.spi->get_device(devname);
445  if (!dev) {
446  printf("%s: no device\n", devname);
447  return false;
448  }
449  dev->set_read_flag(read_flag);
450  uint8_t v;
451  if (!dev->read_registers(regnum, &v, 1)) {
452  printf("%s: reg %02x read fail\n", devname, (unsigned)regnum);
453  return false;
454  }
455  printf("%s: reg %02x %02x %02x\n", devname, (unsigned)regnum, (unsigned)value, (unsigned)v);
456  return v == value;
457 }

494  spi_check_register(HAL_INS_MPU9250_NAME, MPUREG_WHOAMI, MPU_WHOAMI_MPU9250)) {

496  px4.board_type.set(PX4_BOARD_PHMINI);
497  hal.console->printf("Detected PixhawkMini\n");

502  spi_check_register(HAL_INS_MPU9250_NAME, MPUREG_WHOAMI, MPU_WHOAMI_MPU9250))) {

504  // classic or upgraded Pixhawk1

510 #elif defined(CONFIG_ARCH_BOARD_PX4FMU_V4)
