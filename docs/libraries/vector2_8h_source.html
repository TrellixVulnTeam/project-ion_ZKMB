165  // given a position p1 and a velocity v1 produce a vector
166  // perpendicular to v1 maximising distance from p1
167  static Vector2<T> perpendicular(const Vector2<T> &pos_delta, const Vector2<T> &v1)
168  {
169  Vector2<T> perpendicular1 = Vector2<T>(-v1[1], v1[0]);
170  Vector2<T> perpendicular2 = Vector2<T>(v1[1], -v1[0]);
171  T d1 = perpendicular1 * pos_delta;
172  T d2 = perpendicular2 * pos_delta;
173  if (d1 > d2) {
174  return perpendicular1;
175  }
176  return perpendicular2;
177  }

179  /*
180  * Returns the point closest to p on the line segment (v,w).
181  *
182  * Comments and implementation taken from
183  * http://stackoverflow.com/questions/849211/shortest-distance-between-a-point-and-a-line-segment
184  */
185  static Vector2<T> closest_point(const Vector2<T> &p, const Vector2<T> &v, const Vector2<T> &w)
186  {
187  // length squared of line segment
188  const float l2 = (v - w).length_squared();
189  if (l2 < FLT_EPSILON) {
190  // v == w case
191  return v;
192  }
193  // Consider the line extending the segment, parameterized as v + t (w - v).
194  // We find projection of point p onto the line.
195  // It falls where t = [(p-v) . (w-v)] / |w-v|^2
196  // We clamp t from [0,1] to handle points outside the segment vw.
197  const float t = ((p - v) * (w - v)) / l2;
198  if (t <= 0) {
199  return v;
200  } else if (t >= 1) {
201  return w;
202  } else {
203  return v + (w - v)*t;
204  }
205  }
206 
207  // w defines a line segment from the origin
208  // p is a point
209  // returns the closest distance between the radial and the point
210  static float closest_distance_between_radial_and_point(const Vector2<T> &w,
211  const Vector2<T> &p)
212  {
213  const Vector2<T> closest = closest_point(p, Vector2<T>(0,0), w);
214  const Vector2<T> delta = closest - p;
215  return delta.length();
216  }
