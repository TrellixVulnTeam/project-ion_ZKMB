50 void UARTDriver::begin(uint32_t baud, uint16_t rxSpace, uint16_t txSpace)
51 {
52  const char *path = _sitlState->_uart_path[_portNumber];
53 
54  if (strcmp(path, "GPS1") == 0) {
55  /* gps */
56  _connected = true;
57  _fd = _sitlState->gps_pipe();
58  } else if (strcmp(path, "GPS2") == 0) {
59  /* 2nd gps */
60  _connected = true;
61  _fd = _sitlState->gps2_pipe();
62  } else {
63  /* parse type:args:flags string for path.
64  For example:
65  tcp:5760:wait // tcp listen on port 5760
66  tcp:0:wait // tcp listen on use base_port + 0
67  tcpclient:192.168.2.15:5762
68  uart:/dev/ttyUSB0:57600
69  */
70  char *saveptr = nullptr;
71  char *s = strdup(path);
72  char *devtype = strtok_r(s, ":", &saveptr);
73  char *args1 = strtok_r(nullptr, ":", &saveptr);
74  char *args2 = strtok_r(nullptr, ":", &saveptr);
75  if (strcmp(devtype, "tcp") == 0) {
76  uint16_t port = atoi(args1);
77  bool wait = (args2 && strcmp(args2, "wait") == 0);
78  _tcp_start_connection(port, wait);
79  } else if (strcmp(devtype, "tcpclient") == 0) {
80  if (args2 == nullptr) {
81  AP_HAL::panic("Invalid tcp client path: %s", path);
82  }
83  uint16_t port = atoi(args2);
84  _tcp_start_client(args1, port);
85  } else if (strcmp(devtype, "uart") == 0) {
86  uint32_t baudrate = args2? atoi(args2) : baud;
87  ::printf("UART connection %s:%u\n", args1, baudrate);
88  _uart_path = strdup(args1);
89  _uart_baudrate = baudrate;
90  _uart_start_connection();
91  } else {
92  AP_HAL::panic("Invalid device path: %s", path);
93  }
94  free(s);
95  }
96 
97  _set_nonblocking(_fd);
98 }

161  start a TCP connection for the serial port. If wait_for_connection
162  is true then block until a client connects
163  */
164 void UARTDriver::_tcp_start_connection(uint16_t port, bool wait_for_connection)
165 {
166  int one=1;
167  struct sockaddr_in sockaddr;
168  int ret;
169 
170  if (_connected) {
171  return;
172  }
173 
174  _use_send_recv = true;
175 
176  if (_console) {
177  // hack for console access
178  _connected = true;
179  _use_send_recv = false;
180  _listen_fd = -1;
181  _fd = 1;
182  return;
183  }
184 
185  if (_fd != -1) {
186  close(_fd);
187  }
188 
189  if (_listen_fd == -1) {
190  memset(&sockaddr,0,sizeof(sockaddr));
191 
192 #ifdef HAVE_SOCK_SIN_LEN
193  sockaddr.sin_len = sizeof(sockaddr);
194 #endif
195  if (port > 1000) {
196  sockaddr.sin_port = htons(port);
197  } else {
198  sockaddr.sin_port = htons(_sitlState->base_port() + port);
199  }
200  sockaddr.sin_family = AF_INET;
201 
202  _listen_fd = socket(AF_INET, SOCK_STREAM, 0);
203  if (_listen_fd == -1) {
204  fprintf(stderr, "socket failed - %s\n", strerror(errno));
205  exit(1);
206  }
207 
208  /* we want to be able to re-use ports quickly */
209  setsockopt(_listen_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
210 
211  fprintf(stderr, "bind port %u for %u\n",
212  (unsigned)ntohs(sockaddr.sin_port),
213  (unsigned)_portNumber);
214 
215  ret = bind(_listen_fd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
216  if (ret == -1) {
217  fprintf(stderr, "bind failed on port %u - %s\n",
218  (unsigned)ntohs(sockaddr.sin_port),
219  strerror(errno));
220  exit(1);
221  }
222 
223  ret = listen(_listen_fd, 5);
224  if (ret == -1) {
225  fprintf(stderr, "listen failed - %s\n", strerror(errno));
226  exit(1);
227  }
228 
229  fprintf(stderr, "Serial port %u on TCP port %u\n", _portNumber,
230  _sitlState->base_port() + _portNumber);
231  fflush(stdout);
232  }
233 
234  if (wait_for_connection) {
235  fprintf(stdout, "Waiting for connection ....\n");
236  fflush(stdout);
237  _fd = accept(_listen_fd, nullptr, nullptr);
238  if (_fd == -1) {
239  fprintf(stderr, "accept() error - %s", strerror(errno));
240  exit(1);
241  }
242  setsockopt(_fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof(one));
243  setsockopt(_fd, IPPROTO_TCP, TCP_NODELAY, &one, sizeof(one));
244  _connected = true;
245  }
246 }

301  start a UART connection for the serial port
302  */
303 void UARTDriver::_uart_start_connection(void)
304 {
305  struct termios t {};
306  if (!_connected) {
307  _fd = ::open(_uart_path, O_RDWR | O_CLOEXEC);
308  if (_fd == -1) {
309  return;
310  }
311  // use much smaller buffer sizes on real UARTs
312  _writebuffer.set_size(1024);
313  _readbuffer.set_size(512);
314  ::printf("Opened %s\n", _uart_path);
315  }
316 
317  if (_fd == -1) {
318  AP_HAL::panic("Unable to open UART %s", _uart_path);
319  }
320 
321  // set non-blocking
322  int flags = fcntl(_fd, F_GETFL, 0);
323  flags = flags | O_NONBLOCK;
324  fcntl(_fd, F_SETFL, flags);
325 
326  // disable LF -> CR/LF
327  tcgetattr(_fd, &t);
328  t.c_iflag &= ~(BRKINT | ICRNL | IMAXBEL | IXON | IXOFF);
329  t.c_oflag &= ~(OPOST | ONLCR);
330  t.c_lflag &= ~(ISIG | ICANON | IEXTEN | ECHO | ECHOE | ECHOK | ECHOCTL | ECHOKE);
331  t.c_cc[VMIN] = 0;
332  if (_sitlState->use_rtscts()) {
333  t.c_cflag |= CRTSCTS;
334  }
335  tcsetattr(_fd, TCSANOW, &t);
336 
337  // set baudrate
338  tcgetattr(_fd, &t);
339  cfsetspeed(&t, _uart_baudrate);
340  tcsetattr(_fd, TCSANOW, &t);
341 
342  _connected = true;
343  _use_send_recv = false;
344 }

350 {

363  }
364  }

368  use select() to see if something is pending
369  */
370 bool UARTDriver::_select_check(int fd)
371 {
372  if (fd == -1) {
373  return false;
374  }
375  fd_set fds;
376  struct timeval tv;
377 
378  FD_ZERO(&fds);
379  FD_SET(fd, &fds);
380 
381  // zero time means immediate return from select()
382  tv.tv_sec = 0;
383  tv.tv_usec = 0;
384 
385  if (select(fd+1, &fds, nullptr, nullptr, &tv) == 1) {
386  return true;
387  }
388  return false;
389 }

405 void UARTDriver::_timer_tick(void)
406 {
407  if (!_connected) {
408  _check_reconnect();
409  return;
410  }
411  uint32_t navail;
412  ssize_t nwritten;
413 
414  const uint8_t *readptr = _writebuffer.readptr(navail);
415  if (readptr && navail > 0) {
416  if (!_use_send_recv) {
417  nwritten = ::write(_fd, readptr, navail);
418  if (nwritten == -1 && errno != EAGAIN && _uart_path) {
419  close(_fd);
420  _fd = -1;
421  _connected = false;
422  }
423  } else {
424  nwritten = send(_fd, readptr, navail, MSG_DONTWAIT);
425  }
426  if (nwritten > 0) {
427  _writebuffer.advance(nwritten);
428  }
429  }
430 
431  uint32_t space = _readbuffer.space();
432  if (space == 0) {
433  return;
434  }
435 
436  char buf[space];
437  ssize_t nread = 0;
438  if (!_use_send_recv) {
439  int fd = _console?0:_fd;
440  nread = ::read(fd, buf, space);
441  if (nread == -1 && errno != EAGAIN && _uart_path) {
442  close(_fd);
443  _fd = -1;
444  _connected = false;
445  }
446  } else {
447  if (_select_check(_fd)) {
448  nread = recv(_fd, buf, space, MSG_DONTWAIT);
449  if (nread <= 0) {
450  // the socket has reached EOF
451  close(_fd);
452  _connected = false;
453  fprintf(stdout, "Closed connection on serial port %u\n", _portNumber);
454  fflush(stdout);
455  return;
456  }
457  } else {
458  nread = 0;
459  }
460  }
461  if (nread > 0) {
462  _readbuffer.write((uint8_t *)buf, nread);
463  }
464 }
