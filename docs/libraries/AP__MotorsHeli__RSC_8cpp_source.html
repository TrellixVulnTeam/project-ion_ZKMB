39 // output - update value to send to ESC/Servo
40 void AP_MotorsHeli_RSC::output(RotorControlState state)
41 {
42  float dt;
43  uint64_t now = AP_HAL::micros64();
44  float last_control_output = _control_output;
45 
46  if (_last_update_us == 0) {
47  _last_update_us = now;
48  dt = 0.001f;
49  } else {
50  dt = 1.0e-6f * (now - _last_update_us);
51  _last_update_us = now;
52  }
53 
54  switch (state){
55  case ROTOR_CONTROL_STOP:
56  // set rotor ramp to decrease speed to zero, this happens instantly inside update_rotor_ramp()
57  update_rotor_ramp(0.0f, dt);
58 
59  // control output forced to zero
60  _control_output = 0.0f;
61  break;
62 
63  case ROTOR_CONTROL_IDLE:
64  // set rotor ramp to decrease speed to zero
65  update_rotor_ramp(0.0f, dt);
66 
67  // set rotor control speed to idle speed parameter, this happens instantly and ignore ramping
68  _control_output = _idle_output;
69  break;
70 
71  case ROTOR_CONTROL_ACTIVE:
72  // set main rotor ramp to increase to full speed
73  update_rotor_ramp(1.0f, dt);
74 
75  if ((_control_mode == ROTOR_CONTROL_MODE_SPEED_PASSTHROUGH) || (_control_mode == ROTOR_CONTROL_MODE_SPEED_SETPOINT)) {
76  // set control rotor speed to ramp slewed value between idle and desired speed
77  _control_output = _idle_output + (_rotor_ramp_output * (_desired_speed - _idle_output));
78  } else if (_control_mode == ROTOR_CONTROL_MODE_OPEN_LOOP_POWER_OUTPUT) {
79  // throttle output depending on estimated power demand. Output is ramped up from idle speed during rotor runup. A negative load
80  // is for the left side of the V-curve (-ve collective) A positive load is for the right side (+ve collective)
81  if (_load_feedforward >= 0) {
82  float range = _power_output_high - _power_output_low;
83  _control_output = _idle_output + (_rotor_ramp_output * ((_power_output_low - _idle_output) + (range * _load_feedforward)));
84  } else {
85  float range = _power_output_negc - _power_output_low;
86  _control_output = _idle_output + (_rotor_ramp_output * ((_power_output_low - _idle_output) - (range * _load_feedforward)));
87  }
88  }
89  break;
90  }
91 
92  // update rotor speed run-up estimate
93  update_rotor_runup(dt);
94 
95  if (_power_slewrate > 0) {
96  // implement slew rate for throttle
97  float max_delta = dt * _power_slewrate * 0.01f;
98  _control_output = constrain_float(_control_output, last_control_output-max_delta, last_control_output+max_delta);
99  }
100 
101  // output to rsc servo
102  write_rsc(_control_output);
103 }

105 // update_rotor_ramp - slews rotor output scalar between 0 and 1, outputs float scalar to _rotor_ramp_output
106 void AP_MotorsHeli_RSC::update_rotor_ramp(float rotor_ramp_input, float dt)
107 {
108  // sanity check ramp time
109  if (_ramp_time <= 0) {
110  _ramp_time = 1;
111  }
112 
113  // ramp output upwards towards target
114  if (_rotor_ramp_output < rotor_ramp_input) {
115  // allow control output to jump to estimated speed
116  if (_rotor_ramp_output < _rotor_runup_output) {
117  _rotor_ramp_output = _rotor_runup_output;
118  }
119  // ramp up slowly to target
120  _rotor_ramp_output += (dt / _ramp_time);
121  if (_rotor_ramp_output > rotor_ramp_input) {
122  _rotor_ramp_output = rotor_ramp_input;
123  }
124  }else{
125  // ramping down happens instantly
126  _rotor_ramp_output = rotor_ramp_input;
127  }
128 }

130 // update_rotor_runup - function to slew rotor runup scalar, outputs float scalar to _rotor_runup_ouptut
131 void AP_MotorsHeli_RSC::update_rotor_runup(float dt)
132 {
133  // sanity check runup time
134  if (_runup_time < _ramp_time) {
135  _runup_time = _ramp_time;
136  }
137  if (_runup_time <= 0 ) {
138  _runup_time = 1;
139  }
140 
141  // ramp speed estimate towards control out
142  float runup_increment = dt / _runup_time;
143  if (_rotor_runup_output < _rotor_ramp_output) {
144  _rotor_runup_output += runup_increment;
145  if (_rotor_runup_output > _rotor_ramp_output) {
146  _rotor_runup_output = _rotor_ramp_output;
147  }
148  }else{
149  _rotor_runup_output -= runup_increment;
150  if (_rotor_runup_output < _rotor_ramp_output) {
151  _rotor_runup_output = _rotor_ramp_output;
152  }
153  }
154 
155  // update run-up complete flag
156 
157  // if control mode is disabled, then run-up complete always returns true
158  if ( _control_mode == ROTOR_CONTROL_MODE_DISABLED ){
159  _runup_complete = true;
160  return;
161  }
162 
163  // if rotor ramp and runup are both at full speed, then run-up has been completed
164  if (!_runup_complete && (_rotor_ramp_output >= 1.0f) && (_rotor_runup_output >= 1.0f)) {
165  _runup_complete = true;
166  }
167  // if rotor speed is less than critical speed, then run-up is not complete
168  // this will prevent the case where the target rotor speed is less than critical speed
169  if (_runup_complete && (get_rotor_speed() <= _critical_speed)) {
170  _runup_complete = false;
171  }
172 }

181 // write_rsc - outputs pwm onto output rsc channel
182 // servo_out parameter is of the range 0 ~ 1
183 void AP_MotorsHeli_RSC::write_rsc(float servo_out)
184 {
185  if (_control_mode == ROTOR_CONTROL_MODE_DISABLED){
186  // do not do servo output to avoid conflicting with other output on the channel
187  // ToDo: We should probably use RC_Channel_Aux to avoid this problem
188  return;
189  } else {
190  // calculate PWM value based on H_RSC_PWM_MIN, H_RSC_PWM_MAX and H_RSC_PWM_REV
191  uint16_t pwm = servo_out * (_pwm_max - _pwm_min);
192  if (_pwm_rev >= 0) {
193  pwm = _pwm_min + pwm;
194  } else {
195  pwm = _pwm_max - pwm;
196  }
197  SRV_Channels::set_output_pwm(_aux_fn, pwm);
198  }
