157 /*
158  low level transfer function
159  */
160 void SPIDevice::do_transfer(const uint8_t *send, uint8_t *recv, uint32_t len)
161 {
162  /*
163  to accomodate the method in PX4 drivers of using interrupt
164  context for SPI device transfers we need to check if PX4 has
165  registered a driver on this bus. If not then we can avoid the
166  irqsave/irqrestore and get bus parallelism for DMA enabled
167  buses.
168 
169  There is a race in this if a PX4 driver starts while we are
170  running this, but that would only happen at early boot and is
171  very unlikely
172 
173  yes, this is a nasty hack. Suggestions for a better method
174  appreciated.
175  */
176  bool use_irq_save = up_spi_use_irq_save(bus.dev);
177  irqstate_t state;
178  if (use_irq_save) {
179  state = irqsave();
180  }
181  perf_begin(perf);
182  SPI_LOCK(bus.dev, true);
183  SPI_SETFREQUENCY(bus.dev, frequency);
184  SPI_SETMODE(bus.dev, device_desc.mode);
185  SPI_SETBITS(bus.dev, 8);
186  SPI_SELECT(bus.dev, device_desc.device, true);
187  SPI_EXCHANGE(bus.dev, send, recv, len);
188  if (!cs_forced) {
189  SPI_SELECT(bus.dev, device_desc.device, false);
190  }
191  SPI_LOCK(bus.dev, false);
192  perf_end(perf);
193  if (use_irq_save) {
194  irqrestore(state);
195  }
196 }

198 bool SPIDevice::transfer(const uint8_t *send, uint32_t send_len,
199  uint8_t *recv, uint32_t recv_len)
200 {
201  if (send_len == recv_len && send == recv) {
202  // simplest cases, needed for DMA
203  do_transfer(send, recv, recv_len);
204  return true;
205  }
206  uint8_t buf[send_len+recv_len];
207  if (send_len > 0) {
208  memcpy(buf, send, send_len);
209  }
210  if (recv_len > 0) {
211  memset(&buf[send_len], 0, recv_len);
212  }
213  do_transfer(buf, buf, send_len+recv_len);
214  if (recv_len > 0) {
215  memcpy(recv, &buf[send_len], recv_len);
216  }
217  return true;
218 }

257  return a SPIDevice given a string device name
258  */
259 AP_HAL::OwnPtr<AP_HAL::SPIDevice>
260 SPIDeviceManager::get_device(const char *name)
261 {
262  /* Find the bus description in the table */
263  uint8_t i;
264  for (i = 0; device_table[i].name; i++) {
265  if (strcmp(device_table[i].name, name) == 0) {
266  break;
267  }
268  }
269  if (device_table[i].name == nullptr) {
270  printf("SPI: Invalid device name: %s\n", name);
271  return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(nullptr);
272  }
273 
274  SPIDesc &desc = device_table[i];
275 
276  // find the bus
277  SPIBus *busp;
278  for (busp = buses; busp; busp = (SPIBus *)busp->next) {
279  if (busp->bus == desc.bus) {
280  break;
281  }
282  }
283  if (busp == nullptr) {
284  // create a new one
285  busp = new SPIBus;
286  if (busp == nullptr) {
287  return nullptr;
288  }
289  busp->next = buses;
290  busp->bus = desc.bus;
291  busp->dev = up_spiinitialize(desc.bus);
292  buses = busp;
293  }
294 
295  return AP_HAL::OwnPtr<AP_HAL::SPIDevice>(new SPIDevice(*busp, desc));
296 }
