38 /*
39  scan a module for hook symbols
40  */
41 void AP_Module::module_scan(const char *path)
42 {
43 #if defined(HAVE_LIBDL)
44  void *m = dlopen(path, RTLD_NOW);
45  if (m == nullptr) {
46  printf("dlopen(%s) -> %s\n", path, dlerror());
47  return;
48  }
49  uint8_t found_hooks = 0;
50  for (uint16_t i=0; i<NUM_HOOKS; i++) {
51  void *s = dlsym(m, hook_names[i]);
52  if (s != nullptr) {
53  // found a hook in this module, add it to the list
54  struct hook_list *h = new hook_list;
55  if (h == nullptr) {
56  AP_HAL::panic("Failed to allocate hook for %s", hook_names[i]);
57  }
58  h->next = hooks[i];
59  h->symbol = s;
60  hooks[i] = h;
61  found_hooks++;
62  }
63  }
64  if (found_hooks == 0) {
65  // we don't need this module
66  dlclose(m);
67  } else {
68  printf("AP_Module: Loaded %u hooks from %s\n", (unsigned)found_hooks, path);
69  }
70 #endif
71 }

73 /*
74  initialise AP_Module, looking for shared libraries in the given module path
75 */
76 void AP_Module::init(const char *module_path)
77 {
78  // scan through module directory looking for *.so
79  DIR *d;
80  struct dirent *de;
81  d = opendir(module_path);
82  if (d == nullptr) {
83  return;
84  }
85  while ((de = readdir(d))) {
86  const char *extension = strrchr(de->d_name, '.');
87  if (extension == nullptr || strcmp(extension, ".so") != 0) {
88  continue;
89  }
90  char *path = nullptr;
91  if (asprintf(&path, "%s/%s", module_path, de->d_name) == -1) {
92  continue;
93  }
94  module_scan(path);
95  free(path);
96  }
97  closedir(d);
98 }

132 void AP_Module::call_hook_AHRS_update(const AP_AHRS_NavEKF &ahrs)
133 {
134 #if AP_MODULE_SUPPORTED
135  if (hooks[HOOK_AHRS_UPDATE] == nullptr) {
136  // avoid filling in AHRS_state
137  return;
138  }
139 
140  /*
141  construct AHRS_state structure
142  */
143  struct AHRS_state state {};
144  state.structure_version = AHRS_state_version;
145  state.time_us = AP_HAL::micros64();
146 
147  if (!ahrs.initialised()) {
148  state.status = AHRS_STATUS_INITIALISING;
149  } else if (ahrs.healthy()) {
150  state.status = AHRS_STATUS_HEALTHY;
151  } else {
152  state.status = AHRS_STATUS_UNHEALTHY;
153  }
154 
155  Quaternion q;
156  q.from_rotation_matrix(ahrs.get_rotation_body_to_ned());
157  state.quat[0] = q[0];
158  state.quat[1] = q[1];
159  state.quat[2] = q[2];
160  state.quat[3] = q[3];
161 
162  state.eulers[0] = ahrs.roll;
163  state.eulers[1] = ahrs.pitch;
164  state.eulers[2] = ahrs.yaw;
165 
166  Location loc;
167  if (ahrs.get_origin(loc)) {
168  state.origin.initialised = true;
169  state.origin.latitude = loc.lat;
170  state.origin.longitude = loc.lng;
171  state.origin.altitude = loc.alt*0.01f;
172  }
173 
174  if (ahrs.get_position(loc)) {
175  state.position.available = true;
176  state.position.latitude = loc.lat;
177  state.position.longitude = loc.lng;
178  state.position.altitude = loc.alt*0.01f;
179  }
180 
181  Vector3f pos;
182  if (ahrs.get_relative_position_NED_origin(pos)) {
183  state.relative_position[0] = pos[0];
184  state.relative_position[1] = pos[1];
185  state.relative_position[2] = pos[2];
186  }
187 
188  const Vector3f &gyro = ahrs.get_gyro();
189  state.gyro_rate[0] = gyro[0];
190  state.gyro_rate[1] = gyro[1];
191  state.gyro_rate[2] = gyro[2];
192 
193  const Vector3f &accel_ef = ahrs.get_accel_ef();
194  state.accel_ef[0] = accel_ef[0];
195  state.accel_ef[1] = accel_ef[0];
196  state.accel_ef[2] = accel_ef[0];
197 
198  state.primary_accel = ahrs.get_primary_accel_index();
199  state.primary_gyro = ahrs.get_primary_gyro_index();
200 
201  const Vector3f &gyro_bias = ahrs.get_gyro_drift();
202  state.gyro_bias[0] = gyro_bias[0];
203  state.gyro_bias[1] = gyro_bias[1];
204  state.gyro_bias[2] = gyro_bias[2];
205 
206  Vector3f vel;
207  if (ahrs.get_velocity_NED(vel)) {
208  state.velocity_ned[0] = vel.x;
209  state.velocity_ned[1] = vel.y;
210  state.velocity_ned[2] = vel.z;
211  }
212 
213  for (const struct hook_list *h=hooks[HOOK_AHRS_UPDATE]; h; h=h->next) {
214  ap_hook_AHRS_update_fn_t fn = reinterpret_cast<ap_hook_AHRS_update_fn_t>(h->symbol);
215  fn(&state);
216  }
217 #endif
218 }
