45 // update internal state if new GPS information is available
46 bool
47 AP_GPS_PX4::read(void)
48 {
49  bool updated = false;
50  orb_check(_gps_sub, &updated);
51 
52  if (updated) {
53  if (OK == orb_copy(ORB_ID(vehicle_gps_position), _gps_sub, &_gps_pos)) {
54  state.last_gps_time_ms = AP_HAL::millis();
55  state.status = (AP_GPS::GPS_Status) (_gps_pos.fix_type | AP_GPS::NO_FIX);
56  state.num_sats = _gps_pos.satellites_used;
57  state.hdop = uint16_t(_gps_pos.eph*100.0f + .5f);
58 
59  if (_gps_pos.fix_type >= 2) {
60  state.location.lat = _gps_pos.lat;
61  state.location.lng = _gps_pos.lon;
62  state.location.alt = _gps_pos.alt/10;
63 
64  state.ground_speed = _gps_pos.vel_m_s;
65  state.ground_course = wrap_360(degrees(_gps_pos.cog_rad));
66  state.hdop = _gps_pos.eph*100;
67 
68  // convert epoch timestamp back to gps epoch - evil hack until we get the genuine
69  // raw week information (or APM switches to Posix epoch ;-) )
70  const uint64_t posix_offset = 3657ULL * 24 * 3600 * 1000 - GPS_LEAPSECONDS_MILLIS;
71  const uint64_t ms_per_week = 7ULL * 24 * 3600 * 1000;
72  uint64_t epoch_ms = _gps_pos.time_utc_usec/1000;
73  uint64_t gps_ms = epoch_ms - posix_offset;
74  state.time_week = (uint16_t)(gps_ms / ms_per_week);
75  state.time_week_ms = (uint32_t)(gps_ms - (state.time_week) * ms_per_week);
76 
77  if (_gps_pos.time_utc_usec == 0) {
78  // This is a work-around for https://github.com/PX4/Firmware/issues/1474
79  // reject position reports with invalid time, as APM adjusts it's clock after the first lock has been acquired
80  state.status = AP_GPS::NO_FIX;
81  }
82  }
83  if (_gps_pos.fix_type >= 3) {
84  state.have_vertical_velocity = _gps_pos.vel_ned_valid;
85  state.velocity.x = _gps_pos.vel_n_m_s;
86  state.velocity.y = _gps_pos.vel_e_m_s;
87  state.velocity.z = _gps_pos.vel_d_m_s;
88  state.speed_accuracy = _gps_pos.s_variance_m_s;
89  state.have_speed_accuracy = true;
90  }
91  else {
92  state.have_vertical_velocity = false;
93  }
94  }
95  }
96 
97  return updated;
98 }
