94 void AP_Compass_UAVCAN::read(void)
95 {
96  // avoid division by zero if we haven't received any mag reports
97  if (_count == 0) {
98  return;
99  }
100 
101  if (_mag_baro->take(0)) {
102  _sum /= _count;
103 
104  publish_filtered_field(_sum, _instance);
105 
106  _sum.zero();
107  _count = 0;
108  _mag_baro->give();
109  }

112 void AP_Compass_UAVCAN::handle_mag_msg(Vector3f &mag)
113 {
114  Vector3f raw_field = mag * 1000.0;
115 
116  // rotate raw_field from sensor frame to body frame
117  rotate_field(raw_field, _instance);
118 
119  _last_timestamp = AP_HAL::micros64();
120  // publish raw_field (uncorrected point sample) for calibration use
121  publish_raw_field(raw_field, (uint32_t) _last_timestamp, _instance);
122 
123  // correct raw_field for known errors
124  correct_field(raw_field, _instance);
125 
126  if (_mag_baro->take(0)) {
127  // accumulate into averaging filter
128  _sum += raw_field;
129  _count++;
130  _mag_baro->give();
131  }
132 }
