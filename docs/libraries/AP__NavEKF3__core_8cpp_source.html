584  * not used by the EKF equations, which instead estimate the error in the attitude of

588 void NavEKF3_core::UpdateStrapdownEquationsNED()
589 {
590  // update the quaternion states by rotating from the previous attitude through
591  // the delta angle rotation quaternion and normalise
592  // apply correction for earth's rotation rate
593  // % * - and + operators have been overloaded
594  stateStruct.quat.rotate(delAngCorrected - prevTnb * earthRateNED*imuDataDelayed.delAngDT);
595  stateStruct.quat.normalize();
596 
597  // transform body delta velocities to delta velocities in the nav frame
598  // use the nav frame from previous time step as the delta velocities
599  // have been rotated into that frame
600  // * and + operators have been overloaded
601  Vector3f delVelNav; // delta velocity vector in earth axes
602  delVelNav = prevTnb.mul_transpose(delVelCorrected);
603  delVelNav.z += GRAVITY_MSS*imuDataDelayed.delVelDT;
604 
605  // calculate the body to nav cosine matrix
606  stateStruct.quat.inverse().rotation_matrix(prevTnb);
607 
608  // calculate the rate of change of velocity (used for launch detect and other functions)
609  velDotNED = delVelNav / imuDataDelayed.delVelDT;
610 
611  // apply a first order lowpass filter
612  velDotNEDfilt = velDotNED * 0.05f + velDotNEDfilt * 0.95f;
613 
614  // calculate a magnitude of the filtered nav acceleration (required for GPS
615  // variance estimation)
616  accNavMag = velDotNEDfilt.length();
617  accNavMagHoriz = norm(velDotNEDfilt.x , velDotNEDfilt.y);
618 
619  // if we are not aiding, then limit the horizontal magnitude of acceleration
620  // to prevent large manoeuvre transients disturbing the attitude
621  if ((PV_AidingMode == AID_NONE) && (accNavMagHoriz > 5.0f)) {
622  float gain = 5.0f/accNavMagHoriz;
623  delVelNav.x *= gain;
624  delVelNav.y *= gain;
625  }
626 
627  // save velocity for use in trapezoidal integration for position calcuation
628  Vector3f lastVelocity = stateStruct.velocity;
629 
630  // sum delta velocities to get velocity
631  stateStruct.velocity += delVelNav;
632 
633  // apply a trapezoidal integration to velocities to calculate position
634  stateStruct.position += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
635 
636  // accumulate the bias delta angle and time since last reset by an OF measurement arrival
637  delAngBodyOF += delAngCorrected;
638  delTimeOF += imuDataDelayed.delAngDT;
639 
640  // limit states to protect against divergence
641  ConstrainStates();
642 
643  // If main filter velocity states are valid, update the range beacon receiver position states
644  if (filterStatus.flags.horiz_vel) {

646  }

649 /*
650  * Propagate PVA solution forward from the fusion time horizon to the current time horizon
651  * using simple observer which performs two functions:
652  * 1) Corrects for the delayed time horizon used by the EKF.
653  * 2) Applies a LPF to state corrections to prevent 'stepping' in states due to measurement
654  * fusion introducing unwanted noise into the control loops.
655  * The inspiration for using a complementary filter to correct for time delays in the EKF
656  * is based on the work by A Khosravian.
657  *
658  * “Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements”
659  * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
660 */
661 void NavEKF3_core::calcOutputStates()
662 {
663  // apply corrections to the IMU data
664  Vector3f delAngNewCorrected = imuDataNew.delAng;
665  Vector3f delVelNewCorrected = imuDataNew.delVel;
666  correctDeltaAngle(delAngNewCorrected, imuDataNew.delAngDT);
667  correctDeltaVelocity(delVelNewCorrected, imuDataNew.delVelDT);
668 
669  // apply corections to track EKF solution
670  Vector3f delAng = delAngNewCorrected + delAngCorrection;
671 
672  // convert the rotation vector to its equivalent quaternion
673  Quaternion deltaQuat;
674  deltaQuat.from_axis_angle(delAng);
675 
676  // update the quaternion states by rotating from the previous attitude through
677  // the delta angle rotation quaternion and normalise
678  outputDataNew.quat *= deltaQuat;
679  outputDataNew.quat.normalize();
680 
681  // calculate the body to nav cosine matrix
682  Matrix3f Tbn_temp;
683  outputDataNew.quat.rotation_matrix(Tbn_temp);
684 
685  // transform body delta velocities to delta velocities in the nav frame
686  Vector3f delVelNav = Tbn_temp*delVelNewCorrected;
687  delVelNav.z += GRAVITY_MSS*imuDataNew.delVelDT;
688 
689  // save velocity for use in trapezoidal integration for position calcuation
690  Vector3f lastVelocity = outputDataNew.velocity;
691 
692  // sum delta velocities to get velocity
693  outputDataNew.velocity += delVelNav;
694 
695  // apply a trapezoidal integration to velocities to calculate position
696  outputDataNew.position += (outputDataNew.velocity + lastVelocity) * (imuDataNew.delVelDT*0.5f);
697 
698  // If the IMU accelerometer is offset from the body frame origin, then calculate corrections
699  // that can be added to the EKF velocity and position outputs so that they represent the velocity
700  // and position of the body frame origin.
701  // Note the * operator has been overloaded to operate as a dot product
702  if (!accelPosOffset.is_zero()) {
703  // calculate the average angular rate across the last IMU update
704  // note delAngDT is prevented from being zero in readIMUData()
705  Vector3f angRate = imuDataNew.delAng * (1.0f/imuDataNew.delAngDT);
706 
707  // Calculate the velocity of the body frame origin relative to the IMU in body frame
708  // and rotate into earth frame. Note % operator has been overloaded to perform a cross product
709  Vector3f velBodyRelIMU = angRate % (- accelPosOffset);
710  velOffsetNED = Tbn_temp * velBodyRelIMU;
711 
712  // calculate the earth frame position of the body frame origin relative to the IMU
713  posOffsetNED = Tbn_temp * (- accelPosOffset);
714  } else {
715  velOffsetNED.zero();
716  posOffsetNED.zero();
717  }
718 
719  // store INS states in a ring buffer that with the same length and time coordinates as the IMU data buffer
720  if (runUpdates) {
721  // store the states at the output time horizon
722  storedOutput[storedIMU.get_youngest_index()] = outputDataNew;
723 
724  // recall the states from the fusion time horizon
725  outputDataDelayed = storedOutput[storedIMU.get_oldest_index()];
726 
727  // compare quaternion data with EKF quaternion at the fusion time horizon and calculate correction
728 
729  // divide the demanded quaternion by the estimated to get the error
730  Quaternion quatErr = stateStruct.quat / outputDataDelayed.quat;
731 
732  // Convert to a delta rotation using a small angle approximation
733  quatErr.normalize();
734  Vector3f deltaAngErr;
735  float scaler;
736  if (quatErr[0] >= 0.0f) {
737  scaler = 2.0f;
738  } else {
739  scaler = -2.0f;
740  }
741  deltaAngErr.x = scaler * quatErr[1];
742  deltaAngErr.y = scaler * quatErr[2];
743  deltaAngErr.z = scaler * quatErr[3];
744 
745  // calculate a gain that provides tight tracking of the estimator states and
746  // adjust for changes in time delay to maintain consistent damping ratio of ~0.7
747  float timeDelay = 1e-3f * (float)(imuDataNew.time_ms - imuDataDelayed.time_ms);
748  timeDelay = MAX(timeDelay, dtIMUavg);
749  float errorGain = 0.5f / timeDelay;
750 
751  // calculate a corrrection to the delta angle
752  // that will cause the INS to track the EKF quaternions
753  delAngCorrection = deltaAngErr * errorGain * dtIMUavg;
754 
755  // calculate velocity and position tracking errors
756  Vector3f velErr = (stateStruct.velocity - outputDataDelayed.velocity);
757  Vector3f posErr = (stateStruct.position - outputDataDelayed.position);
758 
759  // collect magnitude tracking error for diagnostics
760  outputTrackError.x = deltaAngErr.length();
761  outputTrackError.y = velErr.length();
762  outputTrackError.z = posErr.length();
763 
764  // convert user specified time constant from centi-seconds to seconds
765  float tauPosVel = constrain_float(0.01f*(float)frontend->_tauVelPosOutput, 0.1f, 0.5f);
766 
767  // calculate a gain to track the EKF position states with the specified time constant
768  float velPosGain = dtEkfAvg / constrain_float(tauPosVel, dtEkfAvg, 10.0f);
769 
770  // use a PI feedback to calculate a correction that will be applied to the output state history
771  posErrintegral += posErr;
772  velErrintegral += velErr;
773  Vector3f velCorrection = velErr * velPosGain + velErrintegral * sq(velPosGain) * 0.1f;
774  Vector3f posCorrection = posErr * velPosGain + posErrintegral * sq(velPosGain) * 0.1f;
775 
776  // loop through the output filter state history and apply the corrections to the velocity and position states
777  // this method is too expensive to use for the attitude states due to the quaternion operations required
778  // but does not introduce a time delay in the 'correction loop' and allows smaller tracking time constants
779  // to be used
780  output_elements outputStates;
781  for (unsigned index=0; index < imu_buffer_length; index++) {
782  outputStates = storedOutput[index];
783 
784  // a constant velocity correction is applied
785  outputStates.velocity += velCorrection;
786 
787  // a constant position correction is applied
788  outputStates.position += posCorrection;
789 
790  // push the updated data to the buffer
791  storedOutput[index] = outputStates;
792  }
793 
794  // update output state to corrected values
795  outputDataNew = storedOutput[storedIMU.get_youngest_index()];
796 
797  }
798 }
