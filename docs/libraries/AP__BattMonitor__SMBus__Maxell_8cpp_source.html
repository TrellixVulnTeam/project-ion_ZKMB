205 uint8_t AP_BattMonitor_SMBus_Maxell::get_PEC(const uint8_t i2c_addr, uint8_t cmd, bool reading, const uint8_t buff[], uint8_t len) const
206 {
207  // exit immediately if no data
208  if (len == 0) {
209  return 0;
210  }
211 
212  // prepare temp buffer for calcing crc
213  uint8_t tmp_buff[len+3];
214  tmp_buff[0] = i2c_addr << 1;
215  tmp_buff[1] = cmd;
216  tmp_buff[2] = tmp_buff[0] | (uint8_t)reading;
217  memcpy(&tmp_buff[3],buff,len);
218 
219  // initialise crc to zero
220  uint8_t crc = 0;
221  uint8_t shift_reg = 0;
222  bool do_invert;
223 
224  // for each byte in the stream
225  for (uint8_t i=0; i<sizeof(tmp_buff); i++) {
226  // load next data byte into the shift register
227  shift_reg = tmp_buff[i];
228  // for each bit in the current byte
229  for (uint8_t j=0; j<8; j++) {
230  do_invert = (crc ^ shift_reg) & 0x80;
231  crc <<= 1;
232  shift_reg <<= 1;
233  if(do_invert) {
234  crc ^= SMBUS_PEC_POLYNOME;
235  }
236  }
237  }
238 
239  // return result
240  return crc;
241 }
