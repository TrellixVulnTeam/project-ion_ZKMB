84 // convert a 0..range_max to a pwm
85 uint16_t SRV_Channel::pwm_from_range(int16_t scaled_value) const
86 {
87  if (servo_max <= servo_min || high_out == 0) {
88  return servo_min;
89  }
90  if (scaled_value >= high_out) {
91  scaled_value = high_out;
92  }
93  if (scaled_value < 0) {
94  scaled_value = 0;
95  }
96  if (reversed) {
97  scaled_value = high_out - scaled_value;
98  }
99  return servo_min + ((int32_t)scaled_value * (int32_t)(servo_max - servo_min)) / (int32_t)high_out;
100 }

102 // convert a -angle_max..angle_max to a pwm
103 uint16_t SRV_Channel::pwm_from_angle(int16_t scaled_value) const
104 {
105  if (reversed) {
106  scaled_value = -scaled_value;
107  }
108  if (scaled_value > 0) {
109  return servo_trim + ((int32_t)scaled_value * (int32_t)(servo_max - servo_trim)) / (int32_t)high_out;
110  } else {
111  return servo_trim - (-(int32_t)scaled_value * (int32_t)(servo_trim - servo_min)) / (int32_t)high_out;
112  }
113 }

115 void SRV_Channel::calc_pwm(int16_t output_scaled)
116 {
117  if (have_pwm_mask & (1U<<ch_num)) {
118  return;

151 /*
152  get normalised output from -1 to 1, assuming 0 at mid point of servo_min/servo_max
153  */
154 float SRV_Channel::get_output_norm(void)
155 {
156  uint16_t mid = (servo_max + servo_min) / 2;
157  float ret;
158  if (mid <= servo_min) {
159  return 0;
160  }
161  if (output_pwm < mid) {
162  ret = (float)(output_pwm - mid) / (float)(mid - servo_min);
163  } else if (output_pwm > mid) {
164  ret = (float)(output_pwm - mid) / (float)(servo_max - mid);
165  } else {
166  ret = 0;
167  }
168  if (get_reversed()) {
169  ret = -ret;
170  }
171  return ret;
172 }
