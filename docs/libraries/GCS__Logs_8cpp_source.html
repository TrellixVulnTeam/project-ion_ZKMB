85 {

88 
89  _log_listing = false;

91  _log_sending = false;
92

94  if (packet.id > num_logs || packet.id < 1) {

145 void GCS_MAVLINK::handle_log_send(DataFlash_Class &dataflash)
146 {
147  if (_log_listing) {
148  handle_log_send_listing(dataflash);
149  }
150  if (!_log_sending) {
151  return;
152  }
153 
154 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
155  // assume USB speeds in SITL for the purposes of log download
156  const uint8_t num_sends = 40;
157 #else
158  uint8_t num_sends = 1;
159  if (chan == MAVLINK_COMM_0 && hal.gpio->usb_connected()) {
160  // when on USB we can send a lot more data
161  num_sends = 250;
162  } else if (have_flow_control()) {
163  #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
164  num_sends = 80;
165  #else
166  num_sends = 10;
167  #endif
168  }
169 #endif
170 
171  for (uint8_t i=0; i<num_sends; i++) {
172  if (_log_sending) {
173  if (!handle_log_send_data(dataflash)) break;
174  }
175  }

181 void GCS_MAVLINK::handle_log_send_listing(DataFlash_Class &dataflash)
182 {
183  if (!HAVE_PAYLOAD_SPACE(chan, LOG_ENTRY)) {
184  // no space
185  return;
186  }
187  if (AP_HAL::millis() - last_heartbeat_time > 3000) {
188  // give a heartbeat a chance
189  return;
190  }
191 
192  uint32_t size, time_utc;
193  if (_log_next_list_entry == 0) {
194  size = 0;
195  time_utc = 0;
196  } else {
197  dataflash.get_log_info(_log_next_list_entry, size, time_utc);
198  }
199  mavlink_msg_log_entry_send(chan, _log_next_list_entry, _log_num_logs, _log_last_list_entry, time_utc, size);
200  if (_log_next_list_entry == _log_last_list_entry) {
201  _log_listing = false;
202  } else {
203  _log_next_list_entry++;
204  }
205 }

210 bool GCS_MAVLINK::handle_log_send_data(DataFlash_Class &dataflash)
211 {
212  if (!HAVE_PAYLOAD_SPACE(chan, LOG_DATA)) {
213  // no space
214  return false;
215  }
216  if (AP_HAL::millis() - last_heartbeat_time > 3000) {
217  // give a heartbeat a chance
218  return false;
219  }
220 
221  int16_t ret = 0;
222  uint32_t len = _log_data_remaining;
223  mavlink_log_data_t packet;
224 
225  if (len > 90) {
226  len = 90;
227  }
228  ret = dataflash.get_log_data(_log_num_data, _log_data_page, _log_data_offset, len, packet.data);
229  if (ret < 0) {
230  // report as EOF on error
231  ret = 0;
232  }
233  if (ret < 90) {
234  memset(&packet.data[ret], 0, 90-ret);
235  }
236 
237  packet.ofs = _log_data_offset;
238  packet.id = _log_num_data;
239  packet.count = ret;
240  _mav_finalize_message_chan_send(chan, MAVLINK_MSG_ID_LOG_DATA, (const char *)&packet,
241  MAVLINK_MSG_ID_LOG_DATA_MIN_LEN,
242  MAVLINK_MSG_ID_LOG_DATA_LEN,
243  MAVLINK_MSG_ID_LOG_DATA_CRC);
244 
245  _log_data_offset += len;
246  _log_data_remaining -= len;
247  if (ret < 90 || _log_data_remaining == 0) {
248  _log_sending = false;
249  }
250  return true;
251 }
