98 void RCOutput_QFLIGHT::timer_update(void)
99 {
100  /*
101  we defer the open to the time to ensure all RPC calls are made
102  from the same thread
103  */
104  if (fd == -1 && device != nullptr) {
105  int ret = qflight_UART_open(device, &fd);
106  printf("Opened ESC UART %s ret=%d fd=%d\n",
107  device, ret, (int)fd);
108  if (fd != -1) {
109  qflight_UART_set_baudrate(fd, baudrate);
110  }
111  }
112  if (!need_write || fd == -1) {
113  return;
114  }
115  /*
116  this implements the PWM over UART prototocol.
117  */
118  struct PACKED {
119  uint8_t magic = 0xF7;
120  uint16_t period[channel_count];
121  uint16_t crc;
122  } frame;
123  memcpy(frame.period, period, sizeof(period));
124  frame.crc = crc_calculate((uint8_t*)frame.period, channel_count*2);
125  int32_t nwritten;
126  qflight_UART_write(fd, (uint8_t *)&frame, sizeof(frame), &nwritten);
127  need_write = false;
128 
129  check_rc_in();
130 }

132 /*
133  we accept RC input from the UART and treat it as RC overrides. This
134  is an lazy way to allow an RCOutput driver to do RCInput. See the
135  RC_UART example for the other end of this protocol
136  */
137 void RCOutput_QFLIGHT::check_rc_in(void)
138 {
139  const uint8_t magic = 0xf6;
140  while (nrcin_bytes != sizeof(rcu.bytes)) {
141  int32_t nread;
142  if (qflight_UART_read(fd, rcu.bytes, sizeof(rcu.bytes)-nrcin_bytes, &nread) != 0 || nread <= 0) {
143  return;
144  }
145  nrcin_bytes += nread;
146  if (rcu.rcin.magic != magic) {
147  for (uint8_t i=1; i<nrcin_bytes; i++) {
148  if (rcu.bytes[i] == magic) {
149  memmove(&rcu.bytes[0], &rcu.bytes[i], nrcin_bytes-i);
150  nrcin_bytes = nrcin_bytes - i;
151  return;
152  }
153  }
154  nrcin_bytes = 0;
155  return;
156  }
157  }
158  if (nrcin_bytes == sizeof(rcu.bytes)) {
159  if (rcu.rcin.magic == 0xf6 &&
160  crc_calculate((uint8_t*)rcu.rcin.rcin, sizeof(rcu.rcin.rcin)) == rcu.rcin.crc) {
161  bool have_data = false;
162  for (uint8_t i=0; i<8; i++) {
163  if (rcu.rcin.rcin[i] != 0) {
164  have_data = true;
165  break;
166  }
167  }
168  if (have_data) {
169  hal.rcin->set_overrides((int16_t*)rcu.rcin.rcin, 8);
170  }
171  }
172  nrcin_bytes = 0;
173  }
174 }
