163 // initialisation
164 void AP_Notify::init(bool enable_external_leds)
165 {
166  // clear all flags and events
167  memset(&AP_Notify::flags, 0, sizeof(AP_Notify::flags));
168  memset(&AP_Notify::events, 0, sizeof(AP_Notify::events));
169 
170  // clear flight mode string and text buffer
171  memset(_flight_mode_str, 0, sizeof(_flight_mode_str));
172  memset(_send_text, 0, sizeof(_send_text));
173  _send_text_updated_millis = 0;
174 
175  AP_Notify::flags.external_leds = enable_external_leds;
176 
177  for (uint8_t i = 0; i < CONFIG_NOTIFY_DEVICES_COUNT; i++) {
178  _devices[i]->pNotify = this;
179  _devices[i]->init();
180  }
181 }
