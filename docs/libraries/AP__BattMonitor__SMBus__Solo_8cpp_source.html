177 {
178  // exit immediately if no data
179  if (len <= 0) {
180  return 0;
181  }
182

184  uint8_t tmp_buff[len+3];
185  tmp_buff[0] = i2c_addr << 1;
186  tmp_buff[1] = cmd;
187  tmp_buff[2] = tmp_buff[0] | (uint8_t)reading;
188  memcpy(&tmp_buff[3],buff,len);
189 
190  // initialise crc to zero
191  uint8_t crc = 0;
192  uint8_t shift_reg = 0;
193  bool do_invert;
194 
195  // for each byte in the stream
196  for (uint8_t i=0; i<sizeof(tmp_buff); i++) {
197  // load next data byte into the shift register
198  shift_reg = tmp_buff[i];
199  // for each bit in the current byte
200  for (uint8_t j=0; j<8; j++) {
201  do_invert = (crc ^ shift_reg) & 0x80;
202  crc <<= 1;
203  shift_reg <<= 1;
204  if(do_invert) {

206  }
207  }
208  }
