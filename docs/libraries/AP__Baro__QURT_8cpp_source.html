53 void AP_Baro_QURT::timer(void)
54 {
55  if (handle == 0) {
56  int ret = bmp280_open("/dev/i2c-2", &handle);
57  if (ret != 0) {
58  AP_HAL::panic("unable to open QURT baro");
59  return;
60  }
61  last_timer_ms = AP_HAL::millis();
62  return;
63  }
64  if (AP_HAL::millis() - last_timer_ms < 10) {
65  return;
66  }
67  if (lock.take(1)) {
68  last_timer_ms = AP_HAL::millis();
69  struct bmp280_sensor_data data;
70  int ret = bmp280_get_sensor_data(handle, &data, false);
71  if (ret != 0 || data.sensor_read_counter == last_counter) {
72  lock.give();
73  return;
74  }
75  last_counter = data.sensor_read_counter;
76  temp_sum += data.temperature_in_c;
77  press_sum += data.pressure_in_pa;
78  count++;
79  lock.give();
80  }
81 }
