444  /*
445  buffer for re-assembling RTCM data for GPS injection.
446  The 8 bit flags field in GPS_RTCM_DATA is interpreted as:
447  1 bit for "is fragmented"
448  2 bits for fragment number
449  5 bits for sequence number
450 
451  The rtcm_buffer is allocated on first use. Once a block of data
452  is successfully reassembled it is injected into all active GPS
453  backends. This assumes we don't want more than 4*180=720 bytes
454  in a RTCM data block
455  */
456  struct rtcm_buffer {
457  uint8_t fragments_received:4;
458  uint8_t sequence:5;
459  uint8_t fragment_count;
460  uint16_t total_length;
461  uint8_t buffer[MAVLINK_MSG_GPS_RTCM_DATA_FIELD_DATA_LEN*4];
462  } *rtcm_buffer;
