38 bool ToshibaLED_I2C::hw_init()
39 {
40  // first look for led on external bus
41  _dev = std::move(hal.i2c_mgr->get_device(TOSHIBA_LED_I2C_BUS_EXTERNAL, TOSHIBA_LED_I2C_ADDR));
42  if (!_dev || !_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
43  return false;
44  }
45 
46  // enable the led
47  bool ret = _dev->write_register(TOSHIBA_LED_ENABLE, 0x03);
48 
49  // on failure try the internal bus
50  if (!ret) {
51  // give back external bus semaphore
52  _dev->get_semaphore()->give();
53  // get internal I2C bus driver
54  _dev = std::move(hal.i2c_mgr->get_device(TOSHIBA_LED_I2C_BUS_INTERNAL, TOSHIBA_LED_I2C_ADDR));
55  if (!_dev || !_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
56  return false;
57  }
58  ret = _dev->write_register(TOSHIBA_LED_ENABLE, 0x03);
59  }
60 
61  // update the red, green and blue values to zero
62  uint8_t val[4] = { TOSHIBA_LED_PWM0, _led_off, _led_off, _led_off };
63  ret &= _dev->transfer(val, sizeof(val), nullptr, 0);
64 
65  // give back i2c semaphore
66  _dev->get_semaphore()->give();
67 
68  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&ToshibaLED_I2C::_timer, void));
69 
70  return ret;
71 }
