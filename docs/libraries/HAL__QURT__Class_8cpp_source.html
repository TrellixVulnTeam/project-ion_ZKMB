73 void HAL_QURT::run(int argc, char* const argv[], Callbacks* callbacks) const
74 {
75  assert(callbacks);
76 
77  int opt;
78  const struct GetOptLong::option options[] = {
79  {"uartB", true, 0, 'B'},
80  {"uartC", true, 0, 'C'},
81  {"uartD", true, 0, 'D'},
82  {"uartE", true, 0, 'E'},
83  {"dsm", true, 0, 'S'},
84  {"ESC", true, 0, 'e'},
85  {0, false, 0, 0}
86  };
87 
88  GetOptLong gopt(argc, argv, "B:C:D:E:e:S",
89  options);
90 
91  /*
92  parse command line options
93  */
94  while ((opt = gopt.getoption()) != -1) {
95  switch (opt) {
96  case 'B':
97  uartBDriver.set_device_path(gopt.optarg);
98  break;
99  case 'C':
100  uartCDriver.set_device_path(gopt.optarg);
101  break;
102  case 'D':
103  uartDDriver.set_device_path(gopt.optarg);
104  break;
105  case 'E':
106  uartEDriver.set_device_path(gopt.optarg);
107  break;
108  case 'e':
109  rcoutDriver.set_device_path(gopt.optarg);
110  break;
111  case 'S':
112  rcinDriver.set_device_path(gopt.optarg);
113  break;
114  default:
115  printf("Unknown option '%c'\n", (char)opt);
116  exit(1);
117  }
118  }
119 
120  /* initialize all drivers and private members here.
121  * up to the programmer to do this in the correct order.
122  * Scheduler should likely come first. */
123  scheduler->init();
124  schedulerInstance.hal_initialized();
125  uartA->begin(115200);
126  rcinDriver.init();
127  callbacks->setup();
128  scheduler->system_initialized();
129 
130  for (;;) {
131  callbacks->loop();
132  }
133 }
