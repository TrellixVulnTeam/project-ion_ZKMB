115 void AP_Compass_AK09916::timer()
116 {
117  struct PACKED {
118  int16_t magx;
119  int16_t magy;
120  int16_t magz;
121  uint8_t tmps;
122  uint8_t status2;
123  } data;
124  const float to_utesla = 4912.0f / 32752.0f;
125  const float utesla_to_milliGauss = 10;
126  const float range_scale = to_utesla * utesla_to_milliGauss;
127  Vector3f field;
128 
129  // check data ready
130  uint8_t st1;
131  if (!dev->read_registers(REG_ST1, &st1, 1) || !(st1 & 1)) {
132  goto check_registers;
133  }
134 
135  if (!dev->read_registers(REG_HXL, (uint8_t *)&data, sizeof(data))) {
136  goto check_registers;
137  }
138 
139  field(data.magx * range_scale, data.magy * range_scale, data.magz * range_scale);
140 
141  /* rotate raw_field from sensor frame to body frame */
142  rotate_field(field, compass_instance);
143 
144  /* publish raw_field (uncorrected point sample) for calibration use */
145  publish_raw_field(field, AP_HAL::micros(), compass_instance);
146 
147  /* correct raw_field for known errors */
148  correct_field(field, compass_instance);
149 
150  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
151  accum += field;
152  accum_count++;
153  _sem->give();
154  }
155 
156 check_registers:
157  dev->check_next_register();
158 }
