94 bool ToneAlarm_PX4::init()
95 {
96  // open the tone alarm device
97  _tonealarm_fd = open(TONEALARM0_DEVICE_PATH, O_WRONLY);
98  if (_tonealarm_fd == -1) {
99  hal.console->printf("ToneAlarm_PX4: Unable to open " TONEALARM0_DEVICE_PATH);
100  return false;
101  }
102 
103  // set initial boot states. This prevents us issuing a arming
104  // warning in plane and rover on every boot
105  flags.armed = AP_Notify::flags.armed;
106  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
107  flags.pre_arm_check = 1;
108  _cont_tone_playing = -1;
109  return true;
110 }

112 // play_tune - play one of the pre-defined tunes
113 void ToneAlarm_PX4::play_tone(const uint8_t tone_index)
114 {
115  uint32_t tnow_ms = AP_HAL::millis();
116  const Tone &tone_requested = _tones[tone_index];
117 
118  if(tone_requested.continuous) {
119  _cont_tone_playing = tone_index;
120  }
121 
122  _tone_playing = tone_index;
123  _tone_beginning_ms = tnow_ms;
124 
125  play_string(tone_requested.str);
126 }

140 void ToneAlarm_PX4::check_cont_tone() {
141  uint32_t tnow_ms = AP_HAL::millis();
142  // if we are supposed to be playing a continuous tone,
143  // and it was interrupted, and the interrupting tone has timed out,
144  // resume the continuous tone
145 
146  if (_cont_tone_playing != -1 && _tone_playing != _cont_tone_playing && tnow_ms-_tone_beginning_ms > AP_NOTIFY_PX4_MAX_TONE_LENGTH_MS) {
147  play_tone(_cont_tone_playing);
148  }
149 }

151 // update - updates led according to timed_updated. Should be called at 50Hz
152 void ToneAlarm_PX4::update()
153 {
154  // exit immediately if we haven't initialised successfully
155  if (_tonealarm_fd == -1) {
156  return;
157  }
158 
159  // exit if buzzer is not enabled
160  if (pNotify->buzzer_enabled() == false) {
161  return;
162  }
163 
164  check_cont_tone();
165 
166  if (AP_Notify::flags.compass_cal_running != flags.compass_cal_running) {
167  if(AP_Notify::flags.compass_cal_running) {
168  play_tone(AP_NOTIFY_PX4_TONE_QUIET_COMPASS_CALIBRATING_CTS);
169  play_tone(AP_NOTIFY_PX4_TONE_QUIET_POS_FEEDBACK);
170  } else {
171  if(_cont_tone_playing == AP_NOTIFY_PX4_TONE_QUIET_COMPASS_CALIBRATING_CTS) {
172  stop_cont_tone();
173  }
174  }
175  }
176  flags.compass_cal_running = AP_Notify::flags.compass_cal_running;
177 
178  if (AP_Notify::events.compass_cal_canceled) {
179  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEU_FEEDBACK);
180  return;
181  }
182 
183  if (AP_Notify::events.initiated_compass_cal) {
184  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEU_FEEDBACK);
185  return;
186  }
187 
188  if (AP_Notify::events.compass_cal_saved) {
189  play_tone(AP_NOTIFY_PX4_TONE_QUIET_READY_OR_FINISHED);
190  return;
191  }
192 
193  if (AP_Notify::events.compass_cal_failed) {
194  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEG_FEEDBACK);
195  return;
196  }
197 
198  // don't play other tones if compass cal is running
199  if (AP_Notify::flags.compass_cal_running) {
200  return;
201  }
202 
203  // notify the user when autotune or mission completes
204  if (AP_Notify::flags.armed && (AP_Notify::events.autotune_complete || AP_Notify::events.mission_complete)) {
205  play_tone(AP_NOTIFY_PX4_TONE_LOUD_READY_OR_FINISHED);
206  }
207 
208  //notify the user when autotune fails
209  if (AP_Notify::flags.armed && (AP_Notify::events.autotune_failed)) {
210  play_tone(AP_NOTIFY_PX4_TONE_LOUD_NEG_FEEDBACK);
211  }
212 
213  // notify the user when a waypoint completes
214  if (AP_Notify::events.waypoint_complete) {
215  play_tone(AP_NOTIFY_PX4_TONE_LOUD_WP_COMPLETE);
216  }
217 
218  // notify the user when their mode change was successful
219  if (AP_Notify::events.user_mode_change) {
220  if (AP_Notify::flags.armed) {
221  play_tone(AP_NOTIFY_PX4_TONE_LOUD_NEU_FEEDBACK);
222  } else {
223  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEU_FEEDBACK);
224  }
225  }
226 
227  // notify the user when their mode change failed
228  if (AP_Notify::events.user_mode_change_failed) {
229  if (AP_Notify::flags.armed) {
230  play_tone(AP_NOTIFY_PX4_TONE_LOUD_NEG_FEEDBACK);
231  } else {
232  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEG_FEEDBACK);
233  }
234  }
235 
236  // failsafe initiated mode change
237  if(AP_Notify::events.failsafe_mode_change) {
238  play_tone(AP_NOTIFY_PX4_TONE_LOUD_ATTENTION_NEEDED);
239  }
240 
241  // notify the user when arming fails
242  if (AP_Notify::events.arming_failed) {
243  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEG_FEEDBACK);
244  }
245 
246  // notify the user when RC contact is lost
247  if (flags.failsafe_radio != AP_Notify::flags.failsafe_radio) {
248  flags.failsafe_radio = AP_Notify::flags.failsafe_radio;
249  if (flags.failsafe_radio) {
250  // armed case handled by events.failsafe_mode_change
251  if (!AP_Notify::flags.armed) {
252  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEG_FEEDBACK);
253  }
254  } else {
255  if (AP_Notify::flags.armed) {
256  play_tone(AP_NOTIFY_PX4_TONE_LOUD_POS_FEEDBACK);
257  } else {
258  play_tone(AP_NOTIFY_PX4_TONE_QUIET_POS_FEEDBACK);
259  }
260  }
261  }
262 
263  // notify the user when pre_arm checks are passing
264  if (flags.pre_arm_check != AP_Notify::flags.pre_arm_check) {
265  flags.pre_arm_check = AP_Notify::flags.pre_arm_check;
266  if (flags.pre_arm_check) {
267  play_tone(AP_NOTIFY_PX4_TONE_QUIET_READY_OR_FINISHED);
268  }
269  }
270 
271  // check if arming status has changed
272  if (flags.armed != AP_Notify::flags.armed) {
273  flags.armed = AP_Notify::flags.armed;
274  if (flags.armed) {
275  // arming tune
276  play_tone(AP_NOTIFY_PX4_TONE_QUIET_ARMING_WARNING);
277  }else{
278  // disarming tune
279  play_tone(AP_NOTIFY_PX4_TONE_QUIET_NEU_FEEDBACK);
280  if (!flags.leak_detected) {
281  stop_cont_tone();
282  }
283  }
284  }
285 
286  // check if battery status has changed
287  if (flags.failsafe_battery != AP_Notify::flags.failsafe_battery) {
288  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
289  if (flags.failsafe_battery) {
290  // battery warning tune
291  play_tone(AP_NOTIFY_PX4_TONE_LOUD_BATTERY_ALERT_CTS);
292  }
293  }
294 
295  // check parachute release
296  if (flags.parachute_release != AP_Notify::flags.parachute_release) {
297  flags.parachute_release = AP_Notify::flags.parachute_release;
298  if (flags.parachute_release) {
299  // parachute release warning tune
300  play_tone(AP_NOTIFY_PX4_TONE_LOUD_ATTENTION_NEEDED);
301  }
302  }
303 
304  // lost vehicle tone
305  if (flags.vehicle_lost != AP_Notify::flags.vehicle_lost) {
306  flags.vehicle_lost = AP_Notify::flags.vehicle_lost;
307  if (flags.vehicle_lost) {
308  play_tone(AP_NOTIFY_PX4_TONE_LOUD_VEHICLE_LOST_CTS);
309  } else {
310  stop_cont_tone();
311  }
312  }
313 
314  // waiting to be thrown vehicle tone
315  if (flags.waiting_for_throw != AP_Notify::flags.waiting_for_throw) {
316  flags.waiting_for_throw = AP_Notify::flags.waiting_for_throw;
317  if (flags.waiting_for_throw) {
318  play_tone(AP_NOTIFY_PX4_TONE_WAITING_FOR_THROW);
319  } else {
320  stop_cont_tone();
321  }
322  }
323 
324  if (flags.leak_detected != AP_Notify::flags.leak_detected) {
325  flags.leak_detected = AP_Notify::flags.leak_detected;
326  if (flags.leak_detected) {
327  play_tone(AP_NOTIFY_PX4_TONE_LEAK_DETECTED);
328  } else {
329  stop_cont_tone();
330  }
331  }
332 
333  if (AP_Notify::events.tune_started) {
334  play_tone(AP_NOTIFY_PX4_TONE_TUNING_START);
335  AP_Notify::events.tune_started = 0;
336  }
337  if (AP_Notify::events.tune_next) {
338  // signify which parameter in the set is starting
339  play_tone(AP_NOTIFY_PX4_TONE_LOUD_1 + (AP_Notify::events.tune_next-1));
340  AP_Notify::events.tune_next = 0;
341  }
342  if (AP_Notify::events.tune_save) {
343  play_tone(AP_NOTIFY_PX4_TONE_TUNING_SAVE);
344  AP_Notify::events.tune_save = 0;
345  }
346  if (AP_Notify::events.tune_error) {
347  play_tone(AP_NOTIFY_PX4_TONE_TUNING_ERROR);
348  AP_Notify::events.tune_error = 0;
349  }
350 }
