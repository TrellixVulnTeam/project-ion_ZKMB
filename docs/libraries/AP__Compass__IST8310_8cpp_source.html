104 bool AP_Compass_IST8310::init()
105 {
106  uint8_t reset_count = 0;
107 
108  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
109  return false;
110  }
111 
112  // high retries for init
113  _dev->set_retries(10);
114 
115  uint8_t whoami;
116  if (!_dev->read_registers(WAI_REG, &whoami, 1) ||
117  whoami != DEVICE_ID) {
118  // not an IST8310
119  goto fail;
120  }
121 
122  for (; reset_count < 5; reset_count++) {
123  if (!_dev->write_register(CNTL2_REG, CNTL2_VAL_SRST)) {
124  hal.scheduler->delay(10);
125  continue;
126  }
127 
128  hal.scheduler->delay(10);
129 
130  uint8_t cntl2 = 0xFF;
131  if (_dev->read_registers(CNTL2_REG, &cntl2, 1) &&
132  (cntl2 & 0x01) == 0) {
133  break;
134  }
135  }
136 
137  if (reset_count == 5) {
138  fprintf(stderr, "IST8310: failed to reset device\n");
139  goto fail;
140  }
141 
142  if (!_dev->write_register(AVGCNTL_REG, AVGCNTL_VAL_Y_16 | AVGCNTL_VAL_XZ_16) ||
143  !_dev->write_register(PDCNTL_REG, PDCNTL_VAL_PULSE_DURATION_NORMAL)) {
144  fprintf(stderr, "IST8310: found device but could not set it up\n");
145  goto fail;
146  }
147 
148  // lower retries for run
149  _dev->set_retries(3);
150 
151  // start state machine: request a sample
152  start_conversion();
153 
154  _dev->get_semaphore()->give();
155 
156  _instance = register_compass();
157 
158  printf("%s found on bus %u id %u address 0x%02x\n", name,
159  _dev->bus_num(), _dev->get_bus_id(), _dev->get_bus_address());
160 
161  set_rotation(_instance, _rotation);
162 
163  _dev->set_device_type(DEVTYPE_IST8310);
164  set_dev_id(_instance, _dev->get_bus_id());
165 
166  _periodic_handle = _dev->register_periodic_callback(SAMPLING_PERIOD_USEC,
167  FUNCTOR_BIND_MEMBER(&AP_Compass_IST8310::timer, void));
168 
169  _perf_xfer_err = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "IST8310_xfer_err");
170  _perf_bad_data = hal.util->perf_alloc(AP_HAL::Util::PC_COUNT, "IST8310_bad_data");
171 
172  return true;
173 
174 fail:
175  _dev->get_semaphore()->give();
176  return false;
177 }

187 void AP_Compass_IST8310::timer()
188 {
189  if (_ignore_next_sample) {
190  _ignore_next_sample = false;
191  start_conversion();
192  return;
193  }
194 
195  struct PACKED {
196  le16_t rx;
197  le16_t ry;
198  le16_t rz;
199  } buffer;
200 
201  bool ret = _dev->read_registers(OUTPUT_X_L_REG, (uint8_t *) &buffer, sizeof(buffer));
202  if (!ret) {
203  hal.util->perf_count(_perf_xfer_err);
204  return;
205  }
206 
207  uint32_t now = AP_HAL::micros();
208 
209  start_conversion();
210 
211  /* same period, but start counting from now */
212  _dev->adjust_periodic_callback(_periodic_handle, SAMPLING_PERIOD_USEC);
213 
214  auto x = static_cast<int16_t>(le16toh(buffer.rx));
215  auto y = static_cast<int16_t>(le16toh(buffer.ry));
216  auto z = static_cast<int16_t>(le16toh(buffer.rz));
217 
218  /*
219  * Check if value makes sense according to the FSR and Resolution of
220  * this sensor, discarding outliers
221  */
222  if (x > IST8310_MAX_VAL_XY || x < IST8310_MIN_VAL_XY ||
223  y > IST8310_MAX_VAL_XY || y < IST8310_MIN_VAL_XY ||
224  z > IST8310_MAX_VAL_Z || z < IST8310_MIN_VAL_Z) {
225  hal.util->perf_count(_perf_bad_data);
226  return;
227  }
228 
229  // flip Z to conform to right-hand rule convention
230  z = -z;
231 
232  /* Resolution: 0.3 µT/LSB - already convert to milligauss */
233  Vector3f field = Vector3f{x * 3.0f, y * 3.0f, z * 3.0f};
234 
235  /* rotate raw_field from sensor frame to body frame */
236  rotate_field(field, _instance);
237 
238  /* publish raw_field (uncorrected point sample) for calibration use */
239  publish_raw_field(field, now, _instance);
240 
241  /* correct raw_field for known errors */
242  correct_field(field, _instance);
243 
244  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
245  _accum += field;
246  _accum_count++;
247  _sem->give();
248  }
249 }
