16 /*
17  IMU driver backend class. Each supported gyro/accel sensor type
18  needs to have an object derived from this class.
19 
20  Note that drivers can implement just gyros or just accels, and can
21  also provide multiple gyro/accel instances.
22  */

40  // we declare a virtual destructor so that drivers can

44  /*
45  * Update the sensor data. Called by the frontend to transfer
46  * accumulated sensor readings to the frontend structure via the
47  * _publish_gyro() and _publish_accel() functions
48  */
49  virtual bool update() = 0;
50 
51  /*
52  * optional function to accumulate more samples. This is needed for drivers that don't use a timer to gather samples
53  */
54  virtual void accumulate() {}
55 
56  /*
57  * Configure and start all sensors. The empty implementation allows
58  * subclasses to already start the sensors when it's detected
59  */
60  virtual void start() { }
61 
62  /*
63  * Return an AuxiliaryBus if backend has another bus it is able to export
64  */
65  virtual AuxiliaryBus *get_auxiliary_bus() { return nullptr; }
66 
67  /*
68  * Return the unique identifier for this backend: it's the same for
69  * several sensors if the backend registers more gyros/accels
70  */
71  int16_t get_id() const { return _id; }
72 
73  /*
74  device driver IDs. These are used to fill in the devtype field
75  of the device ID, which shows up as INS*ID* parameters to
76  users. The values are chosen for compatibility with existing PX4
77  drivers.
78  If a change is made to a driver that would make existing
79  calibration values invalid then this number must be changed.
80  */

189

193  }

195  // note that each backend is also expected to have a static detect()
