36 AP_Baro_BMP085::AP_Baro_BMP085(AP_Baro &baro, AP_HAL::OwnPtr<AP_HAL::I2CDevice> dev)
37  : AP_Baro_Backend(baro)
38  , _dev(std::move(dev))
39 {
40  uint8_t buff[22];
41 
42  // get pointer to i2c bus semaphore
43  AP_HAL::Semaphore *sem = _dev->get_semaphore();
44 
45  // take i2c bus sempahore
46  if (!sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
47  AP_HAL::panic("BMP085: unable to get semaphore");
48  }
49 
50  if (BMP085_EOC >= 0) {
51  _eoc = hal.gpio->channel(BMP085_EOC);
52  _eoc->mode(HAL_GPIO_INPUT);
53  }
54 
55  // We read the calibration data registers
56  if (!_dev->read_registers(0xAA, buff, sizeof(buff))) {
57  AP_HAL::panic("BMP085: bad calibration registers");
58  }
59 
60  ac1 = ((int16_t)buff[0] << 8) | buff[1];
61  ac2 = ((int16_t)buff[2] << 8) | buff[3];
62  ac3 = ((int16_t)buff[4] << 8) | buff[5];
63  ac4 = ((int16_t)buff[6] << 8) | buff[7];
64  ac5 = ((int16_t)buff[8] << 8) | buff[9];
65  ac6 = ((int16_t)buff[10] << 8) | buff[11];
66  b1 = ((int16_t)buff[12] << 8) | buff[13];
67  b2 = ((int16_t)buff[14] << 8) | buff[15];
68  mb = ((int16_t)buff[16] << 8) | buff[17];
69  mc = ((int16_t)buff[18] << 8) | buff[19];
70  md = ((int16_t)buff[20] << 8) | buff[21];
71 
72  _last_press_read_command_time = 0;
73  _last_temp_read_command_time = 0;
74 
75  _instance = _frontend.register_sensor();
76 
77  // Send a command to read temperature
78  _cmd_read_temp();
79 
80  _state = 0;
81 
82  sem->give();
83 
84  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&AP_Baro_BMP085::_timer, void));
85 }

87 /*
88  This is a state machine. Acumulate a new sensor reading.
89  */
90 void AP_Baro_BMP085::_timer(void)
91 {
92  if (!_data_ready()) {
93  return;
94  }
95 
96  if (_state == 0) {
97  _read_temp();
98  } else if (_read_pressure()) {
99  _calculate();
100  }
101 
102  _state++;
103  if (_state == 25) {
104  _state = 0;
105  _cmd_read_temp();
106  } else {
107  _cmd_read_pressure();
108  }
109 }

139 {
140  uint8_t buf[3];
141 
142  if (!_dev->read_registers(0xF6, buf, sizeof(buf))) {
143  _retry_time = AP_HAL::millis() + 1000;
144  _dev->set_speed(AP_HAL::Device::SPEED_LOW);

177 // _calculate Temperature and Pressure in real units.
178 void AP_Baro_BMP085::_calculate()
179 {
180  int32_t x1, x2, x3, b3, b5, b6, p;
181  uint32_t b4, b7;
182  int32_t tmp;
183 
184  // See Datasheet page 13 for this formulas
185  // Based also on Jee Labs BMP085 example code. Thanks for share.
186  // Temperature calculations
187  x1 = ((int32_t)_raw_temp - ac6) * ac5 >> 15;
188  x2 = ((int32_t) mc << 11) / (x1 + md);
189  b5 = x1 + x2;
190  _temp = (b5 + 8) >> 4;
191 
192  // Pressure calculations
193  b6 = b5 - 4000;
194  x1 = (b2 * (b6 * b6 >> 12)) >> 11;
195  x2 = ac2 * b6 >> 11;
196  x3 = x1 + x2;
197  //b3 = (((int32_t) ac1 * 4 + x3)<<OVERSAMPLING + 2) >> 2; // BAD
198  //b3 = ((int32_t) ac1 * 4 + x3 + 2) >> 2; //OK for OVERSAMPLING=0
199  tmp = ac1;
200  tmp = (tmp*4 + x3)<<OVERSAMPLING;
201  b3 = (tmp+2)/4;
202  x1 = ac3 * b6 >> 13;
203  x2 = (b1 * (b6 * b6 >> 12)) >> 16;
204  x3 = ((x1 + x2) + 2) >> 2;
205  b4 = (ac4 * (uint32_t) (x3 + 32768)) >> 15;
206  b7 = ((uint32_t) _raw_pressure - b3) * (50000 >> OVERSAMPLING);
207  p = b7 < 0x80000000 ? (b7 * 2) / b4 : (b7 / b4) * 2;
208 
209  x1 = (p >> 8) * (p >> 8);
210  x1 = (x1 * 3038) >> 16;
211  x2 = (-7357 * p) >> 16;
212  p += ((x1 + x2 + 3791) >> 4);
213 
214  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
215  _pressure_filter.apply(p);
216  _has_sample = true;
217  _sem->give();
218  }
219 }

221 bool AP_Baro_BMP085::_data_ready()
222 {
223  if (BMP085_EOC >= 0) {
224  return _eoc->read();
225  }
226 
227  // No EOC pin: use time from last read instead.
228  if (_state == 0) {
229  return AP_HAL::millis() > _last_temp_read_command_time + 5;
230  }
231 
232  uint32_t conversion_time_msec;
233 
234  switch (OVERSAMPLING) {
235  case BMP085_OVERSAMPLING_ULTRALOWPOWER:
236  conversion_time_msec = 5;
237  break;
238  case BMP085_OVERSAMPLING_STANDARD:
239  conversion_time_msec = 8;
240  break;
241  case BMP085_OVERSAMPLING_HIGHRES:
242  conversion_time_msec = 14;
243  break;
244  case BMP085_OVERSAMPLING_ULTRAHIGHRES:
245  conversion_time_msec = 26;
246  default:
247  break;
248  }
249 
250  return AP_HAL::millis() > _last_press_read_command_time + conversion_time_msec;
251 }
