85 {
86  if (devname == nullptr) {
87  return;
88  }
89  if (rxS < 4096) {

92  if (txS < 4096) {

94  }
95  if (fd == -1) {

102  /*
103  allocate the read buffer
104  we allocate buffers before we successfully open the device as we
105  want to allocate in the early stages of boot, and cause minimum
106  thrashing of the heap once we are up. The ttyACM0 driver may not
107  connect for some time after boot
108  */
109  if (rxS != 0 && (readbuf == nullptr || rxS != readbuf->get_size())) {
110  initialised = false;
111  if (readbuf != nullptr) {
112  delete readbuf;
113  }
114  readbuf = new ByteBuffer(rxS);
115  }
116 
117  /*
118  allocate the write buffer
119  */
120  if (txS != 0 && (writebuf == nullptr || txS != writebuf->get_size())) {
121  initialised = false;
122  if (writebuf != nullptr) {
123  delete writebuf;
124  }
125  writebuf = new ByteBuffer(txS);
126  }

128  struct dspal_serial_ioctl_receive_data_callback callback;
129  callback.context = this;
130  callback.rx_data_callback_func_ptr = read_callback_trampoline;
131  int ret = ioctl(fd, SERIAL_IOCTL_SET_RECEIVE_DATA_CALLBACK, (void *)&callback);
132 
133  if (b != 0) {
134  for (uint8_t i=0; i<ARRAY_SIZE(baudrate_table); i++) {
135  if (b <= baudrate_table[i].baudrate) {
136  struct dspal_serial_ioctl_data_rate rate;
137  rate.bit_rate = baudrate_table[i].arg;
138  ret = ioctl(fd, SERIAL_IOCTL_SET_DATA_RATE, (void *)&rate);
139  break;
140  }
141  }
142  }

145  initialised = true;

243 size_t UARTDriver::write(const uint8_t *buffer, size_t size)
244 {
245  if (!initialised) {
246  return 0;
247  }
248  if (!nonblocking_writes) {
249  /*
250  use the per-byte delay loop in write() above for blocking writes
251  */
252  size_t ret = 0;
253  while (size--) {
254  if (write(*buffer++) != 1) break;
255  ret++;
256  }
257  return ret;
258  }
259 
260  if (!lock.take(0)) {
261  return 0;
262  }
263  size_t ret = writebuf->write(buffer, size);
264  lock.give();
265  return ret;

268 /*
269  push any pending bytes to/from the serial port. This is called at
270  1kHz in the timer thread
271  */
272 void UARTDriver::timer_tick(void)
273 {
274  uint16_t n;
275 
276  if (fd == -1 || !initialised || !lock.take_nonblocking()) {
277  return;
278  }
279 
280  in_timer = true;
281 
282  // write any pending bytes
283  n = writebuf->available();
284  if (n == 0) {
285  in_timer = false;
286  lock.give();
287  return;
288  }
289  if (n > 64) {
290  n = 64;
291  }
292  uint8_t buf[n];
293  writebuf->read(buf, n);
294  ::write(fd, buf, n);
295  lock.give();
296 
297  in_timer = false;
298 }
