38 void AP_MotorsTailsitter::output_to_motors()
39 {
40  if (!_flags.initialised_ok) {
41  return;
42  }
43  switch (_spool_mode) {
44  case SHUT_DOWN:
45  _aileron = 0;
46  _elevator = 0;
47  _rudder = 0;
48  _throttle = 0;
49  break;
50  case SPIN_WHEN_ARMED:
51  // sends output to motors when armed but not flying
52  _aileron = 0;
53  _elevator = 0;
54  _rudder = 0;
55  _throttle = constrain_float(_spin_up_ratio, 0.0f, 1.0f) * _spin_min;
56  break;
57  case SPOOL_UP:
58  case THROTTLE_UNLIMITED:
59  case SPOOL_DOWN:
60  break;
61  }
62  // outputs are setup here, and written to the HAL by the plane servos loop
63  SRV_Channels::set_output_scaled(SRV_Channel::k_aileron, _aileron*SERVO_OUTPUT_RANGE);
64  SRV_Channels::set_output_scaled(SRV_Channel::k_elevator, _elevator*SERVO_OUTPUT_RANGE);
65  SRV_Channels::set_output_scaled(SRV_Channel::k_rudder, _rudder*SERVO_OUTPUT_RANGE);
66  SRV_Channels::set_output_scaled(SRV_Channel::k_throttle, _throttle*THROTTLE_RANGE);
67 
68  // also support differential roll with twin motors
69  float throttle_left = constrain_float(_throttle + _rudder*0.5, 0, 1);
70  float throttle_right = constrain_float(_throttle - _rudder*0.5, 0, 1);
71  SRV_Channels::set_output_scaled(SRV_Channel::k_throttleLeft, throttle_left*THROTTLE_RANGE);
72  SRV_Channels::set_output_scaled(SRV_Channel::k_throttleRight, throttle_right*THROTTLE_RANGE);
73 
74 #if APM_BUILD_TYPE(APM_BUILD_ArduCopter)
75  SRV_Channels::calc_pwm();
76  SRV_Channels::output_ch_all();
77 #endif
78 }

81 void AP_MotorsTailsitter::output_armed_stabilizing()
82 {
83  _aileron = -_yaw_in;
84  _elevator = _pitch_in;
85  _rudder = _roll_in;
86  _throttle = get_throttle();
87 
88  // sanity check throttle is above zero and below current limited throttle
89  if (_throttle <= 0.0f) {
90  _throttle = 0.0f;
91  limit.throttle_lower = true;
92  }
93  if (_throttle >= _throttle_thrust_max) {
94  _throttle = _throttle_thrust_max;
95  limit.throttle_upper = true;
96  }
97 
98  _throttle = constrain_float(_throttle, 0.1, 1);
99 }
