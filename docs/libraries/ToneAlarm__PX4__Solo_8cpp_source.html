76 bool ToneAlarm_PX4_Solo::init()
77 {
78  // open the tone alarm device
79  _tonealarm_fd = open(TONEALARM0_DEVICE_PATH, O_WRONLY);
80  if (_tonealarm_fd == -1) {
81  hal.console->printf("ToneAlarm_PX4_Solo: Unable to open " TONEALARM0_DEVICE_PATH);
82  return false;
83  }
84 
85  // set initial boot states. This prevents us issuing a arming
86  // warning in plane and rover on every boot
87  flags.armed = AP_Notify::flags.armed;
88  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
89  flags.pre_arm_check = 1;
90  flags.gps_connected = 1;
91  _cont_tone_playing = -1;
92  _gps_disconnected_time = 0;
93  _init_time = AP_HAL::millis();
94  return true;
95 }

98 void ToneAlarm_PX4_Solo::play_tone(const uint8_t tone_index)
99 {
100  uint32_t tnow_ms = AP_HAL::millis();
101  const Tone &tone_requested = _tones[tone_index];
102 
103  if(tone_requested.continuous) {
104  _cont_tone_playing = tone_index;
105  }
106 
107  _tone_playing = tone_index;
108  _tone_beginning_ms = tnow_ms;
109 
110  play_string(tone_requested.str);
111 }

125 void ToneAlarm_PX4_Solo::check_cont_tone() {
126  uint32_t tnow_ms = AP_HAL::millis();
127  // if we are supposed to be playing a continuous tone,
128  // and it was interrupted, and the interrupting tone has timed out,
129  // resume the continuous tone
130 
131  if (_cont_tone_playing != -1 && _tone_playing != _cont_tone_playing && tnow_ms-_tone_beginning_ms > AP_NOTIFY_PX4_MAX_TONE_LENGTH_MS) {
132  play_tone(_cont_tone_playing);
133  }
134 }
