139 /*
140  return the center stick position expressed as a control_in value
141  used for thr_mid in copter
142  */
143 int16_t RC_Channel::get_control_mid() const
144 {

146  int16_t r_in = (radio_min.get() + radio_max.get())/2;
147 
148  if (reversed) {

153 
154  return (((int32_t)(high_in) * (int32_t)(r_in - radio_trim_low)) / (int32_t)(radio_max - radio_trim_low));
155  } else {

180 /*
181  return an "angle in centidegrees" (normally -4500 to 4500) from
182  the current radio_in value using the specified dead_zone
183  */
184 int16_t
185 RC_Channel::pwm_to_angle_dz_trim(uint16_t _dead_zone, uint16_t _trim)
186 {
187  int16_t radio_trim_high = _trim + _dead_zone;
188  int16_t radio_trim_low = _trim - _dead_zone;
189 
190  // prevent div by 0
191  if ((radio_trim_low - radio_min) == 0 || (radio_max - radio_trim_high) == 0)
192  return 0;
193 
194  int16_t reverse_mul = (reversed?-1:1);
195  if (radio_in > radio_trim_high) {
196  return reverse_mul * ((int32_t)high_in * (int32_t)(radio_in - radio_trim_high)) / (int32_t)(radio_max - radio_trim_high);
197  } else if (radio_in < radio_trim_low) {
198  return reverse_mul * ((int32_t)high_in * (int32_t)(radio_in - radio_trim_low)) / (int32_t)(radio_trim_low - radio_min);
199  } else {
200  return 0;
201  }
202 }

267 float
268 RC_Channel::norm_input()
269 {
270  float ret;
271  int16_t reverse_mul = (reversed?-1:1);
272  if (radio_in < radio_trim) {
273  if (radio_min >= radio_trim) {
274  return 0.0f;
275  }
276  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_trim - radio_min);
277  } else {
278  if (radio_max <= radio_trim) {
279  return 0.0f;
280  }
281  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_max - radio_trim);
282  }
283  return constrain_float(ret, -1.0f, 1.0f);
284 }

286 float
287 RC_Channel::norm_input_dz()
288 {
289  int16_t dz_min = radio_trim - dead_zone;
290  int16_t dz_max = radio_trim + dead_zone;
291  float ret;
292  int16_t reverse_mul = (reversed?-1:1);
293  if (radio_in < dz_min && dz_min > radio_min) {
294  ret = reverse_mul * (float)(radio_in - dz_min) / (float)(dz_min - radio_min);
295  } else if (radio_in > dz_max && radio_max > dz_max) {
296  ret = reverse_mul * (float)(radio_in - dz_max) / (float)(radio_max - dz_max);
297  } else {
298  ret = 0;
299  }
300  return constrain_float(ret, -1.0f, 1.0f);
301 }

303 /*
304  get percentage input from 0 to 100. This ignores the trim value.
305  */
306 uint8_t
307 RC_Channel::percent_input()
308 {
309  if (radio_in <= radio_min) {
310  return reversed?100:0;
311  }
312  if (radio_in >= radio_max) {
313  return reversed?0:100;
314  }
315  uint8_t ret = 100.0f * (radio_in - radio_min) / (float)(radio_max - radio_min);
316  if (reversed) {
317  ret = 100 - ret;
318  }
319  return ret;
