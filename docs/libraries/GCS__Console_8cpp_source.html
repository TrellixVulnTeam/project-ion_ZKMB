27 void gcs_console_send(mavlink_channel_t chan) {
28  uint8_t cons[32];
29  memset(cons, 0, 32);
30  int16_t txspace = comm_get_txspace(chan) - MAVLINK_NUM_NON_PAYLOAD_BYTES;
31  /* There are two bytes of overhead per packet (on top of the non payload
32  * bytes). We can read up to 32 bytes from the console if we have 34
33  * available for tx, up to 16 bytes if we have more than 18 available.
34  * otherwise we have to try again later. */
35  int readable = 32;
36  if (txspace >= 18 && txspace < 34) {
37  readable = 16;
38  } else if ( txspace < 18 ) {
39  return;
40  }
41  /* Read from the console backend */
42  int avail = hal.console->backend_read(cons, readable);
43  /* Send in the smallest packet available. (Don't send if 0.) */
44  if (avail > 0 && avail <= 16) {
45  mavlink_msg_data16_send(chan, DATAMSG_TYPE_CONSOLE, avail, cons);
46  } else if (avail > 16 && avail <= 32) {
47  mavlink_msg_data32_send(chan, DATAMSG_TYPE_CONSOLE, avail, cons);
48  }
