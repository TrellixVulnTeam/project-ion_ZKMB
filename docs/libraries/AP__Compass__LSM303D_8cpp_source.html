219 // Read Sensor data
220 bool AP_Compass_LSM303D::_read_sample()
221 {
222  struct PACKED {
223  uint8_t status;
224  int16_t x;
225  int16_t y;
226  int16_t z;
227  } rx;
228 
229  if (_register_read(ADDR_CTRL_REG7) != _reg7_expected) {
230  hal.console->printf("LSM303D _read_data_transaction_accel: _reg7_expected unexpected\n");
231  return false;
232  }
233 
234  if (!_data_ready()) {
235  return false;
236  }
237 
238  if (!_block_read(ADDR_STATUS_M, (uint8_t *) &rx, sizeof(rx))) {
239  return false;
240  }
241 
242  /* check for overrun */
243  if ((rx.status & 0x70) != 0) {
244  return false;
245  }
246 
247  if (rx.x == 0 && rx.y == 0 && rx.z == 0) {
248  return false;
249  }
250 
251  _mag_x = rx.x;
252  _mag_y = rx.y;
253  _mag_z = rx.z;
254 
255  return true;
256 }

292 bool AP_Compass_LSM303D::_hardware_init()
293 {
294  if (!_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
295  AP_HAL::panic("LSM303D: Unable to get semaphore");
296  }
297 
298  // initially run the bus at low speed
299  _dev->set_speed(AP_HAL::Device::SPEED_LOW);
300 
301  // Test WHOAMI
302  uint8_t whoami = _register_read(ADDR_WHO_AM_I);
303  if (whoami != WHO_I_AM) {
304  hal.console->printf("LSM303D: unexpected WHOAMI 0x%x\n", (unsigned)whoami);
305  goto fail_whoami;
306  }
307 
308  uint8_t tries;
309  for (tries = 0; tries < 5; tries++) {
310  // ensure the chip doesn't interpret any other bus traffic as I2C
311  _disable_i2c();
312 
313  /* enable mag */
314  _reg7_expected = REG7_CONT_MODE_M;
315  _register_write(ADDR_CTRL_REG7, _reg7_expected);
316  _register_write(ADDR_CTRL_REG5, REG5_RES_HIGH_M);
317 
318  // DRDY on MAG on INT2
319  _register_write(ADDR_CTRL_REG4, 0x04);
320 
321  _mag_set_range(LSM303D_MAG_DEFAULT_RANGE_GA);
322  _mag_set_samplerate(LSM303D_MAG_DEFAULT_RATE);
323 
324  hal.scheduler->delay(10);
325  if (_data_ready()) {
326  break;
327  }
328  }
329  if (tries == 5) {
330  hal.console->printf("Failed to boot LSM303D 5 times\n");
331  goto fail_tries;
332  }
333 
334  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
335  _dev->get_semaphore()->give();
336 
337  return true;
338 
339 fail_tries:
340 fail_whoami:
341  _dev->get_semaphore()->give();
342  _dev->set_speed(AP_HAL::Device::SPEED_HIGH);
343  return false;
344 }

346 void AP_Compass_LSM303D::_update()
347 {
348  if (!_read_sample()) {
349  return;
350  }
351 
352  Vector3f raw_field = Vector3f(_mag_x, _mag_y, _mag_z) * _mag_range_scale;
353 
354  // rotate raw_field from sensor frame to body frame
355  rotate_field(raw_field, _compass_instance);
356 
357  // publish raw_field (uncorrected point sample) for calibration use
358  publish_raw_field(raw_field, AP_HAL::micros(), _compass_instance);
359 
360  // correct raw_field for known errors
361  correct_field(raw_field, _compass_instance);
362 
363  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
364  _mag_x_accum += raw_field.x;
365  _mag_y_accum += raw_field.y;
366  _mag_z_accum += raw_field.z;
367  _accum_count++;
368  if (_accum_count == 10) {
369  _mag_x_accum /= 2;
370  _mag_y_accum /= 2;
371  _mag_z_accum /= 2;
372  _accum_count = 5;
373  }
374  _sem->give();
375  }
376 }

379 void AP_Compass_LSM303D::read()
380 {
381  if (!_initialised) {
382  return;
383  }
384 
385  if (!_sem->take_nonblocking()) {
386  return;
387  }
388 
389  if (_accum_count == 0) {
390  /* We're not ready to publish*/
391  _sem->give();
392  return;
393  }
394 
395  Vector3f field(_mag_x_accum, _mag_y_accum, _mag_z_accum);
396  field /= _accum_count;
397 
398  _accum_count = 0;
399  _mag_x_accum = _mag_y_accum = _mag_z_accum = 0;
400 
401  _sem->give();
402 
403  publish_filtered_field(field, _compass_instance);
404 }
