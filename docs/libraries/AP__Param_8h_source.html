414 /* This header is prepended to a variable stored in EEPROM.
415  * The meaning is as follows:
416  *
417  * - key: the k_param enum value from Parameter.h in the sketch
418  *
419  * - group_element: This is zero for top level parameters. For
420  * parameters stored within an object this is divided
421  * into 3 lots of 6 bits, allowing for three levels
422  * of object to be stored in the eeprom
423  *
424  * - type: the ap_var_type value for the variable
425  */
426  struct Param_header {
427  // to get 9 bits for key we needed to split it into two parts to keep binary compatibility
428  uint32_t key_low : 8;
429  uint32_t type : 5;
430  uint32_t key_high : 1;
431  uint32_t group_element : 18;
432  };
