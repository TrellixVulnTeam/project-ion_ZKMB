182 uint32_t UDPDriver::socket_check(uint8_t *buf, int len, uint32_t *nbytes)
183 {
184  if (!initialised) {
185  return 1;
186  }
187  if (!writebuf) {
188  return 1;
189  }
190  *nbytes = writebuf->available();
191  if (*nbytes == 0) {
192  return 1;
193  }
194  if (*nbytes > len) {
195  *nbytes = len;
196  }
197  uint32_t n = *nbytes;
198  if (writebuf->peek(0) != 254) {
199  /*
200  we have a non-mavlink packet at the start of the
201  buffer. Look ahead for a MAVLink start byte, up to 256 bytes
202  ahead
203  */
204  uint16_t limit = n>256?256:n;
205  uint16_t i;
206  for (i=0; i<limit; i++) {
207  if (writebuf->peek(i) == 254) {
208  n = i;
209  break;
210  }
211  }
212  // if we didn't find a MAVLink marker then limit the send size to 256
213  if (i == limit) {
214  n = limit;
215  }
216  } else {
217  // this looks like a MAVLink packet - try to write on
218  // packet boundaries when possible
219  if (n < 8) {
220  n = 0;
221  } else {
222  // the length of the packet is the 2nd byte, and mavlink
223  // packets have a 6 byte header plus 2 byte checksum,
224  // giving len+8 bytes
225  uint8_t len = writebuf->peek(1);
226  if (n < len+8) {
227  // we don't have a full packet yet
228  n = 0;
229  } else if (n > len+8) {
230  // send just 1 packet at a time (so MAVLink packets
231  // are aligned on UDP boundaries)
232  n = len+8;
233  }
234  }
235  }
236 
237  *nbytes = n;
238  writebuf->read(buf, *nbytes);
239  return 0;
240 }
