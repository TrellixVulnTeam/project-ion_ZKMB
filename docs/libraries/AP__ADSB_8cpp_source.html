158 /*
159  * periodic update to handle vehicle timeouts and trigger collision detection
160  */
161 void AP_ADSB::update(void)
162 {
163  // update _my_loc
164  if (!_ahrs.get_position(_my_loc)) {
165  _my_loc.zero();
166  }
167 
168  if (!_enabled) {
169  if (in_state.vehicle_list != nullptr) {
170  deinit();
171  }
172  // nothing to do
173  return;
174  } else if (in_state.vehicle_list == nullptr) {
175  init();
176  return;
177  } else if (in_state.list_size != in_state.list_size_param) {
178  deinit();
179  return;
180  }
181 
182  uint32_t now = AP_HAL::millis();
183 
184  // check current list for vehicles that time out
185  uint16_t index = 0;
186  while (index < in_state.vehicle_count) {
187  // check list and drop stale vehicles. When disabled, the list will get flushed
188  if (now - in_state.vehicle_list[index].last_update_ms > VEHICLE_TIMEOUT_MS) {
189  // don't increment index, we want to check this same index again because the contents changed
190  // also, if we're disabled then clear the list
191  delete_vehicle(index);
192  } else {
193  index++;
194  }
195  }
196 
197  if (_my_loc.is_zero()) {
198  // if we don't have a GPS lock then there's nothing else to do
199  return;
200  }
201 
202  if (out_state.chan < 0) {
203  // if there's no transceiver detected then do not set ICAO and do not service the transceiver
204  return;
205  }
206 
207 
208  // ensure it's positive 24bit but allow -1
209  if (out_state.cfg.ICAO_id_param <= -1 || out_state.cfg.ICAO_id_param > 0x00FFFFFF) {
210  // icao param of -1 means static information is not sent, transceiver is assumed pre-programmed.
211  // reset timer constantly so it never reaches 10s so it never sends
212  out_state.last_config_ms = now;
213 
214  } else if (out_state.cfg.ICAO_id == 0 ||
215  out_state.cfg.ICAO_id_param_prev != out_state.cfg.ICAO_id_param) {
216 
217  // if param changed then regenerate. This allows the param to be changed back to zero to trigger a re-generate
218  if (out_state.cfg.ICAO_id_param == 0) {
219  out_state.cfg.ICAO_id = genICAO(_my_loc);
220  } else {
221  out_state.cfg.ICAO_id = out_state.cfg.ICAO_id_param;
222  }
223  out_state.cfg.ICAO_id_param_prev = out_state.cfg.ICAO_id_param;
224  set_callsign("PING", true);
225  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "ADSB: Using ICAO_id %d and Callsign %s", out_state.cfg.ICAO_id, out_state.cfg.callsign);
226  out_state.last_config_ms = 0; // send now
227  }
228 
229 
230  // send static configuration data to transceiver, every 10s
231  if (out_state.chan_last_ms > 0 && now - out_state.chan_last_ms > ADSB_CHAN_TIMEOUT_MS) {
232  // haven't gotten a heartbeat health status packet in a while, assume hardware failure
233  // TODO: reset out_state.chan
234  out_state.chan = -1;
235  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_ERROR, "ADSB: Transceiver heartbeat timed out");
236  } else if (out_state.chan < MAVLINK_COMM_NUM_BUFFERS) {
237  mavlink_channel_t chan = (mavlink_channel_t)(MAVLINK_COMM_0 + out_state.chan);
238  if (now - out_state.last_config_ms >= 5000 && HAVE_PAYLOAD_SPACE(chan, UAVIONIX_ADSB_OUT_CFG)) {
239  out_state.last_config_ms = now;
240  send_configure(chan);
241  } // last_config_ms
242 
243  // send dynamic data to transceiver at 5Hz
244  if (now - out_state.last_report_ms >= 200 && HAVE_PAYLOAD_SPACE(chan, UAVIONIX_ADSB_OUT_DYNAMIC)) {
245  out_state.last_report_ms = now;
246  send_dynamic_out(chan);
247  } // last_report_ms
248  } // chan_last_ms
249 }

251 /*
252  * determine index and distance of furthest vehicle. This is
253  * used to bump it off when a new closer aircraft is detected
254  */
255 void AP_ADSB::determine_furthest_aircraft(void)
256 {
257  float max_distance = 0;
258  uint16_t max_distance_index = 0;
259 
260  for (uint16_t index = 0; index < in_state.vehicle_count; index++) {
261  float distance = _my_loc.get_distance(get_location(in_state.vehicle_list[index]));
262  if (max_distance < distance || index == 0) {
263  max_distance = distance;
264  max_distance_index = index;
265  }
266  } // for index
267 
268  furthest_vehicle_index = max_distance_index;
269  furthest_vehicle_distance = max_distance;
270 }

286 /*
287  * delete a vehicle by copying last vehicle to
288  * current index then decrementing count
289  */
290 void AP_ADSB::delete_vehicle(const uint16_t index)
291 {
292  if (index < in_state.vehicle_count) {
293  // if the vehicle is the furthest, invalidate it. It has been bumped
294  if (index == furthest_vehicle_index && furthest_vehicle_distance > 0) {
295  furthest_vehicle_distance = 0;
296  furthest_vehicle_index = 0;
297  }
298  if (index != (in_state.vehicle_count-1)) {
299  in_state.vehicle_list[index] = in_state.vehicle_list[in_state.vehicle_count-1];
300  }
301  // TODO: is memset needed? When we decrement the index we essentially forget about it
302  memset(&in_state.vehicle_list[in_state.vehicle_count-1], 0, sizeof(adsb_vehicle_t));
303  in_state.vehicle_count--;
304  }
305 }

323 /*
324  * Update the vehicle list. If the vehicle is already in the
325  * list then it will update it, otherwise it will be added.
326  */
327 void AP_ADSB::handle_vehicle(const mavlink_message_t* packet)
328 {
329  if (in_state.vehicle_list == nullptr) {
330  // We are only null when disabled. Updating is inhibited.
331  return;
332  }
333 
334  uint16_t index = in_state.list_size + 1; // initialize with invalid index
335  adsb_vehicle_t vehicle {};
336  mavlink_msg_adsb_vehicle_decode(packet, &vehicle.info);
337  Location_Class vehicle_loc = Location_Class(AP_ADSB::get_location(vehicle));
338  bool my_loc_is_zero = _my_loc.is_zero();
339  float my_loc_distance_to_vehicle = _my_loc.get_distance(vehicle_loc);
340  bool out_of_range = in_state.list_radius > 0 && !my_loc_is_zero && my_loc_distance_to_vehicle > in_state.list_radius;
341  bool is_tracked_in_list = find_index(vehicle, &index);
342  uint32_t now = AP_HAL::millis();
343 
344  // note the last time the receiver got a packet from the aircraft
345  vehicle.last_update_ms = now - (vehicle.info.tslc * 1000);
346 
347  const uint16_t required_flags_position = ADSB_FLAGS_VALID_COORDS | ADSB_FLAGS_VALID_ALTITUDE;
348  const bool detected_ourself = (out_state.cfg.ICAO_id != 0) && ((uint32_t)out_state.cfg.ICAO_id == vehicle.info.ICAO_address);
349 
350  if (vehicle_loc.is_zero() ||
351  out_of_range ||
352  detected_ourself ||
353  (vehicle.info.ICAO_address > 0x00FFFFFF) || // ICAO address is 24bits, so ignore higher values.
354  !(vehicle.info.flags & required_flags_position) ||
355  now - vehicle.last_update_ms > VEHICLE_TIMEOUT_MS) {
356 
357  // vehicle is out of range or invalid lat/lng. If we're tracking it, delete from list. Otherwise ignore it.
358  if (is_tracked_in_list) {
359  delete_vehicle(index);
360  }
361  return;
362 
363  } else if (is_tracked_in_list) {
364 
365  // found, update it
366  set_vehicle(index, vehicle);
367 
368  } else if (in_state.vehicle_count < in_state.list_size) {
369 
370  // not found and there's room, add it to the end of the list
371  set_vehicle(in_state.vehicle_count, vehicle);
372  in_state.vehicle_count++;
373 
374  } else {
375  // buffer is full. if new vehicle is closer than furthest, replace furthest with new
376 
377  if (my_loc_is_zero) {
378  // nothing else to do
379  furthest_vehicle_distance = 0;
380  furthest_vehicle_index = 0;
381 
382  } else {
383  if (furthest_vehicle_distance <= 0) {
384  // ensure this is populated
385  determine_furthest_aircraft();
386  }
387 
388  if (my_loc_distance_to_vehicle < furthest_vehicle_distance) { // is closer than the furthest
389  // replace with the furthest vehicle
390  set_vehicle(furthest_vehicle_index, vehicle);
391 
392  // furthest_vehicle_index is now invalid because the vehicle was overwritten, need
393  // to run determine_furthest_aircraft() to determine a new one next time
394  furthest_vehicle_distance = 0;
395  furthest_vehicle_index = 0;
396  }
397  }
398  } // if buffer full
399 
400  const uint16_t required_flags_avoidance =
401  ADSB_FLAGS_VALID_COORDS |
402  ADSB_FLAGS_VALID_ALTITUDE |
403  ADSB_FLAGS_VALID_HEADING |
404  ADSB_FLAGS_VALID_VELOCITY;
405 
406  if (vehicle.info.flags & required_flags_avoidance) {
407  push_sample(vehicle); // note that set_vehicle modifies vehicle
408  }
409 }

422 {
423  if (in_state.vehicle_list == nullptr || in_state.vehicle_count == 0) {
424  return;
425  }
426 
427  uint32_t now = AP_HAL::millis();
428 
429  if (in_state.send_index[chan] >= in_state.vehicle_count) {
430  // we've finished a list
431  if (now - in_state.send_start_ms[chan] < 1000) {
432  // too soon to start a new one

573  Reference: DO282B, 2.2.4.5.1.3.2

575  Note gps.time is the number of seconds since UTC midnight

577 uint32_t AP_ADSB::genICAO(const Location_Class &loc)
578 {
579  // gps_time is not seconds since UTC midnight, but it is an equivalent random number
580  // TODO: use UTC time instead of GPS time
581  AP_GPS gps = _ahrs.get_gps();
582  const uint64_t gps_time = gps.time_epoch_usec();
583 
584  uint32_t timeSum = 0;

586

593 // assign a string to out_state.cfg.callsign but ensure it's null terminated
594 void AP_ADSB::set_callsign(const char* str, const bool append_icao)
595 {
596  bool zero_char_pad = false;
597 
598  // clean slate
599  memset(out_state.cfg.callsign, 0, sizeof(out_state.cfg.callsign));
600 
601  // copy str to cfg.callsign but we can't use strncpy because we need
602  // to restrict values to only 'A' - 'Z' and '0' - '9' and pad
603  for (uint8_t i=0; i<sizeof(out_state.cfg.callsign)-1; i++) {
604  if (!str[i] || zero_char_pad) {
605  // finish early. Either pad the rest with zero char or null terminate and call it a day
606  if ((append_icao && i<4) || zero_char_pad) {
607  out_state.cfg.callsign[i] = '0';
608  zero_char_pad = true;
609  } else {
610  // already null terminated via memset so just stop
611  break;
612  }
613 
614  } else if (('A' <= str[i] && str[i] <= 'Z') ||
615  ('0' <= str[i] && str[i] <= '9')) {
616  // valid as-is
617  // spaces are also allowed but are handled in the last else
618  out_state.cfg.callsign[i] = str[i];
619 
620  } else if ('a' <= str[i] && str[i] <= 'z') {
621  // toupper()
622  out_state.cfg.callsign[i] = str[i] - ('a' - 'A');
623 
624  } else if (i == 0) {
625  // invalid, pad to char zero because first index can't be space
626  out_state.cfg.callsign[i] = '0';
627 
628  } else {
629  // invalid, pad with space
630  out_state.cfg.callsign[i] = ' ';
631  }
632  } // for i
633 
634  if (append_icao) {
635  snprintf(&out_state.cfg.callsign[4], 5, "%04X", out_state.cfg.ICAO_id % 0x10000);
636  }
637 }
