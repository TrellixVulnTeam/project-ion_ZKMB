208 bool Scheduler::_register_timesliced_proc(AP_HAL::MemberProc proc,
209  uint8_t freq_div)
210 {
211  unsigned int i, j;
212  uint8_t distance, min_distance, best_distance;
213  uint8_t best_timeslot;
214 
215  if (_num_timesliced_procs > LINUX_SCHEDULER_MAX_TIMESLICED_PROCS) {
216  hal.console->printf("Out of timesliced processes\n");
217  return false;
218  }
219 
220  /* if max_freq_div increases, update the timeslots accordingly */
221  if (freq_div > _max_freq_div) {
222  for (i = 0; i < _num_timesliced_procs; i++) {
223  _timesliced_proc[i].timeslot = _timesliced_proc[i].timeslot
224  / _max_freq_div * freq_div;
225  }
226  _max_freq_div = freq_div;
227  }
228 
229  best_distance = 0;
230  best_timeslot = 0;
231 
232  /* Look for the timeslot that maximizes the min distance with other timeslots */
233  for (i = 0; i < _max_freq_div; i++) {
234  min_distance = _max_freq_div;
235  for (j = 0; j < _num_timesliced_procs; j++) {
236  distance = std::min(i - _timesliced_proc[j].timeslot,
237  _max_freq_div + _timesliced_proc[j].timeslot - i);
238  if (distance < min_distance) {
239  min_distance = distance;
240  if (min_distance == 0) {
241  break;
242  }
243  }
244  }
245  if (min_distance > best_distance) {
246  best_distance = min_distance;
247  best_timeslot = i;
248  }
249  }
250 
251  _timesliced_proc[_num_timesliced_procs].proc = proc;
252  _timesliced_proc[_num_timesliced_procs].timeslot = best_timeslot;
253  _timesliced_proc[_num_timesliced_procs].freq_div = freq_div;
254  _num_timesliced_procs++;
255  return true;
256 }
