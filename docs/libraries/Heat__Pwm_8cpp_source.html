47 void HeatPwm::set_imu_temp(float current)
48 {
49  float error, output;
50 
51  if (_target == nullptr) {
52  // not configured
53  return;
54  }
55 
56  if (AP_HAL::millis() - _last_temp_update < 5) {
57  return;
58  }
59 
60  /* minimal PI algo without dt */
61  error = ((float)*_target) - current;
62  /* Don't accumulate errors if the integrated error is superior
63  * to the max duty cycle(pwm_period)
64  */
65  if ((fabsf(_sum_error) * _Ki < _period_ns)) {
66  _sum_error = _sum_error + error;
67  }
68 
69  output = _Kp*error + _Ki * _sum_error;
70 
71  if (output > _period_ns) {
72  output = _period_ns;
73  } else if (output < 0) {
74  output = 0;
75  }
76 
77  _pwm->set_duty_cycle(output);
78  _last_temp_update = AP_HAL::millis();
79  // printf("target %.1f current %.1f out %.2f\n", _target, current, output);
80 }
