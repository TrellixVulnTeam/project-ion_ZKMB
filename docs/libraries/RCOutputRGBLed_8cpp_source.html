55 bool RCOutputRGBLed::hw_set_rgb(uint8_t red, uint8_t green, uint8_t blue)
56 {
57  const uint16_t freq_motor = hal.rcout->get_freq(0);
58  const uint16_t freq = hal.rcout->get_freq(_red_channel);
59  const uint16_t usec_period = hz_to_usec(freq);
60 
61  if (freq_motor != freq) {
62  /*
63  * keep at same frequency as the first RCOutput channel, some RCOutput
64  * drivers can not operate in different frequency between channels
65  */
66  const uint32_t mask = 1 << _red_channel | 1 << _green_channel
67  | 1 << _blue_channel;
68  hal.rcout->set_freq(mask, freq_motor);
69  }
70 
71  /*
72  * Not calling push() to have a better performance on RCOutput's that
73  * implements cork()/push(), so this changes will be committed together
74  * with the motors.
75  */
76  hal.rcout->cork();
77 
78  uint16_t usec_duty = usec_period * red / _led_bright;
79  hal.rcout->write(_red_channel, usec_duty);
80 
81  usec_duty = usec_period * green / _led_bright;
82  hal.rcout->write(_green_channel, usec_duty);
83 
84  usec_duty = usec_period * blue / _led_bright;
85  hal.rcout->write(_blue_channel, usec_duty);
86 
87  return true;
88 }
