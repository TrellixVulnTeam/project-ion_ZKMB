124 bool AP_Compass_AK8963::init()
125 {
126  AP_HAL::Semaphore *bus_sem = _bus->get_semaphore();
127 
128  if (!bus_sem || !_bus->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
129  hal.console->printf("AK8963: Unable to get bus semaphore\n");
130  return false;
131  }
132 
133  if (!_bus->configure()) {
134  hal.console->printf("AK8963: Could not configure the bus\n");
135  goto fail;
136  }
137 
138  if (!_check_id()) {
139  hal.console->printf("AK8963: Wrong id\n");
140  goto fail;
141  }
142 
143  if (!_calibrate()) {
144  hal.console->printf("AK8963: Could not read calibration data\n");
145  goto fail;
146  }
147 
148  if (!_setup_mode()) {
149  hal.console->printf("AK8963: Could not setup mode\n");
150  goto fail;
151  }
152 
153  if (!_bus->start_measurements()) {
154  hal.console->printf("AK8963: Could not start measurements\n");
155  goto fail;
156  }
157 
158  _initialized = true;
159 
160  /* register the compass instance in the frontend */
161  _compass_instance = register_compass();
162 
163  set_rotation(_compass_instance, _rotation);
164 
165  _bus->set_device_type(DEVTYPE_AK8963);
166  set_dev_id(_compass_instance, _bus->get_bus_id());
167 
168  bus_sem->give();
169 
170  /* timer needs to be called every 10ms so set the freq_div to 10 */
171  if (!_bus->register_periodic_callback(10000, FUNCTOR_BIND_MEMBER(&AP_Compass_AK8963::_update, void))) {
172  // fallback to timer
173  _timesliced = hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Compass_AK8963::_update_timer, void), 10);
174  }
175 
176  return true;
177 
178 fail:
179  bus_sem->give();
180  return false;
181 }

183 void AP_Compass_AK8963::read()
184 {
185  if (!_initialized) {
186  return;
187  }
188 
189  if (_sem->take_nonblocking()) {
190  if (_accum_count == 0) {
191  /* We're not ready to publish */
192  _sem->give();
193  return;
194  }
195 
196  Vector3f field = Vector3f(_mag_x_accum, _mag_y_accum, _mag_z_accum) / _accum_count;
197  _mag_x_accum = _mag_y_accum = _mag_z_accum = 0;
198  _accum_count = 0;
199  _sem->give();
200  publish_filtered_field(field, _compass_instance);
201  }
202 }

218 void AP_Compass_AK8963::_update()
219 {
220  struct sample_regs regs;
221  Vector3f raw_field;
222  uint32_t time_us = AP_HAL::micros();
223 
224  if (!_bus->block_read(AK8963_HXL, (uint8_t *) &regs, sizeof(regs))) {
225  return;
226  }
227 
228  /* Check for overflow. See AK8963's datasheet, section
229  * 6.4.3.6 - Magnetic Sensor Overflow. */
230  if ((regs.st2 & 0x08)) {
231  return;
232  }
233 
234  raw_field = Vector3f(regs.val[0], regs.val[1], regs.val[2]);
235 
236  if (is_zero(raw_field.x) && is_zero(raw_field.y) && is_zero(raw_field.z)) {
237  return;
238  }
239 
240  _make_factory_sensitivity_adjustment(raw_field);
241  _make_adc_sensitivity_adjustment(raw_field);
242  raw_field *= AK8963_MILLIGAUSS_SCALE;
243 
244  // rotate raw_field from sensor frame to body frame
245  rotate_field(raw_field, _compass_instance);
246 
247  // publish raw_field (uncorrected point sample) for calibration use
248  publish_raw_field(raw_field, time_us, _compass_instance);
249 
250  // correct raw_field for known errors
251  correct_field(raw_field, _compass_instance);
252 
253  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
254  _mag_x_accum += raw_field.x;
255  _mag_y_accum += raw_field.y;
256  _mag_z_accum += raw_field.z;
257  _accum_count++;
258  if (_accum_count == 10) {
259  _mag_x_accum /= 2;
260  _mag_y_accum /= 2;
261  _mag_z_accum /= 2;
262  _accum_count = 5;
263  }
264  _sem->give();
265  }
266 }

315 bool AP_Compass_AK8963::_calibrate()
316 {
317  /* Enable FUSE-mode in order to be able to read calibration data */
318  _bus->register_write(AK8963_CNTL1, AK8963_FUSE_MODE | AK8963_16BIT_ADC);
319 
320  uint8_t response[3];
321 
322  _bus->block_read(AK8963_ASAX, response, 3);
323 
324  for (int i = 0; i < 3; i++) {
325  float data = response[i];
326  _magnetometer_ASA[i] = ((data - 128) / 256 + 1);
327  }
328 
329  return true;
330 }
