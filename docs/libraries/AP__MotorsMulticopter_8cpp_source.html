247 // return current_limit as a number from 0 ~ 1 in the range throttle_min to throttle_max
248 float AP_MotorsMulticopter::get_current_limit_max_throttle()
249 {
250  // return maximum if current limiting is disabled
251  if (_batt_current_max <= 0) {
252  _throttle_limit = 1.0f;
253  return 1.0f;
254  }
255 
256  // remove throttle limit if disarmed
257  if (!_flags.armed) {
258  _throttle_limit = 1.0f;
259  return 1.0f;
260  }
261 
262  // calculate the maximum current to prevent voltage sag below _batt_voltage_min
263  float batt_current_max = MIN(_batt_current_max, _batt_current + (_batt_voltage-_batt_voltage_min)/_batt_resistance);
264 
265  float batt_current_ratio = _batt_current/batt_current_max;
266 
267  float loop_interval = 1.0f/_loop_rate;
268  _throttle_limit += (loop_interval/(loop_interval+_batt_current_time_constant))*(1.0f - batt_current_ratio);
269 
270  // throttle limit drops to 20% between hover and full throttle
271  _throttle_limit = constrain_float(_throttle_limit, 0.2f, 1.0f);
272 
273  // limit max throttle
274  return get_throttle_hover() + ((1.0-get_throttle_hover())*_throttle_limit);
275 }

277 // apply_thrust_curve_and_volt_scaling - returns throttle in the range 0 ~ 1
278 float AP_MotorsMulticopter::apply_thrust_curve_and_volt_scaling(float thrust) const
279 {
280  float throttle_ratio = thrust;
281  // apply thrust curve - domain 0.0 to 1.0, range 0.0 to 1.0
282  if (_thrust_curve_expo > 0.0f){
283  if(!is_zero(_batt_voltage_filt.get())) {
284  throttle_ratio = ((_thrust_curve_expo-1.0f) + safe_sqrt((1.0f-_thrust_curve_expo)*(1.0f-_thrust_curve_expo) + 4.0f*_thrust_curve_expo*_lift_max*thrust))/(2.0f*_thrust_curve_expo*_batt_voltage_filt.get());
285  } else {
286  throttle_ratio = ((_thrust_curve_expo-1.0f) + safe_sqrt((1.0f-_thrust_curve_expo)*(1.0f-_thrust_curve_expo) + 4.0f*_thrust_curve_expo*_lift_max*thrust))/(2.0f*_thrust_curve_expo);
287  }
288  }
289 
290  return constrain_float(throttle_ratio, 0.0f, 1.0f);
291 }

293 // update_lift_max from battery voltage - used for voltage compensation
294 void AP_MotorsMulticopter::update_lift_max_from_batt_voltage()
295 {
296  // sanity check battery_voltage_min is not too small
297  // if disabled or misconfigured exit immediately
298  if((_batt_voltage_max <= 0) || (_batt_voltage_min >= _batt_voltage_max) || (_batt_voltage < 0.25f*_batt_voltage_min)) {
299  _batt_voltage_filt.reset(1.0f);
300  _lift_max = 1.0f;
301  return;
302  }
303 
304  _batt_voltage_min = MAX(_batt_voltage_min, _batt_voltage_max * 0.6f);
305 
306  // add current based voltage sag to battery voltage
307  float batt_voltage = _batt_voltage + (_batt_current-_batt_current_resting) * _batt_resistance;
308  batt_voltage = constrain_float(batt_voltage, _batt_voltage_min, _batt_voltage_max);
309 
310  // filter at 0.5 Hz
311  float batt_voltage_filt = _batt_voltage_filt.apply(batt_voltage/_batt_voltage_max, 1.0f/_loop_rate);
312 
313  // calculate lift max
314  _lift_max = batt_voltage_filt*(1-_thrust_curve_expo) + _thrust_curve_expo*batt_voltage_filt*batt_voltage_filt;
315 }

317 // update_battery_resistance - calculate battery resistance when throttle is above hover_out
318 void AP_MotorsMulticopter::update_battery_resistance()
319 {
320  // if disarmed reset resting voltage and current
321  if (!_flags.armed) {
322  _batt_voltage_resting = _batt_voltage;
323  _batt_current_resting = _batt_current;
324  _batt_timer = 0;
325  } else if (_batt_voltage_resting > _batt_voltage && _batt_current_resting < _batt_current) {
326  // update battery resistance when throttle is over hover throttle
327  float batt_resistance = (_batt_voltage_resting-_batt_voltage)/(_batt_current-_batt_current_resting);
328  if ((_batt_timer < 400) && ((_batt_current_resting*2.0f) < _batt_current)) {
329  if (get_throttle() >= get_throttle_hover()) {
330  // filter reaches 90% in 1/4 the test time
331  _batt_resistance += 0.05f*(batt_resistance - _batt_resistance);
332  _batt_timer += 1;
333  } else {
334  // initialize battery resistance to prevent change in resting voltage estimate
335  _batt_resistance = batt_resistance;
336  }
337  }
338  // make sure battery resistance value doesn't result in the predicted battery voltage exceeding the resting voltage
339  if(batt_resistance < _batt_resistance){
340  _batt_resistance = batt_resistance;
341  }
342  }

345 float AP_MotorsMulticopter::get_compensation_gain() const
346 {
347  // avoid divide by zero
348  if (_lift_max <= 0.0f) {
349  return 1.0f;
350  }
351 
352  float ret = 1.0f / _lift_max;
353 
354 #if AP_MOTORS_DENSITY_COMP == 1
355  // air density ratio is increasing in density / decreasing in altitude
356  if (_air_density_ratio > 0.3f && _air_density_ratio < 1.5f) {
357  ret *= 1.0f / constrain_float(_air_density_ratio,0.5f,1.25f);
358  }
359 #endif
360  return ret;
361 }

417 // run spool logic
418 void AP_MotorsMulticopter::output_logic()
419 {
420  if (_flags.armed) {
421  _disarm_safety_timer = 100;
422  } else if (_disarm_safety_timer != 0) {
423  _disarm_safety_timer--;
424  }
425 
426  // force desired and current spool mode if disarmed or not interlocked
427  if (!_flags.armed || !_flags.interlock) {
428  _spool_desired = DESIRED_SHUT_DOWN;
429  _spool_mode = SHUT_DOWN;
430  }
431 
432  if (_spool_up_time < 0.05) {
433  // prevent float exception
434  _spool_up_time.set(0.05);
435  }
436 
437  switch (_spool_mode) {
438  case SHUT_DOWN:
439  // Motors should be stationary.
440  // Servos set to their trim values or in a test condition.
441 
442  // set limits flags
443  limit.roll_pitch = true;
444  limit.yaw = true;
445  limit.throttle_lower = true;
446  limit.throttle_upper = true;
447 
448  // make sure the motors are spooling in the correct direction
449  if (_spool_desired != DESIRED_SHUT_DOWN) {
450  _spool_mode = SPIN_WHEN_ARMED;
451  break;
452  }
453 
454  // set and increment ramp variables
455  _spin_up_ratio = 0.0f;
456  _throttle_thrust_max = 0.0f;
457  break;
458 
459  case SPIN_WHEN_ARMED: {
460  // Motors should be stationary or at spin when armed.
461  // Servos should be moving to correct the current attitude.
462 
463  // set limits flags
464  limit.roll_pitch = true;
465  limit.yaw = true;
466  limit.throttle_lower = true;
467  limit.throttle_upper = true;
468 
469  // set and increment ramp variables
470  float spool_step = 1.0f/(_spool_up_time*_loop_rate);
471  if (_spool_desired == DESIRED_SHUT_DOWN){
472  _spin_up_ratio -= spool_step;
473  // constrain ramp value and update mode
474  if (_spin_up_ratio <= 0.0f) {
475  _spin_up_ratio = 0.0f;
476  _spool_mode = SHUT_DOWN;
477  }
478  } else if(_spool_desired == DESIRED_THROTTLE_UNLIMITED) {
479  _spin_up_ratio += spool_step;
480  // constrain ramp value and update mode
481  if (_spin_up_ratio >= 1.0f) {
482  _spin_up_ratio = 1.0f;
483  _spool_mode = SPOOL_UP;
484  }
485  } else { // _spool_desired == SPIN_WHEN_ARMED
486  float spin_up_armed_ratio = 0.0f;
487  if (_spin_min > 0.0f) {
488  spin_up_armed_ratio = _spin_arm / _spin_min;
489  }
490  _spin_up_ratio += constrain_float(spin_up_armed_ratio-_spin_up_ratio, -spool_step, spool_step);
491  }
492  _throttle_thrust_max = 0.0f;
493  break;
494  }
495  case SPOOL_UP:
496  // Maximum throttle should move from minimum to maximum.
497  // Servos should exhibit normal flight behavior.
498 
499  // initialize limits flags
500  limit.roll_pitch = false;
501  limit.yaw = false;
502  limit.throttle_lower = false;
503  limit.throttle_upper = false;
504 
505  // make sure the motors are spooling in the correct direction
506  if (_spool_desired != DESIRED_THROTTLE_UNLIMITED ){
507  _spool_mode = SPOOL_DOWN;
508  break;
509  }
510 
511  // set and increment ramp variables
512  _spin_up_ratio = 1.0f;
513  _throttle_thrust_max += 1.0f/(_spool_up_time*_loop_rate);
514 
515  // constrain ramp value and update mode
516  if (_throttle_thrust_max >= MIN(get_throttle(), get_current_limit_max_throttle())) {
517  _throttle_thrust_max = get_current_limit_max_throttle();
518  _spool_mode = THROTTLE_UNLIMITED;
519  } else if (_throttle_thrust_max < 0.0f) {
520  _throttle_thrust_max = 0.0f;
521  }
522  break;
523 
524  case THROTTLE_UNLIMITED:
525  // Throttle should exhibit normal flight behavior.
526  // Servos should exhibit normal flight behavior.
527 
528  // initialize limits flags
529  limit.roll_pitch = false;
530  limit.yaw = false;
531  limit.throttle_lower = false;
532  limit.throttle_upper = false;
533 
534  // make sure the motors are spooling in the correct direction
535  if (_spool_desired != DESIRED_THROTTLE_UNLIMITED) {
536  _spool_mode = SPOOL_DOWN;
537  break;
538  }
539 
540  // set and increment ramp variables
541  _spin_up_ratio = 1.0f;
542  _throttle_thrust_max = get_current_limit_max_throttle();
543  break;
544 
545  case SPOOL_DOWN:
546  // Maximum throttle should move from maximum to minimum.
547  // Servos should exhibit normal flight behavior.
548
