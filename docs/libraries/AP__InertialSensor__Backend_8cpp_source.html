16 {
17  /*

68 void AP_InertialSensor_Backend::_notify_new_gyro_raw_sample(uint8_t instance,
69  const Vector3f &gyro,
70  uint64_t sample_us)
71 {
72  float dt;
73 
74  if (_imu._gyro_raw_sample_rates[instance] <= 0) {
75  return;
76  }
77 
78  dt = 1.0f / _imu._gyro_raw_sample_rates[instance];
79 
80  // call gyro_sample hook if any
81  AP_Module::call_hook_gyro_sample(instance, dt, gyro);
82 
83  // push gyros if optical flow present
84  if (hal.opticalflow)
85  hal.opticalflow->push_gyro(gyro.x, gyro.y, dt);
86 
87  // compute delta angle
88  Vector3f delta_angle = (gyro + _imu._last_raw_gyro[instance]) * 0.5f * dt;
89 
90  // compute coning correction
91  // see page 26 of:
92  // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
93  // Available: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
94  // see also examples/coning.py
95  Vector3f delta_coning = (_imu._delta_angle_acc[instance] +
96  _imu._last_delta_angle[instance] * (1.0f / 6.0f));
97  delta_coning = delta_coning % delta_angle;
98  delta_coning *= 0.5f;
99 
100  if (_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
101  // integrate delta angle accumulator
102  // the angles and coning corrections are accumulated separately in the
103  // referenced paper, but in simulation little difference was found between
104  // integrating together and integrating separately (see examples/coning.py)
105  _imu._delta_angle_acc[instance] += delta_angle + delta_coning;
106  _imu._delta_angle_acc_dt[instance] += dt;
107 
108  // save previous delta angle for coning correction
109  _imu._last_delta_angle[instance] = delta_angle;
110  _imu._last_raw_gyro[instance] = gyro;
111 
112  _imu._gyro_filtered[instance] = _imu._gyro_filter[instance].apply(gyro);
113  if (_imu._gyro_filtered[instance].is_nan() || _imu._gyro_filtered[instance].is_inf()) {
114  _imu._gyro_filter[instance].reset();
115  }
116  _imu._new_gyro_data[instance] = true;
117  _sem->give();
118  }
119 
120  DataFlash_Class *dataflash = get_dataflash();
121  if (dataflash != nullptr) {
122  uint64_t now = AP_HAL::micros64();
123  struct log_GYRO pkt = {
124  LOG_PACKET_HEADER_INIT((uint8_t)(LOG_GYR1_MSG+instance)),
125  time_us : now,
126  sample_us : sample_us?sample_us:now,
127  GyrX : gyro.x,
128  GyrY : gyro.y,
129  GyrZ : gyro.z
130  };
131  dataflash->WriteBlock(&pkt, sizeof(pkt));
132  }
133 }
