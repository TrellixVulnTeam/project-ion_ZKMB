146 void ToneAlarm_Raspilot::_set_pwm0_period(uint32_t time_us)
147 {
148  // stop clock and waiting for busy flag doesn't work, so kill clock
149  *(_clk + RPI_PWMCLK_CNTL) = 0x5A000000 | (1 << 5);
150  usleep(10);
151 
152  // set frequency
153  // DIVI is the integer part of the divisor
154  // the fractional part (DIVF) drops clock cycles to get the output frequency, bad for servo motors
155  // 320 bits for one cycle of 20 milliseconds = 62.5 us per bit = 16 kHz
156  int idiv = (int) (19200000.0f / (320000000.0f / time_us));
157  if (idiv < 1 || idiv > 0x1000) {
158  return;
159  }
160  *(_clk + RPI_PWMCLK_DIV) = 0x5A000000 | (idiv<<12);
161 
162  // source=osc and enable clock
163  *(_clk + RPI_PWMCLK_CNTL) = 0x5A000011;
164 
165  // disable PWM
166  *(_pwm + RPI_PWM_CTL) = 0;
167 
168  // needs some time until the PWM module gets disabled, without the delay the PWM module crashs
169  usleep(10);
170 
171  // filled with 0 for 20 milliseconds = 320 bits
172  *(_pwm + RPI_PWM_RNG1) = 320;
173

175  _set_pwm0_duty(0);
