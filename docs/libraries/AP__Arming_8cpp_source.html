169 bool AP_Arming::ins_checks(bool report)
170 {
171  if ((checks_to_perform & ARMING_CHECK_ALL) ||
172  (checks_to_perform & ARMING_CHECK_INS)) {
173  const AP_InertialSensor &ins = ahrs.get_ins();
174  if (!ins.get_gyro_health_all()) {
175  if (report) {
176  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Gyros not healthy");
177  }
178  return false;
179  }
180  if (!ins.gyro_calibrated_ok_all()) {
181  if (report) {
182  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Gyros not calibrated");
183  }
184  return false;
185  }
186  if (!ins.get_accel_health_all()) {
187  if (report) {
188  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Accels not healthy");
189  }
190  return false;
191  }
192  if (!ins.accel_calibrated_ok_all()) {
193  if (report) {
194  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: 3D Accel calibration needed");
195  }
196  return false;
197  }
198 
199  //check if accelerometers have calibrated and require reboot
200  if (ins.accel_cal_requires_reboot()) {
201  if (report) {
202  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Accels calibrated requires reboot");
203  }
204  return false;
205  }
206 
207  // check all accelerometers point in roughly same direction
208  if (ins.get_accel_count() > 1) {
209  const Vector3f &prime_accel_vec = ins.get_accel();
210  for(uint8_t i=0; i<ins.get_accel_count(); i++) {
211  if (!ins.use_accel(i)) {
212  continue;
213  }
214  // get next accel vector
215  const Vector3f &accel_vec = ins.get_accel(i);
216  Vector3f vec_diff = accel_vec - prime_accel_vec;
217  // allow for user-defined difference, typically 0.75 m/s/s. Has to pass in last 10 seconds
218  float threshold = accel_error_threshold;
219  if (i >= 2) {
220  /*
221  we allow for a higher threshold for IMU3 as it
222  runs at a different temperature to IMU1/IMU2,
223  and is not used for accel data in the EKF
224  */
225  threshold *= 3;
226  }
227 
228  // EKF is less sensitive to Z-axis error
229  vec_diff.z *= 0.5f;
230 
231  if (vec_diff.length() <= threshold) {
232  last_accel_pass_ms[i] = AP_HAL::millis();
233  }
234  if (AP_HAL::millis() - last_accel_pass_ms[i] > 10000) {
235  if (report) {
236  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Accels inconsistent");
237  }
238  return false;
239  }
240  }
241  }
242 
243  // check all gyros are giving consistent readings
244  if (ins.get_gyro_count() > 1) {
245  const Vector3f &prime_gyro_vec = ins.get_gyro();
246  for(uint8_t i=0; i<ins.get_gyro_count(); i++) {
247  if (!ins.use_gyro(i)) {
248  continue;
249  }
250  // get next gyro vector
251  const Vector3f &gyro_vec = ins.get_gyro(i);
252  Vector3f vec_diff = gyro_vec - prime_gyro_vec;
253  // allow for up to 5 degrees/s difference. Pass if it has
254  // been OK in last 10 seconds
255  if (vec_diff.length() <= radians(5)) {
256  last_gyro_pass_ms[i] = AP_HAL::millis();
257  }
258  if (AP_HAL::millis() - last_gyro_pass_ms[i] > 10000) {
259  if (report) {
260  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Gyros inconsistent");
261  }
262  return false;
263  }
264  }
265  }
266  }
267 
268  return true;

271 bool AP_Arming::compass_checks(bool report)
272 {
273  if ((checks_to_perform) & ARMING_CHECK_ALL ||
274  (checks_to_perform) & ARMING_CHECK_COMPASS) {
275 
276  if (!_compass.use_for_yaw()) {
277  // compass use is disabled
278  return true;
279  }
280 
281  if (!_compass.healthy()) {
282  if (report) {
283  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass not healthy");
284  }
285  return false;
286  }
287  // check compass learning is on or offsets have been set
288  if (!_compass.learn_offsets_enabled() && !_compass.configured()) {
289  if (report) {
290  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass not calibrated");
291  }
292  return false;
293  }
294 
295  //check if compass is calibrating
296  if (_compass.is_calibrating()) {
297  if (report) {
298  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "Arm: Compass calibration running");
299  }
300  return false;
301  }
302 
303  //check if compass has calibrated and requires reboot
304  if (_compass.compass_cal_requires_reboot()) {
305  if (report) {
306  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass calibrated requires reboot");
307  }
308  return false;
309  }
310 
311  // check for unreasonable compass offsets
312  Vector3f offsets = _compass.get_offsets();
313  if (offsets.length() > _compass.get_offsets_max()) {
314  if (report) {
315  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass offsets too high");
316  }
317  return false;
318  }
319 
320  // check for unreasonable mag field length
321  float mag_field = _compass.get_field().length();
322  if (mag_field > AP_ARMING_COMPASS_MAGFIELD_MAX || mag_field < AP_ARMING_COMPASS_MAGFIELD_MIN) {
323  if (report) {
324  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Check mag field");
325  }
326  return false;
327  }
328 
329  // check all compasses point in roughly same direction
330  if (!_compass.consistent()) {
331  if (report) {
332  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL,"PreArm: Compasses inconsistent");
333  }
334  return false;
335  }
336  }
337 
338  return true;

366  }
367  }
368 
369  // check GPSs are within 50m of each other and that blending is healthy
370  if ((checks_to_perform & ARMING_CHECK_ALL) || (checks_to_perform & ARMING_CHECK_GPS_CONFIG)) {

373  if (report) {
374  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL,
375  "PreArm: GPS positions differ by %4.1fm",
376  (double)distance_m);
377  }
378  return false;
379  }
380  if (!gps.blend_health_check()) {
381  if (report) {
382  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: GPS blending unhealthy");
