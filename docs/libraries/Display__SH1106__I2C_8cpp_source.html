52 bool Display_SH1106_I2C::hw_init()
53 {
54  struct PACKED {
55  uint8_t reg;
56  uint8_t seq[26];
57  } init_seq = { 0x0, {
58  0xAE, // Display OFF
59  0xA1, // Segment re-map
60  0xC8, // COM Output Scan Direction
61  0xA8, 0x3F, // MUX Ratio
62  0xD5, 0x50, // Display Clock Divide Ratio and Oscillator Frequency: (== +0%)
63  0xD3, 0x00, // Display Offset
64  0xDB, 0x40, // VCOMH Deselect Level
65  0x81, 0xCF, // Contrast Control
66  0xAD, 0x8B, // DC-DC Control Mode: 1b (== internal DC-DC enabled) (AKA: Enable charge pump regulator)
67  0x40, // Display Start Line
68  0xDA, 0x12, // +++ COM Pins hardware configuration
69  0xD9, 0xF1, // +++ Pre-charge Period
70  0xA4, // +++ Entire Display ON (ignoring RAM): 0b (== OFF)
71  0xA6, // +++ Normal/Inverse Display: 0b (== Normal)
72  0xAF, // Display ON
73  0xB0, // Page Address
74  0x02, 0x10 // Column Address
75  } };
76 
77  memset(_displaybuffer, 0, SH1106_COLUMNS * SH1106_ROWS_PER_PAGE);
78 
79  // take i2c bus semaphore
80  if (!_dev || !_dev->get_semaphore()->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
81  return false;
82  }
83 
84  // init display
85  bool success = _dev->transfer((uint8_t *)&init_seq, sizeof(init_seq), nullptr, 0);
86 
87  // give back i2c semaphore
88  _dev->get_semaphore()->give();
89 
90  if (success) {
91  _need_hw_update = true;
92  _dev->register_periodic_callback(20000, FUNCTOR_BIND_MEMBER(&Display_SH1106_I2C::_timer, void));
93  }
94 
95  return success;
96 }

103 void Display_SH1106_I2C::_timer()
104 {
105  if (!_need_hw_update) {
106  return;
107  }
108  _need_hw_update = false;
109 
110  struct PACKED {
111  uint8_t reg;
112  uint8_t column_0_3;
113  uint8_t column_4_7;
114  uint8_t page;
115  } command = { 0x0, 0x2, 0x10, 0xB0 };
116 
117  struct PACKED {
118  uint8_t reg;
119  uint8_t db[SH1106_COLUMNS/2];
120  } display_buffer = { 0x40, {} };
121 
122  // write buffer to display
123  for (uint8_t i = 0; i < (SH1106_ROWS / SH1106_ROWS_PER_PAGE); i++) {
124  command.page = 0xB0 | (i & 0x0F);
125  _dev->transfer((uint8_t *)&command, sizeof(command), nullptr, 0);
126 
127  if (_displaybuffer_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
128  memcpy(&display_buffer.db[0], &_displaybuffer[i * SH1106_COLUMNS], SH1106_COLUMNS/2);
129  _displaybuffer_sem->give();
130  _dev->transfer((uint8_t *)&display_buffer, SH1106_COLUMNS/2 + 1, nullptr, 0);
131  }
132 
133  if (_displaybuffer_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
134  memcpy(&display_buffer.db[0], &_displaybuffer[i * SH1106_COLUMNS + SH1106_COLUMNS/2 ], SH1106_COLUMNS/2);
135  _displaybuffer_sem->give();
136  _dev->transfer((uint8_t *)&display_buffer, SH1106_COLUMNS/2 + 1, nullptr, 0);
137  }
138  }
139 }

141 void Display_SH1106_I2C::set_pixel(uint16_t x, uint16_t y)
142 {
143  // check x, y range
144  if ((x >= SH1106_COLUMNS) || (y >= SH1106_ROWS)) {
145  return;
146  }
147  // set pixel in buffer
148  if (!_displaybuffer_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
149  return;
150  }
151  _displaybuffer[x + (y / 8 * SH1106_COLUMNS)] |= 1 << (y % 8);
152  _displaybuffer_sem->give();
