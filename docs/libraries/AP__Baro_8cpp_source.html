146 // calibrate the barometer. This must be called at least once before
147 // the altitude() or climb_rate() interfaces can be used
148 void AP_Baro::calibrate(bool save)
149 {
150  // reset the altitude offset when we calibrate. The altitude
151  // offset is supposed to be for within a flight
152  _alt_offset.set_and_save(0);
153 
154  // start by assuming all sensors are calibrated (for healthy() test)
155  for (uint8_t i=0; i<_num_sensors; i++) {
156  sensors[i].calibrated = true;
157  sensors[i].alt_ok = true;
158  }
159 
160  // let the barometer settle for a full second after startup
161  // the MS5611 reads quite a long way off for the first second,
162  // leading to about 1m of error if we don't wait
163  for (uint8_t i = 0; i < 10; i++) {
164  uint32_t tstart = AP_HAL::millis();
165  do {
166  update();
167  if (AP_HAL::millis() - tstart > 500) {
168  AP_HAL::panic("PANIC: AP_Baro::read unsuccessful "
169  "for more than 500ms in AP_Baro::calibrate [2]\r\n");
170  }
171  hal.scheduler->delay(10);
172  } while (!healthy());
173  hal.scheduler->delay(100);
174  }
175 
176  // now average over 5 values for the ground pressure and
177  // temperature settings
178  float sum_pressure[BARO_MAX_INSTANCES] = {0};
179  float sum_temperature[BARO_MAX_INSTANCES] = {0};
180  uint8_t count[BARO_MAX_INSTANCES] = {0};
181  const uint8_t num_samples = 5;
182 
183  for (uint8_t c = 0; c < num_samples; c++) {
184  uint32_t tstart = AP_HAL::millis();
185  do {
186  update();
187  if (AP_HAL::millis() - tstart > 500) {
188  AP_HAL::panic("PANIC: AP_Baro::read unsuccessful "
189  "for more than 500ms in AP_Baro::calibrate [3]\r\n");
190  }
191  } while (!healthy());
192  for (uint8_t i=0; i<_num_sensors; i++) {
193  if (healthy(i)) {
194  sum_pressure[i] += sensors[i].pressure;
195  sum_temperature[i] += sensors[i].temperature;
196  count[i] += 1;
197  }
198  }
199  hal.scheduler->delay(100);
200  }
201  for (uint8_t i=0; i<_num_sensors; i++) {
202  if (count[i] == 0) {
203  sensors[i].calibrated = false;
204  } else {
205  if (save) {
206  sensors[i].ground_pressure.set_and_save(sum_pressure[i] / count[i]);
207  }
208  }
209  }
210 
211  _guessed_ground_temperature = get_external_temperature();
212 
213  // panic if all sensors are not calibrated
214  for (uint8_t i=0; i<_num_sensors; i++) {
215  if (sensors[i].calibrated) {
216  return;
217  }
218  }
219  AP_HAL::panic("AP_Baro: all sensors uncalibrated");
220 }

249 // return altitude difference in meters between current pressure and a
250 // given base_pressure in Pascal
251 float AP_Baro::get_altitude_difference(float base_pressure, float pressure) const
252 {
253  float ret;
254  float temp = get_ground_temperature() + C_TO_KELVIN;
255  float scaling = pressure / base_pressure;
256 
257  // This is an exact calculation that is within +-2.5m of the standard
258  // atmosphere tables in the troposphere (up to 11,000 m amsl).
259  ret = 153.8462f * temp * (1.0f - expf(0.190259f * logf(scaling)));
260 
261  return ret;
262 }

265 // return current scale factor that converts from equivalent to true airspeed
266 // valid for altitudes up to 10km AMSL
267 // assumes standard atmosphere lapse rate
268 float AP_Baro::get_EAS2TAS(void)
269 {
270  float altitude = get_altitude();
271  if ((fabsf(altitude - _last_altitude_EAS2TAS) < 25.0f) && !is_zero(_EAS2TAS)) {
272  // not enough change to require re-calculating
273  return _EAS2TAS;
274  }
275 
276  // only estimate lapse rate for the difference from the ground location
277  // provides a more consistent reading then trying to estimate a complete
278  // ISA model atmosphere
279  float tempK = get_ground_temperature() + C_TO_KELVIN - ISA_LAPSE_RATE * altitude;
280  _EAS2TAS = safe_sqrt(1.225f / ((float)get_pressure() / (ISA_GAS_CONSTANT * tempK)));
281  _last_altitude_EAS2TAS = altitude;
282  return _EAS2TAS;
283 }

330 /*
331  get the temperature in degrees C to be used for calibration purposes
332  */
333 float AP_Baro::get_external_temperature(const uint8_t instance) const
334 {
335  // if we have a recent external temperature then use it
336  if (_last_external_temperature_ms != 0 && AP_HAL::millis() - _last_external_temperature_ms < 10000) {
337  return _external_temperature;
338  }
339  // if we don't have an external temperature then use the minimum
340  // of the barometer temperature and 35 degrees C. The reason for
341  // not just using the baro temperature is it tends to read high,
342  // often 30 degrees above the actual temperature. That means the
343  // EAS2TAS tends to be off by quite a large margin, as well as
344  // the calculation of altitude difference betweeen two pressures
345  // reporting a high temperature will cause the aircraft to
346  // estimate itself as flying higher then it actually is.
347  return MIN(get_temperature(instance), INTERNAL_TEMPERATURE_CLAMP);
348 }

499 /*
500  call update on all drivers
501  */
502 void AP_Baro::update(void)
503 {
504  if (fabsf(_alt_offset - _alt_offset_active) > 0.01f) {
505  // If there's more than 1cm difference then slowly slew to it via LPF.
506  // The EKF does not like step inputs so this keeps it happy.
507  _alt_offset_active = (0.95f*_alt_offset_active) + (0.05f*_alt_offset);
508  } else {
509  _alt_offset_active = _alt_offset;
510  }
511 
512  if (!_hil_mode) {
513  for (uint8_t i=0; i<_num_drivers; i++) {
514  drivers[i]->update();
515  }
516  }
517 
518  // consider a sensor as healthy if it has had an update in the
519  // last 0.5 seconds
520  uint32_t now = AP_HAL::millis();
521  for (uint8_t i=0; i<_num_sensors; i++) {
522  sensors[i].healthy = (now - sensors[i].last_update_ms < 500) && !is_zero(sensors[i].pressure);
523  }
524 
525  for (uint8_t i=0; i<_num_sensors; i++) {
526  if (sensors[i].healthy) {
527  // update altitude calculation
528  float ground_pressure = sensors[i].ground_pressure;
529  if (is_zero(ground_pressure) || isnan(ground_pressure) || isinf(ground_pressure)) {
530  sensors[i].ground_pressure = sensors[i].pressure;
531  }
532  float altitude = sensors[i].altitude;
533  if (sensors[i].type == BARO_TYPE_AIR) {
534  altitude = get_altitude_difference(sensors[i].ground_pressure, sensors[i].pressure);
535  } else if (sensors[i].type == BARO_TYPE_WATER) {
536  //101325Pa is sea level air pressure, 9800 Pascal/ m depth in water.
537  //No temperature or depth compensation for density of water.
538  altitude = (sensors[i].ground_pressure - sensors[i].pressure) / 9800.0f / _specific_gravity;
539  }
540  // sanity check altitude
541  sensors[i].alt_ok = !(isnan(altitude) || isinf(altitude));
542  if (sensors[i].alt_ok) {
543  sensors[i].altitude = altitude + _alt_offset_active;
544  }
545  }
546  if (_hil.have_alt) {
547  sensors[0].altitude = _hil.altitude;
548  }
549  if (_hil.have_last_update) {
550  sensors[0].last_update_ms = _hil.last_update_ms;
551  }
552  }
553 
554  // ensure the climb rate filter is updated
555  if (healthy()) {
556  _climb_rate_filter.update(get_altitude(), get_last_update());
557  }
558 
559  // choose primary sensor
560  if (_primary_baro >= 0 && _primary_baro < _num_sensors && healthy(_primary_baro)) {
561  _primary = _primary_baro;
562  } else {
563  _primary = 0;
564  for (uint8_t i=0; i<_num_sensors; i++) {
565  if (healthy(i)) {
566  _primary = i;
567  break;
568  }
569  }
570  }
571 }
