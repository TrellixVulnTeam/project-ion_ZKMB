121 // return origin of position estimate system
122 bool AP_Beacon::get_origin(Location &origin_loc) const
123 {
124  if (!device_ready()) {
125  return false;
126  }
127 
128  // check for unitialised origin
129  if (is_zero(origin_lat) && is_zero(origin_lon) && is_zero(origin_alt)) {
130  return false;
131  }
132 
133  // return origin
134  origin_loc.lat = origin_lat * 1.0e7;
135  origin_loc.lng = origin_lon * 1.0e7;
136  origin_loc.alt = origin_alt * 100;
137  origin_loc.options = 0; // all flags to zero meaning alt-above-sea-level
138 
139  return true;
140 }
