324 void PX4RCOutput::write(uint8_t ch, uint16_t period_us)
325 {
326  if (ch >= PX4_NUM_OUTPUT_CHANNELS) {
327  return;
328  }
329  if (!(_enabled_channels & (1U<<ch))) {
330  // not enabled
331  return;
332  }
333  if (ch >= _max_channel) {
334  _max_channel = ch + 1;
335  }
336 
337  if (_output_mode == MODE_PWM_BRUSHED16KHZ) {
338  // map from the PWM range to 0 t0 100% duty cycle. For 16kHz
339  // this ends up being 0 to 500 pulse width in units of
340  // 125usec.
341  const uint32_t period_max = 1000000UL/(16000/8);
342  if (period_us <= _esc_pwm_min) {
343  period_us = 0;
344  } else if (period_us >= _esc_pwm_max) {
345  period_us = period_max;
346  } else {
347  uint32_t pdiff = period_us - _esc_pwm_min;
348  period_us = pdiff*period_max/(_esc_pwm_max - _esc_pwm_min);
349  }
350  }
351 
352  /*
353  only mark an update is needed if there has been a change, or we
354  are in oneshot mode. In oneshot mode we always need to send the
355  output
356  */
357  if (period_us != _period[ch] ||
358  _output_mode == MODE_PWM_ONESHOT) {
359  _period[ch] = period_us;
360  _need_update = true;
361  }
362 }

480 {
481  uint32_t now = AP_HAL::micros();
482 
483  if ((_enabled_channels & ((1U<<_servo_count)-1)) == 0) {
484  // no channels enabled
485  _arm_actuators(false);
486  goto update_pwm;
487  }
488 
489  // always send at least at 20Hz, otherwise the IO board may think
490  // we are dead
491  if (now - _last_output > 50000) {
492  _need_update = true;
493  }
494 
495  // check for PWM count changing. This can happen then the user changes BRD_PWM_COUNT
496  if (now - _last_config_us > 1000000) {
497  if (_pwm_fd != -1) {
498  ioctl(_pwm_fd, PWM_SERVO_GET_COUNT, (unsigned long)&_servo_count);
499  }
500  if (_alt_fd != -1) {
501  ioctl(_alt_fd, PWM_SERVO_GET_COUNT, (unsigned long)&_alt_servo_count);
502  }
503  _last_config_us = now;
504  }
505 
506  if (_need_update && _pwm_fd != -1) {
507  _need_update = false;
508  perf_begin(_perf_rcout);
509  uint8_t to_send = _max_channel<_servo_count?_max_channel:_servo_count;
510  if (_sbus_enabled) {
511  to_send = _max_channel;
512  }
513  if (to_send > 0) {
514  for (int i=to_send-1; i >= 0; i--) {
515  if (_period[i] == PWM_IGNORE_THIS_CHANNEL) {
516  to_send = i;
517  } else {
518  break;
519  }
520  }
521  }
522  if (to_send > 0) {
523  ::write(_pwm_fd, _period, to_send*sizeof(_period[0]));
524  }
525  if (_max_channel > _servo_count) {
526  // maybe send updates to alt_fd
527  if (_alt_fd != -1 && _alt_servo_count > 0) {
528  uint8_t n = _max_channel - _servo_count;
529  if (n > _alt_servo_count) {
530  n = _alt_servo_count;
531  }
532  if (n > 0) {
533  ::write(_alt_fd, &_period[_servo_count], n*sizeof(_period[0]));
534  }
535  }
536  }
537 
538  if(AP_BoardConfig::get_can_enable() >= 1)
539  {
540  // also publish to actuator_direct (UAVCAN is published via AP_UAVCAN)
541  _publish_actuators();
542 #if HAL_WITH_UAVCAN
543 
544  if(hal.can_mgr != nullptr)
545  {
546  AP_UAVCAN *ap_uc = hal.can_mgr->get_UAVCAN();
547  if(ap_uc != nullptr)
548  {
549  if(ap_uc->rc_out_sem_take())
550  {
551  for(uint8_t i = 0; i < _max_channel; i++)
552  {
553  ap_uc->rco_write(_period[i], i);
554  }
555 
556  bool armed = hal.util->get_soft_armed();
557  if (armed) {
558  ap_uc->rco_arm_actuators(true);
559  } else {
560  ap_uc->rco_arm_actuators(false);
561  }
562 
563  ap_uc->rc_out_sem_give();
564  }
565  }
566  }
567 #endif // HAL_WITH_UAVCAN
568  }
569 
570  perf_end(_perf_rcout);
571  _last_output = now;
572  }
573 
574 update_pwm:
575  for (uint8_t i=0; i<ORB_MULTI_MAX_INSTANCES; i++) {
576  bool rc_updated = false;
577  if (_outputs[i].pwm_sub >= 0 &&
578  orb_check(_outputs[i].pwm_sub, &rc_updated) == 0 &&
579  rc_updated) {
580  orb_copy(ORB_ID(actuator_outputs), _outputs[i].pwm_sub, &_outputs[i].outputs);

643 /*
644  setup output mode
645  */
646 void PX4RCOutput::set_output_mode(enum output_mode mode)
647 {
648  if (_output_mode == mode) {
649  // no change
650  return;
651  }
652  if (mode == MODE_PWM_ONESHOT) {
653  // when using oneshot we don't want the regular pulses. The
654  // best we can do with the current PX4Firmware code is ask for
655  // 1Hz. This does still produce pulses, but the trigger calls
656  // mean the timer is constantly reset, so no pulses are
657  // produced except when triggered by push() when the main loop
658  // is running
659  set_freq_fd(_pwm_fd, _rate_mask_main, 1, _rate_mask_main);
660  if (_alt_fd != -1) {
661  set_freq_fd(_alt_fd, _rate_mask_alt, 1, _rate_mask_alt);
662  }
663  }
664  _output_mode = mode;
665  switch (_output_mode) {
666  case MODE_PWM_ONESHOT:
667  ioctl(_pwm_fd, PWM_SERVO_SET_ONESHOT, 1);
668  if (_alt_fd != -1) {
669  ioctl(_alt_fd, PWM_SERVO_SET_ONESHOT, 1);
670  }
671  break;
672  case MODE_PWM_NORMAL:
673  ioctl(_pwm_fd, PWM_SERVO_SET_ONESHOT, 0);
674  if (_alt_fd != -1) {
675  ioctl(_alt_fd, PWM_SERVO_SET_ONESHOT, 0);
676  }
677  break;
678  case MODE_PWM_BRUSHED16KHZ:
679  // setup an 8MHz clock. This has the effect of scaling all outputs by 8x
680  ioctl(_pwm_fd, PWM_SERVO_SET_UPDATE_CLOCK, 8);
681  if (_alt_fd != -1) {
682  ioctl(_alt_fd, PWM_SERVO_SET_UPDATE_CLOCK, 8);
683  }
684  break;
685  }
686 }
