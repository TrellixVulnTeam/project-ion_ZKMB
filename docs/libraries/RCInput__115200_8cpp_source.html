38 void RCInput_115200::init()
39 {
40  fd = open(device_path, O_RDWR | O_NONBLOCK | O_CLOEXEC);
41  if (fd != -1) {
42  struct termios options;
43 
44  tcgetattr(fd, &options);
45 
46  cfsetispeed(&options, B115200);
47  cfsetospeed(&options, B115200);
48 
49  options.c_cflag &= ~(PARENB|CSTOPB|CSIZE);
50  options.c_cflag |= CS8;
51 
52  options.c_lflag &= ~(ICANON|ECHO|ECHOE|ISIG);
53  options.c_iflag &= ~(IXON|IXOFF|IXANY);
54  options.c_oflag &= ~OPOST;
55 
56  if (tcsetattr(fd, TCSANOW, &options) != 0) {
57  AP_HAL::panic("RCInput_UART: error configuring device: %s",
58  strerror(errno));
59  }
60 
61  tcflush(fd, TCIOFLUSH);
62  }

70 void RCInput_115200::_timer_tick(void)
71 {
72  if (fd == -1) {
73  return;
74  }
75 
76  // read up to 256 bytes at a time
77  uint8_t bytes[256];
78  int nread;
79 
80  fd_set fds;
81  struct timeval tv;
82 
83  FD_ZERO(&fds);
84  FD_SET(fd, &fds);
85 
86  tv.tv_sec = 0;
87  tv.tv_usec = 0;
88 
89  // check if any bytes are available
90  if (select(fd+1, &fds, nullptr, nullptr, &tv) != 1) {
91  return;
92  }
93 
94  // cope with there being a large number of pending bytes at
95  // the start and discard them
96  do {
97  nread = ::read(fd, bytes, sizeof(bytes));
98  } while (nread == sizeof(bytes));
99 
100  if (nread <= 0) {
101  return;
102  }
103  bool got_frame = false;
104 
105  if (decoder == DECODER_SYNC ||
106  decoder == DECODER_SRXL) {
107  // try srxl first as it has a 16 bit CRC
108  if (add_srxl_input(bytes, nread)) {
109  // lock immediately
110  decoder = DECODER_SRXL;
111  got_frame = true;
112  }
113  }
114 
115  if (decoder == DECODER_SYNC ||
116  decoder == DECODER_SUMD) {
117  // SUMD also has a 16 bit CRC
118  if (add_sumd_input(bytes, nread)) {
119  // lock immediately
120  decoder = DECODER_SUMD;
121  got_frame = true;
122  }
123  }
124 
125  if (decoder == DECODER_SYNC ||
126  decoder == DECODER_DSM) {
127  // process as DSM
128  if (add_dsm_input(bytes, nread)) {
129  dsm_count++;
130  if (dsm_count == 10) {
131  // we're confident
132  decoder = DECODER_DSM;
133  }
134  got_frame = true;
135  }
136  }
137 
138  if (decoder == DECODER_SYNC ||
139  decoder == DECODER_ST24) {
140  // process as st24
141  if (add_st24_input(bytes, nread)) {
142  st24_count++;
143  if (st24_count == 10) {
144  // we're confident
145  decoder = DECODER_ST24;
146  }
147  got_frame = true;
148  }
149  }
150 
151  uint32_t now = AP_HAL::millis();
152  if (got_frame) {
153  last_input_ms = now;
154  } else if (now - last_input_ms > 1000 && decoder != DECODER_SYNC) {
155  // start search again
156  decoder = DECODER_SYNC;
157  dsm_count = 0;
158  st24_count = 0;
159  }
160 }
