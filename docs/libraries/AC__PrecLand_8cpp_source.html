215 void AC_PrecLand::run_estimator(float rangefinder_alt_m, bool rangefinder_alt_valid)
216 {
217  const struct inertial_data_frame_s& inertial_data_delayed = _inertial_history.front();
218 
219  switch (_estimator_type) {
220  case ESTIMATOR_TYPE_RAW_SENSOR: {
221  // Return if there's any invalid velocity data
222  for (uint8_t i=0; i<_inertial_history.size(); i++) {
223  const struct inertial_data_frame_s& inertial_data = _inertial_history.peek(i);
224  if (!inertial_data.inertialNavVelocityValid) {
225  _target_acquired = false;
226  return;
227  }
228  }
229 
230  // Predict
231  if (target_acquired()) {
232  _target_pos_rel_est_NE.x -= inertial_data_delayed.inertialNavVelocity.x * inertial_data_delayed.dt;
233  _target_pos_rel_est_NE.y -= inertial_data_delayed.inertialNavVelocity.y * inertial_data_delayed.dt;
234  _target_vel_rel_est_NE.x = -inertial_data_delayed.inertialNavVelocity.x;
235  _target_vel_rel_est_NE.y = -inertial_data_delayed.inertialNavVelocity.y;
236  }
237 
238  // Update if a new LOS measurement is available
239  if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
240  _target_pos_rel_est_NE.x = _target_pos_rel_meas_NED.x;
241  _target_pos_rel_est_NE.y = _target_pos_rel_meas_NED.y;
242  _target_vel_rel_est_NE.x = -inertial_data_delayed.inertialNavVelocity.x;
243  _target_vel_rel_est_NE.y = -inertial_data_delayed.inertialNavVelocity.y;
244 
245  _last_update_ms = AP_HAL::millis();
246  _target_acquired = true;
247  }
248 
249  // Output prediction
250  if (target_acquired()) {
251  run_output_prediction();
252  }
253  break;
254  }
255  case ESTIMATOR_TYPE_KALMAN_FILTER: {
256  // Predict
257  if (target_acquired()) {
258  const float& dt = inertial_data_delayed.dt;
259  const Vector3f& vehicleDelVel = inertial_data_delayed.correctedVehicleDeltaVelocityNED;
260 
261  _ekf_x.predict(dt, -vehicleDelVel.x, _accel_noise*dt);
262  _ekf_y.predict(dt, -vehicleDelVel.y, _accel_noise*dt);
263  }
264 
265  // Update if a new LOS measurement is available
266  if (construct_pos_meas_using_rangefinder(rangefinder_alt_m, rangefinder_alt_valid)) {
267  float xy_pos_var = sq(_target_pos_rel_meas_NED.z*(0.01f + 0.01f*_ahrs.get_gyro().length()) + 0.02f);
268  if (!target_acquired()) {
269  // reset filter state
270  if (inertial_data_delayed.inertialNavVelocityValid) {
271  _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, -inertial_data_delayed.inertialNavVelocity.x, sq(2.0f));
272  _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, -inertial_data_delayed.inertialNavVelocity.y, sq(2.0f));
273  } else {
274  _ekf_x.init(_target_pos_rel_meas_NED.x, xy_pos_var, 0.0f, sq(10.0f));
275  _ekf_y.init(_target_pos_rel_meas_NED.y, xy_pos_var, 0.0f, sq(10.0f));
276  }
277  _last_update_ms = AP_HAL::millis();
278  _target_acquired = true;
279  } else {
280  float NIS_x = _ekf_x.getPosNIS(_target_pos_rel_meas_NED.x, xy_pos_var);
281  float NIS_y = _ekf_y.getPosNIS(_target_pos_rel_meas_NED.y, xy_pos_var);
282  if (MAX(NIS_x, NIS_y) < 3.0f || _outlier_reject_count >= 3) {
283  _outlier_reject_count = 0;
284  _ekf_x.fusePos(_target_pos_rel_meas_NED.x, xy_pos_var);
285  _ekf_y.fusePos(_target_pos_rel_meas_NED.y, xy_pos_var);
286  _last_update_ms = AP_HAL::millis();
287  _target_acquired = true;
288  } else {
289  _outlier_reject_count++;
290  }
291  }
292  }
293 
294  // Output prediction
295  if (target_acquired()) {
296  _target_pos_rel_est_NE.x = _ekf_x.getPos();
297  _target_pos_rel_est_NE.y = _ekf_y.getPos();
298  _target_vel_rel_est_NE.x = _ekf_x.getVel();
299  _target_vel_rel_est_NE.y = _ekf_y.getVel();
300 
301  run_output_prediction();
302  }
303  break;
304  }
305  }

330 bool AC_PrecLand::construct_pos_meas_using_rangefinder(float rangefinder_alt_m, bool rangefinder_alt_valid)
331 {
332  Vector3f target_vec_unit_body;
333  if (retrieve_los_meas(target_vec_unit_body)) {
334  const struct inertial_data_frame_s& inertial_data_delayed = _inertial_history.front();
335 
336  Vector3f target_vec_unit_ned = inertial_data_delayed.Tbn * target_vec_unit_body;
337  bool target_vec_valid = target_vec_unit_ned.z > 0.0f;
338  bool alt_valid = (rangefinder_alt_valid && rangefinder_alt_m > 0.0f) || (_backend->distance_to_target() > 0.0f);
339  if (target_vec_valid && alt_valid) {
340  float dist, alt;
341  if (_backend->distance_to_target() > 0.0f) {
342  dist = _backend->distance_to_target();
343  alt = dist * target_vec_unit_ned.z;
344  } else {
345  alt = MAX(rangefinder_alt_m, 0.0f);
346  dist = alt / target_vec_unit_ned.z;
347  }
348 
349  // Compute camera position relative to IMU
350  Vector3f accel_body_offset = _ahrs.get_ins().get_imu_pos_offset(_ahrs.get_primary_accel_index());
351  Vector3f cam_pos_ned = inertial_data_delayed.Tbn * (_cam_offset.get() - accel_body_offset);
352 
353  // Compute target position relative to IMU
354  _target_pos_rel_meas_NED = Vector3f(target_vec_unit_ned.x*dist, target_vec_unit_ned.y*dist, alt) + cam_pos_ned;
355  return true;
356  }
357  }
358  return false;

361 void AC_PrecLand::run_output_prediction()
362 {
363  _target_pos_rel_out_NE = _target_pos_rel_est_NE;
364  _target_vel_rel_out_NE = _target_vel_rel_est_NE;
365 
366  // Predict forward from delayed time horizon
367  for (uint8_t i=1; i<_inertial_history.size(); i++) {
368  const struct inertial_data_frame_s& inertial_data = _inertial_history.peek(i);
369  _target_vel_rel_out_NE.x -= inertial_data.correctedVehicleDeltaVelocityNED.x;
370  _target_vel_rel_out_NE.y -= inertial_data.correctedVehicleDeltaVelocityNED.y;
371  _target_pos_rel_out_NE.x += _target_vel_rel_out_NE.x * inertial_data.dt;
372  _target_pos_rel_out_NE.y += _target_vel_rel_out_NE.y * inertial_data.dt;
373  }
374 
375  const Matrix3f& Tbn = _inertial_history.peek(_inertial_history.size()-1).Tbn;
376  Vector3f accel_body_offset = _ahrs.get_ins().get_imu_pos_offset(_ahrs.get_primary_accel_index());
377 
378  // Apply position correction for CG offset from IMU
379  Vector3f imu_pos_ned = Tbn * accel_body_offset;
380  _target_pos_rel_out_NE.x += imu_pos_ned.x;
381  _target_pos_rel_out_NE.y += imu_pos_ned.y;
382 
383  // Apply position correction for body-frame horizontal camera offset from CG, so that vehicle lands lens-to-target
384  Vector3f cam_pos_horizontal_ned = Tbn * Vector3f(_cam_offset.get().x, _cam_offset.get().y, 0);
385  _target_pos_rel_out_NE.x -= cam_pos_horizontal_ned.x;
386  _target_pos_rel_out_NE.y -= cam_pos_horizontal_ned.y;
387 
388  // Apply velocity correction for IMU offset from CG
389  Vector3f vel_ned_rel_imu = Tbn * (_ahrs.get_gyro() % (-accel_body_offset));
390  _target_vel_rel_out_NE.x -= vel_ned_rel_imu.x;
391  _target_vel_rel_out_NE.y -= vel_ned_rel_imu.y;
392 
393  // Apply land offset
394  Vector3f land_ofs_ned_m = _ahrs.get_rotation_body_to_ned() * Vector3f(_land_ofs_cm_x,_land_ofs_cm_y,0) * 0.01f;
395  _target_pos_rel_out_NE.x += land_ofs_ned_m.x;
396  _target_pos_rel_out_NE.y += land_ofs_ned_m.y;
