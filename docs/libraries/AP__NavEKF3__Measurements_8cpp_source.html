580 // calculate filtered offset between GPS height measurement and EKF height estimate
581 // offset should be subtracted from GPS measurement to match filter estimate
582 // offset is used to switch reversion to GPS from alternate height data source
583 void NavEKF3_core::calcFiltGpsHgtOffset()
584 {
585  // Estimate the WGS-84 height of the EKF's origin using a Bayes filter
586 
587  // calculate the variance of our a-priori estimate of the ekf origin height
588  float deltaTime = constrain_float(0.001f * (imuDataDelayed.time_ms - lastOriginHgtTime_ms), 0.0f, 1.0f);
589  if (activeHgtSource == HGT_SOURCE_BARO) {
590  // Use the baro drift rate
591  const float baroDriftRate = 0.05f;
592  ekfOriginHgtVar += sq(baroDriftRate * deltaTime);
593  } else if (activeHgtSource == HGT_SOURCE_RNG) {
594  // use the worse case expected terrain gradient and vehicle horizontal speed
595  const float maxTerrGrad = 0.25f;
596  ekfOriginHgtVar += sq(maxTerrGrad * norm(stateStruct.velocity.x , stateStruct.velocity.y) * deltaTime);
597  } else if (activeHgtSource == HGT_SOURCE_GPS) {
598  // by definition we are using GPS height as the EKF datum in this mode
599  // so cannot run this filter
600  return;
601  }
602  lastOriginHgtTime_ms = imuDataDelayed.time_ms;
603 
604  // calculate the observation variance assuming EKF error relative to datum is independant of GPS observation error
605  // when not using GPS as height source
606  float originHgtObsVar = sq(gpsHgtAccuracy) + P[8][8];
607 
608  // calculate the correction gain
609  float gain = ekfOriginHgtVar / (ekfOriginHgtVar + originHgtObsVar);
610 
611  // calculate the innovation
612  float innovation = - stateStruct.position.z - gpsDataDelayed.hgt;
613 
614  // check the innovation variance ratio
615  float ratio = sq(innovation) / (ekfOriginHgtVar + originHgtObsVar);
616 
617  // correct the EKF origin and variance estimate if the innovation variance ratio is < 5-sigma
618  if (ratio < 5.0f) {
619  EKF_origin.alt -= (int)(100.0f * gain * innovation);
620  ekfOriginHgtVar -= MAX(gain * ekfOriginHgtVar , 0.0f);
621  }
622 }
