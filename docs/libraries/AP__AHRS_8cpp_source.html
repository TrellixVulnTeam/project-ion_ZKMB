141 // return airspeed estimate if available
142 bool AP_AHRS::airspeed_estimate(float *airspeed_ret) const
143 {
144  if (airspeed_sensor_enabled()) {
145  *airspeed_ret = _airspeed->get_airspeed();
146  if (_wind_max > 0 && _gps.status() >= AP_GPS::GPS_OK_FIX_2D) {
147  // constrain the airspeed by the ground speed
148  // and AHRS_WIND_MAX
149  float gnd_speed = _gps.ground_speed();
150  float true_airspeed = *airspeed_ret * get_EAS2TAS();
151  true_airspeed = constrain_float(true_airspeed,
152  gnd_speed - _wind_max,
153  gnd_speed + _wind_max);
154  *airspeed_ret = true_airspeed / get_EAS2TAS();
155  }
156  return true;
157  }
158  return false;
159 }

188 // return a ground speed estimate in m/s
189 Vector2f AP_AHRS::groundspeed_vector(void)
190 {
191  // Generate estimate of ground speed vector using air data system
192  Vector2f gndVelADS;
193  Vector2f gndVelGPS;
194  float airspeed;
195  bool gotAirspeed = airspeed_estimate_true(&airspeed);
196  bool gotGPS = (_gps.status() >= AP_GPS::GPS_OK_FIX_2D);
197  if (gotAirspeed) {
198  Vector3f wind = wind_estimate();
199  Vector2f wind2d = Vector2f(wind.x, wind.y);
200  Vector2f airspeed_vector = Vector2f(cosf(yaw), sinf(yaw)) * airspeed;
201  gndVelADS = airspeed_vector - wind2d;
202  }
203 
204  // Generate estimate of ground speed vector using GPS
205  if (gotGPS) {
206  float cog = radians(_gps.ground_course_cd()*0.01f);
207  gndVelGPS = Vector2f(cosf(cog), sinf(cog)) * _gps.ground_speed();
208  }
209  // If both ADS and GPS data is available, apply a complementary filter
210  if (gotAirspeed && gotGPS) {
211  // The LPF is applied to the GPS and the HPF is applied to the air data estimate
212  // before the two are summed
213  //Define filter coefficients
214  // alpha and beta must sum to one
215  // beta = dt/Tau, where
216  // dt = filter time step (0.1 sec if called by nav loop)
217  // Tau = cross-over time constant (nominal 2 seconds)
218  // More lag on GPS requires Tau to be bigger, less lag allows it to be smaller
219  // To-Do - set Tau as a function of GPS lag.
220  const float alpha = 1.0f - beta;
221  // Run LP filters
222  _lp = gndVelGPS * beta + _lp * alpha;
223  // Run HP filters
224  _hp = (gndVelADS - _lastGndVelADS) + _hp * alpha;
225  // Save the current ADS ground vector for the next time step
226  _lastGndVelADS = gndVelADS;
227  // Sum the HP and LP filter outputs
228  return _hp + _lp;
229  }
230  // Only ADS data is available return ADS estimate
231  if (gotAirspeed && !gotGPS) {
232  return gndVelADS;
233  }
234  // Only GPS data is available so return GPS estimate
235  if (!gotAirspeed && gotGPS) {
236  return gndVelGPS;
237  }
238  return Vector2f(0.0f, 0.0f);
239 }

241 /*
242  calculate sin and cos of roll/pitch/yaw from a body_to_ned rotation matrix
243  */
244 void AP_AHRS::calc_trig(const Matrix3f &rot,
245  float &cr, float &cp, float &cy,
246  float &sr, float &sp, float &sy) const
247 {
248  Vector2f yaw_vector;
249 
250  yaw_vector.x = rot.a.x;
251  yaw_vector.y = rot.b.x;
252  if (fabsf(yaw_vector.x) > 0 ||
253  fabsf(yaw_vector.y) > 0) {
254  yaw_vector.normalize();
255  }
256  sy = constrain_float(yaw_vector.y, -1.0, 1.0);
257  cy = constrain_float(yaw_vector.x, -1.0, 1.0);
258 
259  // sanity checks
260  if (yaw_vector.is_inf() || yaw_vector.is_nan()) {
261  sy = 0.0f;
262  cy = 1.0f;
263  }
264 
265  float cx2 = rot.c.x * rot.c.x;
266  if (cx2 >= 1.0f) {
267  cp = 0;
268  cr = 1.0f;
269  } else {
270  cp = safe_sqrt(1 - cx2);
271  cr = rot.c.z / cp;
272  }
273  cp = constrain_float(cp, 0, 1.0);
274  cr = constrain_float(cr, -1.0, 1.0); // this relies on constrain_float() of infinity doing the right thing
275 
276  sp = -rot.c.x;
277 
278  if (!is_zero(cp)) {
279  sr = rot.c.y / cp;
280  }
281 
282  if (is_zero(cp) || isinf(cr) || isnan(cr) || isinf(sr) || isnan(sr)) {
283  float r, p, y;
284  rot.to_euler(&r, &p, &y);
285  cr = cosf(r);
286  sr = sinf(r);
287  }
288 }
