48 
49  // rotational air resistance
50  Vector3f rot_accel = -gyro * radians(400) / terminal_rotation_rate;

58  lift_accel = air_resistance_at_climb_rate + GRAVITY_MSS * dcm.c.z;
59  }
60 
61  accel_body = Vector3f(0, 0, -lift_accel);

63
