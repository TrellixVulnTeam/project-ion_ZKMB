604 // Initialise the filter
605 bool NavEKF3::InitialiseFilter(void)
606 {
607  if (_enable == 0) {
608  return false;
609  }
610 
611  imuSampleTime_us = AP_HAL::micros64();
612 
613  if (core == nullptr) {
614 
615  // see if we will be doing logging
616  DataFlash_Class *dataflash = DataFlash_Class::instance();
617  if (dataflash != nullptr) {
618  logging.enabled = dataflash->log_replay();
619  }
620 
621  // don't run multiple filters for 1 IMU
622  const AP_InertialSensor &ins = _ahrs->get_ins();
623  uint8_t mask = (1U<<ins.get_accel_count())-1;
624  _imuMask.set(_imuMask.get() & mask);
625 
626  // initialise the setup variables
627  for (uint8_t i=0; i<7; i++) {
628  coreSetupRequired[i] = false;
629  coreImuIndex[i] = 0;
630  }
631  num_cores = 0;
632 
633  // count IMUs from mask
634  for (uint8_t i=0; i<7; i++) {
635  if (_imuMask & (1U<<i)) {
636  coreSetupRequired[num_cores] = true;
637  coreImuIndex[num_cores] = i;
638  num_cores++;
639  }
640  }
641 
642  // check if there is enough memory to create the EKF cores
643  if (hal.util->available_memory() < sizeof(NavEKF3_core)*num_cores + 4096) {
644  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "NavEKF3: not enough memory");
645  _enable.set(0);
646  return false;
647  }
648 
649  // create the cores
650  core = new NavEKF3_core[num_cores];
651  if (core == nullptr) {
652  _enable.set(0);
653  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "NavEKF3: allocation failed");
654  return false;
655  }
656 
657  }
658 
659  // Set up any cores that have been created
660  // This specifies the IMU to be used and creates the data buffers
661  // If we are unble to set up a core, return false and try again next time the function is called
662  bool core_setup_success = true;
663  for (uint8_t core_index=0; core_index<num_cores; core_index++) {
664  if (coreSetupRequired[core_index]) {
665  coreSetupRequired[core_index] = !core[core_index].setup_core(this, coreImuIndex[core_index], core_index);
666  if (coreSetupRequired[core_index]) {
667  core_setup_success = false;
668  }
669  }
670  }
671  // exit with failure if any cores could not be setup
672  if (!core_setup_success) {
673  return false;
674  }
675 
676  // Set the primary initially to be the lowest index
677  primary = 0;
678 
679  // initialise the cores. We return success only if all cores
680  // initialise successfully
681  bool ret = true;
682  for (uint8_t i=0; i<num_cores; i++) {
683  ret &= core[i].InitialiseFilterBootstrap();
684  }
685 
686  // zero the structs used capture reset events
687  memset(&yaw_reset_data, 0, sizeof(yaw_reset_data));
688  memset(&pos_reset_data, 0, sizeof(pos_reset_data));
689  memset(&pos_down_reset_data, 0, sizeof(pos_down_reset_data));
690 
691  check_log_write();
692  return ret;
693 }

695 // Update Filter States - this should be called whenever new IMU data is available
696 void NavEKF3::UpdateFilter(void)
697 {
698  if (!core) {
699  return;
700  }
701 
702  imuSampleTime_us = AP_HAL::micros64();
703 
704  const AP_InertialSensor &ins = _ahrs->get_ins();
705 
706  bool statePredictEnabled[num_cores];
707  for (uint8_t i=0; i<num_cores; i++) {
708  // if the previous core has only recently finished a new state prediction cycle, then
709  // don't start a new cycle to allow time for fusion operations to complete if the update
710  // rate is higher than 200Hz
711  if ((i > 0) && (core[i-1].getFramesSincePredict() < 2) && (ins.get_sample_rate() > 200)) {
712  statePredictEnabled[i] = false;
713  } else {
714  statePredictEnabled[i] = true;
715  }
716  core[i].UpdateFilter(statePredictEnabled[i]);
717  }
718 
719  // If the current core selected has a bad error score or is unhealthy, switch to a healthy core with the lowest fault score
720  // Don't start running the check until the primary core has started returned healthy for at least 10 seconds to avoid switching
721  // due to initial alignment fluctuations and race conditions
722  if (!runCoreSelection) {
723  static uint64_t lastUnhealthyTime_us = 0;
724  if (!core[primary].healthy() || lastUnhealthyTime_us == 0) {
725  lastUnhealthyTime_us = imuSampleTime_us;
726  }
727  runCoreSelection = (imuSampleTime_us - lastUnhealthyTime_us) > 1E7;
728  }
729  float primaryErrorScore = core[primary].errorScore();
730  if ((primaryErrorScore > 1.0f || !core[primary].healthy()) && runCoreSelection) {
731  float lowestErrorScore = 0.67f * primaryErrorScore;
732  uint8_t newPrimaryIndex = primary; // index for new primary
733  for (uint8_t coreIndex=0; coreIndex<num_cores; coreIndex++) {
734 
735  if (coreIndex != primary) {
736  // an alternative core is available for selection only if healthy and if states have been updated on this time step
737  bool altCoreAvailable = core[coreIndex].healthy() && statePredictEnabled[coreIndex];
738 
739  // If the primary core is unhealthy and another core is available, then switch now
740  // If the primary core is still healthy,then switching is optional and will only be done if
741  // a core with a significantly lower error score can be found
742  float altErrorScore = core[coreIndex].errorScore();
743  if (altCoreAvailable && (!core[primary].healthy() || altErrorScore < lowestErrorScore)) {
744  newPrimaryIndex = coreIndex;
745  lowestErrorScore = altErrorScore;
746  }
747  }
748  }
749  // update the yaw and position reset data to capture changes due to the lane switch
750  if (newPrimaryIndex != primary) {
751  updateLaneSwitchYawResetData(newPrimaryIndex, primary);
752  updateLaneSwitchPosResetData(newPrimaryIndex, primary);
753  updateLaneSwitchPosDownResetData(newPrimaryIndex, primary);
754  primary = newPrimaryIndex;
755  }
756  }
757 
758  check_log_write();
759 }
