613  // constrain the correction between 0 and gndBaroInnovFloor+gndMaxBaroErr

653  for (uint8_t i= 0; i<=stateIndexLim; i++) {
654  for (uint8_t j= 0; j<=stateIndexLim; j++)

672  }

675  ForceSymmetry();
676  ConstrainVariances();
677 
678  // update states and renormalise the quaternions
679  for (uint8_t i = 0; i<=stateIndexLim; i++) {
680  statesArray[i] = statesArray[i] - Kfusion[i] * innovVelPos[obsIndex];
681  }
682  stateStruct.quat.normalize();
683 
684  // record good fusion status
685  if (obsIndex == 0) {
686  faultStatus.bad_nvel = false;
687  } else if (obsIndex == 1) {

689  } else if (obsIndex == 2) {

691  } else if (obsIndex == 3) {

693  } else if (obsIndex == 4) {

697  }
698  } else {

726 // select the height measurement to be fused from the available baro, range finder and GPS sources
727 void NavEKF3_core::selectHeightForFusion()
728 {
729  // Read range finder data and check for new data in the buffer
730  // This data is used by both height and optical flow fusion processing
731  readRangeFinder();
732  rangeDataToFuse = storedRange.recall(rangeDataDelayed,imuDataDelayed.time_ms);
733 
734  // correct range data for the body frame position offset relative to the IMU
735  // the corrected reading is the reading that would have been taken if the sensor was
736  // co-located with the IMU
737  if (rangeDataToFuse) {
738  Vector3f posOffsetBody = frontend->_rng.get_pos_offset(rangeDataDelayed.sensor_idx) - accelPosOffset;
739  if (!posOffsetBody.is_zero()) {
740  Vector3f posOffsetEarth = prevTnb.mul_transpose(posOffsetBody);
741  rangeDataDelayed.rng += posOffsetEarth.z / prevTnb.c.z;
742  }
743  }
744 
745  // read baro height data from the sensor and check for new data in the buffer
746  readBaroData();
747  baroDataToFuse = storedBaro.recall(baroDataDelayed, imuDataDelayed.time_ms);
748 
749  // select height source
750  if (((frontend->_useRngSwHgt > 0) || (frontend->_altSource == 1)) && (imuSampleTime_ms - rngValidMeaTime_ms < 500)) {
751  if (frontend->_altSource == 1) {
752  // always use range finder
753  activeHgtSource = HGT_SOURCE_RNG;
754  } else {
755  // determine if we are above or below the height switch region
756  float rangeMaxUse = 1e-4f * (float)frontend->_rng.max_distance_cm_orient(ROTATION_PITCH_270) * (float)frontend->_useRngSwHgt;
757  bool aboveUpperSwHgt = (terrainState - stateStruct.position.z) > rangeMaxUse;
758  bool belowLowerSwHgt = (terrainState - stateStruct.position.z) < 0.7f * rangeMaxUse;
759 
760  // If the terrain height is consistent and we are moving slowly, then it can be
761  // used as a height reference in combination with a range finder
762  // apply a hysteresis to the speed check to prevent rapid switching
763  bool dontTrustTerrain, trustTerrain;
764  if (filterStatus.flags.horiz_vel) {
765  // We can use the velocity estimate
766  float horizSpeed = norm(stateStruct.velocity.x, stateStruct.velocity.y);
767  dontTrustTerrain = (horizSpeed > frontend->_useRngSwSpd) || !terrainHgtStable;
768  float trust_spd_trigger = MAX((frontend->_useRngSwSpd - 1.0f),(frontend->_useRngSwSpd * 0.5f));
769  trustTerrain = (horizSpeed < trust_spd_trigger) && terrainHgtStable;
770  } else {
771  // We can't use the velocity estimate
772  dontTrustTerrain = !terrainHgtStable;
773  trustTerrain = terrainHgtStable;
774  }
775 
776  /*
777  * Switch between range finder and primary height source using height above ground and speed thresholds with
778  * hysteresis to avoid rapid switching. Using range finder for height requires a consistent terrain height
779  * which cannot be assumed if the vehicle is moving horizontally.
780  */
781  if ((aboveUpperSwHgt || dontTrustTerrain) && (activeHgtSource == HGT_SOURCE_RNG)) {
782  // cannot trust terrain or range finder so stop using range finder height
783  if (frontend->_altSource == 0) {
784  activeHgtSource = HGT_SOURCE_BARO;
785  } else if (frontend->_altSource == 2) {
786  activeHgtSource = HGT_SOURCE_GPS;
787  }
788  } else if (belowLowerSwHgt && trustTerrain && (activeHgtSource != HGT_SOURCE_RNG)) {
789  // reliable terrain and range finder so start using range finder height
790  activeHgtSource = HGT_SOURCE_RNG;
791  }
792  }
793  } else if ((frontend->_altSource == 2) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) < 500) && validOrigin && gpsAccuracyGood) {
794  activeHgtSource = HGT_SOURCE_GPS;
795  } else if ((frontend->_altSource == 3) && validOrigin && rngBcnGoodToAlign) {
796  activeHgtSource = HGT_SOURCE_BCN;
797  } else {
798  activeHgtSource = HGT_SOURCE_BARO;
799  }
800 
801  // Use Baro alt as a fallback if we lose range finder or GPS
802  bool lostRngHgt = ((activeHgtSource == HGT_SOURCE_RNG) && ((imuSampleTime_ms - rngValidMeaTime_ms) > 500));
803  bool lostGpsHgt = ((activeHgtSource == HGT_SOURCE_GPS) && ((imuSampleTime_ms - lastTimeGpsReceived_ms) > 2000));
804  if (lostRngHgt || lostGpsHgt) {
805  activeHgtSource = HGT_SOURCE_BARO;
806  }
807 
808  // if there is new baro data to fuse, calculate filtered baro data required by other processes
809  if (baroDataToFuse) {
810  // calculate offset to baro data that enables us to switch to Baro height use during operation
811  if (activeHgtSource != HGT_SOURCE_BARO) {
812  calcFiltBaroOffset();
813  }
814  // filtered baro data used to provide a reference for takeoff
815  // it is is reset to last height measurement on disarming in performArmingChecks()
816  if (!getTakeoffExpected()) {
817  const float gndHgtFiltTC = 0.5f;
818  const float dtBaro = frontend->hgtAvg_ms*1.0e-3f;
819  float alpha = constrain_float(dtBaro / (dtBaro+gndHgtFiltTC),0.0f,1.0f);
820  meaHgtAtTakeOff += (baroDataDelayed.hgt-meaHgtAtTakeOff)*alpha;
821  }
822  }
823 
824  // calculate offset to GPS height data that enables us to switch to GPS height during operation
825  if (gpsDataToFuse && (activeHgtSource != HGT_SOURCE_GPS)) {
826  calcFiltGpsHgtOffset();
827  }
828 
829  // Select the height measurement source
830  if (rangeDataToFuse && (activeHgtSource == HGT_SOURCE_RNG)) {
831  // using range finder data
832  // correct for tilt using a flat earth model
833  if (prevTnb.c.z >= 0.7) {
834  // calculate height above ground
835  hgtMea = MAX(rangeDataDelayed.rng * prevTnb.c.z, rngOnGnd);
836  // correct for terrain position relative to datum
837  hgtMea -= terrainState;
838  // enable fusion
839  fuseHgtData = true;
840  // set the observation noise
841  posDownObsNoise = sq(constrain_float(frontend->_rngNoise, 0.1f, 10.0f));
842  // add uncertainty created by terrain gradient and vehicle tilt
843  posDownObsNoise += sq(rangeDataDelayed.rng * frontend->_terrGradMax) * MAX(0.0f , (1.0f - sq(prevTnb.c.z)));
844  } else {
845  // disable fusion if tilted too far
846  fuseHgtData = false;
847  }
848  } else if (gpsDataToFuse && (activeHgtSource == HGT_SOURCE_GPS)) {
849  // using GPS data
850  hgtMea = gpsDataDelayed.hgt;
851  // enable fusion
852  fuseHgtData = true;
853  // set the observation noise using receiver reported accuracy or the horizontal noise scaled for typical VDOP/HDOP ratio
854  if (gpsHgtAccuracy > 0.0f) {
855  posDownObsNoise = sq(constrain_float(gpsHgtAccuracy, 1.5f * frontend->_gpsHorizPosNoise, 100.0f));
856  } else {
857  posDownObsNoise = sq(constrain_float(1.5f * frontend->_gpsHorizPosNoise, 0.1f, 10.0f));
858  }
859  } else if (baroDataToFuse && (activeHgtSource == HGT_SOURCE_BARO)) {
860  // using Baro data
861  hgtMea = baroDataDelayed.hgt - baroHgtOffset;
862  // enable fusion
863  fuseHgtData = true;
864  // set the observation noise
865  posDownObsNoise = sq(constrain_float(frontend->_baroAltNoise, 0.1f, 10.0f));
866  // reduce weighting (increase observation noise) on baro if we are likely to be in ground effect
867  if (getTakeoffExpected() || getTouchdownExpected()) {
868  posDownObsNoise *= frontend->gndEffectBaroScaler;
869  }
870  // If we are in takeoff mode, the height measurement is limited to be no less than the measurement at start of takeoff
871  // This prevents negative baro disturbances due to copter downwash corrupting the EKF altitude during initial ascent
872  if (motorsArmed && getTakeoffExpected()) {
873  hgtMea = MAX(hgtMea, meaHgtAtTakeOff);
874  }
875  } else {
876  fuseHgtData = false;
877  }
878 
879  // If we haven't fused height data for a while, then declare the height data as being timed out
880  // set timeout period based on whether we have vertical GPS velocity available to constrain drift
881  hgtRetryTime_ms = (useGpsVertVel && !velTimeout) ? frontend->hgtRetryTimeMode0_ms : frontend->hgtRetryTimeMode12_ms;
882  if (imuSampleTime_ms - lastHgtPassTime_ms > hgtRetryTime_ms) {
883  hgtTimeout = true;
884  } else {
885  hgtTimeout = false;
886  }
887 }
