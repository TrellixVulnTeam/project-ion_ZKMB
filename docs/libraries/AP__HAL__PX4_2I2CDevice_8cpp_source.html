63  implement wrapper for PX4 I2C driver

65 bool PX4_I2C::do_transfer(uint8_t address, const uint8_t *send, uint32_t send_len, uint8_t *recv, uint32_t recv_len)
66 {
67  set_address(address);
68  if (!init_done) {
69  init_done = true;
70  // we do late init() so we can setup the device paths
71  snprintf(devname, sizeof(devname), "AP_I2C_%u", instance);
72  snprintf(devpath, sizeof(devpath), "/dev/api2c%u", instance);
73  init_ok = (init() == OK);
74  if (init_ok) {
75  instance++;
76  }
77  }
78  if (!init_ok) {
79  return false;
80  }
81  /*
82  splitting the transfer() into two pieces avoids a stop condition
83  with SCL low which is not supported on some devices (such as
84  LidarLite blue label)
85  */
86  if (send && send_len) {
87  if (transfer(send, send_len, nullptr, 0) != OK) {
88  return false;
89  }
90  }
91  if (recv && recv_len) {
92  if (transfer(nullptr, 0, recv, recv_len) != OK) {
93  return false;
94  }
95  }
96  return true;
97 }
