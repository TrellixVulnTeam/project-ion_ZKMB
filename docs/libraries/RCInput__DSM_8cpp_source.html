43 void RCInput_DSM::_timer_tick(void)
44 {
45  if (device_path == nullptr) {
46  return;
47  }
48  int ret;
49  /*
50  we defer the open to the timer tick to ensure all RPC calls are
51  made in the same thread
52  */
53  if (fd == -1) {
54  ret = qflight_UART_open(device_path, &fd);
55  if (ret == 0) {
56  printf("Opened DSM input %s fd=%d\n", device_path, (int)fd);
57  fflush(stdout);
58  qflight_UART_set_baudrate(fd, 115200);
59  }
60  }
61  if (fd != -1) {
62  uint8_t bytes[16];
63  int32_t nread;
64  ret = qflight_UART_read(fd, bytes, sizeof(bytes), &nread);
65  if (ret == 0 && nread > 0) {
66  // printf("Read %u DSM bytes at %u\n", (unsigned)nread, AP_HAL::millis());
67  fflush(stdout);
68  add_dsm_input(bytes, nread);
69  }
70  }
71 }
