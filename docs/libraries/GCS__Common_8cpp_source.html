66 void
67 GCS_MAVLINK::setup_uart(const AP_SerialManager& serial_manager, AP_SerialManager::SerialProtocol protocol, uint8_t instance)
68 {
69  serialmanager_p = &serial_manager;
70 
71  // search for serial port
72 
73  AP_HAL::UARTDriver *uart;
74  uart = serial_manager.find_serial(protocol, instance);
75  if (uart == nullptr) {
76  // return immediately if not found
77  return;
78  }
79 
80  // get associated mavlink channel
81  mavlink_channel_t mav_chan;
82  if (!serial_manager.get_mavlink_channel(protocol, instance, mav_chan)) {
83  // return immediately in unlikely case mavlink channel cannot be found
84  return;
85  }
86 
87  /*
88  Now try to cope with SiK radios that may be stuck in bootloader
89  mode because CTS was held while powering on. This tells the
90  bootloader to wait for a firmware. It affects any SiK radio with
91  CTS connected that is externally powered. To cope we send 0x30
92  0x20 at 115200 on startup, which tells the bootloader to reset
93  and boot normally
94  */
95  uart->begin(115200);
96  AP_HAL::UARTDriver::flow_control old_flow_control = uart->get_flow_control();
97  uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
98  for (uint8_t i=0; i<3; i++) {
99  hal.scheduler->delay(1);
100  uart->write(0x30);
101  uart->write(0x20);
102  }
103  // since tcdrain() and TCSADRAIN may not be implemented...
104  hal.scheduler->delay(1);
105 
106  uart->set_flow_control(old_flow_control);
107 
108  // now change back to desired baudrate
109  uart->begin(serial_manager.find_baudrate(protocol, instance));
110 
111  // and init the gcs instance
112  init(uart, mav_chan);
113 
114  AP_SerialManager::SerialProtocol mavlink_protocol = serialmanager_p->get_mavlink_protocol(mav_chan);
115  mavlink_status_t *status = mavlink_get_channel_status(chan);
116  if (status == nullptr) {
117  return;
118  }
119 
120  if (mavlink_protocol == AP_SerialManager::SerialProtocol_MAVLink2) {
121  // load signing key
122  load_signing_key();
123 
124  if (status->signing == nullptr) {
125  // if signing is off start by sending MAVLink1.
126  status->flags |= MAVLINK_STATUS_FLAG_OUT_MAVLINK1;
127  }
128  // announce that we are MAVLink2 capable
129  hal.util->set_capabilities(MAV_PROTOCOL_CAPABILITY_MAVLINK2);
130  } else if (status) {
131  // user has asked to only send MAVLink1
132  status->flags |= MAVLINK_STATUS_FLAG_OUT_MAVLINK1;
133  }
134 
135  if (chan == MAVLINK_COMM_0) {
136  // Always start with MAVLink1 on first port for now, to allow for recovery
137  // after experiments with MAVLink2
138  status->flags |= MAVLINK_STATUS_FLAG_OUT_MAVLINK1;
139  }
140 }

146 void
147 GCS_MAVLINK::queued_param_send()
148 {
149  if (!initialised || _queued_parameter == nullptr) {
150  return;
151  }
152 
153  uint16_t bytes_allowed;
154  uint8_t count;
155  uint32_t tnow = AP_HAL::millis();
156 
157  // use at most 30% of bandwidth on parameters. The constant 26 is
158  // 1/(1000 * 1/8 * 0.001 * 0.3)
159  bytes_allowed = 57 * (tnow - _queued_parameter_send_time_ms) * 26;
160  if (bytes_allowed > comm_get_txspace(chan)) {
161  bytes_allowed = comm_get_txspace(chan);
162  }
163  count = bytes_allowed / (MAVLINK_MSG_ID_PARAM_VALUE_LEN + packet_overhead());
164 
165  // when we don't have flow control we really need to keep the
166  // param download very slow, or it tends to stall
167  if (!have_flow_control() && count > 5) {
168  count = 5;
169  }
170 
171  while (_queued_parameter != nullptr && count--) {
172  AP_Param *vp;
173  float value;
174 
175  // copy the current parameter and prepare to move to the next
176  vp = _queued_parameter;
177 
178  // if the parameter can be cast to float, report it here and break out of the loop
179  value = vp->cast_to_float(_queued_parameter_type);
180 
181  char param_name[AP_MAX_NAME_SIZE];
182  vp->copy_name_token(_queued_parameter_token, param_name, sizeof(param_name), true);
183 
184  mavlink_msg_param_value_send(
185  chan,
186  param_name,
187  value,
188  mav_var_type(_queued_parameter_type),
189  _queued_parameter_count,
190  _queued_parameter_index);
191 
192  _queued_parameter = AP_Param::next_scalar(&_queued_parameter_token, &_queued_parameter_type);
193  _queued_parameter_index++;
194  }
195  _queued_parameter_send_time_ms = tnow;
196 }

287  handle a MISSION_REQUEST mavlink packet
288  */
289 void GCS_MAVLINK::handle_mission_request(AP_Mission &mission, mavlink_message_t *msg)
290 {
291  AP_Mission::Mission_Command cmd;
292 
293  if (msg->msgid == MAVLINK_MSG_ID_MISSION_REQUEST_INT) {
294  // decode
295  mavlink_mission_request_int_t packet;
296  mavlink_msg_mission_request_int_decode(msg, &packet);
297 
298  // retrieve mission from eeprom
299  if (!mission.read_cmd_from_storage(packet.seq, cmd)) {
300  goto mission_item_send_failed;
301  }
302 
303  mavlink_mission_item_int_t ret_packet;
304  memset(&ret_packet, 0, sizeof(ret_packet));
305  if (!AP_Mission::mission_cmd_to_mavlink_int(cmd, ret_packet)) {
306  goto mission_item_send_failed;
307  }
308 
309  // set packet's current field to 1 if this is the command being executed
310  if (cmd.id == (uint16_t)mission.get_current_nav_cmd().index) {
311  ret_packet.current = 1;
312  } else {
313  ret_packet.current = 0;
314  }
315 
316  // set auto continue to 1
317  ret_packet.autocontinue = 1; // 1 (true), 0 (false)
318 
319  /*
320  avoid the _send() function to save memory, as it avoids
321  the stack usage of the _send() function by using the already
322  declared ret_packet above
323  */
324  ret_packet.target_system = msg->sysid;
325  ret_packet.target_component = msg->compid;
326  ret_packet.seq = packet.seq;
327  ret_packet.command = cmd.id;
328 
329  _mav_finalize_message_chan_send(chan,
330  MAVLINK_MSG_ID_MISSION_ITEM_INT,
331  (const char *)&ret_packet,
332  MAVLINK_MSG_ID_MISSION_ITEM_MIN_LEN,
333  MAVLINK_MSG_ID_MISSION_ITEM_INT_LEN,
334  MAVLINK_MSG_ID_MISSION_ITEM_INT_CRC);
335  } else {
336  // decode
337  mavlink_mission_request_t packet;
338  mavlink_msg_mission_request_decode(msg, &packet);
339 
340  if (packet.seq != 0 && // always allow HOME to be read
341  packet.seq >= mission.num_commands()) {
342  // try to educate the GCS on the actual size of the mission:
343  mavlink_msg_mission_count_send(chan,msg->sysid, msg->compid, mission.num_commands());
344  goto mission_item_send_failed;
345  }
346 
347  // retrieve mission from eeprom
348  if (!mission.read_cmd_from_storage(packet.seq, cmd)) {
349  goto mission_item_send_failed;
350  }
351 
352  mavlink_mission_item_t ret_packet;
353  memset(&ret_packet, 0, sizeof(ret_packet));
354  if (!AP_Mission::mission_cmd_to_mavlink(cmd, ret_packet)) {
355  goto mission_item_send_failed;
356  }
357 
358  // set packet's current field to 1 if this is the command being executed
359  if (cmd.id == (uint16_t)mission.get_current_nav_cmd().index) {
360  ret_packet.current = 1;
361  } else {
362  ret_packet.current = 0;
363  }
364 
365  // set auto continue to 1
366  ret_packet.autocontinue = 1; // 1 (true), 0 (false)
367 
368  /*
369  avoid the _send() function to save memory, as it avoids
370  the stack usage of the _send() function by using the already
371  declared ret_packet above
372  */
373  ret_packet.target_system = msg->sysid;
374  ret_packet.target_component = msg->compid;
375  ret_packet.seq = packet.seq;
376  ret_packet.command = cmd.id;
377 
378  _mav_finalize_message_chan_send(chan,
379  MAVLINK_MSG_ID_MISSION_ITEM,
380  (const char *)&ret_packet,
381  MAVLINK_MSG_ID_MISSION_ITEM_MIN_LEN,
382  MAVLINK_MSG_ID_MISSION_ITEM_LEN,
383  MAVLINK_MSG_ID_MISSION_ITEM_CRC);
384  }
385 
386  return;
387 
388 mission_item_send_failed:
389  // send failure message
390  mavlink_msg_mission_ack_send(chan, msg->sysid, msg->compid, MAV_MISSION_ERROR);
391 }

681 // see if we should send a stream now. Called at 50Hz
682 bool GCS_MAVLINK::stream_trigger(enum streams stream_num)
683 {
684  if (stream_num >= NUM_STREAMS) {
685  return false;
686  }
687  float rate = (uint8_t)streamRates[stream_num].get();
688 
689  rate *= adjust_rate_for_stream_trigger(stream_num);
690 
691  if (rate <= 0) {
692  if (chan_is_streaming & (1U<<(chan-MAVLINK_COMM_0))) {
693  // if currently streaming then check if all streams are disabled
694  // to allow runtime detection of user disabling streaming
695  bool is_streaming = false;
696  for (uint8_t i=0; i<stream_num; i++) {
697  if (streamRates[stream_num] > 0) {
698  is_streaming = true;
699  }
700  }
701  if (!is_streaming) {
702  // all streams have been turned off, clear the bit flag
703  chan_is_streaming &= ~(1U<<(chan-MAVLINK_COMM_0));
704  }
705  }
706  return false;
707  } else {
708  chan_is_streaming |= (1U<<(chan-MAVLINK_COMM_0));
709  }
710 
711  if (stream_ticks[stream_num] == 0) {
712  // we're triggering now, setup the next trigger point
713  if (rate > 50) {
714  rate = 50;
715  }
716  stream_ticks[stream_num] = (50 / rate) - 1 + stream_slowdown;
717  return true;
718  }
719 
720  // count down at 50Hz
721  stream_ticks[stream_num]--;
722  return false;
723 }

766  handle an incoming mission item
767  return true if this is the last mission item, otherwise false
768  */
769 bool GCS_MAVLINK::handle_mission_item(mavlink_message_t *msg, AP_Mission &mission)
770 {
771  MAV_MISSION_RESULT result = MAV_MISSION_ACCEPTED;
772  struct AP_Mission::Mission_Command cmd = {};
773  bool mission_is_complete = false;
774  uint16_t seq=0;
775  uint16_t current = 0;
776 
777  if (msg->msgid == MAVLINK_MSG_ID_MISSION_ITEM) {
778  mavlink_mission_item_t packet;
779  mavlink_msg_mission_item_decode(msg, &packet);
780 
781  // convert mavlink packet to mission command
782  result = AP_Mission::mavlink_to_mission_cmd(packet, cmd);
783  if (result != MAV_MISSION_ACCEPTED) {
784  goto mission_ack;
785  }
786 
787  seq = packet.seq;
788  current = packet.current;
789  } else {
790  mavlink_mission_item_int_t packet;
791  mavlink_msg_mission_item_int_decode(msg, &packet);
792 
793  // convert mavlink packet to mission command
794  result = AP_Mission::mavlink_int_to_mission_cmd(packet, cmd);
795  if (result != MAV_MISSION_ACCEPTED) {
796  goto mission_ack;
797  }
798 
799  seq = packet.seq;
800  current = packet.current;
801  }
802 
803  if (current == 2) {
804  // current = 2 is a flag to tell us this is a "guided mode"
805  // waypoint and not for the mission
806  result = (handle_guided_request(cmd) ? MAV_MISSION_ACCEPTED
807  : MAV_MISSION_ERROR) ;
808 
809  // verify we received the command
810  goto mission_ack;
811  }
812 
813  if (current == 3) {
814  //current = 3 is a flag to tell us this is a alt change only
815  // add home alt if needed
816  handle_change_alt_request(cmd);
817 
818  // verify we recevied the command
819  result = MAV_MISSION_ACCEPTED;
820  goto mission_ack;
821  }
822 
823  // Check if receiving waypoints (mission upload expected)
824  if (!waypoint_receiving) {
825  result = MAV_MISSION_ERROR;
826  goto mission_ack;
827  }
828 
829  // check if this is the requested waypoint
830  if (seq != waypoint_request_i) {
831  result = MAV_MISSION_INVALID_SEQUENCE;
832  goto mission_ack;
833  }
834 
835  // sanity check for DO_JUMP command
836  if (cmd.id == MAV_CMD_DO_JUMP) {
837  if ((cmd.content.jump.target >= mission.num_commands() && cmd.content.jump.target >= waypoint_request_last) || cmd.content.jump.target == 0) {
838  result = MAV_MISSION_ERROR;
839  goto mission_ack;
840  }
841  }
842 
843  // if command index is within the existing list, replace the command
844  if (seq < mission.num_commands()) {
845  if (mission.replace_cmd(seq,cmd)) {
846  result = MAV_MISSION_ACCEPTED;
847  }else{
848  result = MAV_MISSION_ERROR;
849  goto mission_ack;
850  }
851  // if command is at the end of command list, add the command
852  } else if (seq == mission.num_commands()) {
853  if (mission.add_cmd(cmd)) {
854  result = MAV_MISSION_ACCEPTED;
855  }else{
856  result = MAV_MISSION_ERROR;
857  goto mission_ack;
858  }
859  // if beyond the end of the command list, return an error
860  } else {
861  result = MAV_MISSION_ERROR;
862  goto mission_ack;
863  }
864 
865  // update waypoint receiving state machine
866  waypoint_timelast_receive = AP_HAL::millis();
867  waypoint_request_i++;
868 
869  if (waypoint_request_i >= waypoint_request_last) {
870  mavlink_msg_mission_ack_send_buf(
871  msg,
872  chan,
873  msg->sysid,
874  msg->compid,
875  MAV_MISSION_ACCEPTED);
876 
877  send_text(MAV_SEVERITY_INFO,"Flight plan received");
878  waypoint_receiving = false;
879  mission_is_complete = true;
880  // XXX ignores waypoint radius for individual waypoints, can
881  // only set WP_RADIUS parameter
882  } else {
883  waypoint_timelast_request = AP_HAL::millis();
884  // if we have enough space, then send the next WP immediately
885  if (HAVE_PAYLOAD_SPACE(chan, MISSION_ITEM)) {
886  queued_waypoint_send();
887  } else {
888  send_message(MSG_NEXT_WAYPOINT);
889  }
890  }
891  return mission_is_complete;
892 
893 mission_ack:
894  // we are rejecting the mission/waypoint
895  mavlink_msg_mission_ack_send_buf(
896  msg,
897  chan,
898  msg->sysid,
899  msg->compid,
900  result);
901 
902  return mission_is_complete;
903 }

916 // send a message using mavlink, handling message queueing
917 void GCS_MAVLINK::send_message(enum ap_message id)
918 {
919  uint8_t i, nextid;
920 
921  if (id == MSG_HEARTBEAT) {
922  save_signing_timestamp(false);
923  }
924 
925  // see if we can send the deferred messages, if any
926  while (num_deferred_messages != 0) {
927  if (!try_send_message(deferred_messages[next_deferred_message])) {
928  break;
929  }
930  next_deferred_message++;
931  if (next_deferred_message == MSG_RETRY_DEFERRED) {
932  next_deferred_message = 0;
933  }
934  num_deferred_messages--;
935  }
936 
937  if (id == MSG_RETRY_DEFERRED) {
938  return;
939  }
940 
941  // this message id might already be deferred
942  for (i=0, nextid = next_deferred_message; i < num_deferred_messages; i++) {
943  if (deferred_messages[nextid] == id) {
944  // it's already deferred, discard
945  return;
946  }
947  nextid++;
948  if (nextid == MSG_RETRY_DEFERRED) {
949  nextid = 0;
950  }
951  }
952 
953  if (num_deferred_messages != 0 ||
954  !try_send_message(id)) {
955  // can't send it now, so defer it
956  if (num_deferred_messages == MSG_RETRY_DEFERRED) {
957  // the defer buffer is full, discard
958  return;
959  }
960  nextid = next_deferred_message + num_deferred_messages;
961  if (nextid >= MSG_RETRY_DEFERRED) {
962  nextid -= MSG_RETRY_DEFERRED;
963  }
964  deferred_messages[nextid] = id;
965  num_deferred_messages++;
966  }
967 }

1381 /*
1382  send a statustext message to specific MAVLink connections in a bitmask
1383  */
1384 void GCS::service_statustext(void)
1385 {
1386  // create bitmask of what mavlink ports we should send this text to.
1387  // note, if sending to all ports, we only need to store the bitmask for each and the string only once.
1388  // once we send over a link, clear the port but other busy ports bit may stay allowing for faster links
1389  // to clear the bit and send quickly but slower links to still store the string. Regardless of mixed
1390  // bitrates of ports, a maximum of _status_capacity strings can be buffered. Downside
1391  // is if you have a super slow link mixed with a faster port, if there are _status_capacity
1392  // strings in the slow queue then the next item can not be queued for the faster link
1393 
1394  if (_statustext_queue.empty()) {
1395  // nothing to do
1396  return;
1397  }
1398 
1399  for (uint8_t idx=0; idx<_status_capacity; ) {
1400  statustext_t *statustext = _statustext_queue[idx];
1401  if (statustext == nullptr) {
1402  break;
1403  }
1404 
1405  // try and send to all active mavlink ports listed in the statustext.bitmask
1406  for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
1407  uint8_t chan_bit = (1U<<i);
1408  // logical AND (&) to mask them together
1409  if (statustext->bitmask & chan_bit) {
1410  // something is queued on a port and that's the port index we're looped at
1411  mavlink_channel_t chan_index = (mavlink_channel_t)(MAVLINK_COMM_0+i);
1412  if (HAVE_PAYLOAD_SPACE(chan_index, STATUSTEXT)) {
1413  // we have space so send then clear that channel bit on the mask
1414  mavlink_msg_statustext_send(chan_index, statustext->msg.severity, statustext->msg.text);
1415  statustext->bitmask &= ~chan_bit;
1416  }
1417  }
1418  }
1419 
1420  if (statustext->bitmask == 0) {
1421  _statustext_queue.remove(idx);
1422  } else {
1423  // move to next index
1424  idx++;
1425  }
1426  }
1427 }
