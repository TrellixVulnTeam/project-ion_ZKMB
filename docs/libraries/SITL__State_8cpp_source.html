323 void SITL_State::_simulator_servos(SITL::Aircraft::sitl_input &input)
324 {
325  static uint32_t last_update_usec;
326 
327  /* this maps the registers used for PWM outputs. The RC
328  * driver updates these whenever it wants the channel output
329  * to change */
330  uint8_t i;
331 
332  if (last_update_usec == 0) {
333  for (i=0; i<SITL_NUM_CHANNELS; i++) {
334  pwm_output[i] = 1000;
335  }
336  if (_vehicle == ArduPlane) {
337  pwm_output[0] = pwm_output[1] = pwm_output[3] = 1500;
338  }
339  if (_vehicle == APMrover2) {
340  pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] = 1500;
341  }
342  }
343 
344  // output at chosen framerate
345  uint32_t now = AP_HAL::micros();
346  last_update_usec = now;
347 
348  // pass wind into simulators, using a wind gradient below 60m
349  float altitude = _barometer?_barometer->get_altitude():0;
350  float wind_speed = 0;
351  float wind_direction = 0;
352  if (_sitl) {
353  // The EKF does not like step inputs so this LPF keeps it happy.
354  wind_speed = _sitl->wind_speed_active = (0.95f*_sitl->wind_speed_active) + (0.05f*_sitl->wind_speed);
355  wind_direction = _sitl->wind_direction_active = (0.95f*_sitl->wind_direction_active) + (0.05f*_sitl->wind_direction);
356  }
357 
358  if (altitude < 0) {
359  altitude = 0;
360  }
361  if (altitude < 60) {
362  wind_speed *= sqrtf(MAX(altitude / 60, 0));
363  }
364 
365  input.wind.speed = wind_speed;
366  input.wind.direction = wind_direction;
367  input.wind.turbulence = _sitl?_sitl->wind_turbulance:0;
368 
369  for (i=0; i<SITL_NUM_CHANNELS; i++) {
370  if (pwm_output[i] == 0xFFFF) {
371  input.servos[i] = 0;
372  } else {
373  input.servos[i] = pwm_output[i];
374  }
375  }
376 
377  float engine_mul = _sitl?_sitl->engine_mul.get():1;
378  uint8_t engine_fail = _sitl?_sitl->engine_fail.get():0;
379  bool motors_on = false;
380 
381  if (engine_fail >= ARRAY_SIZE(input.servos)) {
382  engine_fail = 0;
383  }
384  // apply engine multiplier to motor defined by the SIM_ENGINE_FAIL parameter
385  input.servos[engine_fail] = ((input.servos[engine_fail]-1000) * engine_mul) + 1000;
386 
387  if (_vehicle == ArduPlane) {
388  motors_on = ((input.servos[2]-1000)/1000.0f) > 0;
389  } else if (_vehicle == APMrover2) {
390  motors_on = ((input.servos[2]-1500)/500.0f) != 0;
391  } else {
392  motors_on = false;
393  // run checks on each motor
394  for (i=0; i<4; i++) {
395  // check motors do not exceed their limits
396  if (input.servos[i] > 2000) input.servos[i] = 2000;
397  if (input.servos[i] < 1000) input.servos[i] = 1000;
398  // update motor_on flag
399  if ((input.servos[i]-1000)/1000.0f > 0) {
400  motors_on = true;
401  }
402  }
403  }
404  if (_sitl) {
405  _sitl->motors_on = motors_on;
406  }
407 
408  float voltage = 0;
409  _current = 0;
410 
411  if (_sitl != nullptr) {
412  if (_sitl->state.battery_voltage <= 0) {
413  // simulate simple battery setup
414  float throttle = motors_on?(input.servos[2]-1000) / 1000.0f:0;
415  // lose 0.7V at full throttle
416  voltage = _sitl->batt_voltage - 0.7f*fabsf(throttle);
417 
418  // assume 50A at full throttle
419  _current = 50.0f * fabsf(throttle);
420  } else {
421  // FDM provides voltage and current
422  voltage = _sitl->state.battery_voltage;
423  _current = _sitl->state.battery_current;
424  }
425  }
426 
427  // assume 3DR power brick
428  voltage_pin_value = ((voltage / 10.1f) / 5.0f) * 1024;
429  current_pin_value = ((_current / 17.0f) / 5.0f) * 1024;
430 }
