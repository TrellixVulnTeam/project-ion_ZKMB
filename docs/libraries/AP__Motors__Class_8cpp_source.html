81 /*
82  write to an output channel
83  */
84 void AP_Motors::rc_write(uint8_t chan, uint16_t pwm)
85 {
86  if (_motor_map_mask & (1U<<chan)) {
87  // we have a mapped motor number for this channel
88  chan = _motor_map[chan];
89  }
90  if (_pwm_type == PWM_TYPE_ONESHOT125 && (_motor_fast_mask & (1U<<chan))) {
91  // OneShot125 uses a PWM range from 125 to 250 usec
92  pwm /= 8;
93  /*
94  OneShot125 ESCs can be confused by pulses below 125 or above
95  250, making them fail the pulse type auto-detection. This
96  happens at least with BLHeli
97  */
98  if (pwm < 125) {
99  pwm = 125;
100  } else if (pwm > 250) {
101  pwm = 250;
102  }
103  }
104  hal.rcout->write(chan, pwm);
105 }

107 /*
108  set frequency of a set of channels
109  */
110 void AP_Motors::rc_set_freq(uint32_t mask, uint16_t freq_hz)
111 {
112  mask = rc_map_mask(mask);
113  if (freq_hz > 50) {
114  _motor_fast_mask |= mask;
115  }
116  hal.rcout->set_freq(mask, freq_hz);
117  if ((_pwm_type == PWM_TYPE_ONESHOT ||
118  _pwm_type == PWM_TYPE_ONESHOT125) &&
119  freq_hz > 50 &&
120  mask != 0) {
121  // tell HAL to do immediate output
122  hal.rcout->set_output_mode(AP_HAL::RCOutput::MODE_PWM_ONESHOT);
123  } else if (_pwm_type == PWM_TYPE_BRUSHED16kHz) {
124  hal.rcout->set_output_mode(AP_HAL::RCOutput::MODE_PWM_BRUSHED16KHZ);
125  }
126 }

137 /*
138  map an internal motor mask to real motor mask
139  */
140 uint32_t AP_Motors::rc_map_mask(uint32_t mask) const
141 {
142  uint32_t mask2 = 0;
143  for (uint8_t i=0; i<32; i++) {
144  uint32_t bit = 1UL<<i;
145  if (mask & bit) {
146  if ((i < AP_MOTORS_MAX_NUM_MOTORS) && (_motor_map_mask & bit)) {
147  // we have a mapped motor number for this channel
148  mask2 |= (1UL << _motor_map[i]);
149  } else {
150  mask2 |= bit;
151  }
152  }
153  }
154  return mask2;
155 }

157 // convert input in -1 to +1 range to pwm output
158 int16_t AP_Motors::calc_pwm_output_1to1(float input, const SRV_Channel *servo)
159 {
160  int16_t ret;
161 
162  input = constrain_float(input, -1.0f, 1.0f);
163 
164  if (servo->get_reversed()) {
165  input = -input;
166  }
167 
168  if (input >= 0.0f) {
169  ret = ((input * (servo->get_output_max() - servo->get_trim())) + servo->get_trim());
170  } else {
171  ret = ((input * (servo->get_trim() - servo->get_output_min())) + servo->get_trim());
172  }
173 
174  return constrain_int16(ret, servo->get_output_min(), servo->get_output_max());
175 }

177 // convert input in 0 to +1 range to pwm output
178 int16_t AP_Motors::calc_pwm_output_0to1(float input, const SRV_Channel *servo)
179 {
180  int16_t ret;
181 
182  input = constrain_float(input, 0.0f, 1.0f);
183 
184  if (servo->get_reversed()) {
185  input = 1.0f-input;
186  }
187 
188  ret = input * (servo->get_output_max() - servo->get_output_min()) + servo->get_output_min();
189 
190  return constrain_int16(ret, servo->get_output_min(), servo->get_output_max());
191 }
