134 // The attitude controller works around the concept of the desired attitude, target attitude
135 // and measured attitude. The desired attitude is the attitude input into the attitude controller
136 // that expresses where the higher level code would like the aircraft to move to. The target attitude is moved
137 // to the desired attitude with jerk, acceleration, and velocity limits. The target angular velocities are fed
138 // directly into the rate controllers. The angular error between the measured attitude and the target attitude is
139 // fed into the angle controller and the output of the angle controller summed at the input of the rate controllers.
140 // By feeding the target angular velocity directly into the rate controllers the measured and target attitudes
141 // remain very close together.
142 //
143 // All input functions below follow the same procedure
144 // 1. define the desired attitude the aircraft should attempt to achieve using the input variables
145 // 2. using the desired attitude and input variables, define the target angular velocity so that it should
146 // move the target attitude towards the desired attitude
147 // 3. if _rate_bf_ff_enabled & _use_ff_and_input_shaping are not being used then make the target attitude
148 // and target angular velocities equal to the desired attitude and desired angular velocities.
149 // 4. ensure _attitude_target_quat, _attitude_target_euler_angle, _attitude_target_euler_rate and
150 // _attitude_target_ang_vel have been defined. This ensures input modes can be changed without discontinuity.
151 // 5. attitude_controller_run_quat is then run to pass the target angular velocities to the rate controllers and
152 // integrate them into the target attitude. Any errors between the target attitude and the measured attitude are
153 // corrected by first correcting the thrust vector until the angle between the target thrust vector measured
154 // trust vector drops below 2*AC_ATTITUDE_THRUST_ERROR_ANGLE. At this point the heading is also corrected.

427 // thrust_heading_rotation_angles - calculates two ordered rotations to move the att_from_quat quaternion to the att_to_quat quaternion.
428 // The first rotation corrects the thrust vector and the second rotation corrects the heading vector.
429 void AC_AttitudeControl::thrust_heading_rotation_angles(Quaternion& att_to_quat, const Quaternion& att_from_quat, Vector3f& att_diff_angle, float& thrust_vec_dot)
430 {
431  Matrix3f att_to_rot_matrix; // earth frame to target frame
432  att_to_quat.rotation_matrix(att_to_rot_matrix);
433  Vector3f att_to_thrust_vec = att_to_rot_matrix*Vector3f(0.0f,0.0f,1.0f);
434 
435  Matrix3f att_from_rot_matrix; // earth frame to target frame
436  att_from_quat.rotation_matrix(att_from_rot_matrix);
437  Vector3f att_from_thrust_vec = att_from_rot_matrix*Vector3f(0.0f,0.0f,1.0f);
438 
439  // the cross product of the desired and target thrust vector defines the rotation vector
440  Vector3f thrust_vec_cross = att_from_thrust_vec % att_to_thrust_vec;
441 
442  // the dot product is used to calculate the angle between the target and desired thrust vectors
443  thrust_vec_dot = acosf(constrain_float(att_from_thrust_vec * att_to_thrust_vec,-1.0f,1.0f));
444 
445  // Normalize the thrust rotation vector
446  float thrust_vector_length = thrust_vec_cross.length();
447  if(is_zero(thrust_vector_length) || is_zero(thrust_vec_dot)){
448  thrust_vec_cross = Vector3f(0,0,1);
449  thrust_vec_dot = 0.0f;
450  }else{
451  thrust_vec_cross /= thrust_vector_length;
452  }
453  Quaternion thrust_vec_correction_quat;
454  thrust_vec_correction_quat.from_axis_angle(thrust_vec_cross, thrust_vec_dot);
455  thrust_vec_correction_quat = att_from_quat.inverse()*thrust_vec_correction_quat*att_from_quat;
456 
457  // calculate the remaining rotation required after thrust vector is rotated
458  Quaternion heading_quat = thrust_vec_correction_quat.inverse()*att_from_quat.inverse()*att_to_quat;
459 
460  Vector3f rotation;
461  thrust_vec_correction_quat.to_axis_angle(rotation);
462  att_diff_angle.x = rotation.x;
463  att_diff_angle.y = rotation.y;
464 
465  heading_quat.to_axis_angle(rotation);
466  att_diff_angle.z = rotation.z;
467  if(!is_zero(_p_angle_yaw.kP()) && fabsf(att_diff_angle.z) > AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS/_p_angle_yaw.kP()){
468  att_diff_angle.z = constrain_float(wrap_PI(att_diff_angle.z), -AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS/_p_angle_yaw.kP(), AC_ATTITUDE_ACCEL_Y_CONTROLLER_MAX_RADSS/_p_angle_yaw.kP());
469  heading_quat.from_axis_angle(Vector3f(0.0f,0.0f,att_diff_angle.z));
470  att_to_quat = att_from_quat*thrust_vec_correction_quat*heading_quat;
471  }
472 }

532 // Convert a 321-intrinsic euler angle derivative to an angular velocity vector
533 void AC_AttitudeControl::euler_rate_to_ang_vel(const Vector3f& euler_rad, const Vector3f& euler_rate_rads, Vector3f& ang_vel_rads)
534 {
535  float sin_theta = sinf(euler_rad.y);
536  float cos_theta = cosf(euler_rad.y);
537  float sin_phi = sinf(euler_rad.x);
538  float cos_phi = cosf(euler_rad.x);

541  ang_vel_rads.y = cos_phi * euler_rate_rads.y + sin_phi * cos_theta * euler_rate_rads.z;
542  ang_vel_rads.z = -sin_phi * euler_rate_rads.y + cos_theta * cos_phi * euler_rate_rads.z;

545 // Convert an angular velocity vector to a 321-intrinsic euler angle derivative
546 // Returns false if the vehicle is pitched 90 degrees up or down
547 bool AC_AttitudeControl::ang_vel_to_euler_rate(const Vector3f& euler_rad, const Vector3f& ang_vel_rads, Vector3f& euler_rate_rads)
548 {
549  float sin_theta = sinf(euler_rad.y);
550  float cos_theta = cosf(euler_rad.y);
551  float sin_phi = sinf(euler_rad.x);
552  float cos_phi = cosf(euler_rad.x);
553 
554  // When the vehicle pitches all the way up or all the way down, the euler angles become discontinuous. In this case, we just return false.
555  if (is_zero(cos_theta)) {
556  return false;
557  }
558 
559  euler_rate_rads.x = ang_vel_rads.x + sin_phi * (sin_theta/cos_theta) * ang_vel_rads.y + cos_phi * (sin_theta/cos_theta) * ang_vel_rads.z;
560  euler_rate_rads.y = cos_phi * ang_vel_rads.y - sin_phi * ang_vel_rads.z;
561  euler_rate_rads.z = (sin_phi / cos_theta) * ang_vel_rads.y + (cos_phi / cos_theta) * ang_vel_rads.z;
562  return true;
563 }

689 // Proportional controller with piecewise sqrt sections to constrain second derivative
690 float AC_AttitudeControl::sqrt_controller(float error, float p, float second_ord_lim)
691 {
692  if (second_ord_lim < 0.0f || is_zero(second_ord_lim) || is_zero(p)) {
693  return error*p;
694  }
695 
696  float linear_dist = second_ord_lim/sq(p);
697 
698  if (error > linear_dist) {
699  return safe_sqrt(2.0f*second_ord_lim*(error-(linear_dist/2.0f)));
700  } else if (error < -linear_dist) {
701  return -safe_sqrt(2.0f*second_ord_lim*(-error-(linear_dist/2.0f)));
702  } else {
703  return error*p;
704  }
705 }

707 // Inverse proportional controller with piecewise sqrt sections to constrain second derivative
708 float AC_AttitudeControl::stopping_point(float first_ord_mag, float p, float second_ord_lim)
709 {
710  if (second_ord_lim > 0.0f && !is_zero(second_ord_lim) && is_zero(p)) {
711  return (first_ord_mag*first_ord_mag)/(2.0f*second_ord_lim);
712  } else if ((second_ord_lim < 0.0f || is_zero(second_ord_lim)) && !is_zero(p)) {
713  return first_ord_mag/p;
714  } else if ((second_ord_lim < 0.0f || is_zero(second_ord_lim)) && is_zero(p)) {
715  return 0.0f;
716  }
717 
718  // calculate the velocity at which we switch from calculating the stopping point using a linear function to a sqrt function
719  float linear_velocity = second_ord_lim/p;
720 
721  if (fabsf(first_ord_mag) < linear_velocity) {
722  // if our current velocity is below the cross-over point we use a linear function
723  return first_ord_mag/p;
724  } else {
725  float linear_dist = second_ord_lim/sq(p);
726  float overshoot = (linear_dist/2.0f) + sq(first_ord_mag)/(2.0f*second_ord_lim);
727  if (first_ord_mag > 0){
728  return overshoot;
729  } else {
730  return -overshoot;
731  }
732  }
