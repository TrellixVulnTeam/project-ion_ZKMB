56 // handle mavlink DISTANCE_SENSOR messages
57 void AP_Proximity_MAV::handle_msg(mavlink_message_t *msg)
58 {
59  mavlink_distance_sensor_t packet;
60  mavlink_msg_distance_sensor_decode(msg, &packet);
61 
62  // store distance to appropriate sector based on orientation field
63  if (packet.orientation <= MAV_SENSOR_ROTATION_YAW_315) {
64  uint8_t sector = packet.orientation;
65  _angle[sector] = sector * 45;
66  _distance[sector] = packet.current_distance / 100.0f;
67  _distance_valid[sector] = true;
68  _distance_min = packet.min_distance / 100.0f;
69  _distance_max = packet.max_distance / 100.0f;
70  _last_update_ms = AP_HAL::millis();
71  update_boundary_for_sector(sector);
72  }
73 
74  // store upward distance
75  if (packet.orientation == MAV_SENSOR_ROTATION_PITCH_90) {
76  _distance_upward = packet.current_distance / 100.0f;
77  _last_upward_update_ms = AP_HAL::millis();
78  }
79 }
