78 void SITL::Calibration::_attitude_set(float desired_roll, float desired_pitch, float desired_yaw,
79  Vector3f& rot_accel)
80 {
81  float dt = frame_time_us * 1.0e-6f;
82 
83  Quaternion desired_q;
84  desired_q.from_euler(desired_roll, desired_pitch, desired_yaw);
85  desired_q.normalize();
86 
87  Quaternion current_q;
88  current_q.from_rotation_matrix(dcm);
89  current_q.normalize();
90 
91  Quaternion error_q = desired_q / current_q;
92 
93  Vector3f angle_differential;
94  error_q.normalize();
95  error_q.to_axis_angle(angle_differential);
96 
97  Vector3f desired_angvel = angle_differential * (1 / dt);
98  /* Provide a somewhat "smooth" transition */
99  desired_angvel *= .005f;

102  rot_accel = error * (1.0f / dt);

130 void SITL::Calibration::_calibration_poses(Vector3f& rot_accel)
131 {
132  const struct pose {
133  int16_t roll, pitch, yaw;
134  uint8_t axis;
135  } poses[] = {
136  { 0, 0, 0, 0 },
137  { 0, 0, 0, 1 },
138  { 0, 0, 0, 2 },
139  { 90, 0, 0, 1 },
140  { 0, 90, 0, 1 },
141  { 0, 180, 0, 2 },
142  { 45, 0, 0, 1 },
143  { 0, 45, 0, 2 },
144  { 0, 0, 45, 0 },
145  { 30, 0, 0, 1 },
146  { 0, 30, 0, 0 },
147  { 30, 0, 0, 1 },
148  { 0, 0, 30, 0 },
149  { 0, 0, 30, 1 },
150  { 60, 20, 0, 1 },
151  { 0, 50, 10, 0 },
152  { 0, 30, 50, 1 },
153  { 0, 30, 50, 2 },
154  };
155  const float secs_per_pose = 6;
156  const float rate = radians(360 / secs_per_pose);
157  float tnow = AP_HAL::millis() * 1.0e-3;
158  float t_in_pose = fmod(tnow, secs_per_pose);
159  uint8_t pose_num = ((unsigned)(tnow / secs_per_pose)) % ARRAY_SIZE(poses);
160  const struct pose &pose = poses[pose_num];
161 
162  // let the sensor smoothing create sensible gyro values
163  use_smoothing = true;
164  dcm.identity();
165  dcm.from_euler(radians(pose.roll), radians(pose.pitch), radians(pose.yaw));
166 
167  Vector3f axis;
168  axis[pose.axis] = 1;
169  float rot_angle = rate * t_in_pose;
170  Matrix3f r2;
171  r2.from_axis_angle(axis, rot_angle);
172  dcm = r2 * dcm;
173 
174  accel_body(0, 0, -GRAVITY_MSS);
175  accel_body = dcm.transposed() * accel_body;
176 }
