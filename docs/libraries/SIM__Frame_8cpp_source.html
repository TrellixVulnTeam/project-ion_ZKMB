143 void Frame::init(float _mass, float hover_throttle, float _terminal_velocity, float _terminal_rotation_rate)
144 {
145  /*
146  scaling from total motor power to Newtons. Allows the copter
147  to hover against gravity when each motor is at hover_throttle
148  */
149  thrust_scale = (_mass * GRAVITY_MSS) / (num_motors * hover_throttle);
150 
151  terminal_velocity = _terminal_velocity;
152  terminal_rotation_rate = _terminal_rotation_rate;
153 }

158 Frame *Frame::find_frame(const char *name)
159 {
160  for (uint8_t i=0; i < ARRAY_SIZE(supported_frames); i++) {
161  // do partial name matching to allow for frame variants
162  if (strncasecmp(name, supported_frames[i].name, strlen(supported_frames[i].name)) == 0) {
163  return &supported_frames[i];
164  }
165  }
166  return nullptr;
167 }

169 // calculate rotational and linear accelerations
170 void Frame::calculate_forces(const Aircraft &aircraft,
171  const Aircraft::sitl_input &input,
172  Vector3f &rot_accel,
173  Vector3f &body_accel)
174 {
175  Vector3f thrust; // newtons
176 
177  for (uint8_t i=0; i<num_motors; i++) {
178  Vector3f mraccel, mthrust;
179  motors[i].calculate_forces(input, thrust_scale, motor_offset, mraccel, mthrust);
180  rot_accel += mraccel;
181  thrust += mthrust;
182  }
183 
184  body_accel = thrust/aircraft.gross_mass();
185 
186  if (terminal_rotation_rate > 0) {
187  // rotational air resistance
188  const Vector3f &gyro = aircraft.get_gyro();
189  rot_accel.x -= gyro.x * radians(400.0) / terminal_rotation_rate;
190  rot_accel.y -= gyro.y * radians(400.0) / terminal_rotation_rate;
191  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
192  }
193 
194  if (terminal_velocity > 0) {
195  // air resistance
196  Vector3f air_resistance = -aircraft.get_velocity_air_ef() * (GRAVITY_MSS/terminal_velocity);
197  body_accel += aircraft.get_dcm().transposed() * air_resistance;
198  }
199 
200  // add some noise
201  const float gyro_noise = radians(0.1);
202  const float accel_noise = 0.3;
203  const float noise_scale = thrust.length() / (thrust_scale * num_motors);
204  rot_accel += Vector3f(aircraft.rand_normal(0, 1),
205  aircraft.rand_normal(0, 1),
206  aircraft.rand_normal(0, 1)) * gyro_noise * noise_scale;
207  body_accel += Vector3f(aircraft.rand_normal(0, 1),
208  aircraft.rand_normal(0, 1),
209  aircraft.rand_normal(0, 1)) * accel_noise * noise_scale;
210 }
