66 void AP_GPS_Backend::make_gps_time(uint32_t bcd_date, uint32_t bcd_milliseconds)
67 {
68  uint8_t year, mon, day, hour, min, sec;
69  uint16_t msec;
70 
71  year = bcd_date % 100;
72  mon = (bcd_date / 100) % 100;
73  day = bcd_date / 10000;
74 
75  uint32_t v = bcd_milliseconds;
76  msec = v % 1000; v /= 1000;
77  sec = v % 100; v /= 100;
78  min = v % 100; v /= 100;
79  hour = v % 100; v /= 100;
80 
81  int8_t rmon = mon - 2;
82  if (0 >= rmon) {
83  rmon += 12;
84  year -= 1;
85  }
86 
87  // get time in seconds since unix epoch
88  uint32_t ret = (year/4) - (GPS_LEAPSECONDS_MILLIS / 1000UL) + 367*rmon/12 + day;
89  ret += year*365 + 10501;
90  ret = ret*24 + hour;
91  ret = ret*60 + min;
92  ret = ret*60 + sec;
93 
94  // convert to time since GPS epoch
95  ret -= 272764785UL;
96 
97  // get GPS week and time
98  state.time_week = ret / SEC_PER_WEEK;
99  state.time_week_ms = (ret % SEC_PER_WEEK) * MSEC_PER_SEC;
100  state.time_week_ms += msec;
101 }
