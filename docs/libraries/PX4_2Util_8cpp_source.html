183 void PX4Util::set_imu_temp(float current)
184 {
185  if (!_heater.target || *_heater.target == -1) {
186  return;
187  }
188 
189  // average over temperatures to remove noise
190  _heater.count++;
191  _heater.sum += current;
192 
193  // update once a second
194  uint32_t now = AP_HAL::millis();
195  if (now - _heater.last_update_ms < 1000) {
196  return;
197  }
198  _heater.last_update_ms = now;
199 
200  current = _heater.sum / _heater.count;
201  _heater.sum = 0;
202  _heater.count = 0;
203 
204  // experimentally tweaked for Pixhawk2
205  const float kI = 0.3f;
206  const float kP = 200.0f;
207  float target = (float)(*_heater.target);
208 
209  // limit to 65 degrees to prevent damage
210  target = constrain_float(target, 0, 65);
211 
212  float err = target - current;
213 
214  _heater.integrator += kI * err;
215  _heater.integrator = constrain_float(_heater.integrator, 0, 70);
216 
217  float output = constrain_float(kP * err + _heater.integrator, 0, 100);
218 
219  // hal.console->printf("integrator %.1f out=%.1f temp=%.2f err=%.2f\n", _heater.integrator, output, current, err);
220 
221  if (_heater.fd == -1) {
222  _heater.fd = open("/dev/px4io", O_RDWR);
223  }
224  if (_heater.fd != -1) {
225  ioctl(_heater.fd, GPIO_SET_HEATER_DUTY_CYCLE, (unsigned)output);
226  }
