126 void AP_AHRS_NavEKF::update_EKF2(void)
127 {
128  if (!_ekf2_started) {
129  // wait 1 second for DCM to output a valid tilt error estimate
130  if (start_time_ms == 0) {
131  start_time_ms = AP_HAL::millis();
132  }
133  if (AP_HAL::millis() - start_time_ms > startup_delay_ms || _force_ekf) {
134  _ekf2_started = EKF2.InitialiseFilter();
135  if (_force_ekf) {
136  return;
137  }
138  }
139  }
140  if (_ekf2_started) {
141  EKF2.UpdateFilter();
142  if (active_EKF_type() == EKF_TYPE2) {
143  Vector3f eulers;
144  EKF2.getRotationBodyToNED(_dcm_matrix);
145  EKF2.getEulerAngles(-1,eulers);
146  roll = eulers.x;
147  pitch = eulers.y;
148  yaw = eulers.z;
149 
150  update_cd_values();
151  update_trig();
152 
153  // Use the primary EKF to select the primary gyro
154  int8_t primary_imu = EKF2.getPrimaryCoreIMUIndex();
155 
156  // get gyro bias for primary EKF and change sign to give gyro drift
157  // Note sign convention used by EKF is bias = measurement - truth
158  _gyro_drift.zero();
159  EKF2.getGyroBias(-1,_gyro_drift);
160  _gyro_drift = -_gyro_drift;
161 
162  // calculate corrected gyro estimate for get_gyro()
163  _gyro_estimate.zero();
164  if (primary_imu == -1) {
165  // the primary IMU is undefined so use an uncorrected default value from the INS library
166  _gyro_estimate = _ins.get_gyro();
167  } else {
168  // use the same IMU as the primary EKF and correct for gyro drift
169  _gyro_estimate = _ins.get_gyro(primary_imu) + _gyro_drift;
170  }
171 
172  // get z accel bias estimate from active EKF (this is usually for the primary IMU)
173  float abias = 0;
174  EKF2.getAccelZBias(-1,abias);
175 
176  // This EKF is currently using primary_imu, and abias applies to only that IMU
177  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
178  Vector3f accel = _ins.get_accel(i);
179  if (i == primary_imu) {
180  accel.z -= abias;
181  }
182  if (_ins.get_accel_health(i)) {
183  _accel_ef_ekf[i] = _dcm_matrix * get_rotation_autopilot_body_to_vehicle_body() * accel;
184  }
185  }
186  _accel_ef_ekf_blended = _accel_ef_ekf[primary_imu>=0?primary_imu:_ins.get_primary_accel()];
187  }
188  }
189 }

193 {

196  if (start_time_ms == 0) {

210  EKF3.getRotationBodyToNED(_dcm_matrix);

220  int8_t primary_imu = EKF3.getPrimaryCoreIMUIndex();
221 
222  // get gyro bias for primary EKF and change sign to give gyro drift
223  // Note sign convention used by EKF is bias = measurement - truth
224  _gyro_drift.zero();
225  EKF3.getGyroBias(-1,_gyro_drift);
226  _gyro_drift = -_gyro_drift;
227 
228  // calculate corrected gyro estimate for get_gyro()
229  _gyro_estimate.zero();
230  if (primary_imu == -1) {
231  // the primary IMU is undefined so use an uncorrected default value from the INS library
232  _gyro_estimate = _ins.get_gyro();
233  } else {
234  // use the same IMU as the primary EKF and correct for gyro drift
235  _gyro_estimate = _ins.get_gyro(primary_imu) + _gyro_drift;
236  }
237 
238  // get 3-axis accel bias festimates for active EKF (this is usually for the primary IMU)
239  Vector3f abias;
240  EKF3.getAccelBias(-1,abias);
241 
242  // This EKF uses the primary IMU
243  // Eventually we will run a separate instance of the EKF for each IMU and do the selection and blending of EKF outputs upstream
244  // update _accel_ef_ekf
245  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
246  Vector3f accel = _ins.get_accel(i);
247  if (i==_ins.get_primary_accel()) {
248  accel -= abias;
249  }
250  if (_ins.get_accel_health(i)) {
251  _accel_ef_ekf[i] = _dcm_matrix * accel;
252  }
253  }
254  _accel_ef_ekf_blended = _accel_ef_ekf[_ins.get_primary_accel()];
255  }
256  }

336 // dead-reckoning support
337 bool AP_AHRS_NavEKF::get_position(struct Location &loc) const
338 {
339  Vector3f ned_pos;
340  Location origin;
341  switch (active_EKF_type()) {
342  case EKF_TYPE_NONE:
343  return AP_AHRS_DCM::get_position(loc);
344 
345  case EKF_TYPE2:
346  if (EKF2.getLLH(loc) && EKF2.getPosD(-1,ned_pos.z) && EKF2.getOriginLLH(origin)) {
347  // fixup altitude using relative position from EKF origin
348  loc.alt = origin.alt - ned_pos.z*100;
349  return true;
350  }
351  break;
352 
353  case EKF_TYPE3:
354  if (EKF3.getLLH(loc) && EKF3.getPosD(-1,ned_pos.z) && EKF3.getOriginLLH(origin)) {
355  // fixup altitude using relative position from EKF origin
356  loc.alt = origin.alt - ned_pos.z*100;
357  return true;
358  }
359  break;
360 
361 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
362  case EKF_TYPE_SITL: {
363  const struct SITL::sitl_fdm &fdm = _sitl->state;
364  memset(&loc, 0, sizeof(loc));
365  loc.lat = fdm.latitude * 1e7;
366  loc.lng = fdm.longitude * 1e7;
367  loc.alt = fdm.altitude*100;
368  return true;
369  }
370 #endif
371 
372  default:
373  break;
374  }
375  return AP_AHRS_DCM::get_position(loc);
376 }

826 AP_AHRS_NavEKF::EKF_TYPE AP_AHRS_NavEKF::active_EKF_type(void) const
827 {
828  EKF_TYPE ret = EKF_TYPE_NONE;
829 
830  switch (ekf_type()) {
831  case 0:
832  return EKF_TYPE_NONE;
833 
834  case 2: {
835  // do we have an EKF2 yet?
836  if (!_ekf2_started) {
837  return EKF_TYPE_NONE;
838  }
839  if (always_use_EKF()) {
840  uint16_t ekf2_faults;
841  EKF2.getFilterFaults(-1,ekf2_faults);
842  if (ekf2_faults == 0) {
843  ret = EKF_TYPE2;
844  }
845  } else if (EKF2.healthy()) {
846  ret = EKF_TYPE2;
847  }
848  break;
849  }
850 
851  case 3: {
852  // do we have an EKF3 yet?
853  if (!_ekf3_started) {
854  return EKF_TYPE_NONE;
855  }
856  if (always_use_EKF()) {
857  uint16_t ekf3_faults;
858  EKF3.getFilterFaults(-1,ekf3_faults);
859  if (ekf3_faults == 0) {
860  ret = EKF_TYPE3;
861  }
862  } else if (EKF3.healthy()) {
863  ret = EKF_TYPE3;
864  }
865  break;
866  }
867 
868 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
869  case EKF_TYPE_SITL:
870  ret = EKF_TYPE_SITL;
871  break;
872 #endif
873  }
874 
875  /*
876  fixed wing and rover when in fly_forward mode will fall back to
877  DCM if the EKF doesn't have GPS. This is the safest option as
878  DCM is very robust. Note that we also check the filter status
879  when fly_forward is false and we are disarmed. This is to ensure
880  that the arming checks do wait for good GPS position on fixed
881  wing and rover
882  */
883  if (ret != EKF_TYPE_NONE &&
884  (_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
885  _vehicle_class == AHRS_VEHICLE_GROUND) &&
886  (_flags.fly_forward || !hal.util->get_soft_armed())) {
887  nav_filter_status filt_state;
888  if (ret == EKF_TYPE2) {
889  EKF2.getFilterStatus(-1,filt_state);
890  } else if (ret == EKF_TYPE3) {
891  EKF3.getFilterStatus(-1,filt_state);
892 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
893  } else if (ret == EKF_TYPE_SITL) {
894  get_filter_status(filt_state);
895 #endif
896  }
897  if (hal.util->get_soft_armed() && !filt_state.flags.using_gps && _gps.status() >= AP_GPS::GPS_OK_FIX_3D) {
898  // if the EKF is not fusing GPS and we have a 3D lock, then
899  // plane and rover would prefer to use the GPS position from
900  // DCM. This is a safety net while some issues with the EKF
901  // get sorted out
902  return EKF_TYPE_NONE;
903  }
904  if (hal.util->get_soft_armed() && filt_state.flags.const_pos_mode) {
905  return EKF_TYPE_NONE;
906  }
907  if (!filt_state.flags.attitude ||
908  !filt_state.flags.vert_vel ||
909  !filt_state.flags.vert_pos) {
910  return EKF_TYPE_NONE;
911  }
912  if (!filt_state.flags.horiz_vel ||
913  !filt_state.flags.horiz_pos_abs) {
914  if ((!_compass || !_compass->use_for_yaw()) &&
915  _gps.status() >= AP_GPS::GPS_OK_FIX_3D &&
916  _gps.ground_speed() < 2) {
917  /*
918  special handling for non-compass mode when sitting
919  still. The EKF may not yet have aligned its yaw. We
920  accept EKF as healthy to allow arming. Once we reach
921  speed the EKF should get yaw alignment
922  */
923  if (filt_state.flags.pred_horiz_pos_abs &&
924  filt_state.flags.pred_horiz_pos_rel) {
925  return ret;
926  }
927  }
928  return EKF_TYPE_NONE;
929  }
930  }
931  return ret;
932 }
