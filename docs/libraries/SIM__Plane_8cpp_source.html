70 float Plane::liftCoeff(float alpha) const
71 {
72  const float alpha0 = coefficient.alpha_stall;
73  const float M = coefficient.mcoeff;
74  const float c_lift_0 = coefficient.c_lift_0;
75  const float c_lift_a0 = coefficient.c_lift_a;
76 
77  double sigmoid = ( 1+exp(-M*(alpha-alpha0))+exp(M*(alpha+alpha0)) ) / (1+exp(-M*(alpha-alpha0))) / (1+exp(M*(alpha+alpha0)));
78  double linear = (1.0-sigmoid) * (c_lift_0 + c_lift_a0*alpha); //Lift at small AoA
79  double flatPlate = sigmoid*(2*copysign(1,alpha)*pow(sin(alpha),2)*cos(alpha)); //Lift beyond stall
80 
81  float result = linear+flatPlate;
82  return result;
83 }

85 float Plane::dragCoeff(float alpha) const
86 {
87  const float b = coefficient.b;
88  const float s = coefficient.s;
89  const float c_drag_p = coefficient.c_drag_p;
90  const float c_lift_0 = coefficient.c_lift_0;
91  const float c_lift_a0 = coefficient.c_lift_a;
92  const float oswald = coefficient.oswald;
93 
94  double AR = pow(b,2)/s;
95  double c_drag_a = c_drag_p + pow(c_lift_0+c_lift_a0*alpha,2)/(M_PI*oswald*AR);
96 
97  return c_drag_a;
98 }

101 Vector3f Plane::getTorque(float inputAileron, float inputElevator, float inputRudder, const Vector3f &force) const
102 {
103  const float alpha = angle_of_attack;
104  const float s = coefficient.s;
105  const float c = coefficient.c;
106  const float b = coefficient.b;
107  const float c_l_0 = coefficient.c_l_0;
108  const float c_l_b = coefficient.c_l_b;
109  const float c_l_p = coefficient.c_l_p;
110  const float c_l_r = coefficient.c_l_r;
111  const float c_l_deltaa = coefficient.c_l_deltaa;
112  const float c_l_deltar = coefficient.c_l_deltar;
113  const float c_m_0 = coefficient.c_m_0;
114  const float c_m_a = coefficient.c_m_a;
115  const float c_m_q = coefficient.c_m_q;
116  const float c_m_deltae = coefficient.c_m_deltae;
117  const float c_n_0 = coefficient.c_n_0;
118  const float c_n_b = coefficient.c_n_b;
119  const float c_n_p = coefficient.c_n_p;
120  const float c_n_r = coefficient.c_n_r;
121  const float c_n_deltaa = coefficient.c_n_deltaa;
122  const float c_n_deltar = coefficient.c_n_deltar;
123  const Vector3f &CGOffset = coefficient.CGOffset;
124 
125  float rho = air_density;
126 
127  //read angular rates
128  double p = gyro.x;
129  double q = gyro.y;
130  double r = gyro.z;
131 
132  //calculate aerodynamic torque
133  double qbar = 1.0/2.0*rho*pow(airspeed,2)*s; //Calculate dynamic pressure
134  double la, na, ma;
135  if (is_zero(airspeed))
136  {
137  la = 0;
138  ma = 0;
139  na = 0;
140  }
141  else
142  {
143  la = qbar*b*(c_l_0 + c_l_b*beta + c_l_p*b*p/(2*airspeed) + c_l_r*b*r/(2*airspeed) + c_l_deltaa*inputAileron + c_l_deltar*inputRudder);
144  ma = qbar*c*(c_m_0 + c_m_a*alpha + c_m_q*c*q/(2*airspeed) + c_m_deltae*inputElevator);
145  na = qbar*b*(c_n_0 + c_n_b*beta + c_n_p*b*p/(2*airspeed) + c_n_r*b*r/(2*airspeed) + c_n_deltaa*inputAileron + c_n_deltar*inputRudder);
146  }
147 
148 
149  // Add torque to to force misalignment with CG
150  // r x F, where r is the distance from CoG to CoL
151  la += CGOffset.y * force.z - CGOffset.z * force.y;
152  ma += -CGOffset.x * force.z + CGOffset.z * force.x;
153  na += -CGOffset.y * force.x + CGOffset.x * force.y;
154 
155  return Vector3f(la, ma, na);
156 }

158 // Force calculation function from last_letter
159 Vector3f Plane::getForce(float inputAileron, float inputElevator, float inputRudder) const
160 {
161  const float alpha = angle_of_attack;
162  const float c_drag_q = coefficient.c_drag_q;
163  const float c_lift_q = coefficient.c_lift_q;
164  const float s = coefficient.s;
165  const float c = coefficient.c;
166  const float b = coefficient.b;
167  const float c_drag_deltae = coefficient.c_drag_deltae;
168  const float c_lift_deltae = coefficient.c_lift_deltae;
169  const float c_y_0 = coefficient.c_y_0;
170  const float c_y_b = coefficient.c_y_b;
171  const float c_y_p = coefficient.c_y_p;
172  const float c_y_r = coefficient.c_y_r;
173  const float c_y_deltaa = coefficient.c_y_deltaa;
174  const float c_y_deltar = coefficient.c_y_deltar;
175 
176  float rho = air_density;
177 
178  //request lift and drag alpha-coefficients from the corresponding functions
179  double c_lift_a = liftCoeff(alpha);
180  double c_drag_a = dragCoeff(alpha);
181 
182  //convert coefficients to the body frame
183  double c_x_a = -c_drag_a*cos(alpha)+c_lift_a*sin(alpha);
184  double c_x_q = -c_drag_q*cos(alpha)+c_lift_q*sin(alpha);
185  double c_z_a = -c_drag_a*sin(alpha)-c_lift_a*cos(alpha);
186  double c_z_q = -c_drag_q*sin(alpha)-c_lift_q*cos(alpha);
187 
188  //read angular rates
189  double p = gyro.x;
190  double q = gyro.y;
191  double r = gyro.z;
192 
193  //calculate aerodynamic force
194  double qbar = 1.0/2.0*rho*pow(airspeed,2)*s; //Calculate dynamic pressure
195  double ax, ay, az;
196  if (is_zero(airspeed))
197  {
198  ax = 0;
199  ay = 0;
200  az = 0;
201  }
202  else
203  {
204  ax = qbar*(c_x_a + c_x_q*c*q/(2*airspeed) - c_drag_deltae*cos(alpha)*fabs(inputElevator) + c_lift_deltae*sin(alpha)*inputElevator);
205  // split c_x_deltae to include "abs" term
206  ay = qbar*(c_y_0 + c_y_b*beta + c_y_p*b*p/(2*airspeed) + c_y_r*b*r/(2*airspeed) + c_y_deltaa*inputAileron + c_y_deltar*inputRudder);
207  az = qbar*(c_z_a + c_z_q*c*q/(2*airspeed) - c_drag_deltae*sin(alpha)*fabs(inputElevator) - c_lift_deltae*cos(alpha)*inputElevator);
208  // split c_z_deltae to include "abs" term
209  }
210  return Vector3f(ax, ay, az);
211 }

213 void Plane::calculate_forces(const struct sitl_input &input, Vector3f &rot_accel, Vector3f &body_accel)
214 {
215  float aileron = filtered_servo_angle(input, 0);
216  float elevator = filtered_servo_angle(input, 1);
217  float rudder = filtered_servo_angle(input, 3);
218  float throttle;
219  if (reverse_elevator_rudder) {
220  elevator = -elevator;
221  rudder = -rudder;
222  }
223  if (elevons) {
224  // fake an elevon plane
225  float ch1 = aileron;
226  float ch2 = elevator;
227  aileron = (ch2-ch1)/2.0f;
228  // the minus does away with the need for RC2_REV=-1
229  elevator = -(ch2+ch1)/2.0f;
230 
231  // assume no rudder
232  rudder = 0;
233  } else if (vtail) {
234  // fake a vtail plane
235  float ch1 = elevator;
236  float ch2 = rudder;
237  // this matches VTAIL_OUTPUT==2
238  elevator = (ch2-ch1)/2.0f;
239  rudder = (ch2+ch1)/2.0f;
240  }
241 
242  if (reverse_thrust) {
243  throttle = filtered_servo_angle(input, 2);
244  } else {
245  throttle = filtered_servo_range(input, 2);
246  }
247 
248  float thrust = throttle;
249 
250  if (ice_engine) {
251  thrust = icengine.update(input);
252  }
253 
254  // calculate angle of attack
255  angle_of_attack = atan2f(velocity_air_bf.z, velocity_air_bf.x);
256  beta = atan2f(velocity_air_bf.y,velocity_air_bf.x);
257 
258  Vector3f force = getForce(aileron, elevator, rudder);
259  rot_accel = getTorque(aileron, elevator, rudder, force);
260 
261  // simulate engine RPM
262  rpm1 = thrust * 7000;
263 
264  // scale thrust to newtons
265  thrust *= thrust_scale;
266 
267  accel_body = Vector3f(thrust, 0, 0) + force;
268  accel_body /= mass;
269 
270  // add some noise
271  if (thrust_scale > 0) {
272  add_noise(fabsf(thrust) / thrust_scale);
273  }
274 
275  if (on_ground() && !tailsitter) {
276  // add some ground friction
277  Vector3f vel_body = dcm.transposed() * velocity_ef;
278  accel_body.x -= vel_body.x * 0.3f;
279  }
280 }
