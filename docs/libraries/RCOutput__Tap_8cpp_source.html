318 bool RCOutput_Tap::_uart_open()
319 {
320  // open uart
321  _uart_fd = open(UART_DEVICE_PATH, O_RDWR | O_NOCTTY | O_NONBLOCK);
322  int termios_state = -1;
323 
324  if (_uart_fd < 0) {
325  PX4_ERR("failed to open uart device!");
326  return -1;
327  }
328 
329  // set baud rate
330  int speed = 250000;
331  struct termios uart_config;
332  tcgetattr(_uart_fd, &uart_config);
333 
334  // clear ONLCR flag (which appends a CR for every LF)
335  uart_config.c_oflag &= ~ONLCR;
336 
337  // set baud rate
338  if (cfsetispeed(&uart_config, speed) < 0 || cfsetospeed(&uart_config, speed) < 0) {
339  ::fprintf(stderr, "failed to set baudrate for %s: %d\n",
340  UART_DEVICE_PATH, termios_state);
341  _uart_close();
342  return false;
343  }
344 
345  if ((termios_state = tcsetattr(_uart_fd, TCSANOW, &uart_config)) < 0) {
346  fprintf(stderr, "tcsetattr failed for %s\n", UART_DEVICE_PATH);
347  _uart_close();
348  return false;
349  }
350 
351  return true;
352 }

354 void RCOutput_Tap::init()
355 {
356  _perf_rcout = perf_alloc(PC_ELAPSED, "APM_rcout");
357 
358  if (!_uart_open()) {
359  AP_HAL::panic("Unable to open " UART_DEVICE_PATH);
360  return;
361  }
362 
363  hrt_abstime uptime_usec = hrt_absolute_time();
364  if (uptime_usec < MIN_BOOT_TIME_USEC) {
365  hal.scheduler->delay((MIN_BOOT_TIME_USEC - uptime_usec) / USEC_PER_MSEC);
366  }
367 
368  /* Issue Basic Config */
369 
370  EscPacket packet = {0xfe, sizeof(ConfigInfoBasicRequest), ESCBUS_MSG_ID_CONFIG_BASIC};
371  ConfigInfoBasicRequest &config = packet.d.reqConfigInfoBasic;
372  memset(&config, 0, sizeof(ConfigInfoBasicRequest));
373  config.maxChannelInUse = _channels_count;
374  config.controlMode = 1;
375 
376  /* Assign the id's to the ESCs to match the mux */
377  for (uint8_t phy_chan_index = 0; phy_chan_index < _channels_count; phy_chan_index++) {
378  config.channelMapTable[phy_chan_index] = device_mux_map[phy_chan_index] & ESC_CHANNEL_MAP_CHANNEL;
379  config.channelMapTable[phy_chan_index] |= (device_dir_map[phy_chan_index] << 4) & ESC_CHANNEL_MAP_RUNNING_DIRECTION;
380  }
381 
382  config.maxChannelValue = RPMMAX;
383  config.minChannelValue = RPMMIN;
384 
385  int ret = _send_packet(packet);
386  if (ret < 0) {
387  _uart_close();
388  AP_HAL::panic("Unable to send configuration to " UART_DEVICE_PATH);
389  return;
390  }
391 
392  /*
393  * To Unlock the ESC from the Power up state we need to issue 10
394  * ESCBUS_MSG_ID_RUN request with all the values 0;
395  */
396  EscPacket unlock_packet = {0xfe, _channels_count, ESCBUS_MSG_ID_RUN};
397  unlock_packet.len *= sizeof(unlock_packet.d.reqRun.value[0]);
398  memset(unlock_packet.d.bytes, 0, sizeof(packet.d.bytes));
399 
400  for (uint8_t i = 0; i < 10; i++) {
401  _send_packet(unlock_packet);
402  /* Min Packet to Packet time is 1 Ms so use 2 */
403  hal.scheduler->delay(2);
404  }
405 }

503 void RCOutput_Tap::push()
504 {
505  _corking = false;
506 
507  perf_begin(_perf_rcout);
508 
509  uint16_t out[TAP_ESC_MAX_MOTOR_NUM];
510  uint8_t motor_cnt = _channels_count;
511 
512  uint8_t motor_mapping[] = {
513  [0] = 2,
514  [1] = 1,
515  [2] = 0,
516  [3] = 3,
517  };
518 
519  // map from the RPM range to 0 - 100% duty cycle for the ESCs
520  for (uint8_t i = 0; i < motor_cnt; i++) {
521  uint16_t *val = &out[motor_mapping[i]];
522 
523  if (!(_enabled_channels & (1U << i))) {
524  *val = RPMSTOPPED;
525  } else if (_period[i] < _esc_pwm_min) {
526  *val = RPMSTOPPED;
527  } else if (_period[i] >= _esc_pwm_max) {
528  *val = RPMMAX;
529  } else {
530  float period_us = constrain_int16(_period[i], _esc_pwm_min, _esc_pwm_max);
531 
532  /*
533  * Map to [ RPMSTOPPED, RPMMAX ] range rather than
534  * [ RPMMIN, RPMMAX ] because AP_Motors will send us _esc_pwm_min
535  * when it's disarmed
536  */
537  float rpm = (period_us - _esc_pwm_min)/(_esc_pwm_max - _esc_pwm_min)
538  * (RPMMAX - RPMSTOPPED) + RPMSTOPPED;
539 
540  *val = (uint16_t) rpm;
541  }
542  }
543  for (uint8_t i = motor_cnt; i < TAP_ESC_MAX_MOTOR_NUM; i++) {
544  out[i] = RPMSTOPPED;
545  }
546 
547 /*
548  * Value packet format, little endian
549  *
550  * | 15 | 14 | 13 | 12 | 11 | 10 | 11 | 12 | 11 | 10 ...... 0 |
551  * --------------------------------------------------------------
552  * | -- | -- | -- | -- | REV | FEN | BL | GL | RL | RPM value |
553  *
554  * RPM value: [ RPMMIN, RPMMAX ]
555  * RL: LED1
556  * GL: LED2 (ESC may have only one LED that works, independent of the color)
557  * BL: LED3
558  * FEN: Feedback enable
559  * REV: Reverse direction
560  */
561 
562 // TODO: enable feedback from 1 ESC and read data back
563 #if 0
564  out[_next_channel_reply] |= RUN_FEEDBACK_ENABLE_MASK;
565 #endif
566 
567  EscPacket packet = {0xfe, _channels_count, ESCBUS_MSG_ID_RUN};
568  packet.len *= sizeof(packet.d.reqRun.value[0]);
569 
570  for (uint8_t i = 0; i < _channels_count; i++) {
571  packet.d.reqRun.value[i] = out[i] & RUN_CHANNEL_VALUE_MASK;
572  }
573 
574  int ret = _send_packet(packet);
575  if (ret < 1) {
576  debug("TX ERROR: ret: %d, errno: %d", ret, errno);
577  }
578 
579  _next_channel_reply = (_next_channel_reply + 1) % _channels_count;
580 
581  perf_end(_perf_rcout);
582 }
